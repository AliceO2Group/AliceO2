90 void Digitizer::flush(DigitContainer& digits, o2::dataformats::MCTruthContainer<MCLabel>& labels)
91 {
92  if (mPileupSignals.size() > 0) {
93  // Add the signals, all chambers are keept in the same signal container
94  SignalContainer smc = addSignalsFromPileup();
95  if (smc.size() > 0) {
96  bool status = convertSignalsToADC(smc, digits);
97  if (!status) {
98  LOG(WARN) << "TRD conversion of signals to digits failed";
99  }
100  for (const auto& iter : smc) {
101  if (iter.second.isDigit)
102  labels.addElements(labels.getIndexedSize(), iter.second.labels);
103  }
104  }
105  } else {
106  // since we don't have any pileup signals just flush the signals for each chamber
107  // we avoid flattening the array<map, ndets> to a single map
108  for (auto& smc : mSignalsMapCollection) {
109  bool status = convertSignalsToADC(smc, digits);
110  if (!status) {
111  LOG(WARN) << "TRD conversion of signals to digits failed";
112  }
113  for (const auto& iter : smc) {
114  if (iter.second.isDigit)
115  labels.addElements(labels.getIndexedSize(), iter.second.labels);
116  }
117  }
118  }
119  clearContainers();
120 }

122 SignalContainer Digitizer::addSignalsFromPileup()
123 {
124  int count = 0;
125  SignalContainer addedSignalsMap;
126  for (const auto& collection : mPileupSignals) {
127  for (int det = 0; det < MAXCHAMBER; ++det) {
128  const auto& signalMap = collection[det]; //--> a map with active pads only for this chamber
129  for (const auto& signal : signalMap) { // loop over active pads only, if there is any
130  const int& key = signal.first;
131  const SignalArray& signalArray = signal.second;
132  // check if the signal is from a previous event
133  if (signalArray.firstTBtime < mCurrentTriggerTime) {
134  if ((mCurrentTriggerTime - signalArray.firstTBtime) < BUSY_TIME) {
135  continue; // ignore the signal if it is too old.
136  }
137  // add only what's leftover from this signal
138  // 0.01 = samplingRate/1000, 1/1000 to go from ns to micro-s, the sampling rate is in 1/micro-s
139  int idx = (int)((mCurrentTriggerTime - signalArray.firstTBtime) * 0.01); // number of bins to add, from the tail
140  auto it0 = signalArray.signals.begin() + idx;
141  auto it1 = addedSignalsMap[key].signals.begin();
142  while (it0 < signalArray.signals.end()) {
143  *it1 += *it0;
144  it0++;
145  it1++;
146  }
147  } else {
148  // the signal is from a triggered event
149  int idx = (int)((signalArray.firstTBtime - mCurrentTriggerTime) * 0.01); // number of bins to add, on the tail of the final signal
150  auto it0 = signalArray.signals.begin();
151  auto it1 = addedSignalsMap[key].signals.begin() + idx;
152  while (it1 < addedSignalsMap[key].signals.end()) {
153  *it1 += *it0;
154  it0++;
155  it1++;
156  }
157  if (it0 < signalArray.signals.end()) {
158  count++; // add one more element to keep from the tail of the deque
159  }
160  }
161 
162  // do we need to set this for further processing?, it is ok for setting up the full-signal map
163  addedSignalsMap[key].firstTBtime = mCurrentTriggerTime;
164 
165  // keep the labels
166  for (const auto& label : signalArray.labels) {
167  (addedSignalsMap[key].labels).push_back(label); // maybe check if the label already exists? is that even possible?
168  }
169  } // loop over active pads in detector
170  } // loop over detectors
171  } // loop over pileup container
172  // remove all used added signals, keep those that can pileup to newer events.
173  const int numberOfElementsToPop = mPileupSignals.size() - count;
174  for (int i = 0; i < numberOfElementsToPop; ++i) {
175  mPileupSignals.pop_front();
176  }
177  return addedSignalsMap;
178 }

193 void Digitizer::process(std::vector<Hit> const& hits, DigitContainer& digits,
194  o2::dataformats::MCTruthContainer<MCLabel>& labels)
195 {
196  if (!mCalib) {
197  LOG(FATAL) << "TRD Calibration database not available";
198  }
199 
200  // Get the a hit container for all the hits in a given detector then call convertHits for a given detector (0 - 539)
201  std::array<std::vector<Hit>, MAXCHAMBER> hitsPerDetector;
202  getHitContainerPerDetector(hits, hitsPerDetector);
203 
204 #ifdef WITH_OPENMP
205  omp_set_num_threads(mNumThreads);
206 // Loop over all TRD detectors (in a parallel fashion)
207 #pragma omp parallel for schedule(dynamic)
208 #endif
209  for (int det = 0; det < MAXCHAMBER; ++det) {
210 #ifdef WITH_OPENMP
211  const int threadid = omp_get_thread_num();
212 #else
213  const int threadid = 0;
214 #endif
215  auto& signalsMap = mSignalsMapCollection[det];
216  // Jump to the next detector if the detector is
217  // switched off, not installed, etc
218  if (mCalib->isChamberNoData(det)) {
219  continue;
220  }
221  if (!mGeo->chamberInGeometry(det)) {
222  continue;
223  }
224 
225  // Go to the next detector if there are no hits
226  if (hitsPerDetector[det].size() == 0) {
227  continue;
228  }
229 
230  if (!convertHits(det, hitsPerDetector[det], signalsMap, threadid)) {
231  LOG(WARN) << "TRD conversion of hits failed for detector " << det;
232  continue; // go to the next chamber
233  }
234  }
235 }

249 bool Digitizer::convertHits(const int det, const std::vector<Hit>& hits, SignalContainer& signalMapCont, int thread)
250 {
251  //
252  // Convert the detector-wise sorted hits to detector signals
253  //
254 
255  double padSignal[mNpad];
256 
257  const double calExBDetValue = mCalib->getExB(det); // T * V/cm (check units)
258  const PadPlane* padPlane = mGeo->getPadPlane(det);
259  const int layer = mGeo->getLayer(det);
260  const float rowEndROC = padPlane->getRowEndROC();
261  const float row0 = padPlane->getRow0ROC();
262  const int nRowMax = padPlane->getNrows();
263  const int nColMax = padPlane->getNcols();
264 
265  // Loop over hits
266  for (const auto& hit : hits) {
267  const int qTotal = hit.GetCharge();
268  /*
269  Now the real local coordinate system of the ROC
270  column direction: locC
271  row direction: locR
272  time direction: locT
273  locR and locC are identical to the coordinates of the corresponding
274  volumina of the drift or amplification region.
275  locT is defined relative to the wire plane (i.e. middle of amplification
276  region), meaning locT = 0, and is negative for hits coming from the
277  drift region.
278  */
279  double locC = hit.getLocalC(); // col direction in amplification or drift volume
280  double locR = hit.getLocalR(); // row direction in amplification or drift volume
281  double locT = hit.getLocalT(); // time direction in amplification or drift volume
282  const double driftLength = -1 * locT; // The drift length in cm without diffusion
283  // Patch to take care of TR photons that are absorbed
284  // outside the chamber volume. A real fix would actually need
285  // a more clever implementation of the TR hit generation
286  if (qTotal < 0) {
287  if ((locR < rowEndROC) || (locR > row0)) {
288  continue;
289  }
290  if ((driftLength < DrMin) || (driftLength > DrMax)) {
291  continue;
292  }
293  }
294 
295  int rowE = padPlane->getPadRowNumberROC(locR);
296  if (rowE < 0) {
297  continue;
298  }
299 
300  double rowOffset = padPlane->getPadRowOffsetROC(rowE, locR);
301  double offsetTilt = padPlane->getTiltOffset(rowOffset);
302  int colE = padPlane->getPadColNumber(locC + offsetTilt);
303  if (colE < 0) {
304  continue;
305  }
306 
307  double absDriftLength = std::fabs(driftLength); // Normalized drift length
308  if (mCommonParam->ExBOn()) {
309  absDriftLength /= std::sqrt(1 / (1 + calExBDetValue * calExBDetValue));
310  }
311 
312  float driftVelocity = mCalib->getVDrift(det, colE, rowE); // The drift velocity
313  float t0 = mCalib->getT0(det, colE, rowE); // The T0 velocity
314 
315  // Loop over all created electrons
316  const int nElectrons = std::fabs(qTotal);
317  for (int el = 0; el < nElectrons; ++el) {
318  // Electron attachment
319  if (mSimParam->ElAttachOn()) {
320  if (mFlatRandomRings[thread].getNextValue() < absDriftLength * mElAttachProp) {
321  continue;
322  }
323  }
324  // scoped diffused coordinates for each electron
325  double locRd{locR}, locCd{locC}, locTd{locT};
326 
327  // Apply diffusion smearing
328  if (mSimParam->DiffusionOn()) {
329  if (!diffusion(driftVelocity, absDriftLength, calExBDetValue, locR, locC, locT, locRd, locCd, locTd, thread)) {
330  continue;
331  }
332  }
333 
334  // Apply E x B effects
335  if (mCommonParam->ExBOn()) {
336  locCd = locCd + calExBDetValue * driftLength;
337  }
338  // The electron position after diffusion and ExB in pad coordinates.
339  rowE = padPlane->getPadRowNumberROC(locRd);
340  if (rowE < 0) {
341  continue;
342  }
343  rowOffset = padPlane->getPadRowOffsetROC(rowE, locRd);
344  // The pad column (rphi-direction)
345  offsetTilt = padPlane->getTiltOffset(rowOffset);
346  colE = padPlane->getPadColNumber(locCd + offsetTilt);
347  if (colE < 0) {
348  continue;
349  }
350  const double colOffset = padPlane->getPadColOffset(colE, locCd + offsetTilt);
351  driftVelocity = mCalib->getVDrift(det, colE, rowE); // The drift velocity for the updated col and row
352  t0 = mCalib->getT0(det, colE, rowE); // The T0 velocity for the updated col and row
353  // Convert the position to drift time [mus], using either constant drift velocity or
354  // time structure of drift cells (non-isochronity, GARFIELD calculation).
355  // Also add absolute time of hits to take pile-up events into account properly
356  double driftTime;
357  if (mSimParam->TimeStructOn()) {
358  // Get z-position with respect to anode wire
359  double zz = row0 - locRd + padPlane->getAnodeWireOffset();
360  zz -= ((int)(2 * zz)) * 0.5;
361  if (zz > 0.25) {
362  zz = 0.5 - zz;
363  }
364  // Use drift time map (GARFIELD)
365  driftTime = mDriftEstimators[thread].TimeStruct(driftVelocity, 0.5 * AmWidth - 1.0 * locTd, zz) + hit.GetTime();
366  } else {
367  // Use constant drift velocity
368  driftTime = std::fabs(locTd) / driftVelocity + hit.GetTime(); // drift time in microseconds
369  }
370 
371  // Apply the gas gain including fluctuations
372  const double signal = -(mSimParam->GetGasGain()) * mLogRandomRings[thread].getNextValue();
373 
374  // Apply the pad response
375  if (mSimParam->PRFOn()) {
376  // The distance of the electron to the center of the pad in units of pad width
377  double dist = (colOffset - 0.5 * padPlane->getColSize(colE)) / padPlane->getColSize(colE);
378  // ********************************************************************************
379  // This is a fixed parametrization, i.e. not dependent on calibration values !
380  // ********************************************************************************
381  if (!(mPRF->getPRF(signal, dist, layer, padSignal))) {
382  continue;
383  }
384  } else {
385  padSignal[0] = 0;
386  padSignal[1] = signal;
387  padSignal[2] = 0;
388  }
389 
390  // The time bin (always positive), with t0 distortion
391  double timeBinIdeal = driftTime * mSamplingRate + t0;
392  // Protection
393  if (std::fabs(timeBinIdeal) > (2 * mMaxTimeBins)) {
394  timeBinIdeal = 2 * mMaxTimeBins;
395  }
396  int timeBinTruncated = ((int)timeBinIdeal);
397  // The distance of the position to the middle of the timebin
398  double timeOffset = ((float)timeBinTruncated + 0.5 - timeBinIdeal) / mSamplingRate;
399  // Sample the time response inside the drift region + additional time bins before and after.
400  // The sampling is done always in the middle of the time bin
401  const int firstTimeBin = std::max(timeBinTruncated, 0);
402  const int lastTimeBin = std::min(timeBinTruncated + mTimeBinTRFend, mMaxTimeBins);
403 
404  // loop over pads first then over timebins for better cache friendliness
405  // and less access to signalMapCont
406  for (int pad = 0; pad < mNpad; ++pad) {
407  int colPos = colE + pad - 1;
408  if (colPos < 0) {
409  continue;
410  }
411  if (colPos >= nColMax) {
412  break;
413  }
414 
415  const int key = calculateKey(det, rowE, colPos);
416  auto& currentSignalData = signalMapCont[key]; // Get the old signal or make a new one if it doesn't exist
417  auto& currentSignal = currentSignalData.signals;
418  auto& trackIds = currentSignalData.trackIds;
419  auto& labels = currentSignalData.labels;
420  currentSignalData.firstTBtime = mCurrentTriggerTime;
421  addLabel(hit, labels, trackIds); // add a label record only if needed
422 
423  // add signal with crosstalk for the non-central pads only
424  if (colPos != colE) {
425  for (int tb = firstTimeBin; tb < lastTimeBin; ++tb) {
426  const double t = (tb - timeBinTruncated) / mSamplingRate + timeOffset;
427  const double timeResponse = mSimParam->TRFOn() ? mSimParam->TimeResponse(t) : 1;
428  const double crossTalk = mSimParam->CTOn() ? mSimParam->CrossTalk(t) : 0;
429  currentSignal[tb] += padSignal[pad] * (timeResponse + crossTalk);
430  } // end of loop time bins
431  } else {
432  for (int tb = firstTimeBin; tb < lastTimeBin; ++tb) {
433  const double t = (tb - timeBinTruncated) / mSamplingRate + timeOffset;
434  const double timeResponse = mSimParam->TRFOn() ? mSimParam->TimeResponse(t) : 1;
435  currentSignal[tb] += padSignal[pad] * timeResponse;
436  } // end of loop time bins
437  }
438  } // end of loop over pads
439  } // end of loop over electrons
440  } // end of loop over hits
441  return true;
442 }

460 bool Digitizer::convertSignalsToADC(SignalContainer& signalMapCont, DigitContainer& digits, int thread)
461 {
462  //
463  // Converts the sampled electron signals to ADC values for a given chamber
464  //
465 
466  constexpr double kEl2fC = 1.602e-19 * 1.0e15; // Converts number of electrons to fC
467  double coupling = mSimParam->GetPadCoupling() * mSimParam->GetTimeCoupling(); // Coupling factor
468  double convert = kEl2fC * mSimParam->GetChipGain(); // Electronics conversion factor
469  double adcConvert = mSimParam->GetADCoutRange() / mSimParam->GetADCinRange(); // ADC conversion factor
470  double baseline = mSimParam->GetADCbaseline() / adcConvert; // The electronics baseline in mV
471  double baselineEl = baseline / convert; // The electronics baseline in electrons
472 
473  for (auto& signalMapIter : signalMapCont) {
474  const auto key = signalMapIter.first;
475  const int det = getDetectorFromKey(key);
476  const int row = getRowFromKey(key);
477  const int col = getColFromKey(key);
478  // halfchamber masking
479  int mcm = (int)(col / 18); // current group of 18 col pads
480  int halfchamberside = (mcm > 3 ? 1 : 0); // 0=Aside, 1=Bside
481 
482  // Halfchambers that are switched off, masked by mCalib
483  /* Something is wrong with isHalfChamberNoData - deactivated for now
484  if (mCalib->isHalfChamberNoData(det, halfchamberside)) {
485  continue;
486  }
487  */
488 
489  // Check whether pad is masked
490  // Bridged pads are not considered yet!!!
491  if (mCalib->isPadMasked(det, col, row) || mCalib->isPadNotConnected(det, col, row)) {
492  continue;
493  }
494 
495  float padgain = mCalib->getPadGainFactor(det, row, col); // The gain factor
496  if (padgain <= 0) {
497  LOG(FATAL) << "Not a valid gain " << padgain << ", " << det << ", " << col << ", " << row;
498  }
499 
500  signalMapIter.second.isDigit = true; // flag the signal as digit
501  // Loop over the all timebins in the ADC array
502  const auto& signalArray = signalMapIter.second.signals;
503  ArrayADC adcs{};
504  for (int tb = 0; tb < mMaxTimeBinsTRAP; ++tb) {
505  float signalAmp = (float)signalArray[tb]; // The signal amplitude
506  signalAmp *= coupling; // Pad and time coupling
507  signalAmp *= padgain; // Gain factors
508  // Add the noise, starting from minus ADC baseline in electrons
509  signalAmp = std::max((double)drawGaus(mGausRandomRings[thread], signalAmp, mSimParam->GetNoise()), -baselineEl);
510  signalAmp *= convert; // Convert to mV
511  signalAmp += baseline; // Add ADC baseline in mV
512  // Convert to ADC counts
513  // Set the overflow-bit fADCoutRange if the signal is larger than fADCinRange
514  ADC_t adc = 0;
515  if (signalAmp >= mSimParam->GetADCinRange()) {
516  adc = ((ADC_t)mSimParam->GetADCoutRange());
517  } else {
518  adc = std::lround(signalAmp * adcConvert);
519  }
520  // update the adc array value
521  adcs[tb] = adc;
522  } // loop over timebins
523  // Convert the map to digits here, and push them to the container
524  digits.emplace_back(det, row, col, adcs);
525  } // loop over digits
526  return true;
527 }

529 bool Digitizer::diffusion(float vdrift, float absdriftlength, float exbvalue,
530  float lRow0, float lCol0, float lTime0,
531  double& lRow, double& lCol, double& lTime, int thread)
532 {
533  //
534  // Applies the diffusion smearing to the position of a single electron.
535  // Depends on absolute drift length.
536  //
537  float diffL = 0.0;
538  float diffT = 0.0;
539  if (mDriftEstimators[thread].GetDiffCoeff(diffL, diffT, vdrift)) {
540  float driftSqrt = std::sqrt(absdriftlength);
541  float sigmaT = driftSqrt * diffT;
542  float sigmaL = driftSqrt * diffL;
543  lRow = drawGaus(mGausRandomRings[thread], lRow0, sigmaT);
544  if (mCommonParam->ExBOn()) {
545  const float exbfactor = 1.f / (1.f + exbvalue * exbvalue);
546  lCol = drawGaus(mGausRandomRings[thread], lCol0, sigmaT * exbfactor);
547  lTime = drawGaus(mGausRandomRings[thread], lTime0, sigmaL * exbfactor);
548  } else {
549  lCol = drawGaus(mGausRandomRings[thread], lCol0, sigmaT);
550  lTime = drawGaus(mGausRandomRings[thread], lTime0, sigmaL);
551  }
552  return true;
553  } else {
554  return false;
555  }
