220 size_t current_time_with_ms()
221 {
222  long ms; // Milliseconds
223  time_t s; // Seconds
224  struct timespec spec;
225 
226  clock_gettime(CLOCK_REALTIME, &spec);
227 
228  s = spec.tv_sec;
229  ms = round(spec.tv_nsec / 1.0e6); // Convert nanoseconds to milliseconds
230  if (ms > 999) {
231  s++;
232  ms = 0;
233  }
234  return s * 1000 + ms;
235 }

242 void getChildData(int infd, DeviceInfo& outinfo)
243 {

246  // NOTE: do not quite understand read ends up blocking if I read more than
247  // once. Oh well... Good enough for now.
248  O2_SIGNPOST_START(DriverStatus::ID, DriverStatus::BYTES_READ, outinfo.pid, infd, 0);
249  while (true) {
250  bytes_read = read(infd, buffer, 1024 * 16);
251  if (bytes_read == 0) {
252  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
253  return;
254  }
255  if (bytes_read < 0) {
256  switch (errno) {
257  case EWOULDBLOCK:
258  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
259  return;
260  default:
261  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
262  return;
263  }
264  }
265  assert(bytes_read > 0);
266  outinfo.unprinted.append(buffer, bytes_read);
267  }
268 }

331 void createPipes(int* pipes)
332 {
333  auto p = pipe(pipes);
334 
335  if (p == -1) {
336  std::cerr << "Unable to create PIPE: ";
337  switch (errno) {
338  case EFAULT:
339  assert(false && "EFAULT while reading from pipe");
340  break;
341  case EMFILE:
342  std::cerr << "Too many active descriptors";
343  break;
344  case ENFILE:
345  std::cerr << "System file table is full";
346  break;
347  default:
348  std::cerr << "Unknown PIPE" << std::endl;
349  };
350  // Kill immediately both the parent and all the children
351  kill(-1 * getpid(), SIGKILL);
352  }
353 }

649 void spawnDevice(std::string const& forwardedStdin,
650  DeviceSpec const& spec,
651  DriverInfo& driverInfo,
652  DeviceControl& control,
653  DeviceExecution& execution,
654  std::vector<DeviceInfo>& deviceInfos,
655  ServiceRegistry& serviceRegistry,
656  boost::program_options::variables_map& varmap,
657  uv_loop_t* loop,
658  std::vector<uv_poll_t*> handles,
659  unsigned parentCPU,
660  unsigned parentNode)
661 {
662  int childstdin[2];
663  int childstdout[2];
664  int childstderr[2];
665 
666  createPipes(childstdin);
667  createPipes(childstdout);
668  createPipes(childstderr);
669 
670  // FIXME: this might not work when more than one DPL driver on the same
671  // machine. Hopefully we do not care.
672  // Not how the first port is actually used to broadcast clients.
673  driverInfo.tracyPort++;
674 
675  for (auto& service : spec.services) {
676  if (service.preFork != nullptr) {
677  service.preFork(serviceRegistry, varmap);
678  }
679  }
680  // If we have a framework id, it means we have already been respawned
681  // and that we are in a child. If not, we need to fork and re-exec, adding
682  // the framework-id as one of the options.
683  pid_t id = 0;
684  id = fork();
685  // We are the child: prepare options and reexec.

842  for (size_t di = 0, de = infos.size(); di < de; ++di) {
843  DeviceInfo& info = infos[di];
844  DeviceControl& control = controls[di];
845  DeviceMetricsInfo& metrics = metricsInfos[di];
846  assert(specs.size() == infos.size());
847  DeviceSpec const& spec = specs[di];

850  continue;

856  size_t pos = 0;
857  info.history.resize(info.historySize);
858  info.historyLevel.resize(info.historySize);

861  Metric2DViewIndex::getUpdater({&info.dataRelayerViewIndex,

866  updateMetricsViews(name, metric, value, metricIndex);
867  hasNewMetric = true;
868  };
869 
870  while ((pos = s.find(delimiter)) != std::string::npos) {
871  std::string token{s.substr(0, pos)};
872  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
873 
874  // Check if the token is a metric from SimpleMetricsService
875  // if yes, we do not print it out and simply store it to be displayed
876  // in the GUI.
877  // Then we check if it is part of our Poor man control system
878  // if yes, we execute the associated command.
879  if (DeviceMetricsHelper::parseMetric(token, metricMatch)) {
880  // We use this callback to cache which metrics are needed to provide a
881  // the DataRelayer view.
882  DeviceMetricsHelper::processMetric(metricMatch, metrics, newMetricCallback);
883  result.didProcessMetric = true;
884  } else if (logLevel == LogParsingHelpers::LogLevel::Info && ControlServiceHelpers::parseControl(token, match)) {
885  ControlServiceHelpers::processCommand(infos, info.pid, match[1].str(), match[2].str());
886  result.didProcessControl = true;
887  } else if (logLevel == LogParsingHelpers::LogLevel::Info && DeviceConfigHelper::parseConfig(token, configMatch)) {
888  DeviceConfigHelper::processConfig(configMatch, info);
889  result.didProcessConfig = true;
890  } else if (!control.quiet && (token.find(control.logFilter) != std::string::npos) &&
891  logLevel >= control.logLevel) {
892  assert(info.historyPos >= 0);
893  assert(info.historyPos < info.history.size());
894  info.history[info.historyPos] = token;
895  info.historyLevel[info.historyPos] = logLevel;
896  info.historyPos = (info.historyPos + 1) % info.history.size();

898  result.didProcessLog = true;
899  }
900  // We keep track of the maximum log error a
901  // device has seen.
902  bool maxLogLevelIncreased = false;
903  if (logLevel > info.maxLogLevel && logLevel > LogParsingHelpers::LogLevel::Info &&
904  logLevel != LogParsingHelpers::LogLevel::Unknown) {
905  info.maxLogLevel = logLevel;
906  maxLogLevelIncreased = true;
907  }
908  if (logLevel >= driverInfo.minFailureLevel) {
909  info.lastError = token;
910  if (info.firstSevereError.empty() || maxLogLevelIncreased) {
911  info.firstSevereError = token;
912  }
913  }
914  s.remove_prefix(pos + delimiter.length());
915  }
916  size_t oldSize = info.unprinted.size();
917  info.unprinted = std::string(s);
918  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_PROCESSED, oldSize - info.unprinted.size(), 0, 0);
919  }
920  result.hasNewMetric = hasNewMetric;
921  if (hasNewMetric) {
922  hasNewMetric = false;
923  updateMetricsNames(driverInfo, metricsInfos);
924  }
925  return result;
926 }

1265  {
1266  auto currentTime = uv_hrtime();
1267  uint64_t diff = (currentTime - driverInfo.startTime) / 1000000000LL;
1268  if ((graceful_exit == false) && (driverInfo.timeout > 0) && (diff > driverInfo.timeout)) {
1269  LOG(INFO) << "Timout ellapsed. Requesting to quit.";
1270  graceful_exit = true;
1271  }
1272  }
1273  // Move to exit loop if sigint was sent we execute this only once.
1274  if (graceful_exit == true && driverInfo.sigintRequested == false) {
1275  driverInfo.sigintRequested = true;
1276  driverInfo.states.resize(0);
1277  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1278  }
1279  // If one of the children dies and sigint was not requested
1280  // we should decide what to do.
1281  if (sigchld_requested == true && driverInfo.sigchldRequested == false) {
1282  driverInfo.sigchldRequested = true;
1283  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
1284  }
1285  if (driverInfo.states.empty() == false) {
1286  previous = current;
1287  current = driverInfo.states.back();
1288  } else {
1289  current = DriverState::UNKNOWN;
1290  }
1291  driverInfo.states.pop_back();
1292  switch (current) {
1293  case DriverState::INIT:
1294  LOGP(info, "Initialising O2 Data Processing Layer. Driver PID: {}.", getpid());
1295 
1296  // Install signal handler for quitting children.
1297  driverInfo.sa_handle_child.sa_handler = &handle_sigchld;
1298  sigemptyset(&driverInfo.sa_handle_child.sa_mask);
1299  driverInfo.sa_handle_child.sa_flags = SA_RESTART | SA_NOCLDSTOP;
1300  if (sigaction(SIGCHLD, &driverInfo.sa_handle_child, nullptr) == -1) {
1301  perror(nullptr);
1302  exit(1);
1303  }
1304 
1307  if (driverInfo.noSHMCleanup) {
1308  LOGP(warning, "Not cleaning up shared memory.");
1309  } else {
1310  cleanupSHM(driverInfo.uniqueWorkflowId);
1311  }
1316  driverInfo.states.push_back(DriverState::RUNNING);
1317  // driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
1318  LOG(INFO) << "O2 Data Processing Layer initialised. We brake for nobody.";
1319 #ifdef NDEBUG
1320  LOGF(info, "Optimised build. O2DEBUG / LOG(DEBUG) / LOGF(DEBUG) / assert statement will not be shown.");
1321 #endif
1322  break;
1323  case DriverState::IMPORT_CURRENT_WORKFLOW:
1324  // This state is needed to fill the metadata structure
1325  // which contains how to run the current workflow
1326  dataProcessorInfos = previousDataProcessorInfos;
1327  for (auto const& device : deviceSpecs) {
1328  auto exists = std::find_if(dataProcessorInfos.begin(),
1329  dataProcessorInfos.end(),
1330  [id = device.id](DataProcessorInfo const& info) -> bool { return info.name == id; });
1331  if (exists != dataProcessorInfos.end()) {
1332  continue;
1333  }
1334  std::vector<std::string> channels;
1335  for (auto channel : device.inputChannels) {
1336  channels.push_back(channel.name);
1337  }
1338  for (auto channel : device.outputChannels) {
1339  channels.push_back(channel.name);
1340  }
1341  dataProcessorInfos.push_back(
1342  DataProcessorInfo{
1343  device.id,
1344  workflowInfo.executable,
1345  workflowInfo.args,
1346  workflowInfo.options,
1347  channels});
1348  }
1349  break;
1350  case DriverState::MATERIALISE_WORKFLOW:
1351  try {
1352  auto workflowState = WorkflowHelpers::verifyWorkflow(workflow);
1353  if (driverInfo.batch == true && workflowState == WorkflowParsingState::Empty) {
1354  throw runtime_error("Empty workflow provided while running in batch mode.");
1355  }
1356  DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(workflow,
1357  driverInfo.channelPolicies,
1358  driverInfo.completionPolicies,
1359  driverInfo.dispatchPolicies,
1360  deviceSpecs,
1361  *resourceManager,
1362  driverInfo.uniqueWorkflowId,
1363  !varmap["no-IPC"].as<bool>(),
1364  driverInfo.resourcesMonitoringInterval,
1365  varmap["channel-prefix"].as<std::string>());
1366  metricProcessingCallbacks.clear();
1367  for (auto& device : deviceSpecs) {
1368  for (auto& service : device.services) {
1369  if (service.metricHandling) {
1370  metricProcessingCallbacks.push_back(service.metricHandling);
1371  }
1372  }
1373  }
1374  preScheduleCallbacks.clear();
1375  for (auto& device : deviceSpecs) {
1376  for (auto& service : device.services) {
1377  if (service.preSchedule) {
1378  preScheduleCallbacks.push_back(service.preSchedule);
1379  }
1380  }
1381  }
1382  postScheduleCallbacks.clear();
1383  for (auto& device : deviceSpecs) {
1384  for (auto& service : device.services) {
1385  if (service.postSchedule) {
1386  postScheduleCallbacks.push_back(service.postSchedule);
1387  }
1388  }
1389  }
1390 
1391  // This should expand nodes so that we can build a consistent DAG.
1392  } catch (std::runtime_error& e) {
1393  std::cerr << "Invalid workflow: " << e.what() << std::endl;
1394  return 1;
1395  } catch (o2::framework::RuntimeErrorRef ref) {
1396  auto& err = o2::framework::error_from_ref(ref);
1397 #ifdef DPL_ENABLE_BACKTRACE
1398  backtrace_symbols_fd(err.backtrace, err.maxBacktrace, STDERR_FILENO);
1399 #endif
1400  std::cerr << "Invalid workflow: " << err.what << std::endl;
1401  return 1;
1402  } catch (...) {
1403  std::cerr << "Unknown error while materialising workflow";
1404  return 1;
1405  }
1406  break;
1407  case DriverState::DO_CHILD:
1408  // We do not start the process if by default we are stopped.
1409  if (driverControl.defaultStopped) {
1410  kill(getpid(), SIGSTOP);
1411  }
1412  for (auto& spec : deviceSpecs) {
1413  if (spec.id == frameworkId) {
1414  return doChild(driverInfo.argc, driverInfo.argv,
1415  serviceRegistry, spec,
1416  driverInfo.errorPolicy,
1417  driverInfo.defaultDriverClient,
1418  loop);
1419  }
1420  }
1421  {
1422  std::ostringstream ss;
1423  for (auto& processor : workflow) {
1424  ss << " - " << processor.name << "\n";
1425  }
1426  for (auto& spec : deviceSpecs) {
1427  ss << " - " << spec.name << "(" << spec.id << ")"
1428  << "\n";
1429  }
1430  LOG(ERROR) << "Unable to find component with id "
1431  << frameworkId << ". Available options:\n"
1432  << ss.str();
1433  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1434  }
1435  break;
1436  case DriverState::REDEPLOY_GUI:
1437  // The callback for the GUI needs to be recalculated every time
1438  // the deployed configuration changes, e.g. a new device
1439  // has been added to the topology.
1440  // We need to recreate the GUI callback every time we reschedule
1441  // because getGUIDebugger actually recreates the GUI state.
1442  if (window) {
1443  uv_timer_stop(&gui_timer);
1444  guiContext.callback = debugGUI->getGUIDebugger(infos, deviceSpecs, dataProcessorInfos, metricsInfos, driverInfo, controls, driverControl);
1445  guiContext.window = window;
1446  gui_timer.data = &guiContext;
1447  uv_timer_start(&gui_timer, gui_callback, 0, 20);
1448  guiDeployedOnce = true;
1449  }
1450  break;
1451  case DriverState::MERGE_CONFIGS: {
1452  try {
1453  controls.resize(deviceSpecs.size());
1454  deviceExecutions.resize(deviceSpecs.size());
1455 
1456  // Options which should be uniform across all
1457  // teh subworkflow invokations.
1458  const auto uniformOptions = {
1459  "--aod-file",
1460  "--aod-memory-rate-limit",
1461  "--aod-writer-json",
1462  "--aod-writer-ntfmerge",
1463  "--aod-writer-resfile",
1464  "--aod-writer-resmode",
1465  "--aod-writer-keep",
1466  "--driver-client-backend",
1467  "--fairmq-ipc-prefix",
1468  "--readers",
1469  "--resources-monitoring",
1470  "--time-limit",
1471  };
1472 
1473  for (auto& option : uniformOptions) {
1474  DeviceSpecHelpers::reworkHomogeneousOption(dataProcessorInfos, option, nullptr);
1475  }
1476 
1477  DeviceSpecHelpers::reworkShmSegmentSize(dataProcessorInfos);
1478  DeviceSpecHelpers::prepareArguments(driverControl.defaultQuiet,
1479  driverControl.defaultStopped,
1480  driverInfo.port,
1481  dataProcessorInfos,
1482  deviceSpecs,
1483  deviceExecutions, controls,
1484  driverInfo.uniqueWorkflowId);
1485  } catch (o2::framework::RuntimeErrorRef& ref) {
1486  auto& err = o2::framework::error_from_ref(ref);
1487  LOG(ERROR) << "Unable to merge configurations: " << err.what;
1488 #ifdef DPL_ENABLE_BACKTRACE
1489  std::cerr << "\nStacktrace follows:\n\n";
1490  backtrace_symbols_fd(err.backtrace, err.maxBacktrace, STDERR_FILENO);
1491 #endif
1492  return 1;
1493  }
1494  } break;
1495  case DriverState::SCHEDULE: {
1496  // FIXME: for the moment modifying the topology means we rebuild completely
1497  // all the devices and we restart them. This is also what DDS does at
1498  // a larger scale. In principle one could try to do a delta and only
1499  // restart the data processors which need to be restarted.
1500  LOG(INFO) << "Redeployment of configuration asked.";
1501  std::ostringstream forwardedStdin;
1502  WorkflowSerializationHelpers::dump(forwardedStdin, workflow, dataProcessorInfos);
1503  infos.reserve(deviceSpecs.size());
1504 
1505  // This is guaranteed to be a single CPU.
1506  unsigned parentCPU = -1;
1507  unsigned parentNode = -1;
1508 #if defined(__linux__) && __has_include(<sched.h>)
1509  parentCPU = sched_getcpu();
1510 #elif __has_include(<linux/getcpu.h>)
1511  getcpu(&parentCPU, &parentNode, nullptr);
1512 #elif __has_include(<cpuid.h>) && (__x86_64__ || __i386__)
1513  // FIXME: this is a last resort as it is apparently buggy
1514  // on some Intel CPUs.
1515  GETCPU(parentCPU);
1516 #endif
1517  for (auto& callback : preScheduleCallbacks) {
1518  callback(serviceRegistry, varmap);
1519  }
1520  for (size_t di = 0; di < deviceSpecs.size(); ++di) {
1521  if (deviceSpecs[di].resource.hostname != driverInfo.deployHostname) {
1522  spawnRemoteDevice(forwardedStdin.str(),
1523  deviceSpecs[di], controls[di], deviceExecutions[di], infos);
1524  } else {
1525  spawnDevice(forwardedStdin.str(),
1526  deviceSpecs[di], driverInfo,
1527  controls[di], deviceExecutions[di], infos,
1528  serviceRegistry, varmap, loop, pollHandles, parentCPU, parentNode);
1529  }
1530  }
1531  for (auto& callback : postScheduleCallbacks) {
1532  callback(serviceRegistry, varmap);
1533  }
1534  assert(infos.empty() == false);
1535  LOG(INFO) << "Redeployment of configuration done.";
1536  } break;
1537  case DriverState::RUNNING:
1538  // Run any pending libUV event loop, block if
1539  // any, so that we do not consume CPU time when the driver is
1540  // idle.
1541  uv_run(loop, once ? UV_RUN_ONCE : UV_RUN_NOWAIT);
1542  once = true;
1543  // Calculate what we should do next and eventually
1544  // show the GUI
1545  if (guiQuitRequested ||
1546  (driverInfo.terminationPolicy == TerminationPolicy::QUIT && (checkIfCanExit(infos) == true))) {
1547  // Something requested to quit. This can be a user
1548  // interaction with the GUI or (if --completion-policy=quit)
1549  // it could mean that the workflow does not have anything else to do.
1550  // Let's update the GUI one more time and then EXIT.
1551  LOG(INFO) << "Quitting";
1552  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1553  } else if (infos.size() != deviceSpecs.size()) {
1554  // If the number of deviceSpecs is different from
1555  // the DeviceInfos it means the speicification
1556  // does not match what is running, so we need to do
1557  // further scheduling.
1558  driverInfo.states.push_back(DriverState::RUNNING);
1559  driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
1560  driverInfo.states.push_back(DriverState::SCHEDULE);
1561  driverInfo.states.push_back(DriverState::MERGE_CONFIGS);
1562  } else if (deviceSpecs.empty() && driverInfo.batch == true) {
1563  LOG(INFO) << "No device resulting from the workflow. Quitting.";
1564  // If there are no deviceSpecs, we exit.
1565  driverInfo.states.push_back(DriverState::EXIT);
1566  } else if (deviceSpecs.empty() && driverInfo.batch == false && !guiDeployedOnce) {
1567  // In case of an empty workflow, we need to deploy the GUI at least once.
1568  driverInfo.states.push_back(DriverState::RUNNING);
1569  driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
1570  } else {
1571  driverInfo.states.push_back(DriverState::RUNNING);
1572  }
1573  {
1574  uint64_t inputProcessingStart = uv_hrtime();
1575  auto inputProcessingLatency = inputProcessingStart - inputProcessingLast;
1576  auto outputProcessing = processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
1577  if (outputProcessing.didProcessMetric) {
1578  size_t timestamp = current_time_with_ms();
1579  for (auto& callback : metricProcessingCallbacks) {
1580  callback(serviceRegistry, metricsInfos, deviceSpecs, infos, driverInfo.metrics, timestamp);
1581  }
1582  for (auto& metricsInfo : metricsInfos) {
1583  std::fill(metricsInfo.changed.begin(), metricsInfo.changed.end(), false);
1584  }
1585  }
1586  auto inputProcessingEnd = uv_hrtime();
1587  driverInfo.inputProcessingCost = (inputProcessingEnd - inputProcessingStart) / 1000000;
1588  driverInfo.inputProcessingLatency = (inputProcessingLatency) / 1000000;
1589  inputProcessingLast = inputProcessingStart;
1590  }
1591  break;
1592  case DriverState::QUIT_REQUESTED:
1593  LOG(INFO) << "QUIT_REQUESTED";
1594  guiQuitRequested = true;
1595  // We send SIGCONT to make sure stopped children are resumed
1596  killChildren(infos, SIGCONT);
1597  // We send SIGTERM to make sure we do the STOP transition in FairMQ
1598  killChildren(infos, SIGTERM);
1599  // We have a timer to send SIGUSR1 to make sure we advance all devices
1600  // in a timely manner.
1601  force_step_timer.data = &infos;
1602  uv_timer_start(&force_step_timer, single_step_callback, 0, 300);
1603  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
1604  break;
1605  case DriverState::HANDLE_CHILDREN: {
1606  // Run any pending libUV event loop, block if
1607  // any, so that we do not consume CPU time when the driver is
1608  // idle.
1609  uv_run(loop, once ? UV_RUN_ONCE : UV_RUN_NOWAIT);
1610  once = true;
1611  // I allow queueing of more sigchld only when
1612  // I process the previous call
1613  if (forceful_exit == true) {
1614  static bool forcefulExitMessage = true;
1615  if (forcefulExitMessage) {
1616  LOG(INFO) << "Forceful exit requested.";
1617  forcefulExitMessage = false;
1618  }
1619  killChildren(infos, SIGCONT);
1620  killChildren(infos, SIGKILL);
1621  }
1622  sigchld_requested = false;
1623  driverInfo.sigchldRequested = false;
1624  auto outputProcessing = processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
1625  if (outputProcessing.didProcessMetric) {
1626  size_t timestamp = current_time_with_ms();
1627  for (auto& callback : metricProcessingCallbacks) {
1628  callback(serviceRegistry, metricsInfos, deviceSpecs, infos, driverInfo.metrics, timestamp);
1629  }
1630  }
1631  hasError = processSigChild(infos);
1632  if (areAllChildrenGone(infos) == true &&
1633  (guiQuitRequested || (checkIfCanExit(infos) == true) || graceful_exit)) {
1634  // We move to the exit, regardless of where we were
1635  driverInfo.states.resize(0);
1636  driverInfo.states.push_back(DriverState::EXIT);
1637  } else if (areAllChildrenGone(infos) == false &&
1638  (guiQuitRequested || checkIfCanExit(infos) == true || graceful_exit)) {
1639  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
1640  } else if (hasError && driverInfo.errorPolicy == TerminationPolicy::QUIT &&
1641  !(guiQuitRequested || checkIfCanExit(infos) == true || graceful_exit)) {
1642  graceful_exit = 1;
1643  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1644  } else {
1645  }
1646  } break;
1647  case DriverState::EXIT: {
1648  if (ResourcesMonitoringHelper::isResourcesMonitoringEnabled(driverInfo.resourcesMonitoringInterval)) {
1649  LOG(INFO) << "Dumping performance metrics to performanceMetrics.json file";
1650  auto performanceMetrics = o2::monitoring::ProcessMonitor::getAvailableMetricsNames();
1651  performanceMetrics.push_back("arrow-bytes-delta");
1652  performanceMetrics.push_back("aod-bytes-read-uncompressed");
1653  performanceMetrics.push_back("aod-bytes-read-compressed");
1654  performanceMetrics.push_back("aod-file-read-info");
1655  ResourcesMonitoringHelper::dumpMetricsToJSON(metricsInfos, driverInfo.metrics, deviceSpecs, performanceMetrics);
1656  }
1657  // This is a clean exit. Before we do so, if required,
1658  // we dump the configuration of all the devices so that
1659  // we can reuse it. Notice we do not dump anything if
1660  // the workflow was not really run.
1661  // NOTE: is this really what we want? should we run
1662  // SCHEDULE and dump the full configuration as well?
1663  if (infos.empty()) {
1664  return 0;
1665  }
1666  boost::property_tree::ptree finalConfig;
1667  assert(infos.size() == deviceSpecs.size());
1668  for (size_t di = 0; di < infos.size(); ++di) {
1669  auto info = infos[di];
1670  auto spec = deviceSpecs[di];
1671  finalConfig.put_child(spec.name, info.currentConfig);
1672  }
1673  LOG(INFO) << "Dumping used configuration in dpl-config.json";
1674  boost::property_tree::write_json("dpl-config.json", finalConfig);
1675  if (driverInfo.noSHMCleanup) {
1676  LOGP(warning, "Not cleaning up shared memory.");
1677  } else {
1678  cleanupSHM(driverInfo.uniqueWorkflowId);
1679  }
1680  return calculateExitCode(driverInfo, deviceSpecs, infos);
1681  }
1682  case DriverState::PERFORM_CALLBACKS:
1683  for (auto& callback : driverControl.callbacks) {
1684  callback(workflow, deviceSpecs, deviceExecutions, dataProcessorInfos);
1685  }
1686  driverControl.callbacks.clear();
1687  break;
1688  default:
1689  LOG(ERROR) << "Driver transitioned in an unknown state("
1690  << "current: " << (int)current
1691  << ", previous: " << (int)previous
1692  << "). Shutting down.";
1693  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
1694  }
1695  }
1696 }

1743 void overrideCloning(ConfigContext& ctx, WorkflowSpec& workflow)
1744 {
1745  struct CloningSpec {
1746  std::string templateMatcher;
1747  std::string cloneName;
1748  };
1749  auto s = ctx.options().get<std::string>("clone");
1750  std::vector<CloningSpec> specs;
1751  std::string delimiter = ",";
1752 
1753  size_t pos = 0;
1754  while (s.empty() == false) {
1755  auto newPos = s.find(delimiter);
1756  auto token = s.substr(0, newPos);
1757  auto split = token.find(":");
1758  if (split == std::string::npos) {
1759  throw std::runtime_error("bad clone definition. Syntax <template-processor>:<clone-name>");
1760  }
1761  auto key = token.substr(0, split);
1762  token.erase(0, split + 1);
1763  size_t error;
1764  std::string value = "";
1765  try {
1766  auto numValue = std::stoll(token, &error, 10);
1767  if (token[error] != '\0') {
1768  throw std::runtime_error("bad name for clone:" + token);
1769  }
1770  value = key + "_c" + std::to_string(numValue);
1771  } catch (std::invalid_argument& e) {
1772  value = token;
1773  }
1774  specs.push_back({key, value});
1775  s.erase(0, newPos + (newPos == std::string::npos ? 0 : 1));
1776  }
1777  if (s.empty() == false && specs.empty() == true) {
1778  throw std::runtime_error("bad pipeline definition. Syntax <processor>:<pipeline>");
1779  }
1780 
1781  std::vector<DataProcessorSpec> extraSpecs;
1782  for (auto& spec : specs) {
1783  for (auto& processor : workflow) {
1784  if (processor.name == spec.templateMatcher) {
1785  auto clone = processor;
1786  clone.name = spec.cloneName;
1787  extraSpecs.push_back(clone);
1788  }
1789  }
1790  }
1791  workflow.insert(workflow.end(), extraSpecs.begin(), extraSpecs.end());
1792 }
