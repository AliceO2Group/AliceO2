64 TGeoVolume* V3Services::createInnerBarrelSupports(const TGeoManager* mgr)
65 {
66  //
67  // Creates the ITS3 Inner Barrel support half cylinder and foam wedges
68  //
69  // Input:
70  // mgr : the GeoManager (used only to get the proper material)
71  //
72  // Output:
73  //
74  // Return:
75  // a TGeoVolume(Assembly) with the half cylinder and wedges
76  //
77  // Created: 11 Feb 2021 Mario Sitta
78  // Updated: 22 Feb 2021 Mario Sitta Impregnated carbon fleece added
79  //
80 
81  // For the time being we put all relevant parameters here
82  // May be changed into static const's when definitively set
83  const Double_t sIBSuppCylRint = 40.0 * sMm;
84  const Double_t sIBSuppCylThick = 2.20 * sMm;
85  const Double_t sIBSuppCylYvoid = 0.50 * sMm;
86  const Double_t sIBSuppCylZLen = 331.20 * sMm;
87 
88  const Int_t sNFoamWedges = 12;
89  const Double_t sFoamZLen = 10.0 * sMm;
90  const Double_t sFoamEdgeZDist = 5.20 * sMm;
91  const Double_t sFoamBaseWidth = 4.19 * sMm;
92  const Double_t sFoamYTrans = 1.59 * sMm;
93 
94  // Local variables
95  TGeoVolume* layerVol;
96  TGeoVolume* foamVol[3];
97  TGeoTube* layerSh;
98 
99  Double_t rmin, rmax, zlen, alpha, phi;
100  Double_t xpos, ypos, zpos;
101 
102  // The support half cylinder: a TubeSeg
103  phi = (sIBSuppCylYvoid / sIBSuppCylRint) * TMath::RadToDeg();
104  zlen = sIBSuppCylZLen / 2;
105  rmin = sIBSuppCylRint;
106  rmax = rmin + sIBSuppCylThick;
107 
108  TGeoTubeSeg* suppCylSh = new TGeoTubeSeg(rmin, rmax, zlen, phi, 180. - phi);
109 
110  // The foam wedges: three TubeSeg's
111  phi = (sFoamBaseWidth / sIBSuppCylRint) * TMath::RadToDeg();
112  zlen = sFoamZLen / 2;
113 
114  // The inner and outer radii of each foam wedge are the outer and inner
115  // radii of the IB Layers where they are located
116  // (except the last one which extends till the support cylinder)
117  for (Int_t j = 0; j < 3; j++) {
118  layerVol = mgr->GetVolume(Form("%s%d", o2::its3::GeometryTGeo::getITSLayerPattern(), j));
119  if (!layerVol) { // Should really never happen
120  LOG(FATAL) << "While building services: ITSU Layer " << j << " does not exist";
121  }
122  layerSh = (TGeoTube*)layerVol->GetShape();
123  rmin = layerSh->GetRmax();
124 
125  if (j == 2) {
126  rmax = sIBSuppCylRint;
127  } else {
128  layerVol = mgr->GetVolume(Form("%s%d", o2::its3::GeometryTGeo::getITSLayerPattern(), j + 1));
129  if (!layerVol) { // Should really never happen
130  LOG(FATAL) << "While building services: ITSU Layer " << j + 1 << " does not exist";
131  }
132  layerSh = (TGeoTube*)layerVol->GetShape();
133  rmax = layerSh->GetRmin();
134  }
135 
136  foamVol[j] = createFoamWedge(rmin, rmax, zlen, phi);
137  foamVol[j]->SetName(Form("IBSupportWedge%d", j));
138  }
139 
140  // We have all shapes: now create the real volumes
141  TGeoMedium* medCarbon = mgr->GetMedium("IT3_F6151B05M$");
142 
143  TGeoVolume* suppCylVol = new TGeoVolume("IBSupportCylinder", suppCylSh, medCarbon);
144  suppCylVol->SetFillColor(kBlue);
145  suppCylVol->SetLineColor(kBlue);
146 
147  // Finally put everything in the container volume
148  TGeoVolumeAssembly* ibSupport = new TGeoVolumeAssembly("IBCylinderSupport");
149 
150  // The support cylinder is aligned to the end of the IB Layers
151  // (layerSh points to the last Layer inside the support cylinder)
152  zpos = -layerSh->GetDz() + suppCylSh->GetDz();
153  ibSupport->AddNode(suppCylVol, 1, new TGeoTranslation(0, 0, zpos));
154 
155  alpha = ((sIBSuppCylYvoid + sFoamYTrans) / sIBSuppCylRint) * TMath::RadToDeg();
156 
157  Double_t emptySpace = 2 * layerSh->GetDz() - sNFoamWedges * sFoamZLen - 2 * sFoamEdgeZDist;
158  emptySpace /= (sNFoamWedges - 1);
159 
160  for (Int_t i = 0; i < 3; i++) {
161  for (Int_t j = 0; j < sNFoamWedges; j++) {
162  zpos = -layerSh->GetDz() + sFoamEdgeZDist + j * (sFoamZLen + emptySpace) + 0.5 * sFoamZLen;
163  ibSupport->AddNode(foamVol[i], j + 1, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", alpha, 0, 0)));
164  ibSupport->AddNode(foamVol[i], sNFoamWedges + j + 1, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", 180. - alpha - phi, 0, 0)));
165  }
166  // There are only two wedges at 90deg located at the far ends
167  zpos = -layerSh->GetDz() + sFoamEdgeZDist + 0.5 * sFoamZLen;
168  ibSupport->AddNode(foamVol[i], 2 * sNFoamWedges + 1, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", 90. - phi / 2, 0, 0)));
169  ibSupport->AddNode(foamVol[i], 2 * sNFoamWedges + 2, new TGeoCombiTrans(0, 0, -zpos, new TGeoRotation("", 90. - phi / 2, 0, 0)));
170  }
171 
172  //
173  return ibSupport;
174 }

176 TGeoVolume* V3Services::createFoamWedge(const Double_t rIn, const Double_t rOut, const Double_t zLen, const Double_t phi, const TGeoManager* mgr)
177 {
178  //
179  // Creates a single foam wedge with glue-impregnated carbon fleece attached
180  //
181  // Input:
182  // rIn : the inner radius
183  // rOut : the outer radius
184  // zLen : the half length along Z
185  // phi : the angular aperture
186  // mgr : the GeoManager (used only to get the proper material)
187  //
188  // Output:
189  //
190  // Return:
191  // a TGeoVolume with all elements, ready to be displaced
192  //
193  // Created: 22 Feb 2021 Mario Sitta
194  //
195 
196  // For the time being we put all relevant parameters here
197  // May be changed into static const's when definitively set
198  const Double_t sIBSuppWedgeCFThick = 20.0 * sMicron;
199 
200  // Local variables
201  Double_t rmin, rmax;
202 
203  // The foam wedge container: a TubeSeg
204  TGeoTubeSeg* wedgeSh = new TGeoTubeSeg(rIn, rOut, zLen, 0, phi);
205 
206  // The foam wedge: a TubeSeg
207  rmin = rIn + sIBSuppWedgeCFThick;
208  rmax = rOut - sIBSuppWedgeCFThick;
209  TGeoTubeSeg* foamSh = new TGeoTubeSeg(rmin, rmax, zLen, 0, phi);
210 
211  // The two carbon fleece layers: two TubeSeg's
212  rmax = rIn + sIBSuppWedgeCFThick;
213  TGeoTubeSeg* fleeceInSh = new TGeoTubeSeg(rIn, rmax, zLen, 0, phi);
214 
215  rmin = rOut - sIBSuppWedgeCFThick;
216  TGeoTubeSeg* fleeceOutSh = new TGeoTubeSeg(rmin, rOut, zLen, 0, phi);
217 
218  // We have all shapes: now create the real volumes
219  TGeoMedium* medAir = mgr->GetMedium("IT3_AIR$");
220  TGeoMedium* medFoam = mgr->GetMedium("IT3_ERGDUOCEL$");
221  TGeoMedium* medFleece = mgr->GetMedium("IT3_IMPREG_FLEECE$");
222 
223  TGeoVolume* wedgeVol = new TGeoVolume("IBSupportWedge", wedgeSh, medAir);
224  wedgeVol->SetFillColor(kGreen);
225  wedgeVol->SetLineColor(kGreen);
226 
227  TGeoVolume* foamVol = new TGeoVolume("IBSupportFoam", foamSh, medFoam);
228  foamVol->SetFillColor(kGreen);
229  foamVol->SetLineColor(kGreen);
230 
231  TGeoVolume* fleeceInVol = new TGeoVolume("IBSupportFleeceI", fleeceInSh, medFleece);
232  fleeceInVol->SetFillColor(kViolet);
233  fleeceInVol->SetLineColor(kViolet);
234 
235  TGeoVolume* fleeceOutVol = new TGeoVolume("IBSupportFleeceO", fleeceOutSh, medFleece);
236  fleeceOutVol->SetFillColor(kViolet);
237  fleeceOutVol->SetLineColor(kViolet);
238 
239  // Finally put everything in the container volume
240  wedgeVol->AddNode(foamVol, 1, nullptr);
241  wedgeVol->AddNode(fleeceInVol, 1, nullptr);
242  wedgeVol->AddNode(fleeceOutVol, 1, nullptr);
243 
244  // Done
245  return wedgeVol;
246 }

304 TGeoVolume* V3Services::createCYSSAssembly(const TGeoManager* mgr)
305 {
306  //
307  // Creates the CYSS Assembly (i.e. the supporting cylinder and cone)
308  //
309  // Input:
310  // mgr : the GeoManager (used only to get the proper material)
311  //
312  // Output:
313  //
314  // Return:
315  // a TGeoVolume(Assembly) with all the elements
316  //
317  // Created: 21 Oct 2019 Mario Sitta
318  // Updated: 02 Dec 2019 Mario Sitta Full cylinder implemented
319  //
320 
321  static const Double_t sCyssFlangeAZpos = 9.0 * sMm;
322  static const Double_t sCyssFlangeCZpos = 1.0 * sMm;
323 
324  Double_t zlen, zpos;
325 
326  TGeoVolume* cyssVol = new TGeoVolumeAssembly("IBCYSSAssembly");
327  cyssVol->SetVisibility(kTRUE);
328 
329  TGeoVolume* cyssCylinder = ibCyssCylinder(mgr);
330  zlen = (static_cast<TGeoTubeSeg*>(cyssCylinder->GetShape()))->GetDz();
331  zpos = sIBCYSSFlangeCZPos - sCyssFlangeCZpos - zlen;
332  cyssVol->AddNode(cyssCylinder, 1, new TGeoTranslation(0, 0, -zpos));
333  cyssVol->AddNode(cyssCylinder, 2, new TGeoCombiTrans(0, 0, -zpos, new TGeoRotation("", 180, 0, 0)));
334 
335  TGeoVolume* cyssCone = ibCyssCone(mgr);
336  zpos = -zpos + zlen - (static_cast<TGeoPcon*>(cyssCone->GetShape()))->GetZ(2);
337  cyssVol->AddNode(cyssCone, 1, new TGeoTranslation(0, 0, zpos));
338  cyssVol->AddNode(cyssCone, 2, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", 180, 0, 0)));
339 
340  TGeoVolume* cyssFlangeA = ibCyssFlangeSideA(mgr);
341  Int_t nZPlanes = (static_cast<TGeoPcon*>(cyssCone->GetShape()))->GetNz();
342  zpos = zpos + (static_cast<TGeoPcon*>(cyssCone->GetShape()))->GetZ(nZPlanes - 1) + sCyssFlangeAZpos;
343  cyssVol->AddNode(cyssFlangeA, 1, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", 180, 180, 0)));
344  cyssVol->AddNode(cyssFlangeA, 2, new TGeoCombiTrans(0, 0, zpos, new TGeoRotation("", 0, 180, 0)));
345 
346  TGeoVolume* cyssFlangeC = ibCyssFlangeSideC(mgr);
347  zpos = sIBCYSSFlangeCZPos;
348  cyssVol->AddNode(cyssFlangeC, 1, new TGeoTranslation(0, 0, -zpos));
349  cyssVol->AddNode(cyssFlangeC, 2, new TGeoCombiTrans(0, 0, -zpos, new TGeoRotation("", 180, 0, 0)));
350 
351  // Return the whole assembly
352  return cyssVol;
353 }
