94 //____________________________________________________________________________________________
95 double CalDet::getRMSRobust(double robust) const
96 {
97  //
98  // Calculate the robust RMS
99  //
100 
101  // sorted
102  std::array<int, MAXCHAMBER> index;
103  TMath::Sort((int)MAXCHAMBER, mData.data(), index.data());
104 
105  // reject
106  double reject = (int)(MAXCHAMBER * (1 - robust) / 2.0);
107  if (reject <= 0) {
108  reject = 0;
109  }
110  if (reject >= MAXCHAMBER) {
111  reject = 0;
112  }
113 
114  std::array<double, MAXCHAMBER> ddata;
115  int nPoints = 0;
116  for (int i = 0; i < MAXCHAMBER; i++) {
117  bool rej = kFALSE;
118  for (int k = 0; k < reject; k++) {
119  if (i == index[k]) {
120  rej = kTRUE;
121  }
122  if (i == index[MAXCHAMBER - (k + 1)]) {
123  rej = kTRUE;
124  }
125  }
126  if (!rej) {
127  ddata[nPoints] = mData[i];
128  nPoints++;
129  }
130  }
131  return TMath::RMS(nPoints, ddata.data());
132 }

175 double CalDet::getLTM(double* sigma, double fraction, CalDet* const outlierDet)
176 {
177  //
178  // Calculate LTM mean and sigma
179  //
180 
181  std::array<double, MAXCHAMBER> ddata;
182  double mean = 0, lsigma = 0;
183  unsigned int nPoints = 0;
184  for (int i = 0; i < MAXCHAMBER; i++) {
185  if (!outlierDet || !(outlierDet->getValue(i))) {
186  ddata[nPoints] = mData[nPoints];
187  nPoints++;
188  }
189  }
190  int hh = TMath::Min(TMath::Nint(fraction * nPoints), int(nPoints));
191  // AliMathBase::EvaluateUni(nPoints, ddata, mean, lsigma, hh);
192  TRobustEstimator tre;
193  tre.EvaluateUni(nPoints, ddata.data(), mean, lsigma, hh);
194 
195  if (sigma) {
196  *sigma = lsigma;
197  }
198  return mean;
199 }

202 TH1F* CalDet::makeHisto1Distribution(float min, float max, int type)
203 {
204  //
205  // make 1D histo
206  // type -1 = user defined range
207  // 0 = nsigma cut nsigma=min
208  // 1 = delta cut around median delta=min
209  //
210 
211  if (type >= 0) {
212  if (type == 0) {
213  // nsigma range
214  float mean = getMean();
215  float sigma = getRMS();
216  float nsigma = TMath::Abs(min);
217  min = mean - nsigma * sigma;
218  max = mean + nsigma * sigma;
219  }
220  if (type == 1) {
221  // fixed range
222  float mean = getMedian();
223  float delta = min;
224  min = mean - delta;
225  max = mean + delta;
226  }
227  if (type == 2) {
228  //
229  // LTM mean +- nsigma
230  //
231  double sigma;
232  float mean = getLTM(&sigma, max);
233  sigma *= min;
234  min = mean - sigma;
235  max = mean + sigma;
236  }
237  }
238  std::stringstream title;
239  title << mTitle.c_str() << " CalDet 1Distribution";
240  std::string titlestr = title.str();
241  TH1F* his = new TH1F(mName.c_str(), titlestr.c_str(), 100, min, max);
242  for (int idet = 0; idet < MAXCHAMBER; idet++) {
243  his->Fill(getValue(idet));
244  }
245  return his;
246 }

297 {
298  //
299  // Make 2D graph
300  // ch - chamber
301  // type -1 = user defined range
302  // 0 = nsigma cut nsigma=min
303  // 1 = delta cut around median delta=min
304  //
305 
306  if (type >= 0) {
307  if (type == 0) {
308  // nsigma range
309  float mean = getMean();
310  float sigma = getRMS();
311  float nsigma = TMath::Abs(min);

316  // fixed range
317  float mean = getMedian();
318  float delta = min;
319  min = mean - delta;
320  max = mean + delta;

323  double sigma;

326  min = mean - sigma;
327  max = mean + sigma;
328  }
329  }
330 
331  Geometry* trdGeo = Geometry::instance();
332 
333  float poslocal[3] = {0, 0, 0};
334  float posglobal[3] = {0, 0, 0};
335 
336  std::stringstream title;
337  title << mTitle.c_str() << " Cal AADet 2D ch " << ch;
338  std::string titlestr = title.str();
339  TH2F* his = new TH2F(mName.c_str(), titlestr.c_str(), 400, -400.0, 400.0, 400, -400.0, 400.0);
340 
341  // Where we begin
342  int offsetch = 6 * ch;
343 
344  for (int isec = 0; isec < NSECTOR; isec++) {
345  for (int ipl = 0; ipl < NLAYER; ipl++) {
346  int det = offsetch + isec * 30 + ipl;
347  const PadPlane* padPlane = trdGeo->getPadPlane(ipl, ch);
348  for (int icol = 0; icol < padPlane->getNcols(); icol++) {
349  poslocal[0] = trdGeo->getTime0(ipl);
350  poslocal[2] = padPlane->getRowPos(0);
351  poslocal[1] = padPlane->getColPos(icol);
352  trdGeo->rotateBack(det, poslocal, posglobal);
353  int binx = 1 + TMath::Nint((posglobal[0] + 400.0) * 0.5);
354  int biny = 1 + TMath::Nint((posglobal[1] + 400.0) * 0.5);
355  his->SetBinContent(binx, biny, mData[det]);
356  }
357  }
358  }
359  his->SetXTitle("x (cm)");
360  his->SetYTitle("y (cm)");
361  his->SetStats(0);
362  his->SetMaximum(max);
363  his->SetMinimum(min);
364  return his;
365 }

493 //_____________________________________________________________________________
494 double CalDet::calcMean(bool wghtPads, int& calib)
495 {
496  // Calculate the mean value after rejection of the chambers not calibrated
497  // wghPads = kTRUE weighted with the number of pads in case of a AliTRDCalPad created (t0)
498  // calib = number of used chambers for the mean calculation
499 
500  double sum = 0.0;
501  int ndet = 0;
502  double meanALL = 0.0;
503  double meanWP = 0.0;
504  double padsALL = (144 * 16 * 24 + 144 * 12 * 6) * 18;
505  std::array<double, 18> meanSM{};
506  std::array<double, 18> meanSMWP{};
507 
508  int det = 0;
509  while (det < 540) {
510  float val = mData[det];
511  int iSM = (int)(det / (6 * 5));
512  double pads = (((int)(det % (6 * 5)) / 6) == 2) ? 144 * 12 : 144 * 16;
513  meanALL += val / 540.;
514  meanSM[iSM] += val / 30.;
515  meanWP += val * (pads / padsALL);
516  meanSMWP[iSM] += val * (pads / (padsALL / 18.));
517  det++;
518  }
519 
520  det = 0;
521  while (det < 540) {
522  float val = mData[det];
523  if (((!wghtPads) &&
524  (TMath::Abs(val - meanALL) > 0.0001) &&
525  (TMath::Abs(val - meanSM[(int)(det / (6 * 5))]) > 0.0001)) ||
526  ((wghtPads) &&
527  (TMath::Abs(val - meanWP) > 0.0001) &&
528  (TMath::Abs(val - meanSMWP[(int)(det / (6 * 5))]) > 0.0001))) {
529  if (val <= 50.) { // get rid of exb alternative mean values
530  sum += val;
531  ndet++;
532  }
533  }
534  det++;
535  }
536  calib = ndet;
537  return (sum != 0.0 ? sum / ndet : -1);
538 }
