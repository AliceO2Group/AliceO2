68 template <typename VEC>
69 void CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
70 {
71  using MD = o2::ctf::Metadata::OptStore;
72  // what to do which each field: see o2::ctd::Metadata explanation
73  constexpr MD optField[CTF::getNBlocks()] = {
74  MD::EENCODE, // BLC_trigger
75  MD::EENCODE, // BLC_bcInc
76  MD::EENCODE, // BLC_orbitInc
77  MD::EENCODE, // BLC_nChan
78 
79  MD::EENCODE, // BLC_idChan
80  MD::EENCODE, // BLC_time
81  MD::EENCODE, // BLC_charge
82  MD::EENCODE // BLC_feeBits
83  };
84  CompressedDigits cd;
85  compress(cd, digitVec, channelVec);
86 
87  // book output size with some margin
88  auto szIni = estimateCompressedSize(cd);
89  buff.resize(szIni);
90 
91  auto ec = CTF::create(buff);
92  using ECB = CTF::base;
93 
94  ec->setHeader(cd.header);
95  ec->getANSHeader().majorVersion = 0;
96  ec->getANSHeader().minorVersion = 1;
97  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
98 #define ENCODEFDD(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
99  // clang-format off
100  ENCODEFDD(cd.trigger, CTF::BLC_trigger, 0);
101  ENCODEFDD(cd.bcInc, CTF::BLC_bcInc, 0);
102  ENCODEFDD(cd.orbitInc, CTF::BLC_orbitInc, 0);
103  ENCODEFDD(cd.nChan, CTF::BLC_nChan, 0);
104 
105  ENCODEFDD(cd.idChan , CTF::BLC_idChan, 0);
106  ENCODEFDD(cd.time, CTF::BLC_time, 0);
107  ENCODEFDD(cd.charge, CTF::BLC_charge, 0);
108  ENCODEFDD(cd.feeBits, CTF::BLC_feeBits, 0);
109  // clang-format on
110  CTF::get(buff.data())->print(getPrefix());
111 }

115 void CTFCoder::decode(const CTF::base& ec, VDIG& digitVec, VCHAN& channelVec)
116 {
117  CompressedDigits cd;
118  cd.header = ec.getHeader();
119  ec.print(getPrefix());
120 #define DECODEFDD(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
121  // clang-format off
122  DECODEFDD(cd.trigger, CTF::BLC_trigger);
123  DECODEFDD(cd.bcInc, CTF::BLC_bcInc);
124  DECODEFDD(cd.orbitInc, CTF::BLC_orbitInc);
125  DECODEFDD(cd.nChan, CTF::BLC_nChan);
126 
127  DECODEFDD(cd.idChan, CTF::BLC_idChan);
128  DECODEFDD(cd.time, CTF::BLC_time);
129  DECODEFDD(cd.charge, CTF::BLC_charge);
130  DECODEFDD(cd.feeBits, CTF::BLC_feeBits);
131  // clang-format on
132  //
133  decompress(cd, digitVec, channelVec);
134 }

137 template <typename VDIG, typename VCHAN>
138 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
139 {
140  digitVec.clear();
141  channelVec.clear();
142  digitVec.reserve(cd.header.nTriggers);
143  channelVec.reserve(cd.idChan.size());
144 
145  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
146  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
147 
148  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
149  // restore ROFRecord
150  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
151  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
152  ir.orbit += cd.orbitInc[idig];
153  } else {
154  ir.bc += cd.bcInc[idig];
155  }
156  Triggers trig;
157  trig.triggersignals = cd.trigger[idig];
158 
159  firstEntry = channelVec.size();
160  uint8_t chID = 0;
161  int amplA = 0, amplC = 0, timeA = 0, timeC = 0;
162  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
163  auto icc = channelVec.size();
164  const auto& chan = channelVec.emplace_back((chID += cd.idChan[icc]), cd.time[icc], cd.charge[icc], cd.feeBits[icc]);
165  //
166  // rebuild digit
167  if (chan.mPMNumber > 7) { // A side
168  amplA += chan.mChargeADC;
169  timeA += chan.mTime;
170  trig.nChanA++;
171 
172  } else {
173  amplC += chan.mChargeADC;
174  timeC += chan.mTime;
175  trig.nChanC++;
176  }
177  }
178  if (trig.nChanA) {
179  trig.timeA = timeA / trig.nChanA;
180  trig.amplA = amplA * 0.125;
181  }
182  if (trig.nChanC) {
183  trig.timeC = timeC / trig.nChanC;
184  trig.amplC = amplC * 0.125;
185  }
186  digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig);
187  }
188 }
