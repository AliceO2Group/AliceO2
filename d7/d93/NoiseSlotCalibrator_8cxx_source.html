26 bool NoiseSlotCalibrator::processTimeFrame(gsl::span<const o2::itsmft::CompClusterExt> const& clusters,
27  gsl::span<const unsigned char> const& patterns,
28  gsl::span<const o2::itsmft::ROFRecord> const& rofs)
29 {
30  calibration::TFType nTF = rofs[0].getBCData().orbit / 256;
31  LOG(INFO) << "Processing TF# " << nTF;
32 
33  auto& slotTF = getSlotForTF(nTF);
34  auto& noiseMap = *(slotTF.getContainer());
35 
36  auto pattIt = patterns.begin();
37  for (const auto& rof : rofs) {
38  auto clustersInFrame = rof.getROFData(clusters);
39  for (const auto& c : clustersInFrame) {
40  if (c.getPatternID() != o2::itsmft::CompCluster::InvalidPatternID) {
41  // For the noise calibration, we use "pass1" clusters...
42  continue;
43  }
44  o2::itsmft::ClusterPattern patt(pattIt);
45 
46  auto id = c.getSensorID();
47  auto row = c.getRow();
48  auto col = c.getCol();
49  auto colSpan = patt.getColumnSpan();
50  auto rowSpan = patt.getRowSpan();
51 
52  // Fast 1-pixel calibration
53  if ((rowSpan == 1) && (colSpan == 1)) {
54  noiseMap.increaseNoiseCount(id, row, col);
55  continue;
56  }
57  if (m1pix) {
58  continue;
59  }
60 
61  // All-pixel calibration
62  auto nBits = rowSpan * colSpan;
63  int ic = 0, ir = 0;
64  for (unsigned int i = 2; i < patt.getUsedBytes() + 2; i++) {
65  unsigned char tempChar = patt.getByte(i);
66  int s = 128; // 0b10000000
67  while (s > 0) {
68  if ((tempChar & s) != 0) {
69  noiseMap.increaseNoiseCount(id, row + ir, col + ic);
70  }
71  ic++;
72  s >>= 1;
73  if ((ir + 1) * ic == nBits) {
74  break;
75  }
76  if (ic == colSpan) {
77  ic = 0;
78  ir++;
79  }
80  }
81  if ((ir + 1) * ic == nBits) {
82  break;
83  }
84  }
85  }
86  }
87 
88  mNumberOfStrobes += rofs.size();
89  return hasEnoughData(slotTF);
90 }
