223 inline CalibRawBase::ProcessStatus CalibRawBase::processEventGBT()
224 {
225  if (!mGBTFrameContainers.size()) {
226  return ProcessStatus::NoReaders;
227  }
228  resetEvent();
229 
230  // loop over raw readers, fill digits for 500 time bins and process
231  // digits
232 
233  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
234 
235  ProcessStatus status = ProcessStatus::Ok;
236 
237  std::vector<Digit> digits(80);
238  for (auto& reader_ptr : mGBTFrameContainers) {
239  auto reader = reader_ptr.get();
240  int readTimeBins = 0;
241  for (int i = 0; i < mTimeBinsPerCall; ++i) {
242  if (reader->getData(digits)) {
243  for (auto& digi : digits) {
244  CRU cru(digi.getCRU());
245  const int roc = cru.roc();
246  // TODO: OROC case needs subtraction of number of pad rows in IROC
247  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
248  const PartitionInfo& partInfo = mMapper.getPartitionInfo(cru.partition());
249 
250  // row is local in region (CRU)
251  const int row = digi.getRow();
252  const int pad = digi.getPad();
253  if (row == 255 || pad == 255) {
254  continue;
255  }
256 
257  int rowOffset = 0;
258  switch (mPadSubset) {
259  case PadSubset::ROC: {
260  rowOffset = regionInfo.getGlobalRowOffset();
261  rowOffset -= (cru.isOROC()) * nRowIROC;
262  break;
263  }
264  case PadSubset::Region: {
265  break;
266  }
267  case PadSubset::Partition: {
268  rowOffset = regionInfo.getGlobalRowOffset();
269  rowOffset -= partInfo.getGlobalRowOffset();
270  break;
271  }
272  }
273 
274  // modify row depending on the calibration type used
275  const int timeBin = i; //digi.getTimeStamp();
276  const float signal = digi.getChargeFloat();
277 
278  updateCRU(cru, row, pad, timeBin, signal);
279  updateROC(roc, row + rowOffset, pad, timeBin, signal);
280  }
281  ++readTimeBins;
282  }
283 
284  digits.clear();
285  }
286 
287  // set status, don't overwrite decision
288  if (status == ProcessStatus::Ok) {
289  if (readTimeBins == 0) {
290  return ProcessStatus::NoMoreData;
291  } else if (readTimeBins < mTimeBinsPerCall) {
292  status = ProcessStatus::Truncated;
293  }
294  }
295 
296  // notify that one raw reader processing finalized for this event
297  endReader();
298  }
299 
300  endEvent();
301  ++mNevents;
302  return status;
303 }

306 inline CalibRawBase::ProcessStatus CalibRawBase::processEventRawReader(int eventNumber)
307 {
308  if (!mRawReaders.size()) {
309  return ProcessStatus::NoReaders;
310  }
311  resetEvent();
312 
313  // loop over raw readers, fill digits for 500 time bins and process
314  // digits
315 
316  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
317 
318  ProcessStatus status = ProcessStatus::Ok;
319 
320  mProcessedTimeBins = 0;
321  size_t processedReaders = 0;
322  bool hasData = false;
323 
324  int64_t lastEvent = 0;
325  for (auto& reader_ptr : mRawReaders) {
326  auto reader = reader_ptr.get();
327 
328  lastEvent = std::max(lastEvent, reader->getLastEvent());
329 
330  if (eventNumber >= 0) {
331  reader->loadEvent(eventNumber);
332  mPresentEventNumber = eventNumber;
333  } else if (eventNumber == -1) {
334  mPresentEventNumber = reader->loadNextEvent();
335  } else if (eventNumber == -2) {
336  mPresentEventNumber = reader->loadPreviousEvent();
337  }
338 
339  o2::tpc::PadPos padPos;
340  while (std::shared_ptr<std::vector<uint16_t>> data = reader->getNextData(padPos)) {
341  if (!data) {
342  continue;
343  }
344 
345  mProcessedTimeBins = std::max(mProcessedTimeBins, data->size());
346 
347  CRU cru(reader->getRegion());
348  const int roc = cru.roc();
349  // TODO: OROC case needs subtraction of number of pad rows in IROC
350  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
351  const PartitionInfo& partInfo = mMapper.getPartitionInfo(cru.partition());
352 
353  // row is local in region (CRU)
354  const int row = padPos.getRow();
355  const int pad = padPos.getPad();
356  if (row == 255 || pad == 255) {
357  continue;
358  }
359 
360  int timeBin = 0;
361  for (const auto& signalI : *data) {
362 
363  int rowOffset = 0;
364  switch (mPadSubset) {
365  case PadSubset::ROC: {
366  rowOffset = regionInfo.getGlobalRowOffset();
367  rowOffset -= (cru.rocType() == RocType::OROC) * nRowIROC;
368  break;
369  }
370  case PadSubset::Region: {
371  break;
372  }
373  case PadSubset::Partition: {
374  rowOffset = regionInfo.getGlobalRowOffset();
375  rowOffset -= partInfo.getGlobalRowOffset();
376  break;
377  }
378  }
379 
380  // modify row depending on the calibration type used
381  const float signal = float(signalI);
382  //const FECInfo& fecInfo = mTPCmapper.getFECInfo(PadSecPos(roc, row, pad));
383  //printf("Call update: %d, %d, %d, %d (%d), %.3f -- reg: %02d -- FEC: %02d, Chip: %02d, Chn: %02d\n", roc, row, pad, timeBin, i, signal, cru.region(), fecInfo.getIndex(), fecInfo.getSampaChip(), fecInfo.getSampaChannel());
384  updateCRU(cru, row, pad, timeBin, signal);
385  updateROC(roc, row + rowOffset, pad, timeBin, signal);
386  ++timeBin;
387  hasData = true;
388  }
389  }
390 
391  // notify that one raw reader processing finalized for this event
392  endReader();
393  ++processedReaders;
394  }
395 
396  // set status, don't overwrite decision
397  if (!hasData) {
398  return ProcessStatus::NoMoreData;
399  } else if (processedReaders < mRawReaders.size()) {
400  status = ProcessStatus::Truncated;
401  } else if (mPresentEventNumber == size_t(lastEvent)) {
402  status = ProcessStatus::LastEvent;
403  }
404 
405  endEvent();
406  ++mNevents;
407  return status;
408 }

411 inline CalibRawBase::ProcessStatus CalibRawBase::processEventRawReaderCRU(int eventNumber)
412 {
413  if (!mRawReaderCRUManager.getNumberOfReaders()) {
414  return ProcessStatus::NoReaders;
415  }
416  resetEvent();
417 
418  mRawReaderCRUManager.init();
419 
420  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
421 
422  ProcessStatus status = ProcessStatus::Ok;
423 
424  mProcessedTimeBins = 0;
425  bool hasData = false;
426 
427  const int64_t numberOfEvents = mRawReaderCRUManager.getNumberOfEvents();
428  const int64_t lastEvent = numberOfEvents - 1;
429 
430  if (eventNumber >= 0) {
431  mPresentEventNumber = eventNumber;
432  } else if (eventNumber == -1) {
433  if (mPresentEventNumber == std::numeric_limits<size_t>::max()) {
434  mPresentEventNumber = 0;
435  } else {
436  mPresentEventNumber = (mPresentEventNumber + 1) % numberOfEvents;
437  }
438  } else if (eventNumber == -2) {
439  if (mPresentEventNumber > 0) {
440  mPresentEventNumber -= 1;
441  } else {
442  mPresentEventNumber = numberOfEvents - 1;
443  }
444  }
445 
446  const bool skipEvent = mSkipIncomplete && !isPresentEventComplete();
447  if (!skipEvent) {
448  hasData = true;
449  mRawReaderCRUManager.processEvent(mPresentEventNumber, [this]() { endReader(); });
450  // set status, don't overwrite decision
451  if (!hasData) {
452  return ProcessStatus::NoMoreData;
453  } else if (!isPresentEventComplete()) {
454  status = ProcessStatus::IncompleteEvent;
455  } else if (mPresentEventNumber == size_t(lastEvent)) {
456  status = ProcessStatus::LastEvent;
457  }
458 
459  endEvent();
460  ++mNevents;
461  } else {
462  status = ProcessStatus::IncompleteEvent;
463  if (mPresentEventNumber == size_t(lastEvent)) {
464  status = ProcessStatus::LastEvent;
465  }
466  }
467 
468  LOG(INFO) << "Present event number : " << mPresentEventNumber << (skipEvent ? " (skipped, incomplete)" : "");
469  LOG(INFO) << "Last event : " << lastEvent;
470  LOG(INFO) << "Status : " << int(status);
471 
472  return status;
473 }
