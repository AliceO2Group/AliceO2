57 template <typename VEC>
58 void CTFCoder::encode(VEC& buff, const gsl::span<const BCData>& trigData, const gsl::span<const ChannelData>& chanData, const gsl::span<const OrbitData>& pedData)
59 {
60  using MD = o2::ctf::Metadata::OptStore;
61  // what to do which each field: see o2::ctd::Metadata explanation
62  constexpr MD optField[CTF::getNBlocks()] = {
63  MD::EENCODE, // _bcIncTrig
64  MD::EENCODE, // _orbitIncTrig,
65  MD::EENCODE, // _moduleTrig,
66  MD::EENCODE, // _channelsHL,
67  MD::EENCODE, // _triggersHL,
68  MD::EENCODE, // _extTriggers,
69  MD::EENCODE, // _nchanTrig,
70  //
71  MD::EENCODE, // _chanID,
72  MD::EENCODE, // _chanData,
73  //
74  MD::EENCODE, // _orbitIncEOD,
75  MD::EENCODE, // _pedData
76  MD::EENCODE, // _sclInc
77  };
78 
79  CTFHelper helper(trigData, chanData, pedData);
80 
81  // book output size with some margin
82  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
83  buff.resize(szIni);
84 
85  auto ec = CTF::create(buff);
86  using ECB = CTF::base;
87 
88  ec->setHeader(helper.createHeader());
89  ec->getANSHeader().majorVersion = 0;
90  ec->getANSHeader().minorVersion = 1;
91  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
92 #define ENCODEZDC(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
93  // clang-format off
94  ENCODEZDC(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
95  ENCODEZDC(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
96  ENCODEZDC(helper.begin_moduleTrig(), helper.end_moduleTrig(), CTF::BLC_moduleTrig, 0);
97  ENCODEZDC(helper.begin_channelsHL(), helper.end_channelsHL(), CTF::BLC_channelsHL, 0);
98  ENCODEZDC(helper.begin_triggersHL(), helper.end_triggersHL(), CTF::BLC_triggersHL, 0);
99  ENCODEZDC(helper.begin_extTriggers(), helper.end_extTriggers(), CTF::BLC_extTriggers, 0);
100  ENCODEZDC(helper.begin_nchanTrig(), helper.end_nchanTrig(), CTF::BLC_nchanTrig, 0);
101  //
102  ENCODEZDC(helper.begin_chanID(), helper.end_chanID(), CTF::BLC_chanID, 0);
103  ENCODEZDC(helper.begin_chanData(), helper.end_chanData(), CTF::BLC_chanData, 0);
104  //
105  ENCODEZDC(helper.begin_orbitIncEOD(), helper.end_orbitIncEOD(), CTF::BLC_orbitIncEOD, 0);
106  ENCODEZDC(helper.begin_pedData(), helper.end_pedData(), CTF::BLC_pedData, 0);
107  ENCODEZDC(helper.begin_sclInc(), helper.end_sclInc(), CTF::BLC_sclInc, 0);
108 
109  // clang-format on
110  CTF::get(buff.data())->print(getPrefix());
111 }

114 template <typename VTRG, typename VCHAN, typename VPED>
115 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCHAN& chanVec, VPED& pedVec)
116 {
117  auto header = ec.getHeader();
118  ec.print(getPrefix());
119  std::vector<uint16_t> bcIncTrig, moduleTrig, nchanTrig, chanData, pedData, scalerInc, triggersHL, channelsHL;
120  std::vector<uint32_t> orbitIncTrig, orbitIncEOD;
121  std::vector<uint8_t> extTriggers, chanID;
122 
123 #define DECODEZDC(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
124  // clang-format off
125  DECODEZDC(bcIncTrig, CTF::BLC_bcIncTrig);
126  DECODEZDC(orbitIncTrig, CTF::BLC_orbitIncTrig);
127  DECODEZDC(moduleTrig, CTF::BLC_moduleTrig);
128  DECODEZDC(channelsHL, CTF::BLC_channelsHL);
129  DECODEZDC(triggersHL, CTF::BLC_triggersHL);
130  DECODEZDC(extTriggers, CTF::BLC_extTriggers);
131  DECODEZDC(nchanTrig, CTF::BLC_nchanTrig);
132  //
133  DECODEZDC(chanID, CTF::BLC_chanID);
134  DECODEZDC(chanData, CTF::BLC_chanData);
135  //
136  DECODEZDC(orbitIncEOD, CTF::BLC_orbitIncEOD);
137  DECODEZDC(pedData, CTF::BLC_pedData);
138  DECODEZDC(scalerInc, CTF::BLC_sclInc);
139  // clang-format on
140  //
141  trigVec.clear();
142  chanVec.clear();
143  pedVec.clear();
144  trigVec.reserve(header.nTriggers);
145  chanVec.reserve(header.nChannels);
146  pedVec.reserve(header.nEOData);
147 
148  // triggers and channels
149  uint32_t firstEntry = 0;
150  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
151  auto chanDataIt = chanData.begin();
152  auto chanIdIt = chanID.begin();
153  auto modTrigIt = moduleTrig.begin();
154  auto pedValIt = pedData.begin();
155  auto sclIncIt = scalerInc.begin();
156  auto channelsHLIt = channelsHL.begin();
157  auto triggersHLIt = triggersHL.begin();
158  auto scalers = header.firstScaler;
159 
160  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
161  // restore TrigRecord
162  if (orbitIncTrig[itrig]) { // non-0 increment => new orbit
163  ir.bc = bcIncTrig[itrig]; // bcInc has absolute meaning
164  ir.orbit += orbitIncTrig[itrig];
165  } else {
166  ir.bc += bcIncTrig[itrig];
167  }
168 
169  auto firstChanEntry = chanVec.size();
170  for (uint16_t ic = 0; ic < nchanTrig[itrig]; ic++) {
171  auto& chan = chanVec.emplace_back();
172  chan.id = *chanIdIt++;
173  std::copy_n(chanDataIt, NTimeBinsPerBC, chan.data.begin());
174  chanDataIt += NTimeBinsPerBC;
175  }
176  uint32_t chHL = (uint32_t(*channelsHLIt++) << 16) + *channelsHLIt++;
177  uint32_t trHL = (uint32_t(*triggersHLIt++) << 16) + *triggersHLIt++;
178 
179  auto& bcTrig = trigVec.emplace_back(firstChanEntry, chanVec.size() - firstChanEntry, ir, chHL, trHL, extTriggers[itrig]);
180  std::copy_n(modTrigIt, NModules, bcTrig.moduleTriggers.begin());
181  modTrigIt += NModules;
182  }
183 
184  // pedestal data
185  ir = {o2::constants::lhc::LHCMaxBunches - 1, header.firstOrbitEOData};
186  for (uint32_t ip = 0; ip < header.nEOData; ip++) {
187  ir.orbit += orbitIncEOD[ip];
188  for (uint32_t ic = 0; ic < NChannels; ic++) {
189  scalers[ic] += *sclIncIt++; // increment scaler
190  }
191  auto& ped = pedVec.emplace_back(OrbitData{ir, {}, scalers});
192  std::copy_n(pedValIt, NChannels, ped.data.begin());
193  pedValIt += NChannels;
194  }
195  // make sure whole data was used
196  assert(chanDataIt == chanData.end());
197  assert(chanIdIt == chanID.end());
198  assert(modTrigIt == moduleTrig.end());
199  assert(pedValIt == pedData.end());
200  assert(channelsHLIt == channelsHL.end());
201  assert(triggersHLIt == triggersHL.end());
202  assert(sclIncIt == scalerInc.end());
203 }
