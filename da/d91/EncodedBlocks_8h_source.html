117 struct Registry {
118  char* head = nullptr;
119  int nFilledBlocks = 0; // number of filled blocks = next block to fill (must be strictly consecutive)
120  size_t offsFreeStart = 0; // offset of the start of the writable space (wrt head), in bytes!!!
121  size_t size = 0; // full size in bytes!!!
122 
124  char* getFreeBlockStart() const
125  {
126  assert(offsFreeStart <= size);
127  return head + offsFreeStart;
128  }
129 
131  size_t getFreeSize() const
132  {
133  return size - offsFreeStart;
134  }

140 template <typename W = uint32_t>
141 struct Block {
142 
143  Registry* registry = nullptr;
144  int nDict = 0; // dictionary length (if any)
145  int nData = 0; // length of data
146  int nLiterals = 0; // length of literals vector (if any)
147  int nStored = 0; // total length
148  W* payload = nullptr; //[nStored];
149 
150  inline const W* getDict() const { return nDict ? payload : nullptr; }
151  inline const W* getData() const { return payload ? (payload + nDict) : nullptr; }
152  inline const W* getLiterals() const { return nLiterals ? (payload + nDict + nData) : nullptr; }
153 
154  inline W* getCreatePayload() { return payload ? payload : (registry ? (payload = reinterpret_cast<W*>(registry->getFreeBlockStart())) : nullptr); }
155  inline W* getCreateDict() { return payload ? payload : getCreatePayload(); }
156  inline W* getCreateData() { return payload ? (payload + nDict) : getCreatePayload(); }
157  inline W* getCreateLiterals() { return payload ? payload + (nDict + nData) : getCreatePayload(); }
158 
159  inline void setNDict(int _ndict)
160  {
161  nDict = _ndict;
162  nStored += nDict;
163  }
164 
165  inline void setNData(int _ndata)
166  {
167  nData = _ndata;
168  nStored += nData;
169  }
170 
171  inline void setNLiterals(int _nliterals)
172  {
173  nLiterals = _nliterals;
174  nStored += nLiterals;
175  }
176 
177  inline int getNDict() const { return nDict; }
178  inline int getNData() const { return nData; }
179  inline int getNLiterals() const { return nLiterals; }
180  inline int getNStored() const { return nStored; }
181 
182  ~Block()
183  {
184  if (!registry) { // this is a standalone block owning its data
185  delete[] payload;
186  }
187  }
188 
190  void clear()
191  {
192  nDict = 0;
193  nData = 0;
194  nLiterals = 0;
195  nStored = 0;
196  payload = nullptr;
197  }
198 
200  static size_t estimateSize(int n)
201  {
202  return alignSize(n * sizeof(W));
203  }
204 
205  // store a dictionary in an empty block
206  void storeDict(int _ndict, const W* _dict)
207  {
208  if (getNStored() > 0) {
209  throw std::runtime_error("trying to write in occupied block");
210  }
211  size_t sz = estimateSize(_ndict);
212  assert(registry); // this method is valid only for flat version, which has a registry
213  assert(sz <= registry->getFreeSize());
214  assert((_ndict > 0) == (_dict != nullptr));
215  setNDict(_ndict);
216  if (nDict) {
217  memcpy(getCreateDict(), _dict, _ndict * sizeof(W));
218  realignBlock();
219  }
220  };
221 
222  // store a dictionary to a block which can either be empty or contain a dict.
223  void storeData(int _ndata, const W* _data)
224  {
225  if (getNStored() > getNDict()) {
226  throw std::runtime_error("trying to write in occupied block");
227  }
228 
229  size_t sz = estimateSize(_ndata);
230  assert(registry); // this method is valid only for flat version, which has a registry
231  assert(sz <= registry->getFreeSize());
232  assert((_ndata > 0) == (_data != nullptr));
233  setNData(_ndata);
234  if (nData) {
235  memcpy(getCreateData(), _data, _ndata * sizeof(W));
236  realignBlock();
237  }
238  }
239 
240  // store a dictionary to a block which can either be empty or contain a dict.
241  void storeLiterals(int _nliterals, const W* _literals)
242  {
243  if (getNStored() > getNDict() + getNData()) {
244  throw std::runtime_error("trying to write in occupied block");
245  }
246 
247  size_t sz = estimateSize(_nliterals);
248  assert(registry); // this method is valid only for flat version, which has a registry
249  assert(sz <= registry->getFreeSize());
250  // assert((_nliterals > 0) == (_literals != nullptr));
251  setNLiterals(_nliterals);
252  if (nLiterals) {
253  memcpy(getCreateLiterals(), _literals, _nliterals * sizeof(W));
254  realignBlock();
255  }
256  }
257 
258  // resize block and free up unused buffer space.
259  void realignBlock()
260  {
261  size_t sz = estimateSize(getNStored());
262  registry->offsFreeStart = (reinterpret_cast<char*>(payload) - registry->head) + sz;
263  }
264 
266  void store(int _ndict, int _ndata, int _nliterals, const W* _dict, const W* _data, const W* _literals)
267  {
268  size_t sz = estimateSize(_ndict + _ndata + _nliterals);
269  assert(registry); // this method is valid only for flat version, which has a registry
270  assert(sz <= registry->getFreeSize());
271  assert((_ndict > 0) == (_dict != nullptr));
272  assert((_ndata > 0) == (_data != nullptr));
273  // assert(_literals == _data + _nliterals);
274  setNDict(_ndict);
275  setNData(_ndata);
276  setNLiterals(_nliterals);
277  getCreatePayload(); // do this even for empty block!!!
278  if (getNStored()) {
279  payload = reinterpret_cast<W*>(registry->getFreeBlockStart());
280  if (getNDict()) {
281  memcpy(getCreateDict(), _dict, _ndict * sizeof(W));
282  }
283  if (getNData()) {
284  memcpy(getCreateData(), _data, _ndata * sizeof(W));
285  }
286  if (getNLiterals()) {
287  memcpy(getCreateLiterals(), _literals, _nliterals * sizeof(W));
288  }
289  }
290  realignBlock();
291  }
292 
294  void relocate(const char* oldHead, char* newHeadData, char* newHeadRegistry)
295  {
296  payload = relocatePointer(oldHead, newHeadData, payload);
297  registry = relocatePointer(oldHead, newHeadRegistry, registry);
298  }

300  ClassDefNV(Block, 1);
301 }; // namespace ctf

471 template <typename H, int N, typename W>
472 template <typename VD>
473 void EncodedBlocks<H, N, W>::readFromTree(VD& vec, TTree& tree, const std::string& name, int ev)
474 {
475  auto tmp = create(vec);
476  readTreeBranch(tree, o2::utils::concat_string(name, "_wrapper."), *tmp, ev);
477  tmp = tmp->expand(vec, tmp->estimateSizeFromMetadata());
478  for (int i = 0; i < N; i++) {
479  Block<W> bl;
480  readTreeBranch(tree, o2::utils::concat_string(name, "_block.", std::to_string(i), "."), bl);
481  tmp->mBlocks[i].store(bl.getNDict(), bl.getNData(), bl.getNLiterals(), bl.getDict(), bl.getData(), bl.getLiterals());
482  }
483 }

487 template <typename H, int N, typename W>
488 void EncodedBlocks<H, N, W>::appendToTree(TTree& tree, const std::string& name) const
489 {
490  fillTreeBranch(tree, o2::utils::concat_string(name, "_wrapper."), const_cast<base&>(*this), WrappersCompressionLevel, WrappersSplitLevel);
491  for (int i = 0; i < N; i++) {
492  int compression = mMetadata[i].opt == Metadata::OptStore::ROOTCompression ? 1 : 0;
493  fillTreeBranch(tree, o2::utils::concat_string(name, "_block.", std::to_string(i), "."), const_cast<Block<W>&>(mBlocks[i]), compression);
494  }
495  tree.SetEntries(tree.GetEntries() + 1);
496 }

581

584  buffer.resize(alignSize(newsizeBytes) / buftypesize);

595 void EncodedBlocks<H, N, W>::relocate(const char* oldHead, char* newHead, char* wrapper, size_t newsize)
596 {

598  for (int i = 0; i < N; i++) {
599  newStr->mBlocks[i].relocate(oldHead, newHead, wrapper);

601  newStr->mRegistry.head = newHead; // newHead points on the real data

604  assert(newStr->estimateSize() <= newsize);
605  newStr->mRegistry.size = newsize;

698 template <typename H, int N, typename W>
699 template <typename D_IT, std::enable_if_t<detail::is_iterator_v<D_IT>, bool>>
700 void EncodedBlocks<H, N, W>::decode(D_IT dest, // iterator to destination
701  int slot, // slot of the block to decode
702  const void* decoderExt) const // optional externally provided decoder
703 {
704  // get references to the right data
705  const auto& block = mBlocks[slot];
706  const auto& md = mMetadata[slot];
707 
708  using dest_t = typename std::iterator_traits<D_IT>::value_type;
709 
710  // decode
711  if (block.getNStored()) {
712  if (md.opt == Metadata::OptStore::EENCODE) {
713  if (!decoderExt && !block.getNDict()) {
714  LOG(ERROR) << "Dictionaty is not saved for slot " << slot << " and no external decoder is provided";
715  throw std::runtime_error("Dictionary is not saved and no external decoder provided");
716  }
717  const o2::rans::LiteralDecoder64<dest_t>* decoder = reinterpret_cast<const o2::rans::LiteralDecoder64<dest_t>*>(decoderExt);
718  std::unique_ptr<o2::rans::LiteralDecoder64<dest_t>> decoderLoc;
719  if (block.getNDict()) { // if dictionaty is saved, prefer it
720  o2::rans::FrequencyTable frequencies;
721  frequencies.addFrequencies(block.getDict(), block.getDict() + block.getNDict(), md.min, md.max);
722  decoderLoc = std::make_unique<o2::rans::LiteralDecoder64<dest_t>>(frequencies, md.probabilityBits);
723  decoder = decoderLoc.get();
724  } else { // verify that decoded corresponds to stored metadata
725  if (md.min != decoder->getMinSymbol() || md.max != decoder->getMaxSymbol()) {
726  LOG(ERROR) << "Mismatch between min=" << md.min << "/" << md.max << " symbols in metadata and those in external decoder "
727  << decoder->getMinSymbol() << "/" << decoder->getMaxSymbol() << " for slot " << slot;
728  throw std::runtime_error("Mismatch between min/max symbols in metadata and those in external decoder");
729  }
730  }
731  // load incompressible symbols if they existed
732  std::vector<dest_t> literals;
733  if (block.getNLiterals()) {
734  // note: here we have to use md.nLiterals (original number of literal words) rather than md.nLiteralWords == block.getNLiterals()
735  // (number of W-words in the EncodedBlock occupied by literals) as we cast literals stored in W-word array
736  // to D-word array
737  literals = std::vector<dest_t>{reinterpret_cast<const dest_t*>(block.getLiterals()), reinterpret_cast<const dest_t*>(block.getLiterals()) + md.nLiterals};
738  }
739  decoder->process(dest, block.getData() + block.getNData(), md.messageLength, literals);
740  } else { // data was stored as is
741  using destPtr_t = typename std::iterator_traits<D_IT>::pointer;
742  destPtr_t srcBegin = reinterpret_cast<destPtr_t>(block.payload);
743  destPtr_t srcEnd = srcBegin + md.messageLength * sizeof(dest_t);
744  std::copy(srcBegin, srcEnd, dest);
745  //std::memcpy(dest, block.payload, md.messageLength * sizeof(dest_t));
746  }
747  }
748 }

751 template <typename H, int N, typename W>
752 template <typename S_IT, typename VB>
753 void EncodedBlocks<H, N, W>::encode(const S_IT srcBegin, // iterator begin of source message
754  const S_IT srcEnd, // iterator end of source message
755  int slot, // slot in encoded data to fill
756  uint8_t probabilityBits, // encoding into
757  Metadata::OptStore opt, // option for data compression
758  VB* buffer, // optional buffer (vector) providing memory for encoded blocks
759  const void* encoderExt) // optional external encoder
760 {
761  // fill a new block
762  assert(slot == mRegistry.nFilledBlocks);
763  mRegistry.nFilledBlocks++;
764  using STYP = typename std::iterator_traits<S_IT>::value_type;
765  using stream_t = typename o2::rans::Encoder64<STYP>::stream_t;
766 
767  const size_t messageLength = std::distance(srcBegin, srcEnd);
768  // cover three cases:
769  // * empty source message: no entropy coding
770  // * source message to pass through without any entropy coding
771  // * source message where entropy coding should be applied
772 
773  // case 1: empty source message
774  if (messageLength == 0) {
775  mMetadata[slot] = Metadata{0, 0, sizeof(uint64_t), sizeof(stream_t), probabilityBits, Metadata::OptStore::NODATA, 0, 0, 0, 0, 0};
776  return;
777  }
778  static_assert(std::is_same<W, stream_t>());
779 
780  Metadata md;
781  auto* bl = &mBlocks[slot];
782  auto* meta = &mMetadata[slot];
783 
784  // resize underlying buffer of block if necessary and update all pointers.
785  auto expandStorage = [&](int nElems) {
786  auto eeb = get(bl->registry->head); // extract pointer from the block, as "this" might be invalid
787  auto szNeed = eeb->estimateBlockSize(nElems); // size in bytes!!!
788  if (szNeed >= bl->registry->getFreeSize()) {
789  LOG(INFO) << "Slot " << slot << ": free size: " << bl->registry->getFreeSize() << ", need " << szNeed << " for " << nElems << " words";
790  if (buffer) {
791  eeb->expand(*buffer, size() + (szNeed - getFreeSize()));
792  meta = &(get(buffer->data())->mMetadata[slot]);
793  bl = &(get(buffer->data())->mBlocks[slot]); // in case of resizing this and any this.xxx becomes invalid
794  } else {
795  throw std::runtime_error("no room for encoded block in provided container");
796  }
797  }
798  };
799 
800  // case 3: message where entropy coding should be applied
801  if (opt == Metadata::OptStore::EENCODE) {
802  // build symbol statistics
803  constexpr size_t SizeEstMarginAbs = 10 * 1024;
804  constexpr float SizeEstMarginRel = 1.05;
805  const o2::rans::LiteralEncoder64<STYP>* encoder = reinterpret_cast<const o2::rans::LiteralEncoder64<STYP>*>(encoderExt);
806  std::unique_ptr<o2::rans::LiteralEncoder64<STYP>> encoderLoc;
807  std::unique_ptr<o2::rans::FrequencyTable> frequencies = nullptr;
808  int dictSize = 0;
809  if (!encoder) { // no external encoder provide, create one on spot
810  frequencies = std::make_unique<o2::rans::FrequencyTable>();
811  frequencies->addSamples(srcBegin, srcEnd);
812  encoderLoc = std::make_unique<o2::rans::LiteralEncoder64<STYP>>(*frequencies, probabilityBits);
813  encoder = encoderLoc.get();
814  dictSize = frequencies->size();
815  }
816 
817  // estimate size of encode buffer
818  int dataSize = rans::calculateMaxBufferSize(messageLength, encoder->getAlphabetRangeBits(), sizeof(STYP)); // size in bytes
819  // preliminary expansion of storage based on dict size + estimated size of encode buffer
820  dataSize = SizeEstMarginAbs + int(SizeEstMarginRel * (dataSize / sizeof(W))) + (sizeof(STYP) < sizeof(W)); // size in words of output stream
821  expandStorage(dictSize + dataSize);
822  //store dictionary first
823  if (dictSize) {
824  bl->storeDict(dictSize, frequencies->data());
825  }
826  // vector of incompressible literal symbols
827  std::vector<STYP> literals;
828  // directly encode source message into block buffer.
829  auto blIn = bl->getCreateData();
830  auto frSize = bl->registry->getFreeSize(); // note: "this" might be not valid after expandStorage call!!!
831  const auto encodedMessageEnd = encoder->process(blIn, blIn + frSize, srcBegin, srcEnd, literals);
832  dataSize = encodedMessageEnd - bl->getData();
833  bl->setNData(dataSize);
834  bl->realignBlock();
835  // update the size claimed by encode message directly inside the block
836  // store incompressible symbols if any
837 
838  int literalSize = 0;
839  if (literals.size()) {
840  literalSize = (literals.size() * sizeof(STYP)) / sizeof(stream_t) + (sizeof(STYP) < sizeof(stream_t));
841  expandStorage(literalSize);
842  bl->storeLiterals(literalSize, reinterpret_cast<const stream_t*>(literals.data()));
843  }
844  *meta = Metadata{messageLength, literals.size(), sizeof(uint64_t), sizeof(stream_t), static_cast<uint8_t>(encoder->getProbabilityBits()), opt,
845  encoder->getMinSymbol(), encoder->getMaxSymbol(), dictSize, dataSize, literalSize};
846 
847  } else { // store original data w/o EEncoding
848  const size_t szb = messageLength * sizeof(STYP);
849  const int dataSize = szb / sizeof(stream_t) + (sizeof(STYP) < sizeof(stream_t));
850  // no dictionary needed
851  expandStorage(dataSize);
852  *meta = Metadata{messageLength, 0, sizeof(uint64_t), sizeof(stream_t), probabilityBits, opt, 0, 0, 0, dataSize, 0};
853  //FIXME: no we don't need an intermediate vector.
854  // provided iterator is not necessarily pointer, need to use intermediate vector!!!
855  std::vector<STYP> vtmp(srcBegin, srcEnd);
856  bl->storeData(meta->nDataWords, reinterpret_cast<const W*>(vtmp.data()));
857  }
858  // resize block if necessary
859 }

862 template <typename H, int N, typename W>
863 std::vector<char> EncodedBlocks<H, N, W>::createDictionaryBlocks(const std::vector<o2::rans::FrequencyTable>& vfreq, const std::vector<Metadata>& vmd)
864 {
865  if (vfreq.size() != N) {
866  throw std::runtime_error("mismatch between the size of frequencies vector and number of blocks");
867  }
868  size_t sz = alignSize(sizeof(EncodedBlocks<H, N, W>));
869  for (int ib = 0; ib < N; ib++) {
870  sz += Block<W>::estimateSize(vfreq[ib].size());
871  }
872  std::vector<char> vdict(sz); // memory space for dictionary
873  auto dictBlocks = create(vdict.data(), sz);
874  for (int ib = 0; ib < N; ib++) {
875  if (vfreq[ib].size()) {
876  LOG(INFO) << "adding dictionary of " << vfreq[ib].size() << " words for block " << ib << ", min/max= " << vfreq[ib].getMinSymbol() << "/" << vfreq[ib].getMaxSymbol();
877  dictBlocks->mBlocks[ib].storeDict(vfreq[ib].size(), vfreq[ib].data());
878  dictBlocks = get(vdict.data()); // !!! rellocation might have invalidated dictBlocks pointer
879  dictBlocks->mMetadata[ib] = vmd[ib];
880  dictBlocks->mMetadata[ib].opt = Metadata::OptStore::ROOTCompression; // we will compress the dictionary with root!
881  dictBlocks->mBlocks[ib].realignBlock();
882  } else {
883  dictBlocks->mMetadata[ib].opt = Metadata::OptStore::NONE;
884  }
885  dictBlocks->mRegistry.nFilledBlocks++;
886  }
887  return std::move(vdict);
888 }
