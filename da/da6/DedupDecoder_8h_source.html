52 template <typename stream_IT, typename source_IT, std::enable_if_t<internal::isCompatibleIter_v<stream_T, stream_IT> && internal::isCompatibleIter_v<source_T, source_IT>, bool>>
53 void DedupDecoder<coder_T, stream_T, source_T>::process(const source_IT outputBegin, const stream_IT inputEnd, size_t messageLength, duplicatesMap_t& duplicates) const
54 {
55  using namespace internal;
56  using ransDecoder = internal::Decoder<coder_T, stream_T>;
57  LOG(trace) << "start decoding";
58  RANSTimer t;
59  t.start();
60  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
61  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
62 
63  if (messageLength == 0) {
64  LOG(warning) << "Empty message passed to decoder, skipping decode process";
65  return;
66  }
67 
68  stream_IT inputIter = inputEnd;
69  source_IT it = outputBegin;
70 
71  // make Iter point to the last last element
72  --inputIter;
73 
74  ransDecoder rans;
75  inputIter = rans.init(inputIter);
76 
77  for (size_t i = 0; i < (messageLength); i++) {
78  const auto s = (*this->mReverseLUT)[rans.get(this->mProbabilityBits)];
79 
80  // deduplication
81  auto duplicatesIter = duplicates.find(i);
82  if (duplicatesIter != duplicates.end()) {
83  LOG(trace) << "pos[" << i << "]: restoring " << duplicatesIter->second << " duplicates of symbol " << (char)s;
84  for (int d = 0; d < duplicatesIter->second; d++) {
85  *it++ = s;
86  i++;
87  }
88  }
89  *it++ = s;
90  inputIter = rans.advanceSymbol(inputIter, (*this->mSymbolTable)[s], this->mProbabilityBits);
91  }
92 
93  t.stop();
94  LOG(debug1) << "Decoder::" << __func__ << " { DecodedSymbols: " << messageLength << ","
95  << "processedBytes: " << messageLength * sizeof(source_T) << ","
96  << " inclusiveTimeMS: " << t.getDurationMS() << ","
97  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (messageLength * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
98 
99  LOG(trace) << "done decoding";
100 }
