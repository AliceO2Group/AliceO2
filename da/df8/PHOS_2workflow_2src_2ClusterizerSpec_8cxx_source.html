28 void ClusterizerSpec::run(framework::ProcessingContext& ctx)
29 {
30  if (mUseDigits) {
31  LOG(DEBUG) << "PHOSClusterizer - run on digits called";
32 
33  auto dataref = ctx.inputs().get("digits");
34  auto const* phosheader = o2::framework::DataRefUtils::getHeader<o2::phos::PHOSBlockHeader*>(dataref);
35  if (!phosheader->mHasPayload) {
36  LOG(DEBUG) << "[PHOSClusterizer - run] No more digits" << std::endl;
37  ctx.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
38  return;
39  }
40 
41  // auto digits = ctx.inputs().get<gsl::span<o2::phos::Digit>>("digits");
42  auto digits = ctx.inputs().get<std::vector<o2::phos::Digit>>("digits");
43  auto digitsTR = ctx.inputs().get<std::vector<o2::phos::TriggerRecord>>("digitTriggerRecords");
44  LOG(DEBUG) << "[PHOSClusterizer - run] Received " << digitsTR.size() << " TR, running clusterizer ...";
45  // const o2::dataformats::MCTruthContainer<MCLabel>* truthcont=nullptr;
46  if (mPropagateMC) {
47  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::phos::MCLabel>> truthcont(ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::phos::MCLabel>*>("digitsmctr"));
48  mClusterizer.process(digits, digitsTR, truthcont.get(), &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters on digits (pass by ref)
49  } else {
50  mClusterizer.process(digits, digitsTR, nullptr, &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters on digits (pass by ref)
51  }
52  } else {
53 
54  LOG(DEBUG) << "PHOSClusterizer - run run on cells called";
55 
56  auto cells = ctx.inputs().get<std::vector<o2::phos::Cell>>("cells");
57  // auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
58  LOG(DEBUG) << "[PHOSClusterizer - run] Received " << cells.size() << " cells, running clusterizer ...";
59  // auto cellsTR = ctx.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cellTriggerRecords");
60  auto cellsTR = ctx.inputs().get<std::vector<o2::phos::TriggerRecord>>("cellTriggerRecords");
61  if (mPropagateMC) {
62  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::phos::MCLabel>> truthcont(ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::phos::MCLabel>*>("cellsmctr"));
63  // truthmap = ctx.inputs().get<gsl::span<uint>>("cellssmcmap");
64  mClusterizer.processCells(cells, cellsTR, truthcont.get(), &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters on digits (pass by ref)
65  } else {
66  mClusterizer.processCells(cells, cellsTR, nullptr, &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters on digits (pass by ref)
67  }
68  }
69 
70  if (mPropagateMC) {
71  LOG(DEBUG) << "[PHOSClusterizer - run] Writing " << mOutputClusters.size() << " clusters, " << mOutputClusterTrigRecs.size() << "TR and " << mOutputTruthCont.getIndexedSize() << " Labels";
72  } else {
73  LOG(DEBUG) << "[PHOSClusterizer - run] Writing " << mOutputClusters.size() << " clusters and " << mOutputClusterTrigRecs.size() << " TR";
74  }
75  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusters);
76  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRIGRECS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusterTrigRecs);
77  if (mPropagateMC) {
78  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
79  }
80  ctx.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
81 }

83 o2::framework::DataProcessorSpec o2::phos::reco_workflow::getClusterizerSpec(bool propagateMC)
84 {
85  std::vector<o2::framework::InputSpec> inputs;
86  std::vector<o2::framework::OutputSpec> outputs;
87  inputs.emplace_back("digits", o2::header::gDataOriginPHS, "DIGITS", 0, o2::framework::Lifetime::Timeframe);
88  inputs.emplace_back("digitTriggerRecords", o2::header::gDataOriginPHS, "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe);
89  if (propagateMC) {
90  inputs.emplace_back("digitsmctr", "PHS", "DIGITSMCTR", 0, o2::framework::Lifetime::Timeframe);
91  }
92  outputs.emplace_back("PHS", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe);
93  outputs.emplace_back("PHS", "CLUSTERTRIGRECS", 0, o2::framework::Lifetime::Timeframe);
94  if (propagateMC) {
95  outputs.emplace_back("PHS", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe);
96  }
97 
98  return o2::framework::DataProcessorSpec{"PHOSClusterizerSpec",
99  inputs,
100  outputs,
101  o2::framework::adaptFromTask<o2::phos::reco_workflow::ClusterizerSpec>(propagateMC, true)};
102 }
