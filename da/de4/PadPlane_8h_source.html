21 // TRD pad plane class //
22 // //
23 // Contains the information on ideal pad positions, pad dimensions, //
24 // tilting angle, etc. //
25 // It also provides methods to identify the current pad number from //
26 // local tracking coordinates. //
27 // //

76  GPUd() double getTiltOffset(double rowOffset) const { return mTiltingTan * (rowOffset - 0.5 * mLengthIPad); };
77  GPUd() double getPadRowOffset(int row, double z) const
78  {
79  if ((row < 0) || (row >= mNrows)) {
80  return -1.0;
81  } else {
82  return mPadRow[row] + mPadRowSMOffset - z;
83  }
84  };
85  GPUd() double getPadRowOffsetROC(int row, double z) const
86  {
87  if ((row < 0) || (row >= mNrows)) {
88  return -1.0;
89  } else {
90  return mPadRow[row] - z;
91  }
92  };
93 
94  GPUd() double getPadColOffset(int col, double rphi) const
95  {
96  if ((col < 0) || (col >= mNcols)) {
97  return -1.0;
98  } else {
99  return rphi - mPadCol[col];
100  }
101  };

111  GPUd() double getColEnd() const { return mPadCol[mNcols - 1] + mWidthOPad; };
112  GPUd() double getRowPos(int row) const { return mPadRow[row] + mPadRowSMOffset; };
113  GPUd() double getRowPosROC(int row) const { return mPadRow[row]; };
114  GPUd() double getColPos(int col) const { return mPadCol[col]; };
115  GPUd() double getRowSize(int row) const
116  {
117  if ((row == 0) || (row == mNrows - 1)) {
118  return mLengthOPad;
119  } else {
120  return mLengthIPad;
121  }
122  };

150  double mWidth; // Width of pad plane in rphi-direction (col)

154

159  double mWidthIPad; // Width of an inner pad in rphi-direction (col)
160 
161  double mRowSpacing; // Spacing between the pad rows
162  double mColSpacing; // Spacing between the pad columns
163 
164  int mNrows; // Number of rows
165  int mNcols; // Number of columns
166 
167  double mTiltingAngle; // Pad tilting angle
168  double mTiltingTan; // Tangens of pad tilting angle
169 
170  double mPadRow[MAXROWS]; // Pad border positions in row direction
171  double mPadCol[MAXCOLS]; // Pad border positions in column direction
172 
173  double mPadRowSMOffset; // To be added to translate local ROC system to local SM system
174 
175  double mAnodeWireOffset; // Distance of first anode wire from pad edge
176 
177  private:
178  ClassDefNV(PadPlane, 1); // TRD ROC pad plane
179 };
