31  // associated ConfigParamSpec. Optionally provide a prototype object @a proto
32  // to use for the defaults.
33  template <typename T>
34  static std::vector<ConfigParamSpec> asConfigParamSpecs(std::string const& mainKey, T const& proto = T{})
35  {
36  auto cl = TClass::GetClass<T>();
37  if (!cl) {
