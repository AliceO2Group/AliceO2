131 template <typename DataT>
132 int Spline2DHelper<DataT>::setSpline(
133  const Spline2DContainer<DataT>& spline, int nAuxiliaryPointsU, int nAuxiliaryPointsV)
134 {
135  // Prepare creation of 2D irregular spline
136  // The should be at least one (better, two) Auxiliary measurements on each segnment between two knots and at least 2*nKnots measurements in total
137  // Returns 0 when the spline can not be constructed with the given nAuxiliaryPoints
138 
139  int ret = 0;
140  mFdimensions = spline.getYdimensions();
141  if (mHelperU1.setSpline(spline.getGridX1(), mFdimensions, nAuxiliaryPointsU) != 0) {
142  ret = storeError(-2, "Spline2DHelper::setSpline2D: error by setting U axis");
143  }
144  if (mHelperU2.setSpline(spline.getGridX2(), mFdimensions, nAuxiliaryPointsV) != 0) {
145  ret = storeError(-3, "Spline2DHelper::setSpline2D: error by setting V axis");
146  }
147  return ret;
148 }
