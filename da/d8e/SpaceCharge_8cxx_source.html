30 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
31 void SpaceCharge<DataT, Nz, Nr, Nphi>::calculateDistortionsCorrections(const o2::tpc::Side side)
32 {
33  using timer = std::chrono::high_resolution_clock;
34  using SC = o2::tpc::SpaceCharge<DataT, Nz, Nr, Nphi>;
35  if (!mIsChargeSet[side]) {
36  LOGP(ERROR, "the charge is not set!");
37  }
38 
39  const std::array<std::string, 2> sglobalType{"local distortion/correction interpolator", "Electric fields"};
40  const std::array<std::string, 2> sglobalDistType{"Standard method", "interpolation of global corrections"};
41  const std::array<std::string, 2> sideName{"A", "C"};
42 
43  LOGP(info, "====== starting calculation of distortions and corrections for Side {} ======", sideName[side]);
44  LOGP(info, "Using {} threads", getNThreads());
45 
46  if (getGlobalDistCorrMethod() == SC::GlobalDistCorrMethod::LocalDistCorr) {
47  LOGP(info, "calculation of global distortions and corrections are performed by using: {}", sglobalType[0]);
48  } else {
49  LOGP(info, "calculation of global distortions and corrections are performed by using: {}", sglobalType[1]);
50  }
51 
52  if (getGlobalDistType() == SC::GlobalDistType::Fast) {
53  LOGP(info, "calculation of global distortions performed by following method: {}", sglobalDistType[1]);
54  } else if (getGlobalDistType() == SC::GlobalDistType::Standard) {
55  LOGP(info, "calculation of global distortions performed by following method: {}", sglobalDistType[0]);
56  } else {
57  LOGP(info, "skipping calculation of global distortions");
58  }
59 
60  auto startTotal = timer::now();
61 
62  auto start = timer::now();
63  poissonSolver(side);
64  auto stop = timer::now();
65  std::chrono::duration<float> time = stop - start;
66  LOGP(info, "Poisson Solver time: {}", time.count());
67 
68  start = timer::now();
69  calcEField(side);
70  stop = timer::now();
71  time = stop - start;
72  LOGP(info, "electric field calculation time: {}", time.count());
73 
74  const auto numEFields = getElectricFieldsInterpolator(side);
75  if (getGlobalDistType() == SC::GlobalDistType::Standard) {
76  start = timer::now();
77  const auto dist = o2::tpc::SpaceCharge<DataT, Nz, Nr, Nphi>::Type::Distortions;
78  calcLocalDistortionsCorrections(dist, numEFields); // local distortion calculation
79  stop = timer::now();
80  time = stop - start;
81  LOGP(info, "local distortions time: {}", time.count());
82  } else {
83  LOGP(info, "skipping local distortions (not needed)");
84  }
85 
86  start = timer::now();
87  const auto corr = o2::tpc::SpaceCharge<DataT, Nz, Nr, Nphi>::Type::Corrections;
88  calcLocalDistortionsCorrections(corr, numEFields); // local correction calculation
89  stop = timer::now();
90  time = stop - start;
91  LOGP(info, "local corrections time: {}", time.count());
92 
93  start = timer::now();
94  const auto lCorrInterpolator = getLocalCorrInterpolator(side);
95  (getGlobalDistCorrMethod() == SC::GlobalDistCorrMethod::LocalDistCorr) ? calcGlobalCorrections(lCorrInterpolator) : calcGlobalCorrections(numEFields);
96  stop = timer::now();
97  time = stop - start;
98  LOGP(info, "global corrections time: {}", time.count());
99  start = timer::now();
100  if (getGlobalDistType() == SC::GlobalDistType::Fast) {
101  const auto globalCorrInterpolator = getGlobalCorrInterpolator(side);
102  calcGlobalDistWithGlobalCorrIterative(globalCorrInterpolator);
103  } else if (getGlobalDistType() == SC::GlobalDistType::Standard) {
104  const auto lDistInterpolator = getLocalDistInterpolator(side);
105  (getGlobalDistCorrMethod() == SC::GlobalDistCorrMethod::LocalDistCorr) ? calcGlobalDistortions(lDistInterpolator) : calcGlobalDistortions(numEFields);
106  } else {
107  }
108 
109  stop = timer::now();
110  time = stop - start;
111  LOGP(info, "global distortions time: {}", time.count());
112 
113  stop = timer::now();
114  time = stop - startTotal;
115  LOGP(info, "everything is done. Total Time: {}", time.count());
116 }

249 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
250 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcEField(const Side side)
251 {
252 #pragma omp parallel for num_threads(sNThreads)
253  for (size_t iPhi = 0; iPhi < Nphi; ++iPhi) {
254  const int symmetry = 0;
255  size_t tmpPlus = iPhi + 1;
256  int signPlus = 1;
257  int tmpMinus = static_cast<int>(iPhi - 1);
258  int signMinus = 1;
259  if (symmetry == 1 || symmetry == -1) { // Reflection symmetry in phi (e.g. symmetry at sector boundaries, or half sectors, etc.)
260  if (tmpPlus > Nphi - 1) {
261  if (symmetry == -1) {
262  signPlus = -1;
263  }
264  tmpPlus = Nphi - 2;
265  }
266  if (tmpMinus < 0) {
267  tmpMinus = 1; // SHOULD IT BE =0?
268  if (symmetry == -1) {
269  signMinus = -1;
270  }
271  }
272  } else { // No Symmetries in phi, no boundaries, the calculations is continuous across all phi
273  if (tmpPlus > Nphi - 1) {
274  tmpPlus = iPhi + 1 - Nphi;
275  }
276  if (tmpMinus < 0) {
277  tmpMinus = static_cast<int>(iPhi - 1 + Nphi);
278  }
279  }
280 
281  // for non-boundary V
282  for (size_t iR = 1; iR < Nr - 1; iR++) {
283  const DataT radius = getRVertex(iR, side);
284  for (size_t iZ = 1; iZ < Nz - 1; iZ++) {
285  mElectricFieldEr[side](iZ, iR, iPhi) = -1 * (mPotential[side](iZ, iR + 1, iPhi) - mPotential[side](iZ, iR - 1, iPhi)) * static_cast<DataT>(0.5) * getInvSpacingR(side); // r direction
286  mElectricFieldEz[side](iZ, iR, iPhi) = -1 * (mPotential[side](iZ + 1, iR, iPhi) - mPotential[side](iZ - 1, iR, iPhi)) * static_cast<DataT>(0.5) * getInvSpacingZ(side); // z direction
287  mElectricFieldEphi[side](iZ, iR, iPhi) = -1 * (signPlus * mPotential[side](iZ, iR, tmpPlus) - signMinus * mPotential[side](iZ, iR, tmpMinus)) * static_cast<DataT>(0.5) * getInvSpacingPhi(side) / radius; // phi direction
288  }
289  }
290 
291  // for boundary-r
292  for (size_t iZ = 0; iZ < Nz; iZ++) {
293  mElectricFieldEr[side](iZ, 0, iPhi) = -1 * (-static_cast<DataT>(0.5) * mPotential[side](iZ, 2, iPhi) + 2 * mPotential[side](iZ, 1, iPhi) - static_cast<DataT>(1.5) * mPotential[side](iZ, 0, iPhi)) * getInvSpacingR(side); // forward difference
294  mElectricFieldEr[side](iZ, Nr - 1, iPhi) = -1 * (static_cast<DataT>(1.5) * mPotential[side](iZ, Nr - 1, iPhi) - 2 * mPotential[side](iZ, Nr - 2, iPhi) + static_cast<DataT>(0.5) * mPotential[side](iZ, Nr - 3, iPhi)) * getInvSpacingR(side); // backward difference
295  }
296 
297  for (size_t iR = 0; iR < Nr; iR += Nr - 1) {
298  const DataT radius = getRVertex(iR, side);
299  for (size_t iZ = 1; iZ < Nz - 1; iZ++) {
300  mElectricFieldEz[side](iZ, iR, iPhi) = -1 * (mPotential[side](iZ + 1, iR, iPhi) - mPotential[side](iZ - 1, iR, iPhi)) * static_cast<DataT>(0.5) * getInvSpacingZ(side); // z direction
301  mElectricFieldEphi[side](iZ, iR, iPhi) = -1 * (signPlus * mPotential[side](iZ, iR, tmpPlus) - signMinus * mPotential[side](iZ, iR, tmpMinus)) * static_cast<DataT>(0.5) * getInvSpacingPhi(side) / radius; // phi direction
302  }
303  }
304 
305  // for boundary-z
306  for (size_t iR = 0; iR < Nr; ++iR) {
307  mElectricFieldEz[side](0, iR, iPhi) = -1 * (-static_cast<DataT>(0.5) * mPotential[side](2, iR, iPhi) + 2 * mPotential[side](1, iR, iPhi) - static_cast<DataT>(1.5) * mPotential[side](0, iR, iPhi)) * getInvSpacingZ(side);
308  mElectricFieldEz[side](Nz - 1, iR, iPhi) = -1 * (static_cast<DataT>(1.5) * mPotential[side](Nz - 1, iR, iPhi) - 2 * mPotential[side](Nz - 2, iR, iPhi) + static_cast<DataT>(0.5) * mPotential[side](Nz - 3, iR, iPhi)) * getInvSpacingZ(side);
309  }
310 
311  for (size_t iR = 1; iR < Nr - 1; ++iR) {
312  const DataT radius = getRVertex(iR, side);
313  for (size_t iZ = 0; iZ < Nz; iZ += Nz - 1) {
314  mElectricFieldEr[side](iZ, iR, iPhi) = -1 * (mPotential[side](iZ, iR + 1, iPhi) - mPotential[side](iZ, iR - 1, iPhi)) * static_cast<DataT>(0.5) * getInvSpacingR(side); // r direction
315  mElectricFieldEphi[side](iZ, iR, iPhi) = -1 * (signPlus * mPotential[side](iZ, iR, tmpPlus) - signMinus * mPotential[side](iZ, iR, tmpMinus)) * static_cast<DataT>(0.5) * getInvSpacingPhi(side) / radius; // phi direction
316  }
317  }
318 
319  // corner points for EPhi
320  for (size_t iR = 0; iR < Nr; iR += Nr - 1) {
321  const DataT radius = getRVertex(iR, side);
322  for (size_t iZ = 0; iZ < Nz; iZ += Nz - 1) {
323  mElectricFieldEphi[side](iZ, iR, iPhi) = -1 * (signPlus * mPotential[side](iZ, iR, tmpPlus) - signMinus * mPotential[side](iZ, iR, tmpMinus)) * static_cast<DataT>(0.5) * getInvSpacingPhi(side) / radius; // phi direction
324  }
325  }
326  }
327  mIsEfieldSet[side] = true;
328 }

330 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
331 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcGlobalDistWithGlobalCorrIterative(const DistCorrInterpolator<DataT, Nz, Nr, Nphi>& globCorr, const int maxIter, const DataT approachZ, const DataT approachR, const DataT approachPhi, const DataT diffCorr)
332 {
333  const Side side = globCorr.getSide();
334 
335 #pragma omp parallel for num_threads(sNThreads)
336  for (unsigned int iPhi = 0; iPhi < Nphi; ++iPhi) {
337  const DataT phi = getPhiVertex(iPhi, side);
338  for (unsigned int iR = 0; iR < Nr; ++iR) {
339  const DataT radius = getRVertex(iR, side);
340  for (unsigned int iZ = 0; iZ < Nz; ++iZ) {
341  const DataT z = getZVertex(iZ, side);
342 
343  unsigned int nearestiZ = iZ;
344  unsigned int nearestiR = iR;
345  unsigned int nearestiPhi = iPhi;
346 
347  DataT nearestZ = getZVertex(nearestiZ, side);
348  DataT nearestR = getRVertex(nearestiR, side);
349  DataT nearestPhi = getPhiVertex(nearestiPhi, side);
350 
351  //
352  //==========================================================================================
353  //==== start algorithm: use tricubic upsampling to numerically approach the query point ====
354  //==========================================================================================
355  //
356  // 1. calculate difference from nearest point to query point with stepwidth factor x
357  // and approach the new point
358  //
359  DataT stepR = (radius - nearestR) * approachR;
360  DataT stepZ = (z - nearestZ) * approachZ;
361  DataT stepPhi = (phi - nearestPhi) * approachPhi;
362 
363  // needed to check for convergence
364  DataT lastCorrdR = std::numeric_limits<DataT>::max();
365  DataT lastCorrdZ = std::numeric_limits<DataT>::max();
366  DataT lastCorrdRPhi = std::numeric_limits<DataT>::max();
367 
368  // interpolated global correction
369  DataT corrdR = 0;
370  DataT corrdRPhi = 0;
371  DataT corrdZ = 0;
372 
373  for (int iter = 0; iter < maxIter; ++iter) {
374  // 2. get new point coordinates
375  const DataT rCurrPos = getRVertex(nearestiR, side) + stepR;
376  const DataT zCurrPos = getZVertex(nearestiZ, side) + stepZ;
377  const DataT phiCurrPos = getPhiVertex(nearestiPhi, side) + stepPhi;
378 
379  // interpolate global correction at new point and calculate position of global correction
380  // corrdR = globCorr.evalSparsedR(zCurrPos, rCurrPos, phiCurrPos);
381  corrdR = globCorr.evaldR(zCurrPos, rCurrPos, phiCurrPos);
382  const DataT rNewPos = rCurrPos + corrdR;
383 
384  // const DataT corrPhi = globCorr.evalSparsedRPhi(zCurrPos, rCurrPos, phiCurrPos) / rCurrPos;
385  const DataT corrPhi = globCorr.evaldRPhi(zCurrPos, rCurrPos, phiCurrPos) / rCurrPos;
386  corrdRPhi = corrPhi * rNewPos; // normalize to new r coordinate
387  const DataT phiNewPos = phiCurrPos + corrPhi;
388 
389  // corrdZ = globCorr.evalSparsedZ(zCurrPos, rCurrPos, phiCurrPos);
390  corrdZ = globCorr.evaldZ(zCurrPos, rCurrPos, phiCurrPos);
391  const DataT zNewPos = zCurrPos + corrdZ;
392 
393  // approach desired coordinate
394  stepR += (radius - rNewPos) * approachR;
395  stepZ += (z - zNewPos) * approachZ;
396  stepPhi += (phi - phiNewPos) * approachPhi;
397 
398  // check for convergence
399  const DataT diffCorrdR = std::abs(corrdR - lastCorrdR);
400  const DataT diffCorrdRZ = std::abs(corrdZ - lastCorrdZ);
401  const DataT diffCorrdRPhi = std::abs(corrdRPhi - lastCorrdRPhi);
402 
403  // stop algorithm if converged
404  if (diffCorrdR < diffCorr && diffCorrdRZ < diffCorr && diffCorrdRPhi < diffCorr) {
405  break;
406  }
407 
408  lastCorrdR = corrdR;
409  lastCorrdZ = corrdZ;
410  lastCorrdRPhi = corrdRPhi;
411  }
412  // set global distortions if algorithm converged or iterations exceed max numbers of iterations
413  mGlobalDistdR[side](iZ, iR, iPhi) = -corrdR;
414  mGlobalDistdRPhi[side](iZ, iR, iPhi) = -corrdRPhi;
415  mGlobalDistdZ[side](iZ, iR, iPhi) = -corrdZ;
416  }
417  }
418  }
419  // set flag that global distortions are set to true
420  mIsGlobalDistSet[side] = true;
421 }

626 TH3D SpaceCharge<DataT, Nz, Nr, Nphi>::rebinDensityHisto(const TH3& hOrig) const
627 {
628  TH3D hRebin{};
629 
630  const int nBinsPhiNew = Nphi;
631  const int nBinsRNew = Nr;
632  const int nBinsZNew = 2 * Nz;
633 
634  const auto phiLow = hOrig.GetXaxis()->GetBinLowEdge(1);
635  const auto phiUp = hOrig.GetXaxis()->GetBinUpEdge(hOrig.GetNbinsX());
636  const auto rLow = hOrig.GetYaxis()->GetBinLowEdge(1);
637  const auto rUp = hOrig.GetYaxis()->GetBinUpEdge(hOrig.GetNbinsY());
638  const auto zLow = hOrig.GetZaxis()->GetBinLowEdge(1);
639  const auto zUp = hOrig.GetZaxis()->GetBinUpEdge(hOrig.GetNbinsZ());
640  hRebin.SetBins(nBinsPhiNew, phiLow, phiUp, nBinsRNew, rLow, rUp, nBinsZNew, zLow, zUp);
641 
642  for (int iBinPhi = 1; iBinPhi <= nBinsPhiNew; ++iBinPhi) {
643  const auto phiLowEdge = hRebin.GetXaxis()->GetBinLowEdge(iBinPhi);
644  const auto phiUpEdge = hRebin.GetXaxis()->GetBinUpEdge(iBinPhi);
645 
646  const int phiLowBinOrig = hOrig.GetXaxis()->FindBin(phiLowEdge);
647  const int phiUpBinOrig = hOrig.GetXaxis()->FindBin(phiUpEdge);
648 
649  // calculate the weights (area of original bin lies in the new bin / binwidthOrig) of the first and last bins
650  const auto binWidthPhiOrig = hOrig.GetXaxis()->GetBinWidth(phiLowBinOrig);
651  const auto lowerBinWeightPhi = std::abs(phiLowEdge - hOrig.GetXaxis()->GetBinUpEdge(phiLowBinOrig)) / binWidthPhiOrig;
652  const auto upperBinWeightPhi = std::abs(phiUpEdge - hOrig.GetXaxis()->GetBinLowEdge(phiUpBinOrig)) / binWidthPhiOrig;
653 
654  for (int iBinR = 1; iBinR <= nBinsRNew; ++iBinR) {
655  const auto rLowEdge = hRebin.GetYaxis()->GetBinLowEdge(iBinR);
656  const auto rUpEdge = hRebin.GetYaxis()->GetBinUpEdge(iBinR);
657 
658  const int rLowBinOrig = hOrig.GetYaxis()->FindBin(rLowEdge);
659  const int rUpBinOrig = hOrig.GetYaxis()->FindBin(rUpEdge);
660 
661  // calculate the weights (area of original bin lies in the new bin / binwidthOrig) of the first and last bins
662  const auto binWidthROrig = hOrig.GetYaxis()->GetBinWidth(rLowBinOrig);
663  const auto lowerBinWeightR = std::abs(rLowEdge - hOrig.GetYaxis()->GetBinUpEdge(rLowBinOrig)) / binWidthROrig;
664  const auto upperBinWeightR = std::abs(rUpEdge - hOrig.GetYaxis()->GetBinLowEdge(rUpBinOrig)) / binWidthROrig;
665 
666  for (int iBinZ = 1; iBinZ <= nBinsZNew; ++iBinZ) {
667  const auto zLowEdge = hRebin.GetZaxis()->GetBinLowEdge(iBinZ);
668  const auto zUpEdge = hRebin.GetZaxis()->GetBinUpEdge(iBinZ);
669  const auto zCenter = hRebin.GetZaxis()->GetBinCenter(iBinZ);
670 
671  int zLowBinOrig = hOrig.GetZaxis()->FindBin(zLowEdge);
672  int zUpBinOrig = hOrig.GetZaxis()->FindBin(zUpEdge);
673  const int currside = getSide(zCenter); // set the side of the current z-bin
674  // get the side of the lowest and uppest bin from the orig histo
675  const int sideLowOrig = getSide(hOrig.GetZaxis()->GetBinCenter(zLowBinOrig));
676  const int sideUpOrig = getSide(hOrig.GetZaxis()->GetBinCenter(zUpBinOrig));
677 
678  // make bounds/side check of the zLowBinOrig and zUpBinOrig bins. They must be on the same side as the currside!!!
679  if (currside != sideLowOrig && zLowBinOrig != zUpBinOrig) {
680  // if the lower bins from the orig histo are not on the same side as the rebinned increase the binnumber until they are on the same side
681  bool notequal = true;
682  do {
683  zLowBinOrig += 1;
684  if (zLowBinOrig > zUpBinOrig) {
685  LOGP(WARNING, "SOMETHING WENT WRONG: SETTING BINS TO: {}", zUpBinOrig);
686  zLowBinOrig = zUpBinOrig;
687  notequal = false;
688  }
689  const int sideTmp = getSide(hOrig.GetZaxis()->GetBinCenter(zLowBinOrig));
690  if (sideTmp == currside) {
691  notequal = false;
692  }
693  } while (notequal);
694  }
695 
696  if (currside != sideUpOrig && zLowBinOrig != zUpBinOrig) {
697  // if the upper bins from the orig histo are not on the same side as the rebinned increase the binnumber until they are on the same side
698  bool notequal = true;
699  do {
700  zUpBinOrig -= 1;
701  if (zUpBinOrig < zLowBinOrig) {
702  LOGP(WARNING, "SOMETHING WENT WRONG: SETTING BINS TO: {}", zLowBinOrig);
703  zUpBinOrig = zLowBinOrig;
704  notequal = false;
705  }
706  const int sideTmp = getSide(hOrig.GetZaxis()->GetBinCenter(zUpBinOrig));
707  if (sideTmp == currside) {
708  notequal = false;
709  }
710  } while (notequal);
711  }
712 
713  const auto binWidthZOrig = hOrig.GetZaxis()->GetBinWidth(zLowBinOrig);
714  const auto lowerBinWeightZ = std::abs(zLowEdge - hOrig.GetZaxis()->GetBinUpEdge(zLowBinOrig)) / binWidthZOrig;
715  const auto upperBinWeightZ = std::abs(zUpEdge - hOrig.GetZaxis()->GetBinLowEdge(zUpBinOrig)) / binWidthZOrig;
716 
717  // get the mean value of the original histogram of the found bin range
718  DataT sum = 0;
719  DataT sumW = 0;
720  for (int iPhi = phiLowBinOrig; iPhi <= phiUpBinOrig; ++iPhi) {
721  DataT weightPhi = 1;
722  if (iPhi == phiLowBinOrig) {
723  weightPhi = lowerBinWeightPhi;
724  } else if (iPhi == phiUpBinOrig) {
725  weightPhi = upperBinWeightPhi;
726  }
727 
728  for (int iR = rLowBinOrig; iR <= rUpBinOrig; ++iR) {
729  DataT weightR = 1;
730  if (iR == rLowBinOrig) {
731  weightR = lowerBinWeightR;
732  } else if (iR == rUpBinOrig) {
733  weightR = upperBinWeightR;
734  }
735 
736  for (int iZ = zLowBinOrig; iZ <= zUpBinOrig; ++iZ) {
737  DataT weightZ = 1;
738  if (iZ == zLowBinOrig) {
739  weightZ = lowerBinWeightZ;
740  } else if (iZ == zUpBinOrig) {
741  weightZ = upperBinWeightZ;
742  }
743  const auto val = hOrig.GetBinContent(iPhi, iR, iZ);
744  // if(val==0){
745  // what to do now???
746  // }
747  const auto totalWeight = weightPhi * weightR * weightZ;
748  sum += val * totalWeight;
749  sumW += totalWeight;
750  }
751  }
752  }
753  sum /= sumW;
754  hRebin.SetBinContent(iBinPhi, iBinR, iBinZ, sum);
755  }
756  }
757  }
758  return hRebin;
759 }

761 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
762 template <typename ElectricFields>
763 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcLocalDistortionsCorrections(const SpaceCharge<DataT, Nz, Nr, Nphi>::Type type, const ElectricFields& formulaStruct)
764 {
765  const Side side = formulaStruct.getSide();
766  // calculate local distortions/corrections for each vertex in the tpc
767 #pragma omp parallel for num_threads(sNThreads)
768  for (size_t iPhi = 0; iPhi < Nphi; ++iPhi) {
769  const DataT phi = getPhiVertex(iPhi, side);
770  for (size_t iR = 0; iR < Nr; ++iR) {
771  const DataT radius = getRVertex(iR, side);
772  for (size_t iZ = 0; iZ < Nz - 1; ++iZ) {
773  // set z coordinate depending on distortions or correction calculation
774  const DataT z0 = type == Type::Corrections ? getZVertex(iZ + 1, side) : getZVertex(iZ, side);
775  const DataT z1 = type == Type::Corrections ? getZVertex(iZ, side) : getZVertex(iZ + 1, side);
776 
777  DataT drTmp = 0; // local distortion dR
778  DataT dPhiTmp = 0; // local distortion dPhi (multiplication with R has to be done at the end)
779  DataT dzTmp = 0; // local distortion dZ
780 
781  const DataT stepSize = (z1 - z0) / sSteps; // the distortions are calculated by leting the elctron drift this distance in z direction
782  for (int iter = 0; iter < sSteps; ++iter) {
783  const DataT z0Tmp = (z0 + iter * stepSize + dzTmp); // starting z position
784  const DataT z1Tmp = (z0Tmp + stepSize); // electron drifts from z0Tmp to z1Tmp
785 
786  DataT ddR = 0; // distortion dR for drift from z0Tmp to z1Tmp
787  DataT ddPhi = 0; // distortion dPhi for drift from z0Tmp to z1Tmp
788  DataT ddZ = 0; // distortion dZ for drift from z0Tmp to z1Tmp
789 
790  const DataT radiusTmp = regulateR(radius + drTmp, side); // current radial position
791  const DataT phiTmp = regulatePhi(phi + dPhiTmp, side); // current phi position
792 
793  // calculate distortions/corrections
794  calcDistCorr(radiusTmp, phiTmp, z0Tmp, z1Tmp, ddR, ddPhi, ddZ, formulaStruct, true);
795 
796  // add temp distortions to local distortions
797  drTmp += ddR;
798  dPhiTmp += ddPhi;
799  dzTmp += ddZ;
800  }
801 
802  // store local distortions/corrections
803  switch (type) {
804  case Type::Corrections:
805  mLocalCorrdR[side](iZ + 1, iR, iPhi) = drTmp;
806  mLocalCorrdRPhi[side](iZ + 1, iR, iPhi) = dPhiTmp * radius;
807  mLocalCorrdZ[side](iZ + 1, iR, iPhi) = dzTmp;
808  break;
809 
810  case Type::Distortions:
811  mLocalDistdR[side](iZ, iR, iPhi) = drTmp;
812  mLocalDistdRPhi[side](iZ, iR, iPhi) = dPhiTmp * radius;
813  mLocalDistdZ[side](iZ, iR, iPhi) = dzTmp;
814  break;
815  }
816  }
817  //extrapolate local distortion/correction to last/first bin using legendre polynoms with x0=0, x1=1, x2=2 and x=-1. This has to be done to ensure correct interpolation in the last,second last/first,second bin!
818  switch (type) {
819  case Type::Corrections:
820  mLocalCorrdR[side](0, iR, iPhi) = 3 * (mLocalCorrdR[side](1, iR, iPhi) - mLocalCorrdR[side](2, iR, iPhi)) + mLocalCorrdR[side](3, iR, iPhi);
821  mLocalCorrdRPhi[side](0, iR, iPhi) = 3 * (mLocalCorrdRPhi[side](1, iR, iPhi) - mLocalCorrdRPhi[side](2, iR, iPhi)) + mLocalCorrdRPhi[side](3, iR, iPhi);
822  mLocalCorrdZ[side](0, iR, iPhi) = 3 * (mLocalCorrdZ[side](1, iR, iPhi) - mLocalCorrdZ[side](2, iR, iPhi)) + mLocalCorrdZ[side](3, iR, iPhi);
823  break;
824 
825  case Type::Distortions:
826  mLocalDistdR[side](Nz - 1, iR, iPhi) = 3 * (mLocalDistdR[side](Nz - 2, iR, iPhi) - mLocalDistdR[side](Nz - 3, iR, iPhi)) + mLocalDistdR[side](Nz - 4, iR, iPhi);
827  mLocalDistdRPhi[side](Nz - 1, iR, iPhi) = 3 * (mLocalDistdRPhi[side](Nz - 2, iR, iPhi) - mLocalDistdRPhi[side](Nz - 3, iR, iPhi)) + mLocalDistdRPhi[side](Nz - 4, iR, iPhi);
828  mLocalDistdZ[side](Nz - 1, iR, iPhi) = 3 * (mLocalDistdZ[side](Nz - 2, iR, iPhi) - mLocalDistdZ[side](Nz - 3, iR, iPhi)) + mLocalDistdZ[side](Nz - 4, iR, iPhi);
829  break;
830  }
831  }
832  }
833  switch (type) {
834  case Type::Corrections:
835  mIsLocalCorrSet[side] = true;
836  break;
837  case Type::Distortions:
838  mIsLocalDistSet[side] = true;
839  break;
840  }
841 }

843 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
844 template <typename Fields>
845 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcGlobalDistortions(const Fields& formulaStruct)
846 {
847  const Side side = formulaStruct.getSide();
848  const DataT stepSize = formulaStruct.getID() == 2 ? getGridSpacingZ(side) : getGridSpacingZ(side) / sSteps; // if one used local distortions then no smaller stepsize is needed. if electric fields are used then smaller stepsize can be used
849  // loop over tpc volume and let the electron drift from each vertex to the readout of the tpc
850 #pragma omp parallel for num_threads(sNThreads)
851  for (size_t iPhi = 0; iPhi < Nphi; ++iPhi) {
852  const DataT phi0 = getPhiVertex(iPhi, side);
853  for (size_t iR = 0; iR < Nr; ++iR) {
854  const DataT r0 = getRVertex(iR, side);
855  for (size_t iZ = 0; iZ < Nz - 1; ++iZ) {
856  const DataT z0 = getZVertex(iZ, side); // the electron starts at z0, r0, phi0
857  DataT drDist = 0.0; // global distortion dR
858  DataT dPhiDist = 0.0; // global distortion dPhi (multiplication with R has to be done at the end)
859  DataT dzDist = 0.0; // global distortion dZ
860  int iter = 0;
861 
862  for (;;) {
863  const DataT z0Tmp = z0 + dzDist + iter * stepSize; // starting z position
864  const DataT z1Tmp = regulateZ(z0Tmp + stepSize, side); // electron drifts from z0Tmp to z1Tmp
865  const DataT radius = regulateR(r0 + drDist, side); // current radial position of the electron
866  const DataT phi = regulatePhi(phi0 + dPhiDist, side); // current phi position of the electron
867 
868  DataT ddR = 0; // distortion dR for drift from z0Tmp to z1Tmp
869  DataT ddPhi = 0; // distortion dPhi for drift from z0Tmp to z1Tmp
870  DataT ddZ = 0; // distortion dZ for drift from z0Tmp to z1Tmp
871 
872  // get the distortion from interpolation of local distortions or calculate distortions with the electric field
873  processGlobalDistCorr(radius, phi, z0Tmp, z1Tmp, ddR, ddPhi, ddZ, formulaStruct);
874 
875  // if one uses local distortions the interpolated value for the last bin has to be scaled.
876  // This has to be done because of the interpolated value is defined for a drift length of one z bin, but in the last bin the distance to the readout can be smaller than one z bin.
877  const bool checkReached = side == Side::A ? z1Tmp >= getZMax(side) : z1Tmp <= getZMax(side);
878  if (formulaStruct.getID() == 2 && checkReached) {
879  const DataT fac = std::abs((getZMax(side) - z0Tmp) * getInvSpacingZ(side));
880  ddR *= fac;
881  ddZ *= fac;
882  ddPhi *= fac;
883  }
884 
885  // add local distortions to global distortions
886  drDist += ddR;
887  dPhiDist += ddPhi;
888  dzDist += ddZ;
889 
890  // set loop to exit if the readout is reached and approximate distortion of 'missing' (one never ends exactly on the readout: z1Tmp + ddZ != ZMAX) drift distance.
891  // approximation is done by the current calculated values of the distortions and scaled linear to the 'missing' distance.
892  if (checkReached) {
893  const DataT endPoint = z1Tmp + ddZ;
894  const DataT deltaZ = getZMax(side) - endPoint; // distance from last point to read out
895  const DataT diff = endPoint - z0Tmp;
896  const DataT fac = diff != 0 ? std::abs(deltaZ / diff) : 0; // approximate the distortions for the 'missing' distance deltaZ
897  drDist += ddR * fac;
898  dPhiDist += ddPhi * fac;
899  dzDist += ddZ * fac;
900  break;
901  }
902  ++iter;
903  }
904  // store global distortions
905  mGlobalDistdR[side](iZ, iR, iPhi) = drDist;
906  mGlobalDistdRPhi[side](iZ, iR, iPhi) = dPhiDist * r0;
907  mGlobalDistdZ[side](iZ, iR, iPhi) = dzDist;
908  }
909  }
910  }
911  // set flag that global distortions are set to true
912  mIsGlobalDistSet[side] = true;
913 }

915 template <typename DataT, size_t Nz, size_t Nr, size_t Nphi>
916 template <typename Formulas>
917 void SpaceCharge<DataT, Nz, Nr, Nphi>::calcGlobalCorrections(const Formulas& formulaStruct)
918 {
919  const Side side = formulaStruct.getSide();
920  const int iSteps = formulaStruct.getID() == 2 ? 1 : sSteps; // if one used local corrections no step width is needed. since it is already used for calculation of the local corrections
921  const DataT stepSize = -getGridSpacingZ(side) / iSteps;
922  // loop over tpc volume and let the electron drift from each vertex to the readout of the tpc
923 #pragma omp parallel for num_threads(sNThreads)
924  for (size_t iPhi = 0; iPhi < Nphi; ++iPhi) {
925  const DataT phi0 = getPhiVertex(iPhi, side);
926  for (size_t iR = 0; iR < Nr; ++iR) {
927 
928  const DataT r0 = getRVertex(iR, side);
929  DataT drCorr = 0;
930  DataT dPhiCorr = 0;
931  DataT dzCorr = 0;
932 
933  // start at the readout and follow electron towards central electrode
934  for (size_t iZ = Nz - 1; iZ >= 1; --iZ) {
935  const DataT z0 = getZVertex(iZ, side); // the electron starts at z0, r0, phi0
936  // flag which is set when the central electrode is reached. if the central electrode is reached the calculation of the global corrections is aborted and the value set is the last calculated value.
937  bool centralElectrodeReached = false;
938  for (int iter = 0; iter < iSteps; ++iter) {
939  if (centralElectrodeReached) {
940  break;
941  }
942  const DataT radius = regulateR(r0 + drCorr, side); // current radial position of the electron
943  const DataT phi = regulatePhi(phi0 + dPhiCorr, side); // current phi position of the electron
944  const DataT z0Tmp = z0 + dzCorr + iter * stepSize; // starting z position
945  const DataT z1Tmp = regulateZ(z0Tmp + stepSize, side); // follow electron from z0Tmp to z1Tmp
946  DataT ddR = 0; // distortion dR for z0Tmp to z1Tmp
947  DataT ddPhi = 0; // distortion dPhi for z0Tmp to z1Tmp
948  DataT ddZ = 0; // distortion dZ for z0Tmp to z1Tmp
949 
950  // get the distortion from interpolation of local distortions or calculate distortions with the electric field
951  processGlobalDistCorr(radius, phi, z0Tmp, z1Tmp, ddR, ddPhi, ddZ, formulaStruct);
952 
953  // if one uses local corrections the interpolated value for the first bin has to be scaled.
954  // This has to be done because of the interpolated value is defined for a drift length of one z bin, but in the first bin the distance to the readout can be smaller than one z bin.
955  centralElectrodeReached = getSign(side) * z1Tmp <= getZMin(side);
956  if (formulaStruct.getID() == 2 && centralElectrodeReached) {
957  const DataT fac = (z0Tmp - getZMin(side)) * getInvSpacingZ(side);
958  ddR *= fac;
959  ddZ *= fac;
960  ddPhi *= fac;
961  }
962 
963  // add local corrections to global corrections
964  drCorr += ddR;
965  dPhiCorr += ddPhi;
966  dzCorr += ddZ;
967 
968  // set loop to exit if the central electrode is reached and approximate correction of 'missing' (one never ends exactly on the central electrode: z1Tmp + ddZ != ZMIN) distance.
969  // approximation is done by the current calculated values of the corrections and scaled linear to the 'missing' distance deltaZ. (NOT TESTED)
970  if (centralElectrodeReached) {
971  const DataT endPoint = z1Tmp + ddZ;
972  const DataT deltaZ = endPoint - getZMin(side);
973  const DataT diff = z0Tmp - endPoint;
974  const DataT fac = diff != 0 ? deltaZ / diff : 0; // approximate the distortions for the 'missing' distance deltaZ
975  drCorr += ddR * fac;
976  dPhiCorr += ddPhi * fac;
977  dzCorr += ddZ * fac;
978  break;
979  }
980  }
981  // store global corrections
982  mGlobalCorrdR[side](iZ - 1, iR, iPhi) = drCorr;
983  mGlobalCorrdRPhi[side](iZ - 1, iR, iPhi) = dPhiCorr * r0;
984  mGlobalCorrdZ[side](iZ - 1, iR, iPhi) = dzCorr;
985  }
986  }
987  }
988  // set flag that global corrections are set to true
989  mIsGlobalCorrSet[side] = true;
990 }
