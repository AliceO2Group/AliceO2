71 template <typename VEC>
72 void CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
73 {
74  using MD = o2::ctf::Metadata::OptStore;
75  // what to do which each field: see o2::ctd::Metadata explanation
76  constexpr MD optField[CTF::getNBlocks()] = {
77  MD::EENCODE, // BLC_trigger
78  MD::EENCODE, // BLC_bcInc
79  MD::EENCODE, // BLC_orbitInc
80  MD::EENCODE, // BLC_nChan
81  MD::EENCODE, // BLC_flags
82  MD::EENCODE, // BLC_idChan
83  MD::EENCODE, // BLC_qtcChain
84  MD::EENCODE, // BLC_cfdTime
85  MD::EENCODE // BLC_qtcAmpl
86  };
87  CompressedDigits cd;
88  compress(cd, digitVec, channelVec);
89 
90  // book output size with some margin
91  auto szIni = estimateCompressedSize(cd);
92  buff.resize(szIni);
93 
94  auto ec = CTF::create(buff);
95  using ECB = CTF::base;
96 
97  ec->setHeader(cd.header);
98  ec->getANSHeader().majorVersion = 0;
99  ec->getANSHeader().minorVersion = 1;
100  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
101 #define ENCODEFT0(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
102  // clang-format off
103  ENCODEFT0(cd.trigger, CTF::BLC_trigger, 0);
104  ENCODEFT0(cd.bcInc, CTF::BLC_bcInc, 0);
105  ENCODEFT0(cd.orbitInc, CTF::BLC_orbitInc, 0);
106  ENCODEFT0(cd.nChan, CTF::BLC_nChan, 0);
107  ENCODEFT0(cd.eventFlags, CTF::BLC_flags, 0);
108  ENCODEFT0(cd.idChan , CTF::BLC_idChan, 0);
109  ENCODEFT0(cd.qtcChain, CTF::BLC_qtcChain, 0);
110  ENCODEFT0(cd.cfdTime, CTF::BLC_cfdTime, 0);
111  ENCODEFT0(cd.qtcAmpl, CTF::BLC_qtcAmpl, 0);
112  // clang-format on
113  CTF::get(buff.data())->print(getPrefix());
114 }

140 template <typename VDIG, typename VCHAN>
141 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
142 {
143  digitVec.clear();
144  channelVec.clear();
145  digitVec.reserve(cd.header.nTriggers);
146  channelVec.reserve(cd.idChan.size());
147 
148  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
149  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
150 
151  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
152  // restore ROFRecord
153  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
154  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
155  ir.orbit += cd.orbitInc[idig];
156  } else {
157  ir.bc += cd.bcInc[idig];
158  }
159  Triggers trig;
160  trig.triggersignals = cd.trigger[idig];
161 
162  firstEntry = channelVec.size();
163  uint8_t chID = 0;
164  int amplA = 0, amplC = 0, timeA = 0, timeC = 0;
165  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
166  auto icc = channelVec.size();
167  const auto& chan = channelVec.emplace_back((chID += cd.idChan[icc]), cd.cfdTime[icc], cd.qtcAmpl[icc], cd.qtcChain[icc]);
168  //
169  if (std::abs(chan.CFDTime) < cd.header.triggerGate) {
170  if (chan.ChId < 4 * uint8_t(Geometry::NCellsA)) { // A side
171  amplA += chan.QTCAmpl;
172  timeA += chan.CFDTime;
173  trig.nChanA++;
174 
175  } else {
176  amplC += chan.QTCAmpl;
177  timeC += chan.CFDTime;
178  trig.nChanC++;
179  }
180  }
181  }
182  if (trig.nChanA) {
183  trig.timeA = timeA / trig.nChanA;
184  trig.amplA = amplA / 8;
185  }
186  if (trig.nChanC) {
187  trig.timeC = timeC / trig.nChanC;
188  trig.amplC = amplC / 8;
189  }
190  digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig, idig);
191  }
192 }
