239 // A matcher for a given O2 Data Model descriptor. We use a variant to hold
240 // the different kind of matchers so that we can have a hierarchy or
241 // DataDescriptionMatcher in the future (e.g. to handle OR / AND clauses) or we
242 // can apply it to the whole DataHeader.
243 class DataDescriptorMatcher
244 {
245  public:
246  enum struct Op { Just,
247  Or,
248  And,
249  Xor };
250 
254  DataDescriptorMatcher(DataDescriptorMatcher const& other);
255  DataDescriptorMatcher(DataDescriptorMatcher&& other) = default;
256  DataDescriptorMatcher& operator=(DataDescriptorMatcher const& other);
257  DataDescriptorMatcher& operator=(DataDescriptorMatcher&& other) = default;
258 
260  DataDescriptorMatcher(Op op, Node&& lhs, Node&& rhs = std::move(ConstantValueMatcher{false}));
261 
262  inline ~DataDescriptorMatcher() = default;
263 
268  bool match(ConcreteDataMatcher const& matcher, VariableContext& context) const;
269  bool match(ConcreteDataTypeMatcher const& matcher, VariableContext& context) const;
270  bool match(header::DataHeader const& header, VariableContext& context) const;
271  bool match(header::Stack const& stack, VariableContext& context) const;
272 
273  // actual polymorphic matcher which is able to cast the pointer to the correct
274  // kind of header.
275  bool match(char const* d, VariableContext& context) const;
276 
277  bool operator==(DataDescriptorMatcher const& other) const;
278 
279  friend std::ostream& operator<<(std::ostream& os, DataDescriptorMatcher const& matcher);
280  friend std::ostream& operator<<(std::ostream& os, Op const& matcher);
281 
282  Node const& getLeft() const { return mLeft; };
283  Node const& getRight() const { return mRight; };
284  Op getOp() const { return mOp; };
285 
286  private:
287  Op mOp;
288  Node mLeft;
289  Node mRight;
290 };
