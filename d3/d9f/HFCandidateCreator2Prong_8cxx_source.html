36 struct HFCandidateCreator2Prong {
37  Produces<aod::HfCandProng2Base> rowCandidateBase;
38 
39  Configurable<double> magneticField{"d_bz", 5., "magnetic field"};
40  Configurable<bool> b_propdca{"b_propdca", true, "create tracks version propagated to PCA"};
41  Configurable<double> d_maxr{"d_maxr", 200., "reject PCA's above this radius"};
42  Configurable<double> d_maxdzini{"d_maxdzini", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
43  Configurable<double> d_minparamchange{"d_minparamchange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
44  Configurable<double> d_minrelchi2change{"d_minrelchi2change", 0.9, "stop iterations is chi2/chi2old > this"};
45  Configurable<bool> b_dovalplots{"b_dovalplots", true, "do validation plots"};
46 
47  OutputObj<TH1F> hmass2{TH1F("hmass2", "2-prong candidates;inv. mass (#pi K) (GeV/#it{c}^{2});entries", 500, 0., 5.)};
48  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "2-prong candidates;XX element of cov. matrix of prim. vtx. position (cm^{2});entries", 100, 0., 1.e-4)};
49  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "2-prong candidates;XX element of cov. matrix of sec. vtx. position (cm^{2});entries", 100, 0., 0.2)};
50 
51  double massPi = RecoDecay::getMassPDG(kPiPlus);
52  double massK = RecoDecay::getMassPDG(kKPlus);
53  double massPiK{0.};
54  double massKPi{0.};

56  void process(aod::Collisions const& collisions,
57  aod::HfTrackIndexProng2 const& rowsTrackIndexProng2,
58  aod::BigTracks const& tracks)
59  {
60  // 2-prong vertex fitter
61  o2::vertexing::DCAFitterN<2> df;
62  df.setBz(magneticField);
63  df.setPropagateToPCA(b_propdca);
64  df.setMaxR(d_maxr);
65  df.setMaxDZIni(d_maxdzini);
66  df.setMinParamChange(d_minparamchange);
67  df.setMinRelChi2Change(d_minrelchi2change);
68  df.setUseAbsDCA(true);
69 
70  // loop over pairs of track indices
71  for (const auto& rowTrackIndexProng2 : rowsTrackIndexProng2) {
72  auto track0 = rowTrackIndexProng2.index0_as<aod::BigTracks>();
73  auto track1 = rowTrackIndexProng2.index1_as<aod::BigTracks>();
74  auto trackParVarPos1 = getTrackParCov(track0);
75  auto trackParVarNeg1 = getTrackParCov(track1);
76  auto collision = track0.collision();
77 
78  // reconstruct the 2-prong secondary vertex
79  if (df.process(trackParVarPos1, trackParVarNeg1) == 0) {
80  continue;
81  }
82  const auto& secondaryVertex = df.getPCACandidate();
83  auto chi2PCA = df.getChi2AtPCACandidate();
84  auto covMatrixPCA = df.calcPCACovMatrix().Array();
85  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
86  auto trackParVar0 = df.getTrack(0);
87  auto trackParVar1 = df.getTrack(1);
88 
89  // get track momenta
90  array<float, 3> pvec0;
91  array<float, 3> pvec1;
92  trackParVar0.getPxPyPzGlo(pvec0);
93  trackParVar1.getPxPyPzGlo(pvec1);
94 
95  // get track impact parameters
96  // This modifies track momenta!
97  auto primaryVertex = getPrimaryVertex(collision);
98  auto covMatrixPV = primaryVertex.getCov();
99  hCovPVXX->Fill(covMatrixPV[0]);
100  o2::dataformats::DCA impactParameter0;
101  o2::dataformats::DCA impactParameter1;
102  trackParVar0.propagateToDCA(primaryVertex, magneticField, &impactParameter0);
103  trackParVar1.propagateToDCA(primaryVertex, magneticField, &impactParameter1);
104 
105  // get uncertainty of the decay length
106  double phi, theta;
107  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
108  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
109  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
110 
111  // fill candidate table rows
112  rowCandidateBase(collision.posX(), collision.posY(), collision.posZ(),
113  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
114  errorDecayLength, errorDecayLengthXY,
115  chi2PCA,
116  pvec0[0], pvec0[1], pvec0[2],
117  pvec1[0], pvec1[1], pvec1[2],
118  impactParameter0.getY(), impactParameter1.getY(),
119  std::sqrt(impactParameter0.getSigmaY2()), std::sqrt(impactParameter1.getSigmaY2()),
120  rowTrackIndexProng2.index0Id(), rowTrackIndexProng2.index1Id(),
121  rowTrackIndexProng2.hfflag());
122 
123  // fill histograms
124  if (b_dovalplots) {
125  // calculate invariant masses
126  auto arrayMomenta = array{pvec0, pvec1};
127  massPiK = RecoDecay::M(arrayMomenta, array{massPi, massK});
128  massKPi = RecoDecay::M(arrayMomenta, array{massK, massPi});
129  hmass2->Fill(massPiK);
130  hmass2->Fill(massKPi);
131  }
132  }
133  }
134 };

143 struct HFCandidateCreator2ProngMC {
144  Produces<aod::HfCandProng2MCRec> rowMCMatchRec;
145  Produces<aod::HfCandProng2MCGen> rowMCMatchGen;
146 
147  void process(aod::HfCandProng2 const& candidates,
148  aod::BigTracksMC const& tracks,
149  aod::McParticles const& particlesMC)
150  {
151  int8_t sign = 0;
152  int8_t flag = 0;
153 
154  // Match reconstructed candidates.
155  for (auto& candidate : candidates) {
156  //Printf("New rec. candidate");
157  flag = 0;
158  auto arrayDaughters = array{candidate.index0_as<aod::BigTracksMC>(), candidate.index1_as<aod::BigTracksMC>()};
159 
160  // D0(bar) → π± K∓
161  //Printf("Checking D0(bar) → π± K∓");
162  if (RecoDecay::getMatchedMCRec(particlesMC, arrayDaughters, 421, array{+kPiPlus, -kKPlus}, true, &sign) > -1) {
163  flag = sign * (1 << D0ToPiK);
164  }
165 
166  // J/ψ → e+ e-
167  if (flag == 0) {
168  //Printf("Checking J/ψ → e+ e-");
169  if (RecoDecay::getMatchedMCRec(particlesMC, std::move(arrayDaughters), 443, array{+kElectron, -kElectron}, true) > -1) {
170  flag = 1 << JpsiToEE;
171  }
172  }
173 
174  rowMCMatchRec(flag);
175  }
176 
177  // Match generated particles.
178  for (auto& particle : particlesMC) {
179  //Printf("New gen. candidate");
180  flag = 0;
181 
182  // D0(bar) → π± K∓
183  //Printf("Checking D0(bar) → π± K∓");
184  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 421, array{+kPiPlus, -kKPlus}, true, &sign)) {
185  flag = sign * (1 << D0ToPiK);
186  }
187 
188  // J/ψ → e+ e-
189  if (flag == 0) {
190  //Printf("Checking J/ψ → e+ e-");
191  if (RecoDecay::isMatchedMCGen(particlesMC, particle, 443, array{+kElectron, -kElectron}, true)) {
192  flag = 1 << JpsiToEE;
193  }
194  }
195 
196  rowMCMatchGen(flag);
197  }
198  }
199 };
