67  void expandFEEId(uint16_t feeID, uint16_t& lr, uint16_t& ruOnLr, uint16_t& link) const
68  {
69  lr = feeID >> 12;
70  ruOnLr = feeID & 0x3f;
71  link = (feeID >> 8) & 0x3;
72 #ifdef _OVERRIDE_RUID_HACK_
73  ruOnLr %= NStavesOnLr[lr];
74 #endif
75  }

81  uint16_t modifyLinkInFEEId(uint16_t feeID, uint16_t linkID) const

83  feeID &= ~(0x3 << 8);

107  void getChipInfoSW(int chipSW, ChipInfo& chInfo) const
108  {
109  chInfo.id = chipSW;
110  if (chipSW > NChipsSB[IB] + NChipsSB[MB] - 1) { // OB
111  chipSW -= NChipsSB[IB] + NChipsSB[MB];
112  chInfo.ruType = OB;
113  auto dvRU = std::div(chipSW, NChipsPerStaveSB[OB]);
114  chInfo.ru = NStavesSB[IB] + NStavesSB[MB] + dvRU.quot; // RU ID == stave ID
115  chInfo.chOnRU = getChipOnRUInfo(OB, dvRU.rem);
116  } else if (chipSW > NChipsSB[IB] - 1) { // MB
117  chipSW -= NChipsSB[IB];
118  chInfo.ruType = MB;
119  auto dvRU = std::div(chipSW, NChipsPerStaveSB[MB]);
120  chInfo.ru = NStavesSB[IB] + dvRU.quot; // RU ID == stave ID
121  chInfo.chOnRU = getChipOnRUInfo(MB, dvRU.rem);
122  } else { // IB
123  chInfo.ruType = IB;
124  auto dvRU = std::div(chipSW, NChipsPerStaveSB[IB]);
125  chInfo.ru = dvRU.quot; // RU ID == stave ID = module ID
126  chInfo.chOnRU = getChipOnRUInfo(IB, dvRU.rem);
127  }
128  }

216  int getRUIDSW(int lr, int ruOnLr) const
217  {
218  int sid = 0;
