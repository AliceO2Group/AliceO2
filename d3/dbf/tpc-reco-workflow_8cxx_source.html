64 // customize dispatch policy, dispatch immediately what is ready
65 void customize(std::vector<o2::framework::DispatchPolicy>& policies)
66 {
67  using DispatchOp = o2::framework::DispatchPolicy::DispatchOp;
68  // we customize all devices to dispatch data immediately
69  auto readerMatcher = [](auto const& spec) {
70  return std::regex_match(spec.name.begin(), spec.name.end(), std::regex(".*-reader"));
71  };
72  auto triggerMatcher = [](auto const& query) {
73  // a bit of a hack but we want this to be configurable from the command line,
74  // however DispatchPolicy is inserted before all other setup. Triggering depending
75  // on the global variable set from the command line option. If scheduled messages
76  // are not triggered they are sent out at the end of the computation
77  return gDispatchTrigger.origin == query.origin && gDispatchTrigger.description == query.description;
78  };
79  policies.push_back({"prompt-for-reader", readerMatcher, DispatchOp::WhenReady, triggerMatcher});
80 }

112 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
113 {
114  //auto tpcSectors = o2::RangeTokenizer::tokenize<int>(cfgc.options().get<std::string>("tpc-sectors"));
115  std::vector<int> tpcSectors(36);
116  std::iota(tpcSectors.begin(), tpcSectors.end(), 0);
117  // the lane configuration defines the subspecification ids to be distributed among the lanes.
118  std::vector<int> laneConfiguration = tpcSectors; // Currently just a copy of the tpcSectors, why?
119  auto nLanes = cfgc.options().get<int>("tpc-lanes");
120  auto inputType = cfgc.options().get<std::string>("input-type");
121 
122  // depending on whether to dispatch early (prompt) and on the input type, we
123  // set the matcher. Note that this has to be in accordance with the OutputSpecs
124  // configured for the PublisherSpec
125  auto dispmode = cfgc.options().get<std::string>("dispatching-mode");
126  if (dispmode == "complete") {
127  // nothing to do we leave the matcher empty which will suppress the dispatch
128  // trigger and all messages will be sent out together at end of computation
129  } else if (inputType == "digits") {
130  gDispatchTrigger = o2::framework::Output{"TPC", "DIGITS"};
131  } else if (inputType == "clustershw") {
132  gDispatchTrigger = o2::framework::Output{"TPC", "CLUSTERHW"};
133  } else if (inputType == "clustersnative") {
134  gDispatchTrigger = o2::framework::Output{"TPC", "CLUSTERNATIVE"};
135  } else if (inputType == "zsraw") {
136  gDispatchTrigger = o2::framework::Output{"TPC", "RAWDATA"};
137  }
138  // set up configuration
139  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
140  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
141  o2::conf::ConfigurableParam::writeINI("o2tpcrecoworkflow_configuration.ini");
142 
143  bool doMC = not cfgc.options().get<bool>("disable-mc");
144  return o2::tpc::reco_workflow::getWorkflow(&gPolicyData, //
145  tpcSectors, // sector configuration
146  laneConfiguration, // lane configuration
147  doMC, //
148  nLanes, //
149  inputType, //
150  cfgc.options().get<std::string>("output-type"), //
151  !cfgc.options().get<bool>("no-ca-clusterer"), //
152  !cfgc.options().get<bool>("no-tpc-zs-on-the-fly"), //
153  cfgc.options().get<bool>("zs-10bit"), //
154  cfgc.options().get<float>("zs-threshold") //
155  );
156 }
