33 
34  int trg_gate = DigitizationParameters::Instance().mTime_trg_gate;
35  constexpr int MAXChan = 4 * (Geometry::NCellsA + Geometry::NCellsC);
36  for (int idig = 0; idig < 1000; idig++) {
37  ir += 1 + gRandom->Integer(200);
38  uint8_t ich = gRandom->Poisson(10);
39  auto start = channels.size();
40  int32_t tMeanA = 0, tMeanC = 0;
41  int32_t ampTotA = 0, ampTotC = 0;
42  uint8_t eventFlag = 10;
43  Triggers trig;
44  trig.triggersignals = gRandom->Integer(128);
45  while (ich < MAXChan) {
46  int16_t t = -2048 + gRandom->Integer(2048 * 2);
47  uint16_t q = gRandom->Integer(4096);
48  uint8_t chain = gRandom->Rndm() > 0.5 ? 0 : 1;
49  channels.emplace_back(ich, t, q, chain);
50  if (std::abs(t) < trg_gate) {
51  if (ich < 4 * uint8_t(Geometry::NCellsA)) {
52  trig.nChanA++;
53  ampTotA += q;
54  tMeanA += t;
55  } else {
56  trig.nChanC++;
57  ampTotC += q;
58  tMeanC += t;
59  }
60  }
61  ich += 1 + gRandom->Poisson(10);
62  }
63  if (trig.nChanA) {
64  trig.timeA = tMeanA / trig.nChanA;
65  trig.amplA = ampTotA * 0.125;
66  }
67  if (trig.nChanC) {
68  trig.timeC = tMeanC / trig.nChanC;
69  trig.amplC = ampTotC * 0.125; // sum/8
70  }
71  auto end = channels.size();
72  digits.emplace_back(start, end - start, ir, trig, idig);
73  digits[idig].setEventStatus(eventFlag);
74  }
75 
76  LOG(INFO) << "Generated " << channels.size() << " channels in " << digits.size() << " digits " << sw.CpuTime() << " s";

78  sw.Start();
