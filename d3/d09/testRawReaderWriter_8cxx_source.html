56 struct TestRawWriter { // simple class to create detector payload for multiple links
57 
58  RawFileWriter writer{"TST"};
59  std::string configName = "rawConf.cfg";
60 
61  //_________________________________________________________________
62  TestRawWriter(o2::header::DataOrigin origin = "TST", bool isCRU = true, const std::string& cfg = "rawConf.cfg") : writer(origin, isCRU), configName(cfg) {}
63 
64  //_________________________________________________________________
65  void init()
66  {
67  // init writer
68  writer.useRDHVersion(6);
69  int feeIDShift = writer.isCRUDetector() ? 8 : 9;
70  // register links
71  for (int icru = 0; icru < NCRU; icru++) {
72  std::string outFileName = o2::utils::concat_string("testdata_", writer.isCRUDetector() ? "cru" : "rorc", std::to_string(icru), ".raw");
73  for (int il = 0; il < NLinkPerCRU; il++) {
74  auto& link = writer.registerLink((icru << feeIDShift) + il, icru, il, 0, outFileName);
75  RDHUtils::setDetectorField(link.rdhCopy, 0xff << icru); // if needed, set extra link info, will be copied to all RDHs
76  }
77  }
78 
79  if (writer.isCRUDetector()) {
80  writer.setContinuousReadout(); // in case we want to issue StartOfContinuous trigger in the beginning
81  writer.setEmptyPageCallBack(this); // we want the writer to ask the detector code what to put in empty HBFs
82  }
83  writer.setCarryOverCallBack(this); // we want that writer to ask the detector code how to split large payloads
84 
85  writer.setApplyCarryOverToLastPage(true); // call CarryOver method also for the last chunk
86  }
87 
88  //_________________________________________________________________
89  void run()
90  {
91  // write payload and close outputs
92  nPreformatPages = 0;
93  // generate interaction records for triggers to write
94  std::vector<o2::InteractionTimeRecord> irs(1000);
95  o2::steer::InteractionSampler irSampler;
96  irSampler.setInteractionRate(12000); // ~1.5 interactions per orbit
97  irSampler.init();
98  irSampler.generateCollisionTimes(irs);
99 
100  std::vector<char> buffer;
101  int feeIDShift = writer.isCRUDetector() ? 8 : 9;
102 
103  // create payload for every interaction and push it to writer
104  for (const auto& ir : irs) {
105  int nCRU2Fill = writer.isCRUDetector() ? NCRU - 1 : NCRU; // in CRU mode we will fill 1 special CRU with preformatted data
106  for (int icru = 0; icru < nCRU2Fill; icru++) {
107  // we will create non-0 payload for all but 1st link of every CRU, the writer should take care
108  // of creating empty HBFs for the links w/o data
109  for (int il = 0; il < NLinkPerCRU; il++) {
110  buffer.clear();
111  int nGBT = gRandom->Poisson(RDHUtils::MAXCRUPage / RDHUtils::GBTWord * (il));
112  if (nGBT) {
113  buffer.resize((nGBT + 2) * RDHUtils::GBTWord, icru * NLinkPerCRU + il); // reserve 16B words accounting for the Header and Trailer
114  std::memcpy(buffer.data(), PLHeader.c_str(), RDHUtils::GBTWord);
115  std::memcpy(buffer.data() + buffer.size() - RDHUtils::GBTWord, PLTrailer.c_str(), RDHUtils::GBTWord);
116  // we don't care here about the content of the payload, except the presence of header and trailer
117  }
118  writer.addData((icru << feeIDShift) + il, icru, il, 0, ir, buffer);
119  }
120  }
121  }
122  if (writer.isCRUDetector()) {
123  // fill special CRU with preformatted pages
124  auto irHB = HBFUtils::Instance().getFirstIR(); // IR of the TF0/HBF0
125  int cruID = NCRU - 1;
126  while (irHB < irs.back()) {
127  for (int il = 0; il < NLinkPerCRU; il++) {
128  buffer.clear();
129  int pgSize = SpecSize[il] - sizeof(RDHAny);
130  buffer.resize(pgSize);
131  for (int ipg = 2 * (NLinkPerCRU - il); ipg--;) { // just to enforce writing multiple pages per selected HBFs
132  writer.addData((cruID << 8) + il, cruID, il, 0, irHB, buffer, true); // last argument is there to enforce a special "preformatted" mode
133  nPreformatPages++;
134  }
135  }
136  irHB.orbit += HBFUtils::Instance().getNOrbitsPerTF() / NPreformHBFPerTF; // we will write 32 such HBFs per TF
137  }
138  }
139 
140  // for further use we write the configuration file
141  writer.writeConfFile(writer.getOrigin().str, "RAWDATA", configName);
142  writer.close(); // flush buffers and close outputs
143  }
144 
145  // optional callback functions to register in the RawFileWriter
146  //_________________________________________________________________
147  void emptyHBFMethod(const RDHAny* rdh, std::vector<char>& toAdd) const
148  {
149  // what we want to add for every empty page
150  toAdd.resize(RDHUtils::GBTWord);
151  std::memcpy(toAdd.data(), HBFEmpty.c_str(), RDHUtils::GBTWord);
152  }
153 
154  //_________________________________________________________________
155  int carryOverMethod(const RDHAny* rdh, const gsl::span<char> data, const char* ptr, int maxSize, int splitID,
156  std::vector<char>& trailer, std::vector<char>& header) const
157  {
158  // how we want to split the large payloads. The data is the full payload which was sent for writing and
159  // it is already equiped with header and trailer
160  static int verboseCount = 0;
161 
162  if (maxSize <= RDHUtils::GBTWord) { // do not carry over trailer or header only
163  return 0;
164  }
165 
166  int bytesLeft = data.size() - (ptr - &data[0]);
167  bool lastPage = bytesLeft <= maxSize;
168  if (verboseCount++ < 100) {
169  LOG(INFO) << "Carry-over method for chunk of size " << bytesLeft << " is called, MaxSize = " << maxSize << (lastPage ? " : last chunk being processed!" : "");
170  }
171  // here we simply copy the header/trailer of the payload to every CRU page of this payload
172  header.resize(RDHUtils::GBTWord);
173  std::memcpy(header.data(), &data[0], RDHUtils::GBTWord);
174  trailer.resize(RDHUtils::GBTWord);
175  std::memcpy(trailer.data(), &data[data.size() - RDHUtils::GBTWord], RDHUtils::GBTWord);
176  // since we write an extra GBT word (trailer) in the end of the CRU page, we ask to write
177  // not the block ptr : ptr+maxSize, but ptr : ptr+maxSize - GBTWord;
178  int sz = maxSize; // if the method is called for the last page, then the trailer is overwritten !!!
179  if (!lastPage) { // otherwise it is added incrementally, so its size must be accounted
180  sz -= trailer.size();
181  }
182  return sz;
183  }
184 };

186 struct TestRawReader { // simple class to read detector raw data for multiple links
187 
188  std::unique_ptr<RawFileReader> reader;
189  std::string confName;
190 
191  //_________________________________________________________________
192  TestRawReader(const std::string& name = "TST", const std::string& cfg = "rawConf.cfg") : confName(cfg) {}
193 
194  //_________________________________________________________________
195  void init()
196  {
197  reader = std::make_unique<RawFileReader>(confName); // init from configuration file
198  uint32_t errCheck = 0xffffffff;
199  errCheck ^= 0x1 << RawFileReader::ErrNoSuperPageForTF; // makes no sense for superpages not interleaved by others
200  reader->setCheckErrors(errCheck);
201  reader->init();
202  }
203 
204  //_________________________________________________________________
205  void run()
206  {
207  // read data of all links
208  if (!reader) {
209  init();
210  }
211  int nLinks = reader->getNLinks();
212  BOOST_CHECK(nLinks == NCRU * NLinkPerCRU);
213 
214  // make sure no errors detected after initialization
215  for (int il = 0; il < nLinks; il++) {
216  auto& lnk = reader->getLink(il);
217  BOOST_CHECK(lnk.nErrors == 0);
218  }
219 
220  std::vector<std::vector<char>> buffers;
221  std::vector<bool> firstHBF;
222  std::string testStr;
223  testStr.resize(RDHUtils::GBTWord);
224  buffers.resize(nLinks); // 1 buffer per link
225  firstHBF.resize(nLinks, true);
226 
227  int nLinksRead = 0, nPreformatRead = 0;
228  do {
229  nLinksRead = 0;
230  for (int il = 0; il < nLinks; il++) {
231  auto& buff = buffers[il];
232  buff.clear();
233  auto& lnk = reader->getLink(il);
234  auto sz = lnk.getNextHBFSize(); // HBF treated as a trigger for RORC detectors
235  if (!sz) {
236  continue;
237  }
238  buff.resize(sz);
239  BOOST_CHECK(lnk.readNextHBF(buff.data()) == sz);
240  nLinksRead++;
241  }
242  if (nLinksRead) {
243  BOOST_CHECK(nLinksRead == nLinks); // all links should have the same number of HBFs or triggers
244 
245  const auto rdhRef = *reinterpret_cast<RDHAny*>(buffers[0].data());
246 
247  for (int il = 0; il < nLinks; il++) {
248  auto& lnk = reader->getLink(il);
249  auto& buff = buffers[il];
250  int hbsize = buff.size();
251  char* ptr = buff.data();
252  while (ptr < &buff.back()) { // check all RDH open/close and optional headers and trailers
253  const auto rdhi = *reinterpret_cast<RDHAny*>(ptr);
254  if (firstHBF[il]) { // make sure SOT or SOC is there
255  BOOST_CHECK(RDHUtils::getTriggerType(rdhi) & (o2::trigger::SOC | o2::trigger::SOT));
256  }
257  auto memSize = RDHUtils::getMemorySize(rdhi);
258  auto rdhSize = RDHUtils::getHeaderSize(rdhi);
259  BOOST_CHECK(RDHUtils::checkRDH(rdhi)); // check RDH validity
260 
261  if (!(RDHUtils::getHeartBeatIR(rdhRef) == RDHUtils::getHeartBeatIR(rdhi))) {
262  RDHUtils::printRDH(rdhRef);
263  RDHUtils::printRDH(rdhi);
264  }
265 
266  BOOST_CHECK(RDHUtils::getHeartBeatIR(rdhRef) == RDHUtils::getHeartBeatIR(rdhi)); // make sure the RDH of each link corresponds to the same BC
267  if (RDHUtils::getStop(rdhi)) { // closing page must be empty
268  BOOST_CHECK(memSize == rdhSize);
269  } else {
270  if (!lnk.cruDetector || RDHUtils::getCRUID(rdhi) < NCRU - 1) { // only last CRU of in non-RORC mode was special
271  if (lnk.cruDetector) {
272  BOOST_CHECK(memSize > rdhSize); // in this model all non-closing pages must contain something
273  }
274  if (memSize - rdhSize == RDHUtils::GBTWord) { // empty HBF will contain just a status word
275  if (lnk.cruDetector) {
276  testStr.assign(ptr + rdhSize, RDHUtils::GBTWord);
277  BOOST_CHECK(testStr == HBFEmpty);
278  }
279  } else if (memSize > rdhSize) {
280  // pages with real payload should have at least header + trailer + some payload
281  BOOST_CHECK(memSize - rdhSize > 2 * RDHUtils::GBTWord);
282  testStr.assign(ptr + rdhSize, RDHUtils::GBTWord);
283  BOOST_CHECK(testStr == PLHeader);
284  testStr.assign(ptr + memSize - RDHUtils::GBTWord, RDHUtils::GBTWord);
285  BOOST_CHECK(testStr == PLTrailer);
286  }
287  } else { // for the special CRU with preformatted data make sure the page sizes were not modified
288  if (memSize > rdhSize + RDHUtils::GBTWord) {
289  auto tfhb = HBFUtils::Instance().getTFandHBinTF({RDHUtils::getHeartBeatBC(rdhi), RDHUtils::getHeartBeatOrbit(rdhi)}); // TF and HBF relative to TF
290  BOOST_CHECK(tfhb.second % (HBFUtils::Instance().getNOrbitsPerTF() / NPreformHBFPerTF) == 0); // we were filling only every NPreformHBFPerTF-th HBF
291  BOOST_CHECK(memSize == SpecSize[RDHUtils::getLinkID(rdhi)]); // check if the size is correct
292  nPreformatRead++;
293  }
294  }
295  }
296  ptr += RDHUtils::getOffsetToNext(rdhi);
297  }
298  firstHBF[il] = false;
299  }
300  }
301  } while (nLinksRead); // read until there is something to read
302 
303  BOOST_CHECK(nPreformatRead == nPreformatPages); // make sure no preformatted page was lost
304 
305  } // run
306 };
