72 void convertDigitsToZSfinal(std::string_view digitsFile, std::string_view outputPath, std::string_view fileFor,
73  bool sectorBySector, uint32_t rdhV, bool stopPage, bool noPadding, bool createParentDir)
74 {
75 
76  // ===| open file and get tree |==============================================
77  std::unique_ptr<TFile> o2simDigits(TFile::Open(digitsFile.data()));
78  if (!o2simDigits || !o2simDigits->IsOpen() || o2simDigits->IsZombie()) {
79  LOGP(error, "Could not open file {}", digitsFile.data());
80  exit(1);
81  }
82  auto treeSim = (TTree*)o2simDigits->Get("o2sim");
83  if (!treeSim) {
84  LOGP(error, "Could not read digits tree from file {}", digitsFile.data());
85  exit(1);
86  }
87 
88  gROOT->cd();
89 
90  // ===| set up output directory |=============================================
91  std::string outDir{outputPath};
92  if (outDir.empty()) {
93  outDir = "./";
94  }
95  if (outDir.back() != '/') {
96  outDir += '/';
97  }
98 
99  if (gSystem->AccessPathName(outDir.data())) {
100  if (createParentDir) {
101  if (gSystem->mkdir(outDir.data(), kTRUE)) {
102  LOGP(error, "could not create output directory {}", outDir.data());
103  exit(1);
104  }
105  } else {
106  LOGP(error, "Requested output directory '{}' does not exists, consider removing '-n'", outDir.data());
107  exit(1);
108  }
109  }
110 
111  // ===| set up raw writer |===================================================
112  std::string inputGRP = o2::base::NameConf::getGRPFileName();
113  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
114 
115  o2::raw::RawFileWriter writer{"TPC"}; // to set the RDHv6.sourceID if V6 is used
116  writer.useRDHVersion(rdhV);
117  writer.setAddSeparateHBFStopPage(stopPage);
118  writer.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::TPC)); // must be set explicitly
119  const unsigned int defaultLink = rdh_utils::UserLogicLinkID;
120 
121  for (unsigned int i = 0; i < NSectors; i++) {
122  for (unsigned int j = 0; j < NEndpoints; j++) {
123  const unsigned int cruInSector = j / 2;
124  const unsigned int cruID = i * 10 + cruInSector;
125  const rdh_utils::FEEIDType feeid = rdh_utils::getFEEID(cruID, j & 1, defaultLink);
126  std::string outfname;
127  if (fileFor == "all") { // single file for all links
128  outfname = fmt::format("{}tpc_all.raw", outDir);
129  } else if (fileFor == "sector") {
130  outfname = fmt::format("{}tpc_sector{}.raw", outDir, i);
131  } else if (fileFor == "link") {
132  outfname = fmt::format("{}cru{}_{}.raw", outDir, cruID, j & 1);
133  } else {
134  throw std::runtime_error("invalid option provided for file grouping");
135  }
136  writer.registerLink(feeid, cruID, defaultLink, j & 1, outfname);
137  }
138  }
139  if (fileFor != "link") { // in case >1 link goes to the file, we must cache to preserve the TFs ordering
140  writer.useCaching();
141  }
142 
143  // ===| set up branch addresses |=============================================
144  std::vector<Digit>* vDigitsPerSectorCollection[Sector::MAXSECTOR] = {nullptr}; // container that keeps Digits per sector
145 
146  treeSim->SetBranchStatus("*", 0);
147  treeSim->SetBranchStatus("TPCDigit_*", 1);
148 
149  ProcessAttributes attr;
150  attr.padding = !noPadding;
151 
152  for (int iSecBySec = 0; iSecBySec < Sector::MAXSECTOR; ++iSecBySec) {
153  treeSim->ResetBranchAddresses();
154  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
155  if (sectorBySector) {
156  iSec = iSecBySec;
157  }
158  vDigitsPerSectorCollection[iSec] = nullptr;
159  treeSim->SetBranchAddress(TString::Format("TPCDigit_%d", iSec), &vDigitsPerSectorCollection[iSec]);
160  if (sectorBySector) {
161  break;
162  }
163  }
164  for (Long64_t ievent = 0; ievent < treeSim->GetEntries(); ++ievent) {
165  DigitArray inputDigits;
166  if (sectorBySector) {
167  treeSim->GetBranch(TString::Format("TPCDigit_%d", iSecBySec))->GetEntry(ievent);
168  } else {
169  treeSim->GetEntry(ievent);
170  }
171 
172  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
173  if (sectorBySector) {
174  iSec = iSecBySec;
175  }
176  inputDigits[iSec] = *vDigitsPerSectorCollection[iSec]; //????
177  if (sectorBySector) {
178  break;
179  }
180  }
181  convert(inputDigits, &attr, writer);
182  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
183  delete vDigitsPerSectorCollection[iSec];
184  vDigitsPerSectorCollection[iSec] = nullptr;
185  }
186  }
187  if (!sectorBySector) {
188  break;
189  }
190  }
191  // for further use we write the configuration file for the output
192  writer.writeConfFile("TPC", "RAWDATA", fmt::format("{}tpcraw.cfg", outDir));
193 }

195 void convert(DigitArray& inputDigits, ProcessAttributes* processAttributes, o2::raw::RawFileWriter& writer)
196 {
197  auto& zsEncoder = processAttributes->zsEncoder;
198  const auto zsThreshold = processAttributes->zsThreshold;
199  const auto zs12bit = processAttributes->zs12bit;
200  GPUParam _GPUParam;
201  _GPUParam.SetDefaults(5.00668);
202  const GPUParam mGPUParam = _GPUParam;
203 
204  //o2::InteractionRecord ir = o2::raw::HBFUtils::Instance().getFirstIR();
205  o2::InteractionRecord ir(0, 0); // we start with the time registered in ditigs, w/o extra offset
206  zsEncoder->RunZSEncoder<o2::tpc::Digit>(inputDigits, nullptr, nullptr, &writer, &ir, mGPUParam, zs12bit, false, zsThreshold, processAttributes->padding);
207 }

215  "Commands / Options");

219 
220  try {

223  add_option("verbose,v", bpo::value<uint32_t>()->default_value(0), "Select verbosity level [0 = no output]");
224  add_option("input-file,i", bpo::value<std::string>()->default_value("tpcdigits.root"), "Specifies input file.");
225  add_option("output-dir,o", bpo::value<std::string>()->default_value("./"), "Specify output directory");
226  add_option("no-parent-directories,n", "Do not create parent directories recursively");
227  add_option("sector-by-sector,s", bpo::value<bool>()->default_value(false)->implicit_value(true), "Run one TPC sector after another");
228  add_option("file-for,f", bpo::value<std::string>()->default_value("sector"), "single file per: link,sector,all");

231  uint32_t defRDH = o2::raw::RDHUtils::getVersion<o2::gpu::RAWDataHeaderGPU>();
232  add_option("rdh-version,r", bpo::value<uint32_t>()->default_value(defRDH), "RDH version to use");
233  add_option("configKeyValues", bpo::value<std::string>()->default_value(""), "comma-separated configKeyValues");
234 
235  opt_all.add(opt_general).add(opt_hidden);
236  bpo::store(bpo::command_line_parser(argc, argv).options(opt_all).positional(opt_pos).run(), vm);
237 
238  if (vm.count("help") || argc == 1) {
239  std::cout << opt_general << std::endl;
240  exit(0);
241  }
242 
243  bpo::notify(vm);
244  } catch (bpo::error& e) {
245  std::cerr << "ERROR: " << e.what() << std::endl
246  << std::endl;
247  std::cerr << opt_general << std::endl;
248  exit(1);
249  } catch (std::exception& e) {
250  std::cerr << e.what() << ", application will now exit" << std::endl;
251  exit(2);
252  }
253  o2::conf::ConfigurableParam::updateFromString(vm["configKeyValues"].as<std::string>());
254  convertDigitsToZSfinal(
255  vm["input-file"].as<std::string>(),
256  vm["output-dir"].as<std::string>(),
257  vm["file-for"].as<std::string>(),
258  vm["sector-by-sector"].as<bool>(),
259  vm["rdh-version"].as<uint32_t>(),
260  vm["stop-page"].as<bool>(),
261  vm["no-padding"].as<bool>(),
262  !vm.count("no-parent-directories"));
263 
264  return 0;
265 }
