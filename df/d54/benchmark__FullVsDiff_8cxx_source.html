33 static void BM_MergingTH1I(benchmark::State& state)
34 {
35  const size_t entries = state.range(0) == FULL_OBJECTS ? entriesInFull : entriesInDiff;
36  const size_t bins = 62500; // makes 250kB
37 
38  TCollection* collection = new TObjArray();
39  collection->SetOwner(true);
40  TF1* uni = new TF1("uni", "1", 0, 1000000);
41  for (size_t i = 0; i < collectionSize; i++) {
42  TH1I* h = new TH1I(("test" + std::to_string(i)).c_str(), "test", bins, 0, 1000000);
43  h->FillRandom("uni", entries);
44  collection->Add(h);
45  }
46 
47  TH1I* m = new TH1I("merged", "merged", bins, 0, 1000000);
48  // avoid memory overcommitment by doing something with data.
49  for (size_t i = 0; i < bins; i++) {
50  m->SetBinContent(i, 1);
51  }
52 
53  for (auto _ : state) {
54  if (state.range(0) == FULL_OBJECTS) {
55  m->Reset();
56  }
57 
58  auto start = std::chrono::high_resolution_clock::now();
59  m->Merge(collection, "-NOCHECK");
60  auto end = std::chrono::high_resolution_clock::now();
61 
62  auto elapsed_seconds = std::chrono::duration_cast<std::chrono::duration<double>>(end - start);
63  state.SetIterationTime(elapsed_seconds.count());
64  }
65 
66  delete collection;
67  delete m;
68  delete uni;
69 }

148 static void BM_MergingTHnSparse(benchmark::State& state)
149 {
150  const size_t entries = state.range(0) == FULL_OBJECTS ? entriesInFull : entriesInDiff;
151 
152  const Double_t min = 0.0;
153  const Double_t max = 1000000.0;
154  const size_t dim = 10;
155  const Int_t bins = 250;
156  const Int_t binsDims[dim] = {bins, bins, bins, bins, bins, bins, bins, bins, bins, bins};
157  const Double_t mins[dim] = {min, min, min, min, min, min, min, min, min, min};
158  const Double_t maxs[dim] = {max, max, max, max, max, max, max, max, max, max};
159 
160  TRandomMixMax gen;
161  gen.SetSeed(std::random_device()());
162  Double_t randomArray[dim];
163  auto* m = new THnSparseI("merged", "merged", dim, binsDims, mins, maxs);
164  for (auto _ : state) {
165 
166  TCollection* collection = new TObjArray();
167  collection->SetOwner(true);
168  for (size_t i = 0; i < collectionSize; i++) {
169 
170  auto* h = new THnSparseI(("test" + std::to_string(i)).c_str(), "test", dim, binsDims, mins, maxs);
171  for (size_t entry = 0; entry < entries; entry++) {
172  gen.RndmArray(dim, randomArray);
173  for (double& r : randomArray) {
174  r *= max;
175  }
176  h->Fill(randomArray);
177  }
178  collection->Add(h);
179  }
180 
181  if (state.range(0) == FULL_OBJECTS) {
182  m->Reset();
183  }
184 
185  auto start = std::chrono::high_resolution_clock::now();
186  m->Merge(collection);
187  auto end = std::chrono::high_resolution_clock::now();
188 
189  auto elapsed_seconds = std::chrono::duration_cast<std::chrono::duration<double>>(end - start);
190  state.SetIterationTime(elapsed_seconds.count());
191 
192  delete collection;
193  }
194  delete m;
195 }

197 static void BM_MergingTTree(benchmark::State& state)
198 {
199  const size_t entries = state.range(0) == FULL_OBJECTS ? entriesInFull : entriesInDiff;
200 
201  struct format1 {
202  Int_t a;
203  Long64_t b;
204  Float_t c;
205  Double_t d;
206  } branch1;
207  ULong64_t branch2;
208 
209  auto createTree = [&](std::string name) -> TTree* {
210  TTree* tree = new TTree();
211  tree->SetName(name.c_str());
212  tree->Branch("b1", &branch1, "a/I:b/L:c/F:d/D");
213  tree->Branch("b2", &branch2);
214  return tree;
215  };
216 
217  TRandomMixMax gen;
218  gen.SetSeed(std::random_device()());
219  Double_t randomArray[5];
220  TTree* merged = createTree("merged");
221 
222  for (auto _ : state) {
223  TCollection* collection = new TObjArray();
224  collection->SetOwner(true);
225  for (size_t i = 0; i < collectionSize; i++) {
226  TTree* t = createTree(std::to_string(i));
227  for (size_t entry = 0; entry < entries; entry++) {
228  gen.RndmArray(5, randomArray);
229  branch1 = {static_cast<Int_t>(randomArray[0]), static_cast<Long64_t>(randomArray[1]), static_cast<Float_t>(randomArray[2]), randomArray[3]};
230  branch2 = randomArray[4];
231  t->Fill();
232  }
233  collection->Add(t);
234  }
235 
236  if (state.range(0) == FULL_OBJECTS) {
237  merged->Reset();
238  }
239
