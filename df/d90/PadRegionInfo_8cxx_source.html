34 void PadRegionInfo::init()
35 {
36 
37  const float ks = mPadHeight / mPadWidth * tan(1.74532925199432948e-01); // tan(10deg)
38  // initialize number of pads per row
39  for (unsigned char irow = 0; irow < mNumberOfPadRows; ++irow) {
40  mPadsPerRow[irow] = 2 * std::floor(ks * (irow + mRowOffset) + mXhelper);
41  mNumberOfPads += mPadsPerRow[irow];
42  }

55 const PadPos PadRegionInfo::findPad(const float localX, const float localY, const Side side /*=Side::A*/) const
56 {
57  if (!isInRegion(localX)) {
58  return PadPos(255, 255);
59  }
60 
61  // the pad coordinate system is for pad-side view.
62  // on the A-Side one looks from the back-side, therefore
63  // the localY-sign must be changed
64  const float localYfactor = (side == Side::A) ? -1.f : 1.f;
65  const unsigned int row = std::floor((localX - mRadiusFirstRow) / mPadHeight);
66  if (row >= mNumberOfPadRows) {
67  return PadPos(255, 255);
68  }
69 
70  const unsigned int npads = getPadsInRowRegion(row);
71  const float padfloat = (npads / 2 * mPadWidth - localYfactor * localY) / mPadWidth;
72  if (padfloat < 0) {
73  return PadPos(255, 255);
74  }
75  const unsigned int pad = static_cast<unsigned int>(padfloat);
76 
77  if (pad >= npads) {
78  return PadPos(255, 255);
79  }
80 
81  return PadPos(row, pad);
82 }
