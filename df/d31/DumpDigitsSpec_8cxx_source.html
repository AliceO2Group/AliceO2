56 void DumpDigitsTask::init(framework::InitContext& ic)
57 {
58  LOG(INFO) << "[HMPID Dump Digits - init() ] ";
59  mPrintDigits = ic.options().get<bool>("print");
60 
61  mIsOutputOnFile = false;
62  mOutputFileName = ic.options().get<std::string>("out-file");
63  if (mOutputFileName != "") {
64  mOsFile.open(mOutputFileName, std::ios::out);
65  if (mOsFile.is_open()) {
66  mIsOutputOnFile = true;
67  }
68  }
69  if (mPrintDigits) {
70  std::cout << "--- HMP Digits : [Chamb,PhoCat,x,y]@(Orbit,BC)=Charge ---" << std::endl;
71  }
72  if (mIsOutputOnFile) {
73  mOsFile << "--- HMP Digits : [Chamb,PhoCat,x,y]@(Orbit,BC)=Charge ---" << std::endl;
74  }
75  mOrbit = -1;
76  mBc = -1;
77  mDigitsReceived = 0;
78 
79  mExTimer.start();
80  return;
81 }

83 void DumpDigitsTask::run(framework::ProcessingContext& pc)
84 {
85  LOG(DEBUG) << "[HMPID Dump Digits - run() ] Enter Dump ...";
86  for (auto const& ref : InputRecordWalker(pc.inputs())) {
87  std::vector<o2::hmpid::Digit> digits = pc.inputs().get<std::vector<o2::hmpid::Digit>>(ref);
88  LOG(DEBUG) << "The size of the vector =" << digits.size();
89  mDigitsReceived += digits.size();
90  if (mPrintDigits) {
91  for (o2::hmpid::Digit Dig : digits) {
92  std::cout << Dig << std::endl;
93  }
94  }
95  if (mIsOutputOnFile) {
96  for (o2::hmpid::Digit Dig : digits) {
97  mOsFile << Dig << std::endl;
98  if (Dig.getOrbit() != mOrbit || Dig.getBC() != mBc) {
99  mOrbit = Dig.getOrbit();
100  mBc = Dig.getBC();
101  // LOG(INFO) << "Event :" << mOrbit << " / " << mBc;
102  }
103  }
104  }
105  }
106  mExTimer.elapseMes("... Dumping Digits received = " + std::to_string(mDigitsReceived));
107  return;
108 }
