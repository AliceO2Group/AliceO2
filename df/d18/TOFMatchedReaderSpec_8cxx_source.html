45 void TOFMatchedReader::connectTree(const std::string& filename)
46 {
47  mTree.reset(nullptr); // in case it was already loaded
48  mFile.reset(TFile::Open(filename.c_str()));
49  assert(mFile && !mFile->IsZombie());
50  mTree.reset((TTree*)mFile->Get(mInTreeName.c_str()));
51  assert(mTree);
52  mTree->SetBranchAddress("TOFMatchInfo", &mMatchesPtr);
53  if (mReadTracks) {
54  if (!mTree->GetBranch("TPCTOFTracks")) {
55  throw std::runtime_error("TPC-TOF tracks are requested but not found in the tree");
56  }
57  mTree->SetBranchAddress("TPCTOFTracks", &mTracksPtr);
58  }
59  if (mUseMC) {
60  mTree->SetBranchAddress("MatchTOFMCTruth", &mLabelTOFPtr);
61  }
62  LOG(INFO) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
63 }

65 void TOFMatchedReader::run(ProcessingContext& pc)
66 {
67  auto currEntry = mTree->GetReadEntry() + 1;
68  assert(currEntry < mTree->GetEntries()); // this should not happen
69  mTree->GetEntry(currEntry);
70  LOG(INFO) << "Pushing " << mMatches.size() << " TOF matchings at entry " << currEntry;
71 
72  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, mTPCMatch ? ddMatchInfo_tpc : ddMatchInfo, 0, Lifetime::Timeframe}, mMatches);
73  if (mReadTracks && mTPCMatch) {
74  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "TOFTRACKS_TPC", 0, Lifetime::Timeframe}, mTracks);
75  }
76  if (mUseMC) {
77  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, mTPCMatch ? ddMCMatchTOF_tpc : ddMCMatchTOF, 0, Lifetime::Timeframe}, mLabelTOF);
78  }
79 
80  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
81  pc.services().get<ControlService>().endOfStream();
82  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
83  }
84 }

86 DataProcessorSpec getTOFMatchedReaderSpec(bool useMC, bool tpcmatch, bool readTracks)
87 {
88  std::vector<OutputSpec> outputs;
89 
90  outputs.emplace_back(o2::header::gDataOriginTOF, tpcmatch ? ddMatchInfo_tpc : ddMatchInfo, 0, Lifetime::Timeframe);
91  if (tpcmatch && readTracks) {
92  outputs.emplace_back(o2::header::gDataOriginTOF, "TOFTRACKS_TPC", 0, Lifetime::Timeframe);
93  }
94  if (useMC) {
95  outputs.emplace_back(o2::header::gDataOriginTOF, tpcmatch ? ddMCMatchTOF_tpc : ddMCMatchTOF, 0, Lifetime::Timeframe);
96  }
97 
98  return DataProcessorSpec{
99  o2::utils::concat_string("TOFMatchedReader_", tpcmatch ? "tpc" : "glo"),
100  Inputs{},
101  outputs,
102  AlgorithmSpec{adaptFromTask<TOFMatchedReader>(useMC, tpcmatch, readTracks)},
103  Options{
104  {"tof-matched-infile", VariantType::String, tpcmatch ? "o2match_toftpc.root" : "o2match_tof.root", {"Name of the input file"}},
105  {"treename", VariantType::String, "matchTOF", {"Name of top-level TTree"}},
106  }};
107 }
