25 Metric2DViewIndex::Updater Metric2DViewIndex::getUpdater(std::vector<Metric2DViewIndex*> views)
26 {
27  return [views](std::string const& name, MetricInfo const& metric, int value, int metricsIndex) -> void {
28  for (auto viewPtr : views) {
29  auto& view = *viewPtr;
30  if (view.prefix.size() > name.size()) {
31  continue;
32  }
33  if (std::mismatch(view.prefix.begin(), view.prefix.end(), name.begin()).first != view.prefix.end()) {
34  continue;
35  }
36 
37  auto extra = name;
38 
39  // +1 is to remove the /
40  extra.erase(0, view.prefix.size() + 1);
41  if (extra == "w") {
42  view.w = value;
43  view.indexes.resize(view.w * view.h);
44  return;
45  } else if (extra == "h") {
46  view.h = value;
47  view.indexes.resize(view.w * view.h);
48  return;
49  }
50  int idx = -1;
51  try {
52  idx = std::stoi(extra, nullptr, 10);
53  } catch (...) {
54  LOG(ERROR) << "Badly formatted metric";
55  }
56  if (idx < 0) {
57  LOG(ERROR) << "Negative metric";
58  return;
59  }
60  if (view.indexes.size() <= idx) {
61  view.indexes.resize(std::max(idx + 1, view.w * view.h));
62  }
63  view.indexes[idx] = metricsIndex;
64  }
65  };
66 }
