179 DataProcessorSpec getTOFChannelCalibDeviceSpec(bool useCCDB, bool attachChannelOffsetToLHCphase = false)
180 {
181  using device = o2::calibration::TOFChannelCalibDevice;
182  using clbUtils = o2::calibration::Utils;
183 
184  std::vector<OutputSpec> outputs;
185  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload});
186  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo});
187 
188  std::vector<InputSpec> inputs;
189  inputs.emplace_back("input", "TOF", "CALIBDATA");
190  if (useCCDB) {
191  inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
192  inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
193  inputs.emplace_back("startTimeLHCphase", o2::header::gDataOriginTOF, "StartLHCphase");
194  inputs.emplace_back("startTimeChCalib", o2::header::gDataOriginTOF, "StartChCalib");
195  }
196 
197  return DataProcessorSpec{
198  "calib-tofchannel-calibration",
199  inputs,
200  outputs,
201  AlgorithmSpec{adaptFromTask<device>(useCCDB, attachChannelOffsetToLHCphase)},
202  Options{
203  {"min-entries", VariantType::Int, 500, {"minimum number of entries to fit channel histos"}},
204  {"nbins", VariantType::Int, 1000, {"number of bins for t-texp"}},
205  {"range", VariantType::Float, 24000.f, {"range for t-text"}},
206  {"do-TOF-channel-calib-in-test-mode", VariantType::Bool, false, {"to run in test mode for simplification"}},
207  {"ccdb-path", VariantType::String, "http://ccdb-test.cern.ch:8080", {"Path to CCDB"}}}};
