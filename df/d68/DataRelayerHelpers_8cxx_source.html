53 std::vector<DataDescriptorMatcher>
54  DataRelayerHelpers::createInputMatchers(std::vector<InputRoute> const& routes)
55 {
56  std::vector<DataDescriptorMatcher> result;
57 
58  for (auto& route : routes) {
59  if (auto pval = std::get_if<ConcreteDataMatcher>(&route.matcher.matcher)) {
60  result.emplace_back(fromConcreteMatcher(*pval));
61  } else if (auto matcher = std::get_if<DataDescriptorMatcher>(&route.matcher.matcher)) {
62  result.push_back(*matcher);
63  } else {
64  throw std::runtime_error("Unsupported InputSpec type");
65  }
66  }
67
