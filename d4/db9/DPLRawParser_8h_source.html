232  bool next()
233  {
234  while (mInputIterator != mEnd) {
235  bool isInitial = mParser == nullptr;
236  while (mPartIterator != mInputIterator.end()) {
237  // first increment on the parser level
238  if (mParser && mCurrent != mParser->end() && ++mCurrent != mParser->end()) {
239  // we have an active parser and there is still data at the incremented iterator
240  return true;
241  }
242  // now increment on the level of one input
243  mParser.reset();
244  if (!isInitial && (mPartIterator == mInputIterator.end() || ++mPartIterator == mInputIterator.end())) {
245  // no more parts, go to next input
246  break;
247  }
248  isInitial = false;
249  // check filter rules
250  if (mFilterSpecs.size() > 0) {
251  bool isSelected = false;
252  for (auto const& spec : mFilterSpecs) {
253  if ((isSelected = DataRefUtils::match(*mPartIterator, spec)) == true) {
254  break;
255  }
256  }
257  if (!isSelected) {
258  continue;
259  }
260  }
261  gsl::span<const char> raw;
262  try {
263  raw = mParent.get<gsl::span<char>>(*mPartIterator);
264  } catch (const std::runtime_error& e) {
265  // TODO: need some better handling to avoid to be spammed by error messages
266  LOG(ERROR) << "failed to read data from " << (*mInputIterator).spec->binding;
267  LOG(ERROR) << e.what();
268  }
269  if (raw.size() == 0) {
270  continue;
271  }
272 
273  try {
274  mParser = std::make_unique<parser_type>(raw.data(), raw.size());
275  } catch (const std::runtime_error& e) {
276  LOG(ERROR) << "can not create raw parser form input data";
277  LOG(ERROR) << e.what();
278  }
279 
280  if (mParser != nullptr) {
281  mCurrent = mParser->begin();
282  return true;
283  }
284  } // end loop over parts on one input
285  ++mInputIterator;
286  mPartIterator = mInputIterator.begin();
287  } // end loop over inputs
288  return false;
289  }
