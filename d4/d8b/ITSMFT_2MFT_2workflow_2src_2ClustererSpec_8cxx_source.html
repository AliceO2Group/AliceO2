40 void ClustererDPL::init(InitContext& ic)
41 {
42 
43  mClusterer = std::make_unique<o2::itsmft::Clusterer>();
44  mClusterer->setNChips(o2::itsmft::ChipMappingMFT::getNChips());
45  LOG(INFO) << "MFT ClustererDPL::init total number of sensors " << o2::itsmft::ChipMappingMFT::getNChips() << "\n";
46 
47  //mClusterer->setMaskOverflowPixels(false);
48 
49  auto filenameGRP = ic.options().get<std::string>("grp-file");
50  const auto grp = o2::parameters::GRPObject::loadFrom(filenameGRP.c_str());
51 
52  if (grp) {
53  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut("MFT"));
54  } else {
55  LOG(ERROR) << "Cannot retrieve GRP from the " << filenameGRP.c_str() << " file !";
56  mState = 0;
57  return;
58  }
59 
60  mPatterns = !ic.options().get<bool>("no-patterns");
61  mNThreads = std::max(1, ic.options().get<int>("nthreads"));
62 
63  // settings for the fired pixel overflow masking
64  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::MFT>::Instance();
65  const auto& clParams = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance();
66  auto nbc = clParams.maxBCDiffToMaskBias;
67  nbc += mClusterer->isContinuousReadOut() ? alpParams.roFrameLengthInBC : (alpParams.roFrameLengthTrig / o2::constants::lhc::LHCBunchSpacingNS);
68  mClusterer->setMaxBCSeparationToMask(nbc);
69  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
70 
71  std::string dictPath = ic.options().get<std::string>("mft-dictionary-path");
72  std::string dictFile = o2::base::NameConf::getDictionaryFileName(o2::detectors::DetID::MFT, dictPath, ".bin");
73  if (o2::base::NameConf::pathExists(dictFile)) {
74  mClusterer->loadDictionary(dictFile);
75  LOG(INFO) << "MFTClusterer running with a provided dictionary: " << dictFile;
76  } else {
77  LOG(INFO) << "Dictionary " << dictFile << " is absent, MFTClusterer expects cluster patterns";
78  }
79  mState = 1;
80  mClusterer->print();
81 }

83 void ClustererDPL::run(ProcessingContext& pc)
84 {
85  auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
86  auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
87 
88  gsl::span<const o2::itsmft::MC2ROFRecord> mc2rofs;
89  gsl::span<const char> labelbuffer;
90  if (mUseMC) {
91  labelbuffer = pc.inputs().get<gsl::span<char>>("labels");
92  mc2rofs = pc.inputs().get<gsl::span<o2::itsmft::MC2ROFRecord>>("MC2ROframes");
93  }
94  const o2::dataformats::ConstMCTruthContainerView<o2::MCCompLabel> labels(labelbuffer);
95 
96  LOG(INFO) << "MFTClusterer pulled " << digits.size() << " digits, in "
97  << rofs.size() << " RO frames";
98 
99  o2::itsmft::DigitPixelReader reader;
100  reader.setDigits(digits);
101  reader.setROFRecords(rofs);
102  if (mUseMC) {
103  reader.setMC2ROFRecords(mc2rofs);
104  reader.setDigitsMCTruth(labels.getIndexedSize() > 0 ? &labels : nullptr);
105  }
106  reader.init();
107  auto orig = o2::header::gDataOriginMFT;
108  std::vector<o2::itsmft::CompClusterExt> clusCompVec;
109  std::vector<o2::itsmft::ROFRecord> clusROFVec;
110  std::vector<unsigned char> clusPattVec;
111 
112  std::unique_ptr<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> clusterLabels;
113  if (mUseMC) {
114  clusterLabels = std::make_unique<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>();
115  }
116  mClusterer->process(mNThreads, reader, &clusCompVec, mPatterns ? &clusPattVec : nullptr, &clusROFVec, clusterLabels.get());
117  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
118  pc.outputs().snapshot(Output{orig, "CLUSTERSROF", 0, Lifetime::Timeframe}, clusROFVec);
119  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
120 
121  if (mUseMC) {
122  pc.outputs().snapshot(Output{orig, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, *clusterLabels.get()); // at the moment requires snapshot
123  std::vector<o2::itsmft::MC2ROFRecord> clusterMC2ROframes(mc2rofs.size());
124  for (int i = mc2rofs.size(); i--;) {
125  clusterMC2ROframes[i] = mc2rofs[i]; // Simply, replicate it from digits ?
126  }
127  pc.outputs().snapshot(Output{orig, "CLUSTERSMC2ROF", 0, Lifetime::Timeframe}, clusterMC2ROframes);
128  }
129 
130  // TODO: in principle, after masking "overflow" pixels the MC2ROFRecord maxROF supposed to change, nominally to minROF
131  // -> consider recalculationg maxROF
132  LOG(INFO) << "MFTClusterer pushed " << clusCompVec.size() << " compressed clusters, in " << clusROFVec.size() << " RO frames";
133 }

135 DataProcessorSpec getClustererSpec(bool useMC)
136 {
137  std::vector<InputSpec> inputs;
138  inputs.emplace_back("digits", "MFT", "DIGITS", 0, Lifetime::Timeframe);
139  inputs.emplace_back("ROframes", "MFT", "DIGITSROF", 0, Lifetime::Timeframe);
140 
141  std::vector<OutputSpec> outputs;
142  outputs.emplace_back("MFT", "COMPCLUSTERS", 0, Lifetime::Timeframe);
143  outputs.emplace_back("MFT", "PATTERNS", 0, Lifetime::Timeframe);
144  outputs.emplace_back("MFT", "CLUSTERSROF", 0, Lifetime::Timeframe);
145 
146  if (useMC) {
147  inputs.emplace_back("labels", "MFT", "DIGITSMCTR", 0, Lifetime::Timeframe);
148  inputs.emplace_back("MC2ROframes", "MFT", "DIGITSMC2ROF", 0, Lifetime::Timeframe);
149  outputs.emplace_back("MFT", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
150  outputs.emplace_back("MFT", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
151  }
152 
153  return DataProcessorSpec{
154  "mft-clusterer",
155  inputs,
156  outputs,
157  AlgorithmSpec{adaptFromTask<ClustererDPL>(useMC)},
158  Options{
159  {"mft-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
160  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
161  {"no-patterns", o2::framework::VariantType::Bool, false, {"Do not save rare cluster patterns"}},
162  {"nthreads", VariantType::Int, 1, {"Number of clustering threads"}}}};
163 }
