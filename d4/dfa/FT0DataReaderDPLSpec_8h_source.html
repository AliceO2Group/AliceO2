36 template <typename RawReader>
37 class FT0DataReaderDPLSpec : public Task
38 {
39  public:
40  FT0DataReaderDPLSpec(const RawReader& rawReader) : mRawReader(rawReader) {}
41  FT0DataReaderDPLSpec() = default;
42  ~FT0DataReaderDPLSpec() override = default;
43  void init(InitContext& ic) final {}
44  void run(ProcessingContext& pc) final
45  {
46  DPLRawParser parser(pc.inputs());
47  mRawReader.clear();
48  LOG(INFO) << "FT0DataReaderDPLSpec";
49  uint64_t count = 0;
50  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
51  //Proccessing each page
52  count++;
53  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
54  gsl::span<const uint8_t> payload(it.data(), it.size());
55  mRawReader.process(rdhPtr->linkID, payload);
56  }
57  LOG(INFO) << "Pages: " << count;
58  mRawReader.accumulateDigits();
59  mRawReader.makeSnapshot(pc);
60  }
61  RawReader mRawReader;
62 };
