43 void Tracker::initConfig(const MFTTrackingParam& trkParam)
44 {
46 
47  mMinTrackPointsLTF = trkParam.MinTrackPointsLTF;
48  mMinTrackPointsCA = trkParam.MinTrackPointsCA;
49  mMinTrackStationsLTF = trkParam.MinTrackStationsLTF;
50  mMinTrackStationsCA = trkParam.MinTrackStationsCA;
51  mLTFclsRCut = trkParam.LTFclsRCut;
52  mROADclsRCut = trkParam.ROADclsRCut;
53  mLTFseed2BinWin = trkParam.LTFseed2BinWin;
54  mLTFinterBinWin = trkParam.LTFinterBinWin;
55 
56  mRBins = trkParam.RBins;
57  mPhiBins = trkParam.PhiBins;
58  mRPhiBins = trkParam.RBins * trkParam.PhiBins;
59  if (mRPhiBins > constants::index_table::MaxRPhiBins) {
60  LOG(WARN) << "To many RPhiBins for this configuration!";
61  mRPhiBins = constants::index_table::MaxRPhiBins;
62  mRBins = sqrt(constants::index_table::MaxRPhiBins);
63  mPhiBins = sqrt(constants::index_table::MaxRPhiBins);
64  LOG(WARN) << "Using instead RBins " << mRBins << " and PhiBins " << mPhiBins;
65  }
66  mRBinSize = (constants::index_table::RMax - constants::index_table::RMin) / mRBins;
67  mPhiBinSize = (constants::index_table::PhiMax - constants::index_table::PhiMin) / mPhiBins;
68 }

70 //_________________________________________________________________________________________________
71 void Tracker::initialize()
72 {
75 
76  Float_t dz, x, y, r, phi, x_proj, y_proj, r_proj, phi_proj;
77  Int_t binIndex1, binIndex2, binIndex2S, binR_proj, binPhi_proj;
78 
79  for (Int_t layer1 = 0; layer1 < (constants::mft::LayersNumber - 1); ++layer1) {
80 
81  for (Int_t iRBin = 0; iRBin < mRBins; ++iRBin) {
82 
83  r = (iRBin + 0.5) * mRBinSize + constants::index_table::RMin;
84 
85  for (Int_t iPhiBin = 0; iPhiBin < mPhiBins; ++iPhiBin) {
86 
87  phi = (iPhiBin + 0.5) * mPhiBinSize + constants::index_table::PhiMin;
88 
89  binIndex1 = getBinIndex(iRBin, iPhiBin);
90 
91  x = r * TMath::Cos(phi);
92  y = r * TMath::Sin(phi);
93 
94  for (Int_t layer2 = (layer1 + 1); layer2 < constants::mft::LayersNumber; ++layer2) {
95 
96  dz = constants::mft::LayerZCoordinate()[layer2] - constants::mft::LayerZCoordinate()[layer1];
97  x_proj = x + dz * x * constants::mft::InverseLayerZCoordinate()[layer1];
98  y_proj = y + dz * y * constants::mft::InverseLayerZCoordinate()[layer1];
99  auto clsPoint2D = math_utils::Point2D<Float_t>(x_proj, y_proj);
100  r_proj = clsPoint2D.R();
101  phi_proj = clsPoint2D.Phi();
102  o2::math_utils::bringTo02PiGen(phi_proj);
103 
104  binR_proj = getRBinIndex(r_proj);
105  binPhi_proj = getPhiBinIndex(phi_proj);
106 
107  int binRS, binPhiS;
108 
109  int binwRS = mLTFseed2BinWin;
110  int binhwRS = binwRS / 2;
111 
112  int binwPhiS = mLTFseed2BinWin;
113  int binhwPhiS = binwPhiS / 2;
114 
115  for (Int_t iR = 0; iR < binwRS; ++iR) {
116  binRS = binR_proj + (iR - binhwRS);
117  if (binRS < 0) {
118  continue;
119  }
120 
121  for (Int_t iPhi = 0; iPhi < binwPhiS; ++iPhi) {
122  binPhiS = binPhi_proj + (iPhi - binhwPhiS);
123  if (binPhiS < 0) {
124  continue;
125  }
126 
127  binIndex2S = getBinIndex(binRS, binPhiS);
128  mBinsS[layer1][layer2 - 1][binIndex1].emplace_back(binIndex2S);
129  }
130  }
131 
132  int binR, binPhi;
133 
134  int binwR = mLTFinterBinWin;
135  int binhwR = binwR / 2;
136 
137  int binwPhi = mLTFinterBinWin;
138  int binhwPhi = binwPhi / 2;
139 
140  for (Int_t iR = 0; iR < binwR; ++iR) {
141  binR = binR_proj + (iR - binhwR);
142  if (binR < 0) {
143  continue;
144  }
145 
146  for (Int_t iPhi = 0; iPhi < binwPhi; ++iPhi) {
147  binPhi = binPhi_proj + (iPhi - binhwPhi);
148  if (binPhi < 0) {
149  continue;
150  }
151 
152  binIndex2 = getBinIndex(binR, binPhi);
153  mBins[layer1][layer2 - 1][binIndex1].emplace_back(binIndex2);
154  }
155  }
156 
157  } // end loop layer2
158  } // end loop PhiBinIndex
159  } // end loop RBinIndex
160  } // end loop layer1
161 
162  mRoad.initialize();
163 }

181 //_________________________________________________________________________________________________
182 void Tracker::findTracksLTF(ROframe& event)
183 {
184  // find (high momentum) tracks by the Linear Track Finder (LTF) method
185 
186  MCCompLabel mcCompLabel;
187  Int_t layer1, layer2, nPointDisks;
188  Int_t binR_proj, binPhi_proj, bin;
189  Int_t binIndex, clsMinIndex, clsMaxIndex, clsMinIndexS, clsMaxIndexS;
190  Int_t extClsIndex;
191  Float_t dR2, dR2min, dR2cut = mLTFclsR2Cut;
192  Bool_t hasDisk[constants::mft::DisksNumber], newPoint, seed;
193 
194  Int_t clsInLayer1, clsInLayer2, clsInLayer;
195 
196  Int_t nPoints;
197  TrackElement trackPoints[constants::mft::LayersNumber];
198 
199  Int_t step = 0;
200  seed = kTRUE;
201  layer1 = 0;
202 
203  while (seed) {
204 
205  layer2 = (step == 0) ? (constants::mft::LayersNumber - 1) : (layer2 - 1);
206  step++;
207 
208  if (layer2 < layer1 + (mMinTrackPointsLTF - 1)) {
209  ++layer1;
210  if (layer1 > (constants::mft::LayersNumber - (mMinTrackPointsLTF - 1))) {
211  break;
212  }
213  step = 0;
214  continue;
215  }
216 
217  for (std::vector<Cluster>::iterator it1 = event.getClustersInLayer(layer1).begin(); it1 != event.getClustersInLayer(layer1).end(); ++it1) {
218  Cluster& cluster1 = *it1;
219  if (cluster1.getUsed()) {
220  continue;
221  }
222  clsInLayer1 = it1 - event.getClustersInLayer(layer1).begin();
223 
224  // loop over the bins in the search window
225  for (auto& binS : mBinsS[layer1][layer2 - 1][cluster1.indexTableBin]) {
226 
227  getBinClusterRange(event, layer2, binS, clsMinIndexS, clsMaxIndexS);
228 
229  for (std::vector<Cluster>::iterator it2 = (event.getClustersInLayer(layer2).begin() + clsMinIndexS); it2 != (event.getClustersInLayer(layer2).begin() + clsMaxIndexS + 1); ++it2) {
230  Cluster& cluster2 = *it2;
231  if (cluster2.getUsed()) {
232  continue;
233  }
234  clsInLayer2 = it2 - event.getClustersInLayer(layer2).begin();
235 
236  // start a TrackLTF
237  nPoints = 0;
238 
239  // add the first seed point
240  trackPoints[nPoints].layer = layer1;
241  trackPoints[nPoints].idInLayer = clsInLayer1;
242  nPoints++;
243 
244  // intermediate layers
245  for (Int_t layer = (layer1 + 1); layer <= (layer2 - 1); ++layer) {
246 
247  newPoint = kTRUE;
248 
249  // loop over the bins in the search window
250  dR2min = dR2cut;
251  for (auto& bin : mBins[layer1][layer - 1][cluster1.indexTableBin]) {
252 
253  getBinClusterRange(event, layer, bin, clsMinIndex, clsMaxIndex);
254 
255  for (std::vector<Cluster>::iterator it = (event.getClustersInLayer(layer).begin() + clsMinIndex); it != (event.getClustersInLayer(layer).begin() + clsMaxIndex + 1); ++it) {
256  Cluster& cluster = *it;
257  if (cluster.getUsed()) {
258  continue;
259  }
260  clsInLayer = it - event.getClustersInLayer(layer).begin();
261 
262  dR2 = getDistanceToSeed(cluster1, cluster2, cluster);
263  // retain the closest point within a radius dR2cut
264  if (dR2 >= dR2min) {
265  continue;
266  }
267  dR2min = dR2;
268 
269  if (newPoint) {
270  trackPoints[nPoints].layer = layer;
271  trackPoints[nPoints].idInLayer = clsInLayer;
272  nPoints++;
273  }
274  // retain only the closest point in DistanceToSeed
275  newPoint = false;
276  } // end clusters bin intermediate layer
277  } // end intermediate layers
278  } // end binRPhi
279 
280  // add the second seed point
281  trackPoints[nPoints].layer = layer2;
282  trackPoints[nPoints].idInLayer = clsInLayer2;
283  nPoints++;
284 
285  // keep only tracks fulfilling the minimum length condition
286  if (nPoints < mMinTrackPointsLTF) {
287  continue;
288  }
289  for (Int_t i = 0; i < (constants::mft::DisksNumber); i++) {
290  hasDisk[i] = kFALSE;
291  }
292  for (Int_t point = 0; point < nPoints; ++point) {
293  auto layer = trackPoints[point].layer;
294  hasDisk[layer / 2] = kTRUE;
295  }
296  nPointDisks = 0;
297  for (Int_t disk = 0; disk < (constants::mft::DisksNumber); ++disk) {
298  if (hasDisk[disk]) {
299  ++nPointDisks;
300  }
301  }
302  if (nPointDisks < mMinTrackStationsLTF) {
303  continue;
304  }
305 
306  // add a new TrackLTF
307  event.addTrackLTF();
308  for (Int_t point = 0; point < nPoints; ++point) {
309  auto layer = trackPoints[point].layer;
310  auto clsInLayer = trackPoints[point].idInLayer;
311  Cluster& cluster = event.getClustersInLayer(layer)[clsInLayer];
312  mcCompLabel = mUseMC ? event.getClusterLabels(layer, cluster.clusterId) : MCCompLabel();
313  extClsIndex = event.getClusterExternalIndex(layer, cluster.clusterId);
314  event.getCurrentTrackLTF().setPoint(cluster, layer, clsInLayer, mcCompLabel, extClsIndex);
315  // mark the used clusters
316  cluster.setUsed(true);
317  }
318  } // end seed clusters bin layer2
319  } // end binRPhi
320  } // end clusters layer1
321 
322  } // end seeding
323 }

325 //_________________________________________________________________________________________________
326 void Tracker::findTracksCA(ROframe& event)
327 {
328  // layers: 0, 1, 2, ..., 9
329  // rules for combining first/last plane in a road:
330  // 0 with 6, 7, 8, 9
331  // 1 with 6, 7, 8, 9
332  // 2 with 8, 9
333  // 3 with 8, 9
334  Int_t layer1Min = 0, layer1Max = 3;
335  Int_t layer2Min[4] = {6, 6, 8, 8};
336  constexpr Int_t layer2Max = constants::mft::LayersNumber - 1;
337 
338  MCCompLabel mcCompLabel;
339  Int_t roadId, nPointDisks;
340  Int_t binR_proj, binPhi_proj, bin;
341  Int_t binIndex, clsMinIndex, clsMaxIndex, clsMinIndexS, clsMaxIndexS;
342  Float_t dR2, dR2cut = mROADclsR2Cut;
343  Bool_t hasDisk[constants::mft::DisksNumber];
344 
345  Int_t clsInLayer1, clsInLayer2, clsInLayer;
346 
347  Int_t nPoints;
348  std::vector<TrackElement> roadPoints;
349 
350  roadId = 0;
351 
352  for (Int_t layer1 = layer1Min; layer1 <= layer1Max; ++layer1) {
353 
354  for (Int_t layer2 = layer2Max; layer2 >= layer2Min[layer1]; --layer2) {
355 
356  for (std::vector<Cluster>::iterator it1 = event.getClustersInLayer(layer1).begin(); it1 != event.getClustersInLayer(layer1).end(); ++it1) {
357  Cluster& cluster1 = *it1;
358  if (cluster1.getUsed()) {
359  continue;
360  }
361  clsInLayer1 = it1 - event.getClustersInLayer(layer1).begin();
362 
363  // loop over the bins in the search window
364  for (auto& binS : mBinsS[layer1][layer2 - 1][cluster1.indexTableBin]) {
365 
366  getBinClusterRange(event, layer2, binS, clsMinIndexS, clsMaxIndexS);
367 
368  for (std::vector<Cluster>::iterator it2 = (event.getClustersInLayer(layer2).begin() + clsMinIndexS); it2 != (event.getClustersInLayer(layer2).begin() + clsMaxIndexS + 1); ++it2) {
369  Cluster& cluster2 = *it2;
370  if (cluster2.getUsed()) {
371  continue;
372  }
373  clsInLayer2 = it2 - event.getClustersInLayer(layer2).begin();
374 
375  // start a road
376  roadPoints.clear();
377 
378  // add the first seed point
379  roadPoints.emplace_back(layer1, clsInLayer1);
380 
381  for (Int_t layer = (layer1 + 1); layer <= (layer2 - 1); ++layer) {
382 
383  // loop over the bins in the search window
384  for (auto& bin : mBins[layer1][layer - 1][cluster1.indexTableBin]) {
385 
386  getBinClusterRange(event, layer, bin, clsMinIndex, clsMaxIndex);
387 
388  for (std::vector<Cluster>::iterator it = (event.getClustersInLayer(layer).begin() + clsMinIndex); it != (event.getClustersInLayer(layer).begin() + clsMaxIndex + 1); ++it) {
389  Cluster& cluster = *it;
390  if (cluster.getUsed()) {
391  continue;
392  }
393  clsInLayer = it - event.getClustersInLayer(layer).begin();
394 
395  dR2 = getDistanceToSeed(cluster1, cluster2, cluster);
396  // add all points within a radius dR2cut
397  if (dR2 >= dR2cut) {
398  continue;
399  }
400 
401  roadPoints.emplace_back(layer, clsInLayer);
402 
403  } // end clusters bin intermediate layer
404  } // end intermediate layers
405  } // end binR
406 
407  // add the second seed point
408  roadPoints.emplace_back(layer2, clsInLayer2);
409  nPoints = roadPoints.size();
410 
411  // keep only roads fulfilling the minimum length condition
412  if (nPoints < mMinTrackPointsCA) {
413  continue;
414  }
415  for (Int_t i = 0; i < (constants::mft::DisksNumber); i++) {
416  hasDisk[i] = kFALSE;
417  }
418  for (Int_t point = 0; point < nPoints; ++point) {
419  auto layer = roadPoints[point].layer;
420  hasDisk[layer / 2] = kTRUE;
421  }
422  nPointDisks = 0;
423  for (Int_t disk = 0; disk < (constants::mft::DisksNumber); ++disk) {
424  if (hasDisk[disk]) {
425  ++nPointDisks;
426  }
427  }
428  if (nPointDisks < mMinTrackStationsCA) {
429  continue;
430  }
431 
432  mRoad.reset();
433  for (Int_t point = 0; point < nPoints; ++point) {
434  auto layer = roadPoints[point].layer;
435  auto clsInLayer = roadPoints[point].idInLayer;
436  mRoad.setPoint(layer, clsInLayer);
437  }
438  mRoad.setRoadId(roadId);
439  ++roadId;
440 
441  computeCellsInRoad(event);
442  runForwardInRoad();
443  runBackwardInRoad(event);
444 
445  } // end clusters in layer2
446  } // end binRPhi
447  } // end clusters in layer1
448  } // end layer2
449  } // end layer1
450 }

452 //_________________________________________________________________________________________________
453 void Tracker::computeCellsInRoad(ROframe& event)
454 {
455  Int_t layer1, layer1min, layer1max, layer2, layer2min, layer2max;
456  Int_t nPtsInLayer1, nPtsInLayer2;
457  Int_t clsInLayer1, clsInLayer2;
458  Int_t cellId;
459  Bool_t noCell;
460 
461  mRoad.getLength(layer1min, layer1max);
462  --layer1max;
463 
464  for (layer1 = layer1min; layer1 <= layer1max; ++layer1) {
465 
466  cellId = 0;
467 
468  layer2min = layer1 + 1;
469  layer2max = std::min(layer1 + (constants::mft::DisksNumber - isDiskFace(layer1)), constants::mft::LayersNumber - 1);
470 
471  nPtsInLayer1 = mRoad.getNPointsInLayer(layer1);
472 
473  for (Int_t point1 = 0; point1 < nPtsInLayer1; ++point1) {
474 
475  clsInLayer1 = mRoad.getClustersIdInLayer(layer1)[point1];
476 
477  layer2 = layer2min;
478 
479  noCell = kTRUE;
480  while (noCell && (layer2 <= layer2max)) {
481 
482  nPtsInLayer2 = mRoad.getNPointsInLayer(layer2);
483  /*
484  if (nPtsInLayer2 > 1) {
485  LOG(INFO) << "BV===== more than one point in road " << mRoad.getRoadId() << " in layer " << layer2 << " : " << nPtsInLayer2 << "\n";
486  }
487  */
488  for (Int_t point2 = 0; point2 < nPtsInLayer2; ++point2) {
489 
490  clsInLayer2 = mRoad.getClustersIdInLayer(layer2)[point2];
491 
492  noCell = kFALSE;
493  // create a cell
494  addCellToCurrentRoad(event, layer1, layer2, clsInLayer1, clsInLayer2, cellId);
495  } // end points in layer2
496  ++layer2;
497 
498  } // end while(noCell && (layer2 <= layer2max))
499  } // end points in layer1
500  } // end layer1
501 }

503 //_________________________________________________________________________________________________
504 void Tracker::runForwardInRoad()
505 {
506  Int_t layerR, layerL, icellR, icellL;
507  Int_t iter = 0;
508  Bool_t levelChange = kTRUE;
509 
510  while (levelChange) {
511 
512  levelChange = kFALSE;
513  ++iter;
514 
515  // R = right, L = left
516  for (layerL = 0; layerL < (constants::mft::LayersNumber - 2); ++layerL) {
517 
518  for (icellL = 0; icellL < mRoad.getCellsInLayer(layerL).size(); ++icellL) {
519 
520  Cell& cellL = mRoad.getCellsInLayer(layerL)[icellL];
521 
522  layerR = cellL.getSecondLayerId();
523 
524  if (layerR == (constants::mft::LayersNumber - 1)) {
525  continue;
526  }
527 
528  for (icellR = 0; icellR < mRoad.getCellsInLayer(layerR).size(); ++icellR) {
529 
530  Cell& cellR = mRoad.getCellsInLayer(layerR)[icellR];
531 
532  if ((cellL.getLevel() == cellR.getLevel()) && getCellsConnect(cellL, cellR)) {
533  if (iter == 1) {
534  mRoad.addRightNeighbourToCell(layerL, icellL, layerR, icellR);
535  mRoad.addLeftNeighbourToCell(layerR, icellR, layerL, icellL);
536  }
537  mRoad.incrementCellLevel(layerR, icellR);
538  levelChange = kTRUE;
539 
540  } // end matching cells
541  } // end loop cellR
542  } // end loop cellL
543  } // end loop layer
544 
545  updateCellStatusInRoad();
546 
547  } // end while (levelChange)
548 }

550 //_________________________________________________________________________________________________
551 void Tracker::runBackwardInRoad(ROframe& event)
552 {
553  if (mMaxCellLevel == 1) {
554  return; // we have only isolated cells
555  }
556 
557  Bool_t addCellToNewTrack, hasDisk[constants::mft::DisksNumber];
558 
559  Int_t lastCellLayer, lastCellId, icell;
560  Int_t cellId, layerC, cellIdC, layerRC, cellIdRC, layerL, cellIdL;
561  Int_t nPointDisks;
562  Float_t deviationPrev, deviation;
563 
564  Int_t minLayer = 6;
565  Int_t maxLayer = 8;
566 
567  Int_t nCells;
568  TrackElement trackCells[constants::mft::LayersNumber - 1];
569 
570  for (Int_t layer = maxLayer; layer >= minLayer; --layer) {
571 
572  for (cellId = 0; cellId < mRoad.getCellsInLayer(layer).size(); ++cellId) {
573 
574  if (mRoad.isCellUsed(layer, cellId) || (mRoad.getCellLevel(layer, cellId) < (mMinTrackPointsCA - 1))) {
575  continue;
576  }
577 
578  // start a TrackCA
579  nCells = 0;
580 
581  trackCells[nCells].layer = layer;
582  trackCells[nCells].idInLayer = cellId;
583  nCells++;
584 
585  // add cells to the new track
586  addCellToNewTrack = kTRUE;
587  while (addCellToNewTrack) {
588 
589  layerRC = trackCells[nCells - 1].layer;
590  cellIdRC = trackCells[nCells - 1].idInLayer;
591 
592  const Cell& cellRC = mRoad.getCellsInLayer(layerRC)[cellIdRC];
593 
594  addCellToNewTrack = kFALSE;
595 
596  // loop over left neighbours
597  deviationPrev = o2::constants::math::TwoPI;
598 
599  for (Int_t iLN = 0; iLN < cellRC.getNLeftNeighbours(); ++iLN) {
600 
601  auto leftNeighbour = cellRC.getLeftNeighbours()[iLN];
602  layerL = leftNeighbour.first;
603  cellIdL = leftNeighbour.second;
604 
605  const Cell& cellL = mRoad.getCellsInLayer(layerL)[cellIdL];
606 
607  if (mRoad.isCellUsed(layerL, cellIdL) || (mRoad.getCellLevel(layerL, cellIdL) != (mRoad.getCellLevel(layerRC, cellIdRC) - 1))) {
608  continue;
609  }
610 
611  deviation = getCellDeviation(cellL, cellRC);
612 
613  if (deviation < deviationPrev) {
614 
615  deviationPrev = deviation;
616 
617  if (iLN > 0) {
618  // delete the last added cell
619  nCells--;
620  }
621 
622  trackCells[nCells].layer = layerL;
623  trackCells[nCells].idInLayer = cellIdL;
624  nCells++;
625 
626  addCellToNewTrack = kTRUE;
627  }
628 
629  } // end loop left neighbour
630  } // end while(addCellToNewTrack)
631 
632  // check the track length
633  for (Int_t i = 0; i < (constants::mft::DisksNumber); ++i) {
634  hasDisk[i] = kFALSE;
635  }
636 
637  layerC = trackCells[0].layer;
638  cellIdC = trackCells[0].idInLayer;
639  const Cell& cellC = mRoad.getCellsInLayer(layerC)[cellIdC];
640  hasDisk[cellC.getSecondLayerId() / 2] = kTRUE;
641  for (icell = 0; icell < nCells; ++icell) {
