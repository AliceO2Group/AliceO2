64 template <typename... Ts>
65 static inline auto doExtractTypedOriginal(framework::pack<Ts...>, ProcessingContext& pc)
66 {
67  if constexpr (sizeof...(Ts) == 1) {
68  return pc.inputs().get<TableConsumer>(aod::MetadataTrait<framework::pack_element_t<0, framework::pack<Ts...>>>::metadata::tableLabel())->asArrowTable();
69  } else {
70  return std::vector{pc.inputs().get<TableConsumer>(aod::MetadataTrait<Ts>::metadata::tableLabel())->asArrowTable()...};
71  }
72 }

86 AlgorithmSpec AODReaderHelpers::indexBuilderCallback(std::vector<InputSpec> requested)
87 {
88  return AlgorithmSpec::InitCallback{[requested](InitContext& ic) {
89 
90  return [requested](ProcessingContext& pc) {
91  auto outputs = pc.outputs();
92  // spawn tables
93  for (auto& input : requested) {
94  auto description = std::visit(
95  overloaded{
96  [](ConcreteDataMatcher const& matcher) { return matcher.description; },
97  [](auto&&) { return header::DataDescription{""}; }},
98  input.matcher);
99 
100  auto origin = std::visit(
101  overloaded{
102  [](ConcreteDataMatcher const& matcher) { return matcher.origin; },
103  [](auto&&) { return header::DataOrigin{""}; }},
104  input.matcher);
105 
106  auto maker = [&](auto metadata) {
107  using metadata_t = decltype(metadata);
108  using Key = typename metadata_t::Key;
109  using index_pack_t = typename metadata_t::index_pack_t;
110  using sources = typename metadata_t::originals;
111  if constexpr (metadata_t::exclusive == true) {
112  return o2::framework::IndexExclusive::indexBuilder(index_pack_t{},
113  extractTypedOriginal<Key>(pc),
114  extractOriginalsTuple(sources{}, pc));
115  } else {
116  return o2::framework::IndexSparse::indexBuilder(index_pack_t{},
117  extractTypedOriginal<Key>(pc),
118  extractOriginalsTuple(sources{}, pc));
119  }
120  };
121 
122  if (description == header::DataDescription{"MA_RN2_EX"}) {
123  outputs.adopt(Output{origin, description}, maker(o2::aod::Run2MatchedExclusiveMetadata{}));
124  } else if (description == header::DataDescription{"MA_RN2_SP"}) {
125  outputs.adopt(Output{origin, description}, maker(o2::aod::Run2MatchedSparseMetadata{}));
126  } else if (description == header::DataDescription{"MA_RN3_EX"}) {
127  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedExclusiveMetadata{}));
128  } else if (description == header::DataDescription{"MA_RN3_SP"}) {
129  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedSparseMetadata{}));
130  } else if (description == header::DataDescription{"MA_BCCOL_EX"}) {
131  outputs.adopt(Output{origin, description}, maker(o2::aod::MatchedBCCollisionsExclusiveMetadata{}));
132  } else if (description == header::DataDescription{"MA_BCCOL_SP"}) {
133  outputs.adopt(Output{origin, description}, maker(o2::aod::MatchedBCCollisionsSparseMetadata{}));
134  } else if (description == header::DataDescription{"MA_RN3_BC_SP"}) {
135  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedToBCSparseMetadata{}));
136  } else if (description == header::DataDescription{"MA_RN3_BC_EX"}) {
137  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedToBCExclusiveMetadata{}));
138  } else {
139  throw std::runtime_error("Not an index table");
140  }
141  }
142  };
143  }};
144 }

146 AlgorithmSpec AODReaderHelpers::aodSpawnerCallback(std::vector<InputSpec> requested)
147 {
148  return AlgorithmSpec::InitCallback{[requested](InitContext& ic) {
149 
150  return [requested](ProcessingContext& pc) {
151  auto outputs = pc.outputs();
152  // spawn tables
153  for (auto& input : requested) {
154  auto description = std::visit(
155  overloaded{
156  [](ConcreteDataMatcher const& matcher) { return matcher.description; },
157  [](auto&&) { return header::DataDescription{""}; }},
158  input.matcher);
159 
160  auto origin = std::visit(
161  overloaded{
162  [](ConcreteDataMatcher const& matcher) { return matcher.origin; },
163  [](auto&&) { return header::DataOrigin{""}; }},
164  input.matcher);
165 
166  auto maker = [&](auto metadata) {
167  using metadata_t = decltype(metadata);
168  using expressions = typename metadata_t::expression_pack_t;
169  auto original_table = pc.inputs().get<TableConsumer>(input.binding)->asArrowTable();
170  return o2::framework::spawner(expressions{}, original_table.get());
171  };
172 
173  if (description == header::DataDescription{"TRACK"}) {
174  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksExtensionMetadata{}));
175  } else if (description == header::DataDescription{"TRACKCOV"}) {
176  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksCovExtensionMetadata{}));
177  } else if (description == header::DataDescription{"MUON"}) {
178  outputs.adopt(Output{origin, description}, maker(o2::aod::MuonsExtensionMetadata{}));
179  } else {
180  throw runtime_error("Not an extended table");
181  }
182  }
183  };
184  }};
185 }
