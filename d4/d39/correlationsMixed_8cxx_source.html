40 struct HashTask {
41  std::vector<float> vtxBins{-7.0f, -5.0f, -3.0f, -1.0f, 1.0f, 3.0f, 5.0f, 7.0f};
42  std::vector<float> multBins{0.0f, 20.0f, 40.0f, 60.0f, 80.0f, 100.0f};
43  Produces<aod::Hashes> hashes;
44 
45  // Calculate hash for an element based on 2 properties and their bins.
46  int getHash(std::vector<float> const& vtxBins, std::vector<float> const& multBins, float vtx, float mult)
47  {
48  // underflow
49  if (vtx < vtxBins[0]) {
50  return -1;
51  }
52  if (mult < multBins[0]) {
53  return -1;
54  }
55 
56  for (int i = 1; i < vtxBins.size(); i++) {
57  if (vtx < vtxBins[i]) {
58  for (int j = 1; j < multBins.size(); j++) {
59  if (mult < multBins[j]) {
60  return i + j * (vtxBins.size() + 1);
61  }
62  }
63  }
64  }
65  // overflow
66  return -1;
67  }
68 
69  void process(soa::Join<aod::Collisions, aod::Cents> const& collisions)
70  {
71  for (auto& collision : collisions) {
72  int hash = getHash(vtxBins, multBins, collision.posZ(), collision.centV0M());
73  LOGF(info, "Collision: %d (%f, %f) hash: %d", collision.index(), collision.posZ(), collision.centV0M(), hash);
74  hashes(hash);
75  }
76  }
77 };

162  void process(soa::Join<aod::Collisions, aod::Hashes, aod::EvSels, aod::Cents>& collisions, myTracks const& tracks)
163  {
164  int bSign = 1; // TODO magnetic field from CCDB
165  const float pTCut = 2.0;
166 
167  collisions.bindExternalIndices(&tracks);
168  auto tracksTuple = std::make_tuple(tracks);
169  AnalysisDataProcessorBuilder::GroupSlicer slicer(collisions, tracksTuple);
170 
171  // Strictly upper categorised collisions, for 5 combinations per bin, skipping those in entry -1
172  for (auto& [collision1, collision2] : selfCombinations("fBin", 5, -1, collisions, collisions)) {
173 
174  LOGF(info, "Collisions bin: %d pair: %d (%f), %d (%f)", collision1.bin(), collision1.index(), collision1.posZ(), collision2.index(), collision2.posZ());
175 
176  auto it1 = slicer.begin();
177  auto it2 = slicer.begin();
178  for (auto& slice : slicer) {
179  if (slice.groupingElement().index() == collision1.index()) {
180  it1 = slice;
181  break;
182  }
183  }
184  for (auto& slice : slicer) {
185  if (slice.groupingElement().index() == collision2.index()) {
186  it2 = slice;
187  break;
188  }
189  }
190 
191  auto tracks1 = std::get<myTracks>(it1.associatedTables());
192  tracks1.bindExternalIndices(&collisions);
193  auto tracks2 = std::get<myTracks>(it2.associatedTables());
194  tracks2.bindExternalIndices(&collisions);
195 
196  // LOGF(info, "Tracks: %d and %d entries", tracks1.size(), tracks2.size());
197 
198  for (auto& track1 : tracks1) {
199 
200  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.sign() < 0) {
201  continue;
202  }
203 
204  //LOGF(info, "TRACK %f %f | %f %f | %f %f", track1.eta(), track1.eta(), track1.phi(), track1.phi(), track1.pt(), track1.pt());
205 
206  mixed->getTriggerHist()->Fill(CorrelationContainer::kCFStepReconstructed, track1.pt(), collision1.centV0M(), collision1.posZ());
207 
208  for (auto& track2 : tracks2) {
209 
210  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.sign() < 0) {
211  continue;
212  }
213  if (cfgPairCharge != 0 && cfgPairCharge * track1.sign() * track2.sign() < 0) {
214  continue;
215  }
216 
217  if (cfg.mPairCuts && conversionCuts(track1, track2)) {
218  continue;
219  }
220 
221  if (cfgTwoTrackCut > 0 && twoTrackCut(track1, track2, bSign)) {
222  continue;
223  }
224 
225  float deltaPhi = track1.phi() - track2.phi();
226  if (deltaPhi > 1.5 * M_PI) {
227  deltaPhi -= M_PI * 2;
228  }
229  if (deltaPhi < -0.5 * M_PI) {
230  deltaPhi += M_PI * 2;
231  }
232 
233  mixed->getPairHist()->Fill(CorrelationContainer::kCFStepReconstructed,
234  track1.eta() - track2.eta(), track2.pt(), track1.pt(), collision1.centV0M(), deltaPhi, collision1.posZ());
235  }
236  }
237  }

240  template <typename T>
241  bool conversionCuts(T const& track1, T const& track2)
242  {
243  // skip if like sign
244  if (track1.sign() * track2.sign() > 0) {
245  return false;
246  }
247 
248  bool decision = false;
249 
250  if (conversionCut(track1, track2, Photon, cfgPairCutPhoton)) {
251  decision = true;
252  }
253  if (conversionCut(track1, track2, K0, cfgPairCutK0)) {
254  decision = true;
255  }
256  if (conversionCut(track1, track2, Lambda, cfgPairCutLambda)) {
257  decision = true;
258  }
259  if (conversionCut(track2, track1, Lambda, cfgPairCutLambda)) {
260  decision = true;
261  }
262  if (conversionCut(track1, track2, Phi, cfgPairCutPhi)) {
263  decision = true;
264  }
265  if (conversionCut(track1, track2, Rho, cfgPairCutRho)) {
266  decision = true;
267  }
268 
269  return decision;
270  }

272  template <typename T>
273  bool conversionCut(T const& track1, T const& track2, PairCuts conv, double cut)
274  {
275  //LOGF(info, "pt is %f %f", track1.pt(), track2.pt());
276 
277  if (cut < 0) {
278  return false;
279  }
280 
281  double massD1, massD2, massM;
282 
283  switch (conv) {
284  case Photon:
285  massD1 = 0.51e-3;
286  massD2 = 0.51e-3;
287  massM = 0;
288  break;
289  case K0:
290  massD1 = 0.1396;
291  massD2 = 0.1396;
292  massM = 0.4976;
293  break;
294  case Lambda:
295  massD1 = 0.9383;
296  massD2 = 0.1396;
297  massM = 1.115;
298  break;
299  case Phi:
300  massD1 = 0.4937;
301  massD2 = 0.4937;
302  massM = 1.019;
303  break;
304  case Rho:
305  massD1 = 0.1396;
306  massD2 = 0.1396;
307  massM = 0.770;
308  break;
309  }
310 
311  auto massC = getInvMassSquaredFast(track1, massD1, track2, massD2);
312 
313  if (TMath::Abs(massC - massM * massM) > cut * 5) {
314  return false;
315  }
316 
317  massC = getInvMassSquared(track1, massD1, track2, massD2);
318  qa.mControlConvResoncances->Fill(static_cast<int>(conv), massC - massM * massM);
319  if (massC > (massM - cut) * (massM - cut) && massC < (massM + cut) * (massM + cut)) {
320  return true;
321  }
322 
323  return false;
324  }

326  template <typename T>
327  double getInvMassSquared(T const& track1, double m0_1, T const& track2, double m0_2)
328  {
329  // calculate inv mass squared
330  // same can be achieved, but with more computing time with
331  /*TLorentzVector photon, p1, p2;
332  p1.SetPtEtaPhiM(triggerParticle->Pt(), triggerEta, triggerParticle->Phi(), 0.510e-3);
333  p2.SetPtEtaPhiM(particle->Pt(), eta[j], particle->Phi(), 0.510e-3);
334  photon = p1+p2;
335  photon.M()*/
336 
337  float tantheta1 = 1e10;
338 
339  if (track1.eta() < -1e-10 || track1.eta() > 1e-10) {
340  float expTmp = TMath::Exp(-track1.eta());
341  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
342  }
343 
344  float tantheta2 = 1e10;
345  if (track2.eta() < -1e-10 || track2.eta() > 1e-10) {
346  float expTmp = TMath::Exp(-track2.eta());
347  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
348  }
349 
350  float e1squ = m0_1 * m0_1 + track1.pt() * track1.pt() * (1.0 + 1.0 / tantheta1 / tantheta1);
351  float e2squ = m0_2 * m0_2 + track2.pt() * track2.pt() * (1.0 + 1.0 / tantheta2 / tantheta2);
352 
353  float mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (track1.pt() * track2.pt() * (TMath::Cos(track1.phi() - track2.phi()) + 1.0 / tantheta1 / tantheta2)));
354 
355  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
356 
357  return mass2;
358  }

360  template <typename T>
361  double getInvMassSquaredFast(T const& track1, double m0_1, T const& track2, double m0_2)
362  {
363  // calculate inv mass squared approximately
364 
365  const float eta1 = track1.eta();
366  const float eta2 = track2.eta();
367  const float phi1 = track1.phi();
368  const float phi2 = track2.phi();
369  const float pt1 = track1.pt();
370  const float pt2 = track2.pt();
371 
372  float tantheta1 = 1e10;
373 
374  if (eta1 < -1e-10 || eta1 > 1e-10) {
375  float expTmp = 1.0 - eta1 + eta1 * eta1 / 2 - eta1 * eta1 * eta1 / 6 + eta1 * eta1 * eta1 * eta1 / 24;
376  tantheta1 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
377  }
378 
379  float tantheta2 = 1e10;
380  if (eta2 < -1e-10 || eta2 > 1e-10) {
381  float expTmp = 1.0 - eta2 + eta2 * eta2 / 2 - eta2 * eta2 * eta2 / 6 + eta2 * eta2 * eta2 * eta2 / 24;
382  tantheta2 = 2.0 * expTmp / (1.0 - expTmp * expTmp);
383  }
384 
385  float e1squ = m0_1 * m0_1 + pt1 * pt1 * (1.0 + 1.0 / tantheta1 / tantheta1);
386  float e2squ = m0_2 * m0_2 + pt2 * pt2 * (1.0 + 1.0 / tantheta2 / tantheta2);
387 
388  // fold onto 0...pi
389  float deltaPhi = TMath::Abs(phi1 - phi2);
390  while (deltaPhi > M_PI * 2) {
391  deltaPhi -= M_PI * 2;
392  }
393  if (deltaPhi > M_PI) {
394  deltaPhi = M_PI * 2 - deltaPhi;
395  }
396 
397  float cosDeltaPhi = 0;
398  if (deltaPhi < M_PI / 3) {
399  cosDeltaPhi = 1.0 - deltaPhi * deltaPhi / 2 + deltaPhi * deltaPhi * deltaPhi * deltaPhi / 24;
400  } else if (deltaPhi < 2 * M_PI / 3) {
401  cosDeltaPhi = -(deltaPhi - M_PI / 2) + 1.0 / 6 * TMath::Power((deltaPhi - M_PI / 2), 3);
402  } else {
403  cosDeltaPhi = -1.0 + 1.0 / 2.0 * (deltaPhi - M_PI) * (deltaPhi - M_PI) - 1.0 / 24.0 * TMath::Power(deltaPhi - M_PI, 4);
404  }
405 
406  double mass2 = m0_1 * m0_1 + m0_2 * m0_2 + 2 * (TMath::Sqrt(e1squ * e2squ) - (pt1 * pt2 * (cosDeltaPhi + 1.0 / tantheta1 / tantheta2)));
407 
408  // Printf(Form("%f %f %f %f %f %f %f %f %f", pt1, eta1, phi1, pt2, eta2, phi2, m0_1, m0_2, mass2));
409 
410  return mass2;
411  }

413  template <typename T>
414  bool twoTrackCut(T const& track1, T const& track2, int bSign)
415  {
416  // the variables & cuthave been developed by the HBT group
417  // see e.g. https://indico.cern.ch/materialDisplay.py?contribId=36&sessionId=6&materialId=slides&confId=142700
418 
419  auto deta = track1.eta() - track2.eta();
420 
421  // optimization
422  if (TMath::Abs(deta) < cfgTwoTrackCut * 2.5 * 3) {
423  // check first boundaries to see if is worth to loop and find the minimum
424  float dphistar1 = getDPhiStar(track1, track2, cfgTwoTrackCutMinRadius, bSign);
425  float dphistar2 = getDPhiStar(track1, track2, 2.5, bSign);
426 
427  const float kLimit = cfgTwoTrackCut * 3;
428 
429  if (TMath::Abs(dphistar1) < kLimit || TMath::Abs(dphistar2) < kLimit || dphistar1 * dphistar2 < 0) {
430  float dphistarminabs = 1e5;
431  float dphistarmin = 1e5;
432  for (Double_t rad = cfgTwoTrackCutMinRadius; rad < 2.51; rad += 0.01) {
433  float dphistar = getDPhiStar(track1, track2, rad, bSign);
434 
435  float dphistarabs = TMath::Abs(dphistar);
436 
437  if (dphistarabs < dphistarminabs) {
438  dphistarmin = dphistar;
439  dphistarminabs = dphistarabs;
440  }
441  }
442 
443  qa.mTwoTrackDistancePt[0]->Fill(deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
444 
445  if (dphistarminabs < cfgTwoTrackCut && TMath::Abs(deta) < cfgTwoTrackCut) {
446  //Printf("Removed track pair %ld %ld with %f %f %f %f %d %f %f %d %d", track1.index(), track2.index(), deta, dphistarminabs, track1.phi(), track1.pt(), track1.sign(), track2.phi(), track2.pt(), track2.sign(), bSign);
447  return true;
448  }
449 
450  qa.mTwoTrackDistancePt[1]->Fill(deta, dphistarmin, TMath::Abs(track1.pt() - track2.pt()));
451  }
