29 struct TrackCovI {
30  float sxx, syy, syz, szz;
31 
32  TrackCovI(const o2::track::TrackParCov& trc, float xerrFactor = 1.) { set(trc, xerrFactor); }
33 
34  TrackCovI() = default;
35 
36  void set(const o2::track::TrackParCov& trc, float xerrFactor = 1)
37  {
38  // we assign Y error to X for DCA calculation
39  // (otherwise for quazi-collinear tracks the X will not be constrained)
40  float cyy = trc.getSigmaY2(), czz = trc.getSigmaZ2(), cyz = trc.getSigmaZY(), cxx = cyy * xerrFactor;
41  float detYZ = cyy * czz - cyz * cyz;
42  if (detYZ > 0.) {
43  auto detYZI = 1. / detYZ;
44  sxx = 1. / cxx;
45  syy = czz * detYZI;
46  syz = -cyz * detYZI;
47  szz = cyy * detYZI;
48  } else {
49  throw std::runtime_error("invalid track covariance");
50  }
51  }
52 };

56 struct TrackDeriv {
57  float dydx, dzdx, d2ydx2, d2zdx2;
58  TrackDeriv() = default;
59  TrackDeriv(const o2::track::TrackPar& trc, float bz) { set(trc, bz); }
60  void set(const o2::track::TrackPar& trc, float bz)
61  {
62  float snp = trc.getSnp(), csp = std::sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz) * cspI;
63  dydx = snp * cspI; // = snp/csp
64  dzdx = trc.getTgl() * cspI; // = tgl/csp
65  d2ydx2 = crv2c * cspI * cspI; // = crv/csp^3
66  d2zdx2 = crv2c * dzdx * dydx; // = crv*tgl*snp/csp^3
67  }
68 };

295 template <int N, typename... Args>
296 template <class... Tr>
297 int DCAFitterN<N, Args...>::process(const Tr&... args)
298 {
299  // This is a main entry point: fit PCA of N tracks
300  static_assert(sizeof...(args) == N, "incorrect number of input tracks");
301  assign(0, args...);
302  clear();
303  for (int i = 0; i < N; i++) {
304  mTrAux[i].set(*mOrigTrPtr[i], mBz);
305  }
306  if (!mCrossings.set(mTrAux[0], *mOrigTrPtr[0], mTrAux[1], *mOrigTrPtr[1])) { // even for N>2 it should be enough to test just 1 loop
307  return 0; // no crossing
308  }
309  if (mUseAbsDCA) {
310  calcRMatrices(); // needed for fast residuals derivatives calculation in case of abs. distance minimization
311  }
312  if (mCrossings.nDCA == MAXHYP) { // if there are 2 candidates and they are too close, chose their mean as a starting point
313  auto dst2 = (mCrossings.xDCA[0] - mCrossings.xDCA[1]) * (mCrossings.xDCA[0] - mCrossings.xDCA[1]) +
314  (mCrossings.yDCA[0] - mCrossings.yDCA[1]) * (mCrossings.yDCA[0] - mCrossings.yDCA[1]);
315  if (dst2 < mMaxDist2ToMergeSeeds) {
316  mCrossings.nDCA = 1;
317  mCrossings.xDCA[0] = 0.5 * (mCrossings.xDCA[0] + mCrossings.xDCA[1]);
318  mCrossings.yDCA[0] = 0.5 * (mCrossings.yDCA[0] + mCrossings.yDCA[1]);
319  }
320  }
321  // check all crossings
322  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
323  // check if radius is acceptable
324  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
325  continue;
326  }
327  mCrossIDCur = ic;
328  mCrossIDAlt = (mCrossings.nDCA == 2 && mAllowAltPreference) ? 1 - ic : -1; // works for max 2 crossings
329  mNIters[mCurHyp] = 0;
330  mTrPropDone[mCurHyp] = false;
331  mChi2[mCurHyp] = -1.;
332  mPCA[mCurHyp][0] = mCrossings.xDCA[ic];
333  mPCA[mCurHyp][1] = mCrossings.yDCA[ic];
334 
335  if (mUseAbsDCA ? minimizeChi2NoErr() : minimizeChi2()) {
336  mOrder[mCurHyp] = mCurHyp;
337  if (mPropagateToPCA && !propagateTracksToVertex(mCurHyp)) {
338  continue; // discard candidate if failed to propagate to it
339  }
340  mCurHyp++;
341  }
342  }
343 
344  for (int i = mCurHyp; i--;) { // order in quality
345  for (int j = i; j--;) {
346  if (mChi2[mOrder[i]] < mChi2[mOrder[j]]) {
347  std::swap(mOrder[i], mOrder[j]);
348  }
349  }
350  }
351  return mCurHyp;
352 }

355 template <int N, typename... Args>
356 bool DCAFitterN<N, Args...>::calcPCACoefs()
357 {
358  //< calculate Ti matrices for global vertex decomposition to V = sum_{0<i<N} Ti pi, see EQ.T in the ref
359  if (!calcInverseWeight()) {
360  return false;
361  }
362  for (int i = N; i--;) { // build Mi*Ei matrix
363  const auto& taux = mTrAux[i];
364  const auto& tcov = mTrcEInv[mCurHyp][i];
365  MatStd3D miei;
366  miei[0][0] = taux.c * tcov.sxx;
367  miei[0][1] = -taux.s * tcov.syy;
368  miei[0][2] = -taux.s * tcov.syz;
369  miei[1][0] = taux.s * tcov.sxx;
370  miei[1][1] = taux.c * tcov.syy;
371  miei[1][2] = taux.c * tcov.syz;
372  //miei[2][0] = 0;
373  miei[2][1] = tcov.syz;
374  miei[2][2] = tcov.szz;
375  mTrCFVT[mCurHyp][i] = mWeightInv * miei;
376  }
377  return true;
378 }

381 template <int N, typename... Args>
382 bool DCAFitterN<N, Args...>::calcInverseWeight()
383 {
384  //< calculate [sum_{0<j<N} M_j*E_j*M_j^T]^-1 used for Ti matrices, see EQ.T
385  auto* arrmat = mWeightInv.Array();
386  memset(arrmat, 0, sizeof(mWeightInv));
387  enum { XX,
388  XY,
389  YY,
390  XZ,
391  YZ,
392  ZZ };
393  for (int i = N; i--;) {
394  const auto& taux = mTrAux[i];
395  const auto& tcov = mTrcEInv[mCurHyp][i];
396  arrmat[XX] += taux.cc * tcov.sxx + taux.ss * tcov.syy;
397  arrmat[XY] += taux.cs * (tcov.sxx - tcov.syy);
398  arrmat[XZ] += -taux.s * tcov.syz;
399  arrmat[YY] += taux.cc * tcov.syy + taux.ss * tcov.sxx;
400  arrmat[YZ] += taux.c * tcov.syz;
401  arrmat[ZZ] += tcov.szz;
402  }
403  // invert 3x3 symmetrix matrix
404  return mWeightInv.Invert();
405 }

408 template <int N, typename... Args>
409 void DCAFitterN<N, Args...>::calcResidDerivatives()
410 {
411  //< calculate matrix of derivatives for weighted chi2: residual i vs parameter X of track j
412  MatStd3D matMT;
413  for (int i = N; i--;) { // residual being differentiated
414  const auto& taux = mTrAux[i];
415  for (int j = N; j--;) { // track over which we differentiate
416  const auto& matT = mTrCFVT[mCurHyp][j]; // coefficient matrix for track J
417  const auto& trDx = mTrDer[mCurHyp][j]; // track point derivs over track X param
418  auto& dr1 = mDResidDx[i][j];
419  auto& dr2 = mD2ResidDx2[i][j];
420  // calculate M_i^tr * T_j
421  matMT[0][0] = taux.c * matT[0][0] + taux.s * matT[1][0];
422  matMT[0][1] = taux.c * matT[0][1] + taux.s * matT[1][1];
423  matMT[0][2] = taux.c * matT[0][2] + taux.s * matT[1][2];
424  matMT[1][0] = -taux.s * matT[0][0] + taux.c * matT[1][0];
425  matMT[1][1] = -taux.s * matT[0][1] + taux.c * matT[1][1];
426  matMT[1][2] = -taux.s * matT[0][2] + taux.c * matT[1][2];
427  matMT[2][0] = matT[2][0];
428  matMT[2][1] = matT[2][1];
429  matMT[2][2] = matT[2][2];
430 
431  // calculate DResid_i/Dx_j = (delta_ij - M_i^tr * T_j) * DTrack_k/Dx_k
432  dr1[0] = -(matMT[0][0] + matMT[0][1] * trDx.dydx + matMT[0][2] * trDx.dzdx);
433  dr1[1] = -(matMT[1][0] + matMT[1][1] * trDx.dydx + matMT[1][2] * trDx.dzdx);
434  dr1[2] = -(matMT[2][0] + matMT[2][1] * trDx.dydx + matMT[2][2] * trDx.dzdx);
435 
436  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ijk - M_i^tr * T_j * delta_jk) * D2Track_k/dx_k^2
437  dr2[0] = -(matMT[0][1] * trDx.d2ydx2 + matMT[0][2] * trDx.d2zdx2);
438  dr2[1] = -(matMT[1][1] * trDx.d2ydx2 + matMT[1][2] * trDx.d2zdx2);
439  dr2[2] = -(matMT[2][1] * trDx.d2ydx2 + matMT[2][2] * trDx.d2zdx2);
440 
441  if (i == j) {
442  dr1[0] += 1.;
443  dr1[1] += trDx.dydx;
444  dr1[2] += trDx.dzdx;
445 
446  dr2[1] += trDx.d2ydx2;
447  dr2[2] += trDx.d2zdx2;
448  }
449  } // track over which we differentiate
450  } // residual being differentiated
451 }

454 template <int N, typename... Args>
455 void DCAFitterN<N, Args...>::calcResidDerivativesNoErr()
456 {
457  //< calculate matrix of derivatives for absolute distance chi2: residual i vs parameter X of track j
458  constexpr double NInv1 = 1. - NInv; // profit from Rii = I/Ninv
459  for (int i = N; i--;) { // residual being differentiated
460  const auto& trDxi = mTrDer[mCurHyp][i]; // track point derivs over track X param
461  auto& dr1ii = mDResidDx[i][i];
462  auto& dr2ii = mD2ResidDx2[i][i];
463  dr1ii[0] = NInv1;
464  dr1ii[1] = NInv1 * trDxi.dydx;
465  dr1ii[2] = NInv1 * trDxi.dzdx;
466 
467  dr2ii[0] = 0;
468  dr2ii[1] = NInv1 * trDxi.d2ydx2;
469  dr2ii[2] = NInv1 * trDxi.d2zdx2;
470 
471  for (int j = i; j--;) { // track over which we differentiate
472  auto& dr1ij = mDResidDx[i][j];
473  auto& dr1ji = mDResidDx[j][i];
474  const auto& trDxj = mTrDer[mCurHyp][j]; // track point derivs over track X param
475  auto cij = mCosDif[i][j], sij = mSinDif[i][j]; // M_i^T*M_j / N matrices non-trivial elements = {ci*cj+si*sj , si*cj-ci*sj }, see 5 in ref.
476 
477  // calculate DResid_i/Dx_j = (delta_ij - R_ij) * DTrack_j/Dx_j for j<i
478  dr1ij[0] = -(cij + sij * trDxj.dydx);
479  dr1ij[1] = -(-sij + cij * trDxj.dydx);
480  dr1ij[2] = -trDxj.dzdx * NInv;
481 
482  // calculate DResid_j/Dx_i = (delta_ij - R_ji) * DTrack_i/Dx_i for j<i
483  dr1ji[0] = -(cij - sij * trDxi.dydx);
484  dr1ji[1] = -(sij + cij * trDxi.dydx);
485  dr1ji[2] = -trDxi.dzdx * NInv;
486 
487  auto& dr2ij = mD2ResidDx2[i][j];
488  auto& dr2ji = mD2ResidDx2[j][i];
489  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ij - Rij) * D2Track_j/dx_j^2 * delta_jk for j<i
490  dr2ij[0] = -sij * trDxj.d2ydx2;
491  dr2ij[1] = -cij * trDxj.d2ydx2;
492  dr2ij[2] = -trDxj.d2zdx2 * NInv;
493 
494  // calculate D2Resid_j/(Dx_i Dx_k) = (delta_ij - Rji) * D2Track_i/dx_i^2 * delta_ik for j<i
495  dr2ji[0] = sij * trDxi.d2ydx2;
496  dr2ji[1] = -cij * trDxi.d2ydx2;
497  dr2ji[2] = -trDxi.d2zdx2 * NInv;
498 
499  } // track over which we differentiate
500  } // residual being differentiated
501 }

519 template <int N, typename... Args>
520 void DCAFitterN<N, Args...>::calcChi2Derivatives()
521 {
522  //< calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters X, see EQ.Chi2 in the ref
523  std::array<std::array<Vec3D, N>, N> covIDrDx; // tempory vectors of covI_j * dres_j/dx_i
524 
525  // chi2 1st derivative
526  for (int i = N; i--;) {
527  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * covI_j * Dres_j/Dx_i }
528  dchi1 = 0;
529  for (int j = N; j--;) {
530  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
531  const auto& covI = mTrcEInv[mCurHyp][j]; // inverse cov matrix of track j
532  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
533  auto& cidr = covIDrDx[i][j]; // vector covI_j * dres_j/dx_i, save for 2nd derivative calculation
534  cidr[0] = covI.sxx * dr1[0];
535  cidr[1] = covI.syy * dr1[1] + covI.syz * dr1[2];
536  cidr[2] = covI.syz * dr1[1] + covI.szz * dr1[2];
537  // calculate res_i * covI_j * dres_j/dx_i
538  dchi1 += ROOT::Math::Dot(res, cidr);
539  }
540  }
541  // chi2 2nd derivative
542  for (int i = N; i--;) {
543  for (int j = i + 1; j--;) { // symmetric matrix
544  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
545  dchi2 = 0;
546  for (int k = N; k--;) {
547  const auto& dr1j = mDResidDx[k][j]; // vector of k-th residuals 1st derivative over X param of track j
548  const auto& cidrkj = covIDrDx[i][k]; // vector covI_k * dres_k/dx_i
549  dchi2 += ROOT::Math::Dot(dr1j, cidrkj);
550  if (k == j) {
551  const auto& res = mTrRes[mCurHyp][k]; // vector of residuals of track k
552  const auto& covI = mTrcEInv[mCurHyp][k]; // inverse cov matrix of track k
553  const auto& dr2ij = mD2ResidDx2[k][j]; // vector of k-th residuals 2nd derivative over X params of track j
554  dchi2 += res[0] * covI.sxx * dr2ij[0] + res[1] * (covI.syy * dr2ij[1] + covI.syz * dr2ij[2]) + res[2] * (covI.syz * dr2ij[1] + covI.szz * dr2ij[2]);
555  }
556  }
557  }
558  }
559 }

563 void DCAFitterN<N, Args...>::calcChi2DerivativesNoErr()
564 {
565  //< calculate 1st and 2nd derivatives of abs DCA (chi2) over track parameters X, see (6) in the ref
566  for (int i = N; i--;) {
567  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * Dres_j/Dx_i }

569  for (int j = N; j--;) {
570  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
571  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
572  dchi1 += ROOT::Math::Dot(res, dr1);
573  if (i >= j) { // symmetrix matrix
574  // chi2 2nd derivative
575  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
576  dchi2 = ROOT::Math::Dot(mTrRes[mCurHyp][i], mD2ResidDx2[i][j]);
577  for (int k = N; k--;) {
578  dchi2 += ROOT::Math::Dot(mDResidDx[k][i], mDResidDx[k][j]);
579  }
580  }
581  }
582  }

750 //___________________________________________________________________
751 template <int N, typename... Args>
752 bool DCAFitterN<N, Args...>::minimizeChi2()
753 {
754  // find best chi2 (weighted DCA) of N tracks in the vicinity of the seed PCA
755  for (int i = N; i--;) {
756  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];
757  auto x = mTrAux[i].c * mPCA[mCurHyp][0] + mTrAux[i].s * mPCA[mCurHyp][1]; // X of PCA in the track frame
758  if (!mCandTr[mCurHyp][i].propagateTo(x, mBz)) {
759  return false;
760  }
761  setTrackPos(mTrPos[mCurHyp][i], mCandTr[mCurHyp][i]); // prepare positions
762  mTrcEInv[mCurHyp][i].set(mCandTr[mCurHyp][i], XerrFactor); // prepare inverse cov.matrices at starting point
763  }
764 
765  if (mMaxDZIni > 0 && !roughDZCut()) { // apply rough cut on tracks Z difference
766  return false;
767  }
768 
769  if (!calcPCACoefs()) { // prepare tracks contribution matrices to the global PCA
770  return false;
771  }
772  calcPCA(); // current PCA
773  calcTrackResiduals(); // current track residuals
774  float chi2Upd, chi2 = calcChi2();
775  do {
776  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
777  calcResidDerivatives(); // current residals derivatives (1st and 2nd)
778  calcChi2Derivatives(); // current chi2 derivatives (1st and 2nd)
779 
780  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
781  if (!mD2Chi2Dx2.Invert()) {
782  LOG(ERROR) << "InversionFailed";
783  return false;
784  }
785  VecND dx = mD2Chi2Dx2 * mDChi2Dx;
786  if (!correctTracks(dx)) {
787  return false;
788  }
789  calcPCA(); // updated PCA
790  if (mCrossIDAlt >= 0 && closerToAlternative()) {
791  mAllowAltPreference = false;
792  return false;
793  }
794  calcTrackResiduals(); // updated residuals
795  chi2Upd = calcChi2(); // updated chi2
796  if (getAbsMax(dx) < mMinParamChange || chi2Upd > chi2 * mMinRelChi2Change) {
797  chi2 = chi2Upd;
798  break; // converged
799  }
800  chi2 = chi2Upd;
801  } while (++mNIters[mCurHyp] < mMaxIter);
802  //
803  mChi2[mCurHyp] = chi2 * NInv;
804  return mChi2[mCurHyp] < mMaxChi2;
805 }
