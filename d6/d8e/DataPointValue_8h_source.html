384  inline void set_payload(
385  const uint64_t* const data,
386  const DeliveryType type)
387  {
388  switch (type) {
389  case RAW_INT:
390  case DPVAL_INT:
391  case RAW_UINT:
392  case DPVAL_UINT:
393  this->payload_pt1 = data[0] & 0x00000000FFFFFFFF;
394  break;
395  case RAW_BOOL:
396  case DPVAL_BOOL:
397  this->payload_pt1 = (data[0] ? 1 : 0);
398  break;
399  case RAW_CHAR:
400  case DPVAL_CHAR:
401  this->payload_pt1 = data[0] & 0x00000000000000FF;
402  break;
403  case RAW_DOUBLE:
404  case DPVAL_DOUBLE:
405  case RAW_TIME:
406  case DPVAL_TIME:
407  this->payload_pt1 = data[0];
408  break;
409  case RAW_STRING:
410  case DPVAL_STRING:
411  std::strncpy((char*)&payload_pt1, (char*)data, 8);
412  std::strncpy((char*)&payload_pt2, (char*)data + 8, 8);
413  std::strncpy((char*)&payload_pt3, (char*)data + 16, 8);
414  std::strncpy((char*)&payload_pt4, (char*)data + 24, 8);
415  std::strncpy((char*)&payload_pt5, (char*)data + 32, 8);
416  std::strncpy((char*)&payload_pt6, (char*)data + 40, 8);
417  std::strncpy((char*)&payload_pt7, (char*)data + 48, 8);
418  break;
419  case RAW_BINARY:
420  case DPVAL_BINARY:
421  memcpy((void*)&payload_pt1, (void*)data, 56);
422  case VOID:
423  break;
424  default:
425  throw std::domain_error("Invalid DeliveryType.");
426  }
427  }

460  inline std::unique_ptr<std::string> get_timestamp() const noexcept
461  {
462 #if defined(__linux__) || defined(__APPLE__)
463  // time_t should be uint64_t (compatible) on 64-bit Linux platforms:
464  char buffer[17];
465  std::time_t ts((uint64_t)sec);
466  std::strftime(buffer, 32, "%FT%T", std::gmtime(&ts));
467  std::ostringstream oss;
468  oss << std::string(buffer) << "." << std::setw(3) << std::setfill('0') << std::to_string(msec) << "Z";
469  return std::make_unique<std::string>(
470  std::move(oss.str()));
471 #else
472  return std::make_unique<std::string>("Unsupported platform");
473 #endif
474  }
