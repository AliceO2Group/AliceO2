35 void Stepper::process(const TVirtualMC& vmc)
36 {
37 
38  if (!(vmc.TrackCharge())) {
39  // Only charged particles
40  return;
41  }
42 
43  o2::SimTrackStatus t{vmc};
44 
45  int detElemId;
46  vmc.CurrentVolOffID(2, detElemId); // go up 2 levels in the hierarchy to get the DE
47 
48  auto stack = static_cast<o2::data::Stack*>(vmc.GetStack());
49 
50  if (t.isEntering() || t.isExiting()) {
51  // generate a track referenced
52  o2::TrackReference tr{vmc, detElemId};
53  stack->addTrackReference(tr);
54  }
55 
56  if (t.isEntering()) {
57  float x, y, z;
58  vmc.TrackPosition(x, y, z);
59  mEntrancePoint.SetXYZ(x, y, z);
60  resetStep();
61  }
62 
63  mTrackEloss += vmc.Edep();
64  mTrackLength += vmc.TrackStep();
65 
66  if (t.isExiting() || t.isStopped()) {
67  float x, y, z;
68  vmc.TrackPosition(x, y, z);
69  mHits->emplace_back(stack->GetCurrentTrackNumber(), detElemId, mEntrancePoint,
70  math_utils::Point3D<float>{x, y, z}, mTrackEloss, mTrackLength);
71  resetStep();
72  }
73 }
