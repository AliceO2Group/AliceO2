85 bool DataInputDescriptor::setFile(int counter)
86 {
87  // no files left
88  if (counter >= getNumberInputfiles()) {
89  return false;
90  }
91 
92  // open file
93  auto filename = mfilenames[counter]->fileName;
94  if (mcurrentFile) {
95  if (mcurrentFile->GetName() != filename) {
96  closeInputFile();
97  mcurrentFile = TFile::Open(filename.c_str());
98  }
99  } else {
100  mcurrentFile = TFile::Open(filename.c_str());
101  }
102  if (!mcurrentFile) {
103  throw std::runtime_error(fmt::format("Couldn't open file \"{}\"!", filename));
104  }
105  mcurrentFile->SetReadaheadSize(50 * 1024 * 1024);
106 
107  // get the directory names
108  if (mfilenames[counter]->numberOfTimeFrames <= 0) {
109  std::regex TFRegex = std::regex("DF_[0-9]+");
110  TList* keyList = mcurrentFile->GetListOfKeys();
111 
112  // extract TF numbers and sort accordingly
113  for (auto key : *keyList) {
114  if (std::regex_match(((TObjString*)key)->GetString().Data(), TFRegex)) {
115  auto folderNumber = std::stoul(std::string(((TObjString*)key)->GetString().Data()).substr(3));
116  mfilenames[counter]->listOfTimeFrameNumbers.emplace_back(folderNumber);
117  }
118  }
119  std::sort(mfilenames[counter]->listOfTimeFrameNumbers.begin(), mfilenames[counter]->listOfTimeFrameNumbers.end());
120 
121  for (auto folderNumber : mfilenames[counter]->listOfTimeFrameNumbers) {
122  auto folderName = "DF_" + std::to_string(folderNumber);
123  mfilenames[counter]->listOfTimeFrameKeys.emplace_back(folderName);
124  }
125  mfilenames[counter]->numberOfTimeFrames = mfilenames[counter]->listOfTimeFrameKeys.size();
126  }
127 
128  return true;
129 }

181 int DataInputDescriptor::fillInputfiles()
182 {
183  if (getNumberInputfiles() > 0) {
184  // 1. mfilenames
185  return getNumberInputfiles();
186  }
187 
188  auto fileName = getInputfilesFilename();
189  if (!fileName.empty()) {
190  // 2. getFilenamesRegex() @ getInputfilesFilename()
191  try {
192  std::ifstream filelist(fileName);
193  while (std::getline(filelist, fileName)) {
194  // remove white spaces, empty lines are skipped
195  fileName.erase(std::remove_if(fileName.begin(), fileName.end(), ::isspace), fileName.end());
196  if (!fileName.empty() && (getFilenamesRegexString().empty() ||
197  std::regex_match(fileName, getFilenamesRegex()))) {
198  addFileNameHolder(makeFileNameHolder(fileName));
199  }
200  }
201  } catch (...) {
202  LOGP(ERROR, "Check the input files file! Unable to process \"{}\"!", getInputfilesFilename());
203  return 0;
204  }
205  } else {
206  // 3. getFilenamesRegex() @ mdefaultFilenamesPtr
207  if (mdefaultFilenamesPtr) {
208  for (auto fileNameHolder : *mdefaultFilenamesPtr) {
209  if (getFilenamesRegexString().empty() ||
210  std::regex_match(fileNameHolder->fileName, getFilenamesRegex())) {
211  addFileNameHolder(fileNameHolder);
212  }
213  }
214  }
215  }
216 
217  return getNumberInputfiles();
218 }

294 bool DataInputDirector::readJsonDocument(Document* jsonDoc)
295 {
296  // initialisations
297  std::string fileName("");
298  int ntfm = -1;
299  const char* itemName;
300 
301  // is it a proper json document?
302  if (jsonDoc->HasParseError()) {
303  LOGP(ERROR, "Check the JSON document! There is a problem with the format!");
304  return false;
305  }
306 
307  // InputDirector
308  itemName = "InputDirector";
309  const Value& didirItem = (*jsonDoc)[itemName];
310  if (!didirItem.IsObject()) {
311  LOGP(INFO, "No \"{}\" object found in the JSON document!", itemName);
312  return true;
313  }
314 
315  // now read various items
316  itemName = "debugmode";
317  if (didirItem.HasMember(itemName)) {
318  if (didirItem[itemName].IsBool()) {
319  mDebugMode = (didirItem[itemName].GetBool());
320  } else {
321  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a boolean!", itemName);
322  return false;
323  }
324  } else {
325  mDebugMode = false;
326  }
327 
328  if (mDebugMode) {
329  StringBuffer buffer;
330  buffer.Clear();
331  PrettyWriter<StringBuffer> writer(buffer);
332  didirItem.Accept(writer);
333  LOGP(INFO, "InputDirector object: {}", std::string(buffer.GetString()));
334  }
335 
336  itemName = "fileregex";
337  if (didirItem.HasMember(itemName)) {
338  if (didirItem[itemName].IsString()) {
339  setFilenamesRegex(didirItem[itemName].GetString());
340  } else {
341  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
342  return false;
343  }
344  }
345 
346  itemName = "resfiles";
347  if (didirItem.HasMember(itemName)) {
348  if (didirItem[itemName].IsString()) {
349  fileName = didirItem[itemName].GetString();
350  if (fileName.size() && fileName[0] == '@') {
351  fileName.erase(0, 1);
352  setInputfilesFile(fileName);
353  } else {
354  setInputfilesFile("");
355  mdefaultInputFiles.emplace_back(makeFileNameHolder(fileName));
356  }
357  } else if (didirItem[itemName].IsArray()) {
358  setInputfilesFile("");
359  auto fns = didirItem[itemName].GetArray();
360  for (auto& fn : fns) {
361  mdefaultInputFiles.emplace_back(makeFileNameHolder(fn.GetString()));
362  }
363  } else {
364  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string or an array!", itemName);
365  return false;
366  }
367  }
368 
369  itemName = "InputDescriptors";
370  if (didirItem.HasMember(itemName)) {
371  if (!didirItem[itemName].IsArray()) {
372  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be an array!", itemName);
373  return false;
374  }
375 
376  // loop over DataInputDescriptors
377  for (auto& didescItem : didirItem[itemName].GetArray()) {
378  if (!didescItem.IsObject()) {
379  LOGP(ERROR, "Check the JSON document! \"{}\" must be objects!", itemName);
380  return false;
381  }
382  // create a new dataInputDescriptor
383  auto didesc = new DataInputDescriptor(mAlienSupport);
384  didesc->setDefaultInputfiles(&mdefaultInputFiles);
385 
386  itemName = "table";
387  if (didescItem.HasMember(itemName)) {
388  if (didescItem[itemName].IsString()) {
389  didesc->tablename = didescItem[itemName].GetString();
390  didesc->matcher = DataDescriptorQueryBuilder::buildNode(didesc->tablename);
391  } else {
392  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
393  return false;
394  }
395  } else {
396  LOGP(ERROR, "Check the JSON document! Item \"{}\" is missing!", itemName);
397  return false;
398  }
399 
400  itemName = "treename";
401  if (didescItem.HasMember(itemName)) {
402  if (didescItem[itemName].IsString()) {
403  didesc->treename = didescItem[itemName].GetString();
404  } else {
405  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
406  return false;
407  }
408  } else {
409  auto m = DataDescriptorQueryBuilder::getTokens(didesc->tablename);
410  didesc->treename = m[2];
411  }
412 
413  itemName = "fileregex";
414  if (didescItem.HasMember(itemName)) {
415  if (didescItem[itemName].IsString()) {
416  if (didesc->getNumberInputfiles() == 0) {
417  didesc->setFilenamesRegex(didescItem[itemName].GetString());
418  }
419  } else {
420  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string!", itemName);
421  return false;
422  }
423  } else {
424  if (didesc->getNumberInputfiles() == 0) {
425  didesc->setFilenamesRegex(mFilenameRegexPtr);
426  }
427  }
428 
429  itemName = "resfiles";
430  if (didescItem.HasMember(itemName)) {
431  if (didescItem[itemName].IsString()) {
432  fileName = didescItem[itemName].GetString();
433  if (fileName.size() && fileName[0] == '@') {
434  didesc->setInputfilesFile(fileName.erase(0, 1));
435  } else {
436  if (didesc->getFilenamesRegexString().empty() ||
437  std::regex_match(fileName, didesc->getFilenamesRegex())) {
438  didesc->addFileNameHolder(makeFileNameHolder(fileName));
439  }
440  }
441  } else if (didescItem[itemName].IsArray()) {
442  auto fns = didescItem[itemName].GetArray();
443  for (auto& fn : fns) {
444  if (didesc->getFilenamesRegexString().empty() ||
445  std::regex_match(fn.GetString(), didesc->getFilenamesRegex())) {
446  didesc->addFileNameHolder(makeFileNameHolder(fn.GetString()));
447  }
448  }
449  } else {
450  LOGP(ERROR, "Check the JSON document! Item \"{}\" must be a string or an array!", itemName);
451  return false;
452  }
453  } else {
454  didesc->setInputfilesFile(minputfilesFilePtr);
455  }
456 
457  // fill mfilenames and add InputDescriptor to InputDirector
458  if (didesc->fillInputfiles() > 0) {
459  mdataInputDescriptors.emplace_back(didesc);
460  } else {
461  didesc->printOut();
462  LOGP(INFO, "This DataInputDescriptor is ignored because its file list is empty!");
463  }
464  mAlienSupport &= didesc->isAlienSupportOn();
465  }
466  }
467 
468  // add a default DataInputDescriptor
469  createDefaultDataInputDescriptor();
470 
471  // check that all DataInputDescriptors have the same number of input files
472  if (!isValid()) {
473  printOut();
474  return false;
475  }
476 
477  // print the DataIputDirector
478  if (mDebugMode) {
479  printOut();
480  }
481 
482  return true;
483 }
