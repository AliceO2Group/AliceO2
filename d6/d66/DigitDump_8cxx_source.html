48 Int_t DigitDump::updateCRU(const CRU& cru, const Int_t row, const Int_t pad,
49  const Int_t timeBin, const Float_t signal)
50 {
51  if ((timeBin < mFirstTimeBin) || (timeBin > mLastTimeBin)) {
52  return 0;
53  }
54 
55  if (!mInitialized) {
56  initInputOutput();
57  }
58 
59  Mapper& mapper = Mapper::instance();
60  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
61  const int globalRow = row + regionInfo.getGlobalRowOffset();
62  const int sectorRow = globalRow - (cru.rocType() == RocType::OROC) * mapper.getNumberOfRowsROC(0);
63 
64  // noise and pedestal values
65  float pedestal = 0;
66  float noise = 0;
67  if (mPedestal) {
68  pedestal = mPedestal->getValue(cru.roc(), sectorRow, pad);
69  }
70  if (mNoise) {
71  noise = mNoise->getValue(cru.roc(), sectorRow, pad);
72  }
73 
74  // check adc thresholds (zero suppression)
75  const Float_t signalCorr = signal - pedestal;
76 
77  if ((signalCorr < mADCMin) || (signalCorr > mADCMax)) {
78  return 0;
79  }
80 
81  if ((mNoiseThreshold > 0) && (signalCorr < noise * mNoiseThreshold)) {
82  return 0;
83  }
84 
85  // check for masked pads
86  if (mPadMask.size() && std::find(mPadMask.begin(), mPadMask.end(), std::array<int, 3>({int(cru.roc()), sectorRow, pad})) != mPadMask.end()) {
87  return 1;
88  }
89  //printf("updateCRU: %d, %d (%d, %d), %d, %d, %f, %f\n", int(cru), row, globalRow, sectorRow, pad, timeBin, signal, pedestal);
90 
91  // fill digits
92  addDigit(cru, signalCorr, globalRow, pad, timeBin);
93 
94  return 0;
95 }

98 void DigitDump::sortDigits()
99 {
100  // sort digits
101  for (auto& digits : mDigits) {
102  std::sort(digits.begin(), digits.end(), [](const auto& a, const auto& b) {
103  if (a.getTimeStamp() < b.getTimeStamp()) {
104  return true;
105  }
106  if ((a.getTimeStamp() == b.getTimeStamp()) && (a.getRow() < b.getRow())) {
107  return true;
108  }
109  return false;
110  });
111  }
112 }

126 void DigitDump::loadNoiseAndPedestal()
127 {
128  if (!mPedestalAndNoiseFile.size()) {
129  LOG(WARNING) << "No pedestal and noise file name set";
130  return;
131  }
132 
133  std::unique_ptr<TFile> f(TFile::Open(mPedestalAndNoiseFile.data()));
134  if (!f || !f->IsOpen() || f->IsZombie()) {
135  LOG(FATAL) << "Could not open pedestal file: " << mPedestalAndNoiseFile;
136  }
137 
138  CalPad* pedestal{nullptr};
139  CalPad* noise{nullptr};
140 
141  f->GetObject("Pedestals", pedestal);
142  f->GetObject("Noise", noise);
143 
144  mPedestal = std::move(std::unique_ptr<CalPad>(pedestal));
145  mNoise = std::move(std::unique_ptr<CalPad>(noise));
146 }
