40 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
41 {
42  // convert digits/channel to their compressed version
43  cd.clear();
44  if (!digitVec.size()) {
45  return;
46  }
47  const auto& dig0 = digitVec[0];
48  cd.header.nTriggers = digitVec.size();
49  cd.header.firstOrbit = dig0.mIntRecord.orbit;
50  cd.header.firstBC = dig0.mIntRecord.bc;
51 
52  cd.trigger.resize(cd.header.nTriggers);
53  cd.bcInc.resize(cd.header.nTriggers);
54  cd.orbitInc.resize(cd.header.nTriggers);
55  cd.nChan.resize(cd.header.nTriggers);
56 
57  cd.idChan.resize(channelVec.size());
58  cd.time.resize(channelVec.size());
59  cd.charge.resize(channelVec.size());
60  cd.feeBits.resize(channelVec.size());
61 
62  uint16_t prevBC = cd.header.firstBC;
63  uint32_t prevOrbit = cd.header.firstOrbit;
64  uint32_t ccount = 0;
65  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
66  const auto& digit = digitVec[idig];
67  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
68 
69  // fill trigger info
70  cd.trigger[idig] = digit.mTriggers.triggersignals;
71  if (prevOrbit == digit.mIntRecord.orbit) {
72  cd.bcInc[idig] = digit.mIntRecord.bc - prevBC;
73  cd.orbitInc[idig] = 0;
74  } else {
75  cd.bcInc[idig] = digit.mIntRecord.bc;
76  cd.orbitInc[idig] = digit.mIntRecord.orbit - prevOrbit;
77  }
78  prevBC = digit.mIntRecord.bc;
79  prevOrbit = digit.mIntRecord.orbit;
80  // fill channels info
81  cd.nChan[idig] = chanels.size();
82  if (!cd.nChan[idig]) {
83  LOG(ERROR) << "Digits with no channels";
84  continue;
85  }
86  uint8_t prevChan = 0;
87  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
88  assert(prevChan <= chanels[ic].mPMNumber);
89  cd.idChan[ccount] = chanels[ic].mPMNumber - prevChan;
90  cd.time[ccount] = chanels[ic].mTime; // make sure it fits to short!!!
91  cd.charge[ccount] = chanels[ic].mChargeADC; // make sure we really need short!!!
92  cd.feeBits[ccount] = chanels[ic].mFEEBits;
93  prevChan = chanels[ic].mPMNumber;
94  ccount++;
95  }
96  }
97 }

100 void CTFCoder::createCoders(const std::string& dictPath, o2::ctf::CTFCoderBase::OpType op)
101 {
102  bool mayFail = true; // RS FIXME if the dictionary file is not there, do not produce exception
103  auto buff = readDictionaryFromFile<CTF>(dictPath, mayFail);
104  if (!buff.size()) {
105  if (mayFail) {
106  return;
107  }
108  throw std::runtime_error("Failed to create CTF dictionaty");
109  }
110  const auto* ctf = CTF::get(buff.data());
111 
112  auto getFreq = [ctf](CTF::Slots slot) -> o2::rans::FrequencyTable {
113  o2::rans::FrequencyTable ft;
114  auto bl = ctf->getBlock(slot);
115  auto md = ctf->getMetadata(slot);
116  ft.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min, md.max);
117  return std::move(ft);
118  };
119  auto getProbBits = [ctf](CTF::Slots slot) -> int {
120  return ctf->getMetadata(slot).probabilityBits;
121  };
122 
123  CompressedDigits cd; // just to get member types
124 #define MAKECODER(part, slot) createCoder<decltype(part)::value_type>(op, getFreq(slot), getProbBits(slot), int(slot))
125  // clang-format off
126  MAKECODER(cd.trigger, CTF::BLC_trigger);
127  MAKECODER(cd.bcInc, CTF::BLC_bcInc);
128  MAKECODER(cd.orbitInc, CTF::BLC_orbitInc);
129  MAKECODER(cd.nChan, CTF::BLC_nChan);
130 
131  MAKECODER(cd.idChan, CTF::BLC_idChan);
132  MAKECODER(cd.time, CTF::BLC_time);
133  MAKECODER(cd.charge, CTF::BLC_charge);
134  MAKECODER(cd.feeBits, CTF::BLC_feeBits);
135  // clang-format on
136 }
