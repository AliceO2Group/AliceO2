60 {
64  while (true) {
65 
67  mParticles.clear();
68

73

77  }
78

106  Generator::addTracks(FairPrimaryGenerator* primGen)
107 {

111  if (!o2primGen) {
112  LOG(FATAL) << "PrimaryGenerator is not a o2::eventgen::PrimaryGenerator";
113  return kFALSE;
114  }
115 
117  for (const auto& particle : mParticles) {
118  o2primGen->AddTrack(particle.GetPdgCode(),
119  particle.Px() * mMomentumUnit,
120  particle.Py() * mMomentumUnit,
121  particle.Pz() * mMomentumUnit,
122  particle.Vx() * mPositionUnit,
123  particle.Vy() * mPositionUnit,
124  particle.Vz() * mPositionUnit,
125  particle.GetMother(0),
126  particle.GetMother(1),
127  particle.GetDaughter(0),
128  particle.GetDaughter(1),
129  particle.GetStatusCode() == 1,
130  particle.Energy() * mEnergyUnit,

132  particle.GetWeight(),

153 Bool_t
154  Generator::triggerEvent()
155 {
159  if (mTriggers.size() == 0 && mDeepTriggers.size() == 0) {
160  return kTRUE;
161  }
162 
164  Bool_t triggered;
165  if (mTriggerMode == kTriggerOFF) {
166  return kTRUE;
167  } else if (mTriggerMode == kTriggerOR) {
168  triggered = kFALSE;
169  } else if (mTriggerMode == kTriggerAND) {
170  triggered = kTRUE;
171  } else {
172  return kTRUE;
173  }
174 
176  for (const auto& trigger : mTriggers) {
177  auto retval = trigger(mParticles);
178  if (mTriggerMode == kTriggerOR) {
179  triggered |= retval;
180  }
181  if (mTriggerMode == kTriggerAND) {
182  triggered &= retval;
183  }
184  }
185 
187  for (const auto& trigger : mDeepTriggers) {
188  auto retval = trigger(mInterface, mInterfaceName);
189  if (mTriggerMode == kTriggerOR) {
190  triggered |= retval;
191  }
192  if (mTriggerMode == kTriggerAND) {
193  triggered &= retval;
194  }
195  }
196 
198  return triggered;
199 }
