57  void run(o2::framework::ProcessingContext& pc) final
58  {
59  int nSlots = pc.inputs().getNofParts(0);
60  assert(pc.inputs().getNofParts(1) == nSlots);
61 
62  for (int isl = 0; isl < nSlots; isl++) {
63  const auto type = pc.inputs().get<int>("clbInfo", isl);
64  const auto pld = pc.inputs().get<gsl::span<char>>("clbPayload", isl); // this is actually an image of TMemFile
65  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().get("clbInfo", isl));
66  const auto subSpec = dh->subSpecification;
67  const int lane = subSpec >> 4;
68  const int calibType = subSpec & 0xf;
69 
70  //const auto& path = wrp->getPath();
71  TMemFile f("file", (char*)&pld[0], pld.size(), "READ");
72  if (!f.IsZombie()) {
73  auto calDet = f.Get<o2::tpc::CalDet<float>>("data");
74  if (calDet) {
75  if (mMergedCalDets.find(type) == mMergedCalDets.end()) {
76  mMergedCalDets[type] = *calDet;
77  } else {
78  mMergedCalDets[type] += *calDet;
79  }
80  }
81  }
82  f.Close();
83 
84  LOGP(info, "getting slot {}, subspec {:#8x}, lane {}, type {}", isl, subSpec, lane, calibType);
85  //if (mReceivedLanes.test(lane)) {
86  //LOGP(warning, "lane {} received multiple times", lane);
87  //}
88  mReceivedLanes.set(lane);
89  }
90 
91  if (mReceivedLanes.count() == mLanesToExpect) {
92  LOGP(info, "data of all lanes received");
93  if (mPublishAfterComplete) {
94  LOGP(info, "publishing after all data was received");
95  dumpCalibData();
96  sendOutput(pc.outputs());
97  mCalibDumped = false;
98  }
99  mReceivedLanes.reset();
100  }
101  }

110  }

123  //____________________________________________________________________________
124  void sendOutput(DataAllocator& output)
125  {
126  //CDBStorage::MetaData_t md;
127 
128  // perhaps should be changed to time of the run
129  const auto now = std::chrono::system_clock::now();
130  const long timeStart = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
131  const long timeEnd = 99999999999999;
132 
133  for (auto& [type, object] : mMergedCalDets) {
134  o2::ccdb::CcdbObjectInfo w;
135  auto image = o2::ccdb::CcdbApi::createObjectImage(&object, &w);
136 
137  w.setPath(CDBTypeMap.at(CDBType(type)));
138  w.setStartValidityTimestamp(timeStart);
139  w.setEndValidityTimestamp(timeEnd);
140 
141  LOG(INFO) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
142  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
143 
144  o2::header::DataHeader::SubSpecificationType subSpec{(o2::header::DataHeader::SubSpecificationType)type};
145  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, subSpec}, *image.get());
146  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, subSpec}, w);
147  }
148  }
149 
150  //____________________________________________________________________________
151  void dumpCalibData()
152  {
153  if (mDirectFileDump && !mCalibDumped) {
154  LOGP(info, "Dumping output to file");
155  TFile f("merged_CalDet.root", "recreate");
156  for (auto& [type, object] : mMergedCalDets) {
157  f.WriteObject(&object, object.getName().data());
158  }
159  mCalibDumped = true;
160  }
161  }
