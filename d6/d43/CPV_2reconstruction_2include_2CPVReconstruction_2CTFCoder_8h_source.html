57 template <typename VEC>
58 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Cluster>& cluData)
59 {
60  using MD = o2::ctf::Metadata::OptStore;
61  // what to do which each field: see o2::ctd::Metadata explanation
62  constexpr MD optField[CTF::getNBlocks()] = {
63  MD::EENCODE, // BLC_bcIncTrig
64  MD::EENCODE, // BLC_orbitIncTrig
65  MD::EENCODE, // BLC_entriesTrig
66  MD::EENCODE, // BLC_posX
67  MD::EENCODE, // BLC_posZ
68  MD::EENCODE, // BLC_energy
69  MD::EENCODE // BLC_status
70  };
71 
72  CTFHelper helper(trigData, cluData);
73 
74  // book output size with some margin
75  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
76  buff.resize(szIni);
77 
78  auto ec = CTF::create(buff);
79  using ECB = CTF::base;
80 
81  ec->setHeader(helper.createHeader());
82  ec->getANSHeader().majorVersion = 0;
83  ec->getANSHeader().minorVersion = 1;
84  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
85 #define ENCODECPV(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
86  // clang-format off
87  ENCODECPV(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
88  ENCODECPV(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
89  ENCODECPV(helper.begin_entriesTrig(), helper.end_entriesTrig(), CTF::BLC_entriesTrig, 0);
90 
91  ENCODECPV(helper.begin_posX(), helper.end_posX(), CTF::BLC_posX, 0);
92  ENCODECPV(helper.begin_posZ(), helper.end_posZ(), CTF::BLC_posZ, 0);
93  ENCODECPV(helper.begin_energy(), helper.end_energy(), CTF::BLC_energy, 0);
94  ENCODECPV(helper.begin_status(), helper.end_status(), CTF::BLC_status, 0);
95  // clang-format on
96  CTF::get(buff.data())->print(getPrefix());
97 }

100 template <typename VTRG, typename VCLUSTER>
101 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCLUSTER& cluVec)
102 {
103  auto header = ec.getHeader();
104  ec.print(getPrefix());
105  std::vector<uint16_t> bcInc, entries, posX, posZ;
106  std::vector<uint32_t> orbitInc;
107  std::vector<uint8_t> energy, status;
108 
109 #define DECODECPV(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
110  // clang-format off
111  DECODECPV(bcInc, CTF::BLC_bcIncTrig);
112  DECODECPV(orbitInc, CTF::BLC_orbitIncTrig);
113  DECODECPV(entries, CTF::BLC_entriesTrig);
114  DECODECPV(posX, CTF::BLC_posX);
115  DECODECPV(posZ, CTF::BLC_posZ);
116  DECODECPV(energy, CTF::BLC_energy);
117  DECODECPV(status, CTF::BLC_status);
118  // clang-format on
119  //
120  trigVec.clear();
121  cluVec.clear();
122  trigVec.reserve(header.nTriggers);
123  status.reserve(header.nClusters);
124 
125  uint32_t firstEntry = 0, cluCount = 0;
126  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
127 
128  Cluster clu;
129  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
130  // restore TrigRecord
131  if (orbitInc[itrig]) { // non-0 increment => new orbit
132  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
133  ir.orbit += orbitInc[itrig];
134  } else {
135  ir.bc += bcInc[itrig];
136  }
137 
138  firstEntry = cluVec.size();
139  for (uint16_t ic = 0; ic < entries[itrig]; ic++) {
140  clu.setPacked(posX[cluCount], posZ[cluCount], energy[cluCount], status[cluCount]);
141  cluVec.emplace_back(clu);
142  cluCount++;
143  }
144  trigVec.emplace_back(ir, firstEntry, entries[itrig]);
145  }
146  assert(cluCount == header.nClusters);
147 }
