36 void ColumnDataToLocalBoard::process(gsl::span<const ColumnData> data)
37 {
39  mLocalBoardsMap.clear();
40  mGBTMap.clear();
41 
42  // First fill the map with the active local boards.
43  // Each local board gets a unique id.
44  for (auto& col : data) {
45  for (int iline = mMapping.getFirstBoardBP(col.columnId, col.deId); iline <= mMapping.getLastBoardBP(col.columnId, col.deId); ++iline) {
46  if (col.getBendPattern(iline) || col.getNonBendPattern()) {
47  auto uniqueLocId = mCrateMapper.deLocalBoardToRO(col.deId, col.columnId, iline);
48  auto& roData = mLocalBoardsMap[uniqueLocId];
49  roData.statusWord = raw::sSTARTBIT | raw::sCARDTYPE;
50  roData.boardId = uniqueLocId;
51  int ich = detparams::getChamber(col.deId);
52  roData.firedChambers |= (1 << ich);
53  roData.patternsBP[ich] = col.getBendPattern(iline);
54  roData.patternsNBP[ich] = col.getNonBendPattern();
55  }
56  }
57  }
58 
59  // Then group the boards belonging to the same GBT link
60  for (auto& item : mLocalBoardsMap) {
61  if (mDebugMode || keepBoard(item.second)) {
62  auto crateId = crateparams::getCrateId(item.first);
63  auto feeId = crateparams::makeROId(crateId, crateparams::getGBTIdFromBoardInCrate(crateparams::getLocId(item.second.boardId)));
64  mGBTMap[feeId].emplace_back(item.second);
65  }
66  }
67 }
