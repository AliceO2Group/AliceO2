52 std::vector<TopoIndexInfo>
53  WorkflowHelpers::topologicalSort(size_t nodeCount,
54  int const* edgeIn,
55  int const* edgeOut,
56  size_t byteStride,
57  size_t edgesCount)
58 {
59  size_t stride = byteStride / sizeof(int);
60  using EdgeIndex = int;
61  // Create the index which will be returned.
62  std::vector<TopoIndexInfo> index(nodeCount);
63  for (auto wi = 0; wi < nodeCount; ++wi) {
64  index[wi] = {wi, 0};
65  }
66  std::vector<EdgeIndex> remainingEdgesIndex(edgesCount);
67  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
68  remainingEdgesIndex[ei] = ei;
69  }
70 
71  // Create a vector where at each position we have true
72  // if the vector has dependencies, false otherwise
73  std::vector<bool> nodeDeps(nodeCount, false);
74  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
75  nodeDeps[*(edgeOut + ei * stride)] = true;
76  }
77 
78  // We start with all those which do not have any dependencies
79  // They are layer 0.
80  std::list<TopoIndexInfo> L;
81  for (auto ii = 0; ii < index.size(); ++ii) {
82  if (nodeDeps[ii] == false) {
83  L.push_back({ii, 0});
84  }
85  }
86 
87  // The final result.
88  std::vector<TopoIndexInfo> S;
89  // The set of vertices which can be reached by the current node
90  std::set<TopoIndexInfo> nextVertex;
91  // The set of edges which are not related to the current node.
92  std::vector<EdgeIndex> nextEdges;
93  while (!L.empty()) {
94  auto node = L.front();
95  S.push_back(node);
96  L.pop_front();
97  nextVertex.clear();
98  nextEdges.clear();
99 
100  // After this, nextVertex will contain all the vertices
101  // which have the current node as incoming.
102  // nextEdges will contain all the edges which are not related
103  // to the current node.
104  for (auto& ei : remainingEdgesIndex) {
105  if (*(edgeIn + ei * stride) == node.index) {
106  nextVertex.insert({*(edgeOut + ei * stride), node.layer + 1});
107  } else {
108  nextEdges.push_back(ei);
109  }
110  }
111  remainingEdgesIndex.swap(nextEdges);
112 
113  // Of all the vertices which have node as incoming,
114  // check if there is any other incoming node.
115  std::set<TopoIndexInfo> hasPredecessors;
116  for (auto& ei : remainingEdgesIndex) {
117  for (auto& m : nextVertex) {
118  if (m.index == *(edgeOut + ei * stride)) {
119  hasPredecessors.insert({m.index, m.layer});
120  }
121  }
122  }
123  std::vector<TopoIndexInfo> withPredecessor;
124  std::set_difference(nextVertex.begin(), nextVertex.end(),
125  hasPredecessors.begin(), hasPredecessors.end(),
126  std::back_inserter(withPredecessor));
127  std::copy(withPredecessor.begin(), withPredecessor.end(), std::back_inserter(L));
128  }
129  return S;
130 }

132 void addMissingOutputsToReader(std::vector<OutputSpec> const& providedOutputs,
133  std::vector<InputSpec> requestedInputs,
134  DataProcessorSpec& publisher)
135 {
136  auto matchingOutputFor = [](InputSpec const& requested) {
137  return [&requested](OutputSpec const& provided) {
138  return DataSpecUtils::match(requested, provided);
139  };
140  };
141  for (InputSpec const& requested : requestedInputs) {
142  auto provided = std::find_if(providedOutputs.begin(),
143  providedOutputs.end(),
144  matchingOutputFor(requested));
145 
146  if (provided != providedOutputs.end()) {
147  continue;
148  }
149 
150  auto inList = std::find_if(publisher.outputs.begin(),
151  publisher.outputs.end(),
152  matchingOutputFor(requested));
153  if (inList != publisher.outputs.end()) {
154  continue;
155  }
156 
157  auto concrete = DataSpecUtils::asConcreteDataMatcher(requested);
158  publisher.outputs.emplace_back(OutputSpec{concrete.origin, concrete.description, concrete.subSpec});
159  }
160 }

174 void addMissingOutputsToBuilder(std::vector<InputSpec>&& requestedIDXs,
175  std::vector<InputSpec>& requestedAODs,
176  DataProcessorSpec& publisher)
177 {
178  auto inputSpecFromString = [](std::string s) {
179  std::regex word_regex("(\\w+)");
180  auto words = std::sregex_iterator(s.begin(), s.end(), word_regex);
181  if (std::distance(words, std::sregex_iterator()) != 3) {
182  throw runtime_error_f("Malformed spec: %s", s.c_str());
183  }
184  std::vector<std::string> data;
185  for (auto i = words; i != std::sregex_iterator(); ++i) {
186  data.emplace_back(i->str());
187  }
188  char origin[4];
189  char description[16];
190  std::memcpy(&origin, data[1].c_str(), 4);
191  std::memcpy(&description, data[2].c_str(), 16);
192  return InputSpec{data[0], header::DataOrigin{origin}, header::DataDescription{description}};
193  };
194 
195  for (auto& input : requestedIDXs) {
196  auto concrete = DataSpecUtils::asConcreteDataMatcher(input);
197  publisher.outputs.emplace_back(OutputSpec{concrete.origin, concrete.description, concrete.subSpec});
198  for (auto& i : input.metadata) {
199  auto spec = inputSpecFromString(i.defaultValue.get<std::string>());
200  auto j = std::find_if(publisher.inputs.begin(), publisher.inputs.end(), [&](auto x) { return x.binding == spec.binding; });
201  if (j == publisher.inputs.end()) {
202  publisher.inputs.push_back(spec);
203  }
204  requestedAODs.push_back(spec);
205  }
206  }

534  extraSpecs.push_back(CommonDataProcessors::getDummySink(ignored));
535  }
536 
537  workflow.insert(workflow.end(), extraSpecs.begin(), extraSpecs.end());
538  extraSpecs.clear();

541 void WorkflowHelpers::constructGraph(const WorkflowSpec& workflow,
542  std::vector<DeviceConnectionEdge>& logicalEdges,
543  std::vector<OutputSpec>& outputs,
544  std::vector<LogicalForwardInfo>& forwardedInputsInfo)
545 {
546  assert(!workflow.empty());
547 
548  // This is the state. Oif is the iterator I use for the searches.
549  std::list<LogicalOutputInfo> availableOutputsInfo;
550  auto const& constOutputs = outputs; // const version of the outputs
551  decltype(availableOutputsInfo.begin()) oif;
552  // Forwards is a local cache to avoid adding forwards before time.
553  std::vector<LogicalOutputInfo> forwards;
554 
555  // Notice that availableOutputsInfo MUST be updated first, since it relies on
556  // the size of outputs to be the one before the update.
557  auto enumerateAvailableOutputs = [&workflow, &outputs, &availableOutputsInfo]() {
558  for (size_t wi = 0; wi < workflow.size(); ++wi) {
559  auto& producer = workflow[wi];
560 
561  for (size_t oi = 0; oi < producer.outputs.size(); ++oi) {
562  auto& out = producer.outputs[oi];
563  auto uniqueOutputId = outputs.size();
564  availableOutputsInfo.emplace_back(LogicalOutputInfo{wi, uniqueOutputId, false});
565  outputs.push_back(out);
566  }
567  }
568  };
569 
570  // Notice that if the output is actually a forward, we need to store that
571  // information so that when we add it at device level we know which output
572  // channel we need to connect it too.
573  auto hasMatchingOutputFor = [&workflow, &constOutputs,
574  &availableOutputsInfo, &oif,
575  &forwardedInputsInfo](size_t ci, size_t ii) {
576  assert(ci < workflow.size());
577  assert(ii < workflow[ci].inputs.size());
578  auto& input = workflow[ci].inputs[ii];
579  auto matcher = [&input, &constOutputs](const LogicalOutputInfo& outputInfo) -> bool {
580  auto& output = constOutputs[outputInfo.outputGlobalIndex];
581  return DataSpecUtils::match(input, output);
582  };
583  oif = std::find_if(availableOutputsInfo.begin(),
584  availableOutputsInfo.end(),
585  matcher);
586  if (oif != availableOutputsInfo.end() && oif->forward) {
587  LogicalForwardInfo forward;
588  forward.consumer = ci;
589  forward.inputLocalIndex = ii;
590  forward.outputGlobalIndex = oif->outputGlobalIndex;
591  forwardedInputsInfo.emplace_back(LogicalForwardInfo{ci, ii, oif->outputGlobalIndex});
592  }
593  return oif != availableOutputsInfo.end();
594  };
595 
596  // We have consumed the input, therefore we remove it from the list.
597  // We will insert the forwarded inputs only at the end of the iteration.
598  auto findNextOutputFor = [&availableOutputsInfo, &constOutputs, &oif, &workflow](
599  size_t ci, size_t& ii) {
600  auto& input = workflow[ci].inputs[ii];
601  auto matcher = [&input, &constOutputs](const LogicalOutputInfo& outputInfo) -> bool {
602  auto& output = constOutputs[outputInfo.outputGlobalIndex];
603  return DataSpecUtils::match(input, output);
604  };
605  oif = availableOutputsInfo.erase(oif);
606  oif = std::find_if(oif, availableOutputsInfo.end(), matcher);
607  return oif;
608  };
609 
610  auto numberOfInputsFor = [&workflow](size_t ci) {
611  auto& consumer = workflow[ci];
612  return consumer.inputs.size();
613  };
614 
615  auto maxInputTimeslicesFor = [&workflow](size_t pi) {
616  auto& processor = workflow[pi];
617  return processor.maxInputTimeslices;
618  };
619 
620  // Trivial, but they make reading easier..
621  auto getOutputAssociatedProducer = [&oif]() {
622  return oif->specIndex;
623  };
624 
625  // Trivial, but they make reading easier..
626  auto getAssociateOutput = [&oif]() {
627  return oif->outputGlobalIndex;
628  };
629 
630  auto isForward = [&oif]() {
631  return oif->forward;
632  };
633 
634  // Trivial but makes reasing easier in the outer loop.
635  auto createEdge = [&logicalEdges](size_t producer,
636  size_t consumer,
637  size_t tpi,
638  size_t ptpi,
639  size_t uniqueOutputId,
640  size_t matchingInputInConsumer,
641  bool doForward) {
642  logicalEdges.emplace_back(
643  DeviceConnectionEdge{producer,
644  consumer,
645  tpi,
646  ptpi,
647  uniqueOutputId,
648  matchingInputInConsumer,
649  doForward});
650  };
651 
652  auto errorDueToMissingOutputFor = [&workflow, &constOutputs](size_t ci, size_t ii) {
653  auto input = workflow[ci].inputs[ii];
654  std::ostringstream str;
655  str << "No matching output found for "
656  << DataSpecUtils::describe(input) << ". Candidates:\n";
657 
658  for (auto& output : constOutputs) {
659  str << "-" << DataSpecUtils::describe(output) << "\n";
660  }
661 
662  throw std::runtime_error(str.str());
663  };
664 
665  // Whenever we have a set of forwards, we need to append it
666  // the the global list of outputs, so that they can be matched
667  // and we need to add a ForwardRoute for the current consumer
668  // because it is the one who will actually do the forwarding.
669  auto appendForwardsToPossibleOutputs = [&availableOutputsInfo, &forwards]() {
670  for (auto& forward : forwards) {
671  availableOutputsInfo.push_back(forward);
672  }
673  };
674 
675  // Given we create a forward every time we match and input and an
676  // output, having no forwards means we did not find any matching.
677  auto noMatchingOutputFound = [&forwards]() {
678  return forwards.empty();
679  };
680 
681  // Forwards is basically a cache to record
682  auto newEdgeBetweenDevices = [&forwards]() {
683  forwards.clear();
684  };
685 
686  auto forwardOutputFrom = [&forwards](size_t consumer, size_t uniqueOutputId) {
687  forwards.push_back(LogicalOutputInfo{consumer, uniqueOutputId, true});
688  };
689 
690  // This is the outer loop
691  //
692  // Here we iterate over dataprocessor items in workflow and we consider them
693  // as consumer, since we are interested in their inputs.
694  // Notice also we need to search for all the matching inputs, since
695  // we could have more than one source that matches (e.g. in the
696  // case of a time merger).
697  // Once consumed, an output is not actually used anymore, however
698  // we append it as a forward.
699  // Finally, If a device has n-way pipelining, we need to create one node per
700  // parallel pipeline and add an edge for each.
701  enumerateAvailableOutputs();
702 
703  for (size_t consumer = 0; consumer < workflow.size(); ++consumer) {
704  for (size_t input = 0; input < numberOfInputsFor(consumer); ++input) {
705  newEdgeBetweenDevices();
706 
707  while (hasMatchingOutputFor(consumer, input)) {
708  auto producer = getOutputAssociatedProducer();
709  auto uniqueOutputId = getAssociateOutput();
710  for (size_t tpi = 0; tpi < maxInputTimeslicesFor(consumer); ++tpi) {
711  for (size_t ptpi = 0; ptpi < maxInputTimeslicesFor(producer); ++ptpi) {
712  createEdge(producer, consumer, tpi, ptpi, uniqueOutputId, input, isForward());
713  }
714  forwardOutputFrom(consumer, uniqueOutputId);
715  }
716  findNextOutputFor(consumer, input);
717  }
718  if (noMatchingOutputFound()) {
719  errorDueToMissingOutputFor(consumer, input);
720  }
721  appendForwardsToPossibleOutputs();
722  }
723  }
724 }

767 void WorkflowHelpers::sortEdges(std::vector<size_t>& inEdgeIndex,
768  std::vector<size_t>& outEdgeIndex,
769  const std::vector<DeviceConnectionEdge>& edges)
770 {
771  inEdgeIndex.resize(edges.size());
772  outEdgeIndex.resize(edges.size());
773  std::iota(inEdgeIndex.begin(), inEdgeIndex.end(), 0);
774  std::iota(outEdgeIndex.begin(), outEdgeIndex.end(), 0);
775 
776  // Two indexes, one to bind the outputs, the other
777  // one to connect the inputs. The
778  auto outSorter = [&edges](size_t i, size_t j) {
779  auto& a = edges[i];
780  auto& b = edges[j];
781  return std::tie(a.producer, a.producerTimeIndex, a.timeIndex, a.consumer) < std::tie(b.producer, b.producerTimeIndex, b.timeIndex, b.consumer);
782  };
783  auto inSorter = [&edges](size_t i, size_t j) {
784  auto& a = edges[i];
785  auto& b = edges[j];
786  return std::tie(a.consumer, a.timeIndex, a.producer, a.producerTimeIndex) < std::tie(b.consumer, b.timeIndex, b.producer, b.producerTimeIndex);
787  };
788 
789  std::sort(inEdgeIndex.begin(), inEdgeIndex.end(), inSorter);
790  std::sort(outEdgeIndex.begin(), outEdgeIndex.end(), outSorter);
791 }

924 std::tuple<std::vector<InputSpec>, std::vector<unsigned char>> WorkflowHelpers::analyzeOutputs(WorkflowSpec const& workflow)
925 {
926  // compute total number of input/output
927  size_t totalInputs = 0;
928  size_t totalOutputs = 0;
929  for (auto& spec : workflow) {
930  totalInputs += spec.inputs.size();
931  totalOutputs += spec.outputs.size();
932  }
933 
934  std::vector<DataMatcherId> inputs;
935  std::vector<DataMatcherId> outputs;
936  inputs.reserve(totalInputs);
937  outputs.reserve(totalOutputs);
938 
939  std::vector<InputSpec> results;
940  std::vector<unsigned char> outputTypes;
941  results.reserve(totalOutputs);
942  outputTypes.reserve(totalOutputs);
943 
945  for (size_t wi = 0, we = workflow.size(); wi != we; ++wi) {
946  auto& spec = workflow[wi];
947  for (size_t ii = 0, ie = spec.inputs.size(); ii != ie; ++ii) {
948  inputs.emplace_back(DataMatcherId{wi, ii});
949  }
950  for (size_t oi = 0, oe = spec.outputs.size(); oi != oe; ++oi) {
951  outputs.emplace_back(DataMatcherId{wi, oi});
952  }
953  }
954 
955  for (size_t oi = 0, oe = outputs.size(); oi != oe; ++oi) {
956  auto& output = outputs[oi];
957  auto& outputSpec = workflow[output.workflowId].outputs[output.id];
958 
959  // compute output type
960  unsigned char outputType = UNKNOWN;
961 
962  // is AOD?
963  if (DataSpecUtils::partialMatch(outputSpec, header::DataOrigin("AOD"))) {
964  outputType |= ANALYSIS;
965  }
966  // is RN2?
967  if (DataSpecUtils::partialMatch(outputSpec, header::DataOrigin("RN2"))) {
968  outputType |= ANALYSIS;
969  }
970 
971  // is dangling output?
972  bool matched = false;
973  for (size_t ii = 0, ie = inputs.size(); ii != ie; ++ii) {
974  auto& input = inputs[ii];
975  // Inputs of the same workflow cannot match outputs
976  if (output.workflowId == input.workflowId) {
977  continue;
978  }
979  auto& inputSpec = workflow[input.workflowId].inputs[input.id];
980  if (DataSpecUtils::match(inputSpec, outputSpec)) {
981  matched = true;
982  break;
983  }
984  }
985  if (!matched) {
986  outputType |= DANGLING;
987  }
988 
989  // update results and outputTypes
990  auto input = DataSpecUtils::matchingInput(outputSpec);
991  char buf[64];
992  input.binding = (snprintf(buf, 63, "output_%zu_%zu", output.workflowId, output.id), buf);
993 
994  // make sure that entries are unique
995  if (std::find(results.begin(), results.end(), input) == results.end()) {
996  results.emplace_back(input);
997  outputTypes.emplace_back(outputType);
998  }
999  }
1000 
1001  // make sure that results is unique
1002 
1003  return std::make_tuple(results, outputTypes);
1004 }
