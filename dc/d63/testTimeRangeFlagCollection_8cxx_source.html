24 BOOST_AUTO_TEST_CASE(test_TimeRangeFlagCollection)
25 {

28 
29  TimeRangeFlagCollection trfc1{"Raw data checks", "TOF"};
30  trfc1.insert(trf1); // by copy
31  trfc1.insert(trf2);
32  trfc1.insert({50, 77, FlagReasonFactory::Invalid()}); // by move
33  BOOST_CHECK_EQUAL(trfc1.size(), 3);
34 
35  TimeRangeFlagCollection trfc2{"Reco checks", "TOF"};
36  trfc2.insert({50, 77, FlagReasonFactory::Invalid()}); // this is a duplicate to an entry in trfc1
37  trfc2.insert({51, 77, FlagReasonFactory::Invalid()});
38  trfc2.insert({1234, 3434, FlagReasonFactory::LimitedAcceptance()});
39  trfc2.insert({50, 77, FlagReasonFactory::LimitedAcceptance()});
40  BOOST_CHECK_EQUAL(trfc2.size(), 4);
41 
42  // Try merging. Duplicate entries should be left in the 'other' objects.
43  // Notice that we merge the two partial TRFCs into the third, which covers all cases
44  TimeRangeFlagCollection trfc3{"ALL", "TOF"};
45  trfc3.merge(trfc1);
46  trfc3.merge(trfc2);
47  BOOST_CHECK_EQUAL(trfc1.size(), 0);
48  BOOST_CHECK_EQUAL(trfc2.size(), 1);
49  BOOST_CHECK_EQUAL(trfc3.size(), 6);
50 
51  // Try const merging. It should copy the elements and keep the 'other' intact.
52  TimeRangeFlagCollection trfc4{"ALL", "TOF"};

54  trfc4.merge(constTrfc3);
55  BOOST_CHECK_EQUAL(trfc3.size(), 6);

58  // Try merging different detectors - it should throw.
