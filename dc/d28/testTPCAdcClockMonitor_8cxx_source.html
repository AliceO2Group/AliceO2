82  }
83 
84  // feed monitor with clock sequence, replace beforehand one sequence of first pattern with something else
85  unsigned short oldSeq;
86  for (int pos = 10; pos < 18; ++pos) {
87  mon.reset();
88  oldSeq = clockSequence[pos];
89  clockSequence[pos] = 0xA;
90  int seq;
91  for (std::vector<unsigned short>::iterator it = clockSequence.begin(); it != clockSequence.end(); ++it) {
92  seq = std::distance(clockSequence.begin(), it);
93  // only the 26th sequence completes the full cycle
94  // 10 times 000 in the beginning
95  // + 8 sequences with with wrong part
96  // + 8 sequences with correct parts to recognize full pattern
97 
98  // one special case for phase = 0 and position of wrong part = 17 (last needed part for full pattern)
99  // phase 0 means the following pattern 0000 FFFF 0000 FFFF ...
100  // ^
101  // |
102  // part 17 wrong means this one wrong (not 0) -------------|
103  //
104  // The finding algorithm neads two clean transitions from 0 to F to recognize
105  // a clock with phase 0. If part 17 is wrong, we won't have this clean transition
106  // at this position, therefore the the clock will be recognized one complete
107  // pattern later. For every other wrong part, we still habe here the first transition
108  // from 0 to F and for every other phase, we don't need two sequences for the
109  // transition just the one with the transition
110  int expGoodSequence = (phase == 0 && pos == 17) ? (18 + 8 + 8) : (18 + 8);
111 
112  BOOST_CHECK_EQUAL(mon.addSequence(*it), (seq >= expGoodSequence ? 0 : 1));
113  }
114  clockSequence[pos] = oldSeq;
115  }
116  }
