96 void PedestalsCalculationTask::endOfStream(framework::EndOfStreamContext& ec)
97 {
98  double Average;
99  double Variance;
100  double Samples;
101  double SumOfCharge;
102  double SumOfSquares;
103 
104  uint32_t Buffer;
105  uint32_t Pedestal;
106  uint32_t Threshold;
107  char padsFileName[1024];
108 
109  for (int e = 0; e < Geo::MAXEQUIPMENTS; e++) {
110  if (mDeco->getAverageEventSize(e) == 0) {
111  continue;
112  }
113  sprintf(padsFileName, "%s_%d.dat", mPedestalsBasePath.c_str(), e);
114  FILE* fpads = fopen(padsFileName, "w");
115  // TODO: Add controls on the file open
116  for (int c = 0; c < Geo::N_COLUMNS; c++) {
117  for (int d = 0; d < Geo::N_DILOGICS; d++) {
118  for (int h = 0; h < Geo::N_CHANNELS; h++) {
119  Samples = (double)mDeco->getChannelSamples(e, c, d, h);
120  SumOfCharge = mDeco->getChannelSum(e, c, d, h);
121  SumOfSquares = mDeco->getChannelSquare(e, c, d, h);
122 
123  if (Samples > 0) {
124  Average = SumOfCharge / Samples;
125  Variance = sqrt(abs((Samples * SumOfSquares) - (SumOfCharge * SumOfCharge))) / Samples;
126  } else {
127  Average = 0;
128  Variance = 0;
129  }
130  Pedestal = (uint32_t)Average;
131  Threshold = (uint32_t)(Variance * mSigmaCut + Average);
132  Buffer = ((Threshold & 0x001FF) << 9) | (Pedestal & 0x001FF);
133  fprintf(fpads, "%05X\n", Buffer);
134  }
135  for (int h = 48; h < 64; h++) {
136  fprintf(fpads, "%05X\n", 0);
137  }
138  }
139  }
140  mExTimer.logMes("End write the equipment = " + std::to_string(e));
141  fprintf(fpads, "%05X\n", 0xA0A0A);
142  fclose(fpads);
143  }
144  mExTimer.logMes("End Writing the pedestals ! Digits decoded = " + std::to_string(mTotalDigits) + " Frames received = " + std::to_string(mTotalFrames));
145 
146  if (mWriteToDB) {
147  recordPedInCcdb();
148  }
149  mExTimer.stop();
150  return;
151 }

153 void PedestalsCalculationTask::recordPedInCcdb()
154 {
155  // create the root structure
156 
157  LOG(INFO) << "Store Pedestals in ccdb ";
158 
159  float xb, yb, ch;
160  double Samples, SumOfCharge, SumOfSquares, Average, Variance;
161 
162  TObjArray aSigmas(Geo::N_MODULES);
163  TObjArray aPedestals(Geo::N_MODULES);
164 
165  for (int i = 0; i < Geo::N_MODULES; i++) {
166  aSigmas.AddAt(new TMatrixF(160, 144), i);
167  aPedestals.AddAt(new TMatrixF(160, 144), i);
168  }
169 
170  for (int m = 0; m < o2::hmpid::Geo::N_MODULES; m++) {
171  if (mDeco->getAverageEventSize(m * 2) == 0 && mDeco->getAverageEventSize(m * 2 + 1) == 0) {
172  continue; // If no events skip the chamber
173  }
174  TMatrixF* pS = (TMatrixF*)aSigmas.At(m);
175  TMatrixF* pP = (TMatrixF*)aPedestals.At(m);
176 
177  for (int x = 0; x < o2::hmpid::Geo::N_XROWS; x++) {
178  for (int y = 0; y < o2::hmpid::Geo::N_YCOLS; y++) {
179 
180  Samples = (double)mDeco->getPadSamples(m, x, y);
181  SumOfCharge = mDeco->getPadSum(m, x, y);
182  SumOfSquares = mDeco->getPadSquares(m, x, y);
183  if (Samples > 0) {
184  (*pP)(x, y) = SumOfCharge / Samples;
185  (*pS)(x, y) = sqrt(abs((Samples * SumOfSquares) - (SumOfCharge * SumOfCharge))) / Samples;
186  } else {
187  (*pP)(x, y) = 0;
188  (*pS)(x, y) = 0;
189  }
190  }
191  }
192  }
193  struct timeval tp;
194  gettimeofday(&tp, nullptr);
195  uint64_t ms = tp.tv_sec * 1000 + tp.tv_usec / 1000;
196 
197  uint64_t minTimeStamp = ms;
198  uint64_t maxTimeStamp = ms + 1;
199 
200  for (int i = 0; i < Geo::N_MODULES; i++) {
201  if (mDeco->getAverageEventSize(i * 2) == 0 && mDeco->getAverageEventSize(i * 2 + 1) == 0) {
202  continue; // If no events skip the chamber
203  }
204  TString filename = TString::Format("HMP/Pedestals/%s/Mean_%d", mPedestalTag.c_str(), i);
205  mDbMetadata.emplace("Tag", mPedestalTag.c_str());
206  mDBapi.storeAsTFileAny(aPedestals.At(i), filename.Data(), mDbMetadata, minTimeStamp, maxTimeStamp);
207  }
208  for (int i = 0; i < Geo::N_MODULES; i++) {
209  if (mDeco->getAverageEventSize(i * 2) == 0 && mDeco->getAverageEventSize(i * 2 + 1) == 0) {
210  continue; // If no events skip the chamber
211  }
212  TString filename = TString::Format("HMP/Pedestals/%s/Sigma_%d", mPedestalTag.c_str(), i);
213  mDbMetadata.emplace("Tag", mPedestalTag.c_str());
214  mDBapi.storeAsTFileAny(aSigmas.At(i), filename.Data(), mDbMetadata, minTimeStamp, maxTimeStamp);
215  }
216  return;
217 }

222 {
223  LOG(DEBUG) << "*********** In decodeTF **************";
224 
225  // get the input buffer
226  auto& inputs = pc.inputs();
227  DPLRawParser parser(inputs, o2::framework::select("TF:HMP/RAWDATA"));
228  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
229  uint32_t* theBuffer = (uint32_t*)it.raw();
230  mDeco->setUpStream(theBuffer, it.size() + it.offset());
231  mDeco->decodePageFast(&theBuffer);
232  mTotalFrames++;
233  }
234  return;

257 void PedestalsCalculationTask::decodeRawFile(framework::ProcessingContext& pc)
258 {
259  LOG(INFO) << "*********** In decode rawfile **************";
260 
261  for (auto&& input : pc.inputs()) {
262  if (input.spec->binding == "file") {
263  const header::DataHeader* header = o2::header::get<header::DataHeader*>(input.header);
264  if (!header) {
265  return;
266  }
267 
268  auto const* raw = input.payload;
269  size_t payloadSize = header->payloadSize;
270 
271  LOG(INFO) << " payloadSize=" << payloadSize;
272  if (payloadSize == 0) {
273  return;
274  }
275 
276  uint32_t* theBuffer = (uint32_t*)input.payload;
277  int pagesize = header->payloadSize;
278  mDeco->setUpStream(theBuffer, pagesize);
279  mDeco->decodePageFast(&theBuffer);
280  }
281  }
282  return;
