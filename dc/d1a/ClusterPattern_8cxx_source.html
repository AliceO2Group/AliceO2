104 int ClusterPattern::getCOG(int rowSpan, int colSpan, const unsigned char patt[MaxPatternBytes], float& xCOG, float& zCOG)
105 {
106  int tempxCOG = 0, tempzCOG = 0, tempFiredPixels = 0, ic = 0, ir = 0;
107  int nBits = rowSpan * colSpan;
108  int nBytes = nBits / 8;
109  if (nBits % 8 != 0) {
110  nBytes++;
111  }
112  for (unsigned int i = 0; i < nBytes; i++) {
113  unsigned char tempChar = patt[i];
114  int s = 128; // 0b10000000
115  while (s > 0) {
116  if ((tempChar & s) != 0) {
117  tempFiredPixels++;
118  tempxCOG += ir;
119  tempzCOG += ic;
120  }
121  ic++;
122  s /= 2;
123  if ((ir + 1) * ic == nBits) {
124  break;
125  }
126  if (ic == colSpan) {
127  ic = 0;
128  ir++;
129  }
130  }
131  if ((ir + 1) * ic == nBits) {
132  break;
133  }
134  }
135  xCOG = float(tempxCOG) / tempFiredPixels;
136  zCOG = float(tempzCOG) / tempFiredPixels;
137 
138  return tempFiredPixels;
139 }
