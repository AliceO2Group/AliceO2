48  aod::JetConstituentsSub const& constituentsSub)
49  {
50  registry.fill(HIST("hJetPt"), jet.pt());
51  if (b_DoConstSub) {
52  registry.fill(HIST("hNJetConstituents"), constituentsSub.size());
53  for (const auto constituent : constituentsSub) {
54  registry.fill(HIST("hConstituentPt"), constituent.pt());
55  }
56  } else {
57  registry.fill(HIST("hNJetConstituents"), constituents.size());
58  for (const auto constituentIndex : constituents) {

60  registry.fill(HIST("hConstituentPt"), constituent.pt());
61  }
62  }
63  }
64 };
