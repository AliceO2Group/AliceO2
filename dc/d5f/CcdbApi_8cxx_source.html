135 void CcdbApi::storeAsBinaryFile(const char* buffer, size_t size, const std::string& filename, const std::string& objectType,
136  const std::string& path, const std::map<std::string, std::string>& metadata,
137  long startValidityTimestamp, long endValidityTimestamp) const
138 {
139  // Store a binary file
140 
141  // Prepare URL
142  long sanitizedStartValidityTimestamp = startValidityTimestamp;
143  if (startValidityTimestamp == -1) {
144  cout << "Start of Validity not set, current timestamp used." << endl;
145  sanitizedStartValidityTimestamp = getCurrentTimestamp();
146  }
147  long sanitizedEndValidityTimestamp = endValidityTimestamp;
148  if (endValidityTimestamp == -1) {
149  cout << "End of Validity not set, start of validity plus 1 year used." << endl;
150  sanitizedEndValidityTimestamp = getFutureTimestamp(60 * 60 * 24 * 365);
151  }
152 
153  // Curl preparation
154  CURL* curl = nullptr;
155  struct curl_httppost* formpost = nullptr;
156  struct curl_httppost* lastptr = nullptr;
157  struct curl_slist* headerlist = nullptr;
158  static const char buf[] = "Expect:";
159  curl_formadd(&formpost,
160  &lastptr,
161  CURLFORM_COPYNAME, "send",
162  CURLFORM_BUFFER, filename.c_str(),
163  CURLFORM_BUFFERPTR, buffer, //.Buffer(),
164  CURLFORM_BUFFERLENGTH, size,
165  CURLFORM_END);
166 
167  curl = curl_easy_init();
168  headerlist = curl_slist_append(headerlist, buf);
169  if (curl != nullptr) {
170  string fullUrl = getFullUrlForStorage(curl, path, objectType, metadata, sanitizedStartValidityTimestamp, sanitizedEndValidityTimestamp);
171  LOG(debug3) << "Full URL Encoded: " << fullUrl;
172  /* what URL that receives this POST */
173  curl_easy_setopt(curl, CURLOPT_URL, fullUrl.c_str());
174  curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headerlist);
175  curl_easy_setopt(curl, CURLOPT_HTTPPOST, formpost);
176 
177  /* Perform the request, res will get the return code */
178  CURLcode res = curl_easy_perform(curl);
179  /* Check for errors */
180  if (res != CURLE_OK) {
181  fprintf(stderr, "curl_easy_perform() failed: %s\n",
182  curl_easy_strerror(res));
183  }
184 
185  /* always cleanup */
186  curl_easy_cleanup(curl);
187 
188  /* then cleanup the formpost chain */
189  curl_formfree(formpost);
190  /* free slist */
191  curl_slist_free_all(headerlist);
192  } else {
193  cerr << "curl initialization failure" << endl;
194  }
195 }

206 string CcdbApi::getFullUrlForStorage(CURL* curl, const string& path, const string& objtype,
207  const map<string, string>& metadata,
208  long startValidityTimestamp, long endValidityTimestamp) const
209 {
210  // Prepare timestamps
211  string startValidityString = getTimestampString(startValidityTimestamp < 0 ? getCurrentTimestamp() : startValidityTimestamp);
212  string endValidityString = getTimestampString(endValidityTimestamp < 0 ? getFutureTimestamp(60 * 60 * 24 * 365) : endValidityTimestamp);
213  // Build URL
214  string fullUrl = mUrl + "/" + path + "/" + startValidityString + "/" + endValidityString + "/";
215  // Add type as part of metadata
216  // we need to URL encode the object type, since in case it has special characters (like the "<", ">" for templated classes) it won't work otherwise
217  char* objtypeEncoded = curl_easy_escape(curl, objtype.c_str(), objtype.size());
218  fullUrl += "ObjectType=" + string(objtypeEncoded) + "/";
219  curl_free(objtypeEncoded);
220  // Add general metadata
221  for (auto& kv : metadata) {
222  string mfirst = kv.first;
223  string msecond = kv.second;
224  // same trick for the metadata as for the object type
225  char* mfirstEncoded = curl_easy_escape(curl, mfirst.c_str(), mfirst.size());
226  char* msecondEncoded = curl_easy_escape(curl, msecond.c_str(), msecond.size());
227  fullUrl += string(mfirstEncoded) + "=" + string(msecondEncoded) + "/";
228  curl_free(mfirstEncoded);
229  curl_free(msecondEncoded);
230  }
231  return fullUrl;
232 }

277 static size_t WriteMemoryCallback(void* contents, size_t size, size_t nmemb, void* userp)
278 {
279  size_t realsize = size * nmemb;
280  auto* mem = (struct MemoryStruct*)userp;
281 
282  mem->memory = (char*)realloc(mem->memory, mem->size + realsize + 1);
283  if (mem->memory == nullptr) {
284  printf("not enough memory (realloc returned NULL)\n");
285  return 0;
286  }
287 
288  memcpy(&(mem->memory[mem->size]), contents, realsize);
289  mem->size += realsize;
290  mem->memory[mem->size] = 0;
291 
292  return realsize;
293 }

310 TObject* CcdbApi::retrieve(std::string const& path, std::map<std::string, std::string> const& metadata,
311  long timestamp) const
312 {
313  // Note : based on https://curl.haxx.se/libcurl/c/getinmemory.html
314  // Thus it does not comply to our coding guidelines as it is a copy paste.
315 
316  // Prepare CURL
317  CURL* curl_handle;
318  CURLcode res;
319  struct MemoryStruct chunk {
320  (char*)malloc(1) /*memory*/, 0 /*size*/
321  };
322  TObject* result = nullptr;
323 
324  /* init the curl session */
325  curl_handle = curl_easy_init();
326 
327  string fullUrl = getFullUrlForRetrieval(curl_handle, path, metadata, timestamp);
328 
329  /* specify URL to get */
330  curl_easy_setopt(curl_handle, CURLOPT_URL, fullUrl.c_str());
331 
332  /* send all data to this function */
333  curl_easy_setopt(curl_handle, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
334 
335  /* we pass our 'chunk' struct to the callback function */
336  curl_easy_setopt(curl_handle, CURLOPT_WRITEDATA, (void*)&chunk);
337 
338  /* some servers don't like requests that are made without a user-agent
339  field, so we provide one */
340  curl_easy_setopt(curl_handle, CURLOPT_USERAGENT, "libcurl-agent/1.0");
341 
342  /* if redirected , we tell libcurl to follow redirection */
343  curl_easy_setopt(curl_handle, CURLOPT_FOLLOWLOCATION, 1L);
344 
345  /* get it! */
346  res = curl_easy_perform(curl_handle);
347 
348  /* check for errors */
349  if (res != CURLE_OK) {
350  fprintf(stderr, "curl_easy_perform() failed: %s\n",
351  curl_easy_strerror(res));
352  } else {
353  /*
354  * Now, our chunk.memory points to a memory block that is chunk.size
355  * bytes big and contains the remote file.
356  */
357 
358  // printf("%lu bytes retrieved\n", (long) chunk.size);
359 
360  long response_code;
361  res = curl_easy_getinfo(curl_handle, CURLINFO_RESPONSE_CODE, &response_code);
362  if ((res == CURLE_OK) && (response_code != 404)) {
363  std::lock_guard<std::mutex> guard(gIOMutex);
364  TMessage mess(kMESS_OBJECT);
365  mess.SetBuffer(chunk.memory, chunk.size, kFALSE);
366  mess.SetReadMode();
367  mess.Reset();
368  result = (TObject*)(mess.ReadObjectAny(mess.GetClass()));
369  if (result == nullptr) {
370  cerr << "couldn't retrieve the object " << path << endl;
371  }
372  } else {
373  cerr << "invalid URL : " << fullUrl << endl;
374  }
375 
376  // Print data
377  // cout << "size : " << chunk.size << endl;
378  // cout << "data : " << endl;
379  // char* mem = (char*)chunk.memory;
380  // for (int i = 0 ; i < chunk.size/4 ; i++) {
381  // cout << mem;
382  // mem += 4;
383  // }
384  }
385 
386  /* cleanup curl stuff */
387  curl_easy_cleanup(curl_handle);
388 
389  free(chunk.memory);
390 
391  return result;

404 template <typename MapType = std::map<std::string, std::string>>
405 size_t header_map_callback(char* buffer, size_t size, size_t nitems, void* userdata)
406 {
407  auto* headers = static_cast<MapType*>(userdata);
408  auto header = std::string(buffer, size * nitems);
409  std::string::size_type index = header.find(':', 0);
410  if (index != std::string::npos) {
411  const auto key = boost::algorithm::trim_copy(header.substr(0, index));
412  const auto value = boost::algorithm::trim_copy(header.substr(index + 1));
413  headers->insert(std::make_pair(key, value));
414  }
415  return size * nitems;
416 }

419 TObject* CcdbApi::retrieveFromTFile(std::string const& path, std::map<std::string, std::string> const& metadata,
420  long timestamp, std::map<std::string, std::string>* headers, std::string const& etag,
421  const std::string& createdNotAfter, const std::string& createdNotBefore) const
422 {
423  // Note : based on https://curl.haxx.se/libcurl/c/getinmemory.html
424  // Thus it does not comply to our coding guidelines as it is a copy paste.
425 
426  // std::map<std::string, std::string> headers2;

428  // Prepare CURL
429  CURL* curl_handle;
430  CURLcode res;
431  struct MemoryStruct chunk {
432  (char*)malloc(1) /*memory*/, 0 /*size*/
433  };
434 
435  /* init the curl session */
436  curl_handle = curl_easy_init();
437 
438  string fullUrl = getFullUrlForRetrieval(curl_handle, path, metadata, timestamp);
439 
440  /* specify URL to get */
441  curl_easy_setopt(curl_handle, CURLOPT_URL, fullUrl.c_str());
442 
443  /* send all data to this function */
444  curl_easy_setopt(curl_handle, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
445 
446  /* we pass our 'chunk' struct to the callback function */
447  curl_easy_setopt(curl_handle, CURLOPT_WRITEDATA, (void*)&chunk);
448 
449  /* some servers don't like requests that are made without a user-agent
450  field, so we provide one */
451  curl_easy_setopt(curl_handle, CURLOPT_USERAGENT, "libcurl-agent/1.0");
452 
453  /* if redirected , we tell libcurl to follow redirection */
454  curl_easy_setopt(curl_handle, CURLOPT_FOLLOWLOCATION, 1L);
455 
456  struct curl_slist* list = nullptr;
457  if (!etag.empty()) {
458  list = curl_slist_append(list, ("If-None-Match: " + etag).c_str());
459  }
460 
461  if (!createdNotAfter.empty()) {
462  list = curl_slist_append(list, ("If-Not-After: " + createdNotAfter).c_str());
463  }
464 
465  if (!createdNotBefore.empty()) {
466  list = curl_slist_append(list, ("If-Not-Before: " + createdNotBefore).c_str());
467  }
468 
469  // setup curl for headers handling
470  if (headers != nullptr) {
471  list = curl_slist_append(list, ("If-None-Match: " + to_string(timestamp)).c_str());
472  curl_easy_setopt(curl_handle, CURLOPT_HEADERFUNCTION, header_map_callback<>);
473  curl_easy_setopt(curl_handle, CURLOPT_HEADERDATA, headers);
474  }
475 
476  if (list) {
477  curl_easy_setopt(curl_handle, CURLOPT_HTTPHEADER, list);
478  }
479 
480  /* get it! */
481  res = curl_easy_perform(curl_handle);
482  std::string errStr;
483  TObject* result = nullptr;
484  if (res == CURLE_OK) {
485  long response_code;
486  res = curl_easy_getinfo(curl_handle, CURLINFO_RESPONSE_CODE, &response_code);
487  if ((res == CURLE_OK) && (response_code != 404)) {
488  Int_t previousErrorLevel = gErrorIgnoreLevel;
489  gErrorIgnoreLevel = kFatal;
490  std::lock_guard<std::mutex> guard(gIOMutex);
491  TMemFile memFile("name", chunk.memory, chunk.size, "READ");
492  gErrorIgnoreLevel = previousErrorLevel;
493  if (!memFile.IsZombie()) {
494  result = (TObject*)extractFromTFile(memFile, TClass::GetClass("TObject"));
495  if (result == nullptr) {
496  errStr = o2::utils::concat_string("Couldn't retrieve the object ", path);
497  LOG(ERROR) << errStr;
498  }
499  memFile.Close();
500  } else {
501  LOG(DEBUG) << "Object " << path << " is stored in a TMemFile";
502  }
503  } else {
504  errStr = o2::utils::concat_string("Invalid URL : ", fullUrl);
505  LOG(ERROR) << errStr;
506  }
507  } else {
508  errStr = o2::utils::concat_string("curl_easy_perform() failed: ", curl_easy_strerror(res));
509  fprintf(stderr, "%s", errStr.c_str());
510  }
511 
512  if (!errStr.empty() && headers) {
513  (*headers)["Error"] = errStr;
514  }
515 
516  curl_easy_cleanup(curl_handle);
517  free(chunk.memory);
518  return result;
519 }

612 void* CcdbApi::extractFromTFile(TFile& file, TClass const* cl)
613 {
614  if (!cl) {
615  return nullptr;
616  }
617  auto object = file.GetObjectChecked(CCDBOBJECT_ENTRY, cl);
618  if (!object) {
619  // it could be that object was stored with previous convention
620  // where the classname was taken as key
621  std::string objectName(cl->GetName());
622  utils::trim(objectName);
623  object = file.GetObjectChecked(objectName.c_str(), cl);
624  LOG(WARN) << "Did not find object under expected name " << CCDBOBJECT_ENTRY;
625  if (!object) {
626  return nullptr;
627  }
628  LOG(WARN) << "Found object under deprecated name " << cl->GetName();
629  }
630  auto result = object;
631  // We need to handle some specific cases as ROOT ties them deeply
632  // to the file they are contained in
633  if (cl->InheritsFrom("TObject")) {
634  // make a clone
635  auto obj = ((TObject*)object)->Clone();
636  // detach from the file

718 // navigate sequence of URLs until TFile content is found; object is extracted and returned
719 void* CcdbApi::navigateURLsAndRetrieveContent(CURL* curl_handle, std::string const& url, std::type_info const& tinfo, std::map<string, string>* headers) const
720 {
721  // a global internal data structure that can be filled with HTTP header information
722  // static --> to avoid frequent alloc/dealloc as optimization
723  // not sure if thread_local takes away that benefit
724  static thread_local std::multimap<std::string, std::string> headerData;
725 
726  // let's see first of all if the url is something specific that curl cannot handle
727  if (url.find("alien:/", 0) != std::string::npos) {
728  return downloadAlienContent(url, tinfo);
729  }
730  // add other final cases here
731  // example root://
732 
733  // otherwise make an HTTP/CURL request
734  // specify URL to get
735  curl_easy_setopt(curl_handle, CURLOPT_URL, url.c_str());
736  // some servers don't like requests that are made without a user-agent
737  // field, so we provide one
738  curl_easy_setopt(curl_handle, CURLOPT_USERAGENT, "libcurl-agent/1.0");
739  // if redirected , we tell libcurl NOT to follow redirection
740  curl_easy_setopt(curl_handle, CURLOPT_FOLLOWLOCATION, 0L);
741  curl_easy_setopt(curl_handle, CURLOPT_HEADERFUNCTION, header_map_callback<decltype(headerData)>);
742  headerData.clear();
743  curl_easy_setopt(curl_handle, CURLOPT_HEADERDATA, (void*)&headerData);
744 
745  MemoryStruct chunk{(char*)malloc(1), 0};
746 
747  // send all data to this function
748  curl_easy_setopt(curl_handle, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
749  curl_easy_setopt(curl_handle, CURLOPT_WRITEDATA, (void*)&chunk);
750 
751  auto res = curl_easy_perform(curl_handle);
752  long response_code = -1;
753  void* content = nullptr;
754  bool errorflag = false;
755  bool cachingflag = false;
756  if (res == CURLE_OK && curl_easy_getinfo(curl_handle, CURLINFO_RESPONSE_CODE, &response_code) == CURLE_OK) {
757  if (headers) {
758  for (auto& p : headerData) {
759  (*headers)[p.first] = p.second;
760  }
761  }
762  if (200 <= response_code && response_code < 300) {
763  // good response and the content is directly provided and should have been dumped into "chunk"
764  content = interpretAsTMemFileAndExtract(chunk.memory, chunk.size, tinfo);
765  } else if (response_code == 304) {
766  // this means the object exist but I am not serving
767  // it since it's already in your possession
768 
769  // there is nothing to be done here
770  cachingflag = true;
771  }
772  // this is a more general redirection
773  else if (300 <= response_code && response_code < 400) {
774  // we try content locations in order of appearance until one succeeds
775  // 1st: The "Location" field
776  // 2nd: Possible "Content-Location" fields - Location field
777 
778  // some locations are relative to the main server so we need to fix/complement them
779  auto complement_Location = [this](std::string const& loc) {
780  if (loc[0] == '/') {
781  // if it's just a path (noticed by trailing '/' we prepend the server url
782  return getURL() + loc;
783  }
784  return loc;
785  };
786 
787  std::vector<std::string> locs;
788  auto iter = headerData.find("Location");
789  if (iter != headerData.end()) {
790  locs.push_back(complement_Location(iter->second));
791  }
792  // add alternative locations (not yet included)
793  auto iter2 = headerData.find("Content-Location");
794  if (iter2 != headerData.end()) {
795  auto range = headerData.equal_range("Content-Location");
796  for (auto it = range.first; it != range.second; ++it) {
797  if (std::find(locs.begin(), locs.end(), it->second) == locs.end()) {
798  locs.push_back(complement_Location(it->second));
799  }
800  }
801  }
802  for (auto& l : locs) {
803  if (l.size() > 0) {
804  LOG(DEBUG) << "Trying content location " << l;
805  content = navigateURLsAndRetrieveContent(curl_handle, l, tinfo, nullptr);
806  if (content /* or other success marker in future */) {
807  break;
808  }
809  }
810  }
811  } else if (response_code == 404) {
812  LOG(ERROR) << "Requested resource does not exist: " << url;
813  errorflag = true;
814  } else {
815  errorflag = true;
816  }
817  // cleanup
818  if (chunk.memory != nullptr) {
819  free(chunk.memory);
820  }
821  } else {
822  LOG(ERROR) << "Curl request to " << url << " failed ";
823  errorflag = true;
824  }
825  // indicate that an error occurred ---> used by caching layers (such as CCDBManager)
826  if (errorflag && headers) {
827  (*headers)["Error"] = "An error occurred during retrieval";
828  }
829  return content;
830 }
