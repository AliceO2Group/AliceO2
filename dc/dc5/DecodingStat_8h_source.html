27 struct ChipStat {
28 
29  enum DecErrors : int {
30  BusyViolation,
31  DataOverrun,
32  Fatal,
33  BusyOn,
34  BusyOff,
35  TruncatedChipEmpty, // Data was truncated after ChipEmpty
36  TruncatedChipHeader, // Data was truncated after ChipHeader
37  TruncatedRegion, // Data was truncated after Region record
38  TruncatedLondData, // Data was truncated in the LongData record
39  WrongDataLongPattern, // LongData pattern has highest bit set
40  NoDataFound, // Region is not followed by Short or Long data
41  UnknownWord, // Unknow word was seen
42  NErrorsDefined
43  };
44 
45  static constexpr std::array<std::string_view, NErrorsDefined> ErrNames = {
46  "BusyViolation flag ON", // BusyViolation
47  "DataOverrun flag ON", // DataOverrun
48  "Fatal flag ON", // Fatal
49  "BusyON", // BusyOn
50  "BusyOFF", // BusyOff
51  "Data truncated after ChipEmpty", // TruncatedChipEmpty
52  "Data truncated after ChipHeader", // TruncatedChipHeader
53  "Data truncated after Region", // TruncatedRegion
54  "Data truncated after LongData", // TruncatedLondData
55  "LongData pattern has highest bit set", // WrongDataLongPattern
56  "Region is not followed by Short or Long data", // NoDataFound
57  "Unknow word" // UnknownWord
58  };
59 
60  uint16_t chipID = 0;
61  size_t nHits = 0;
62  std::array<uint32_t, NErrorsDefined> errorCounts = {};
63  ChipStat() = default;
64  ChipStat(uint16_t id) : chipID(id) {}
65 
66  void clear()
67  {
68  memset(errorCounts.data(), 0, sizeof(uint32_t) * errorCounts.size());
69  nHits = 0;
70  }
71 
72  uint32_t getNErrors() const;
73 
74  void addErrors(uint32_t mask)
75  {
76  if (mask) {
77  for (int i = NErrorsDefined; i--;) {
78  if (mask & (0x1 << i)) {
79  errorCounts[i]++;
80  }
81  }
82  }
83  }
84 
85  void print(bool skipEmpty = true) const;
86 
87  ClassDefNV(ChipStat, 1);
88 };

102  ErrMissingGBTTrailer, // GBT payload trailer was expected but not found

104  ErrUnstoppedLanes, // end of FEE data reached while not all lanes received stop
105  ErrDataForStoppedLane, // data was received for stopped lane
106  ErrNoDataForActiveLane, // no data was seen for lane (which was not in timeout)
107  ErrIBChipLaneMismatch, // chipID (on module) was different from the lane ID on the IB stave
108  ErrCableDataHeadWrong, // cable data does not start with chip header or empty chip
109  ErrInvalidActiveLanes, // active lanes pattern conflicts with expected for given RU type
110  ErrPacketCounterJump, // jump in RDH.packetCounter
111  ErrPacketDoneMissing, // packet done is missing in the trailer while CRU page is not over
112  ErrMissingDiagnosticWord, // missing diagnostic word after RDH with stop
113  ErrGBTWordNotRecognized, // GBT word not recognized
114  NErrorsDefined
115  };
116  static constexpr std::array<std::string_view, NErrorsDefined> ErrNames = {

123  "GBT payload header was expected but not found", // ErrMissingGBTHeader

138  uint32_t ruLinkID = 0; // Link ID within RU
139 
140  // Note: packet here is meant as a group of CRU pages belonging to the same trigger
141  uint32_t nPackets = 0; // total number of packets (RDH pages)
142  uint32_t nTriggers = 0; // total number of triggers (ROFs)
143  std::array<uint32_t, NErrorsDefined> errorCounts = {}; // error counters
144  std::array<uint32_t, GBTDataTrailer::MaxStateCombinations> packetStates = {}; // packet status from the trailer
145 
146  void clear()
147  {
148  nPackets = 0;
149  nTriggers = 0;
150  errorCounts.fill(0);
151  packetStates.fill(0);
152  }
153 
154  void print(bool skipEmpty = true) const;
155 
156  ClassDefNV(GBTLinkDecodingStat, 2);
