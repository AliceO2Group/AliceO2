53 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
54 {
55  DetID::mask_t dets;
56  WorkflowSpec specs;
57 
58  std::string inpNames;
59  if (!configcontext.helpOnCommandLine()) {
60  dets.set(); // by default read all
61  auto mskOnly = DetID::getMask(configcontext.options().get<std::string>("onlyDet"));
62  auto mskSkip = DetID::getMask(configcontext.options().get<std::string>("skipDet"));
63  if (mskOnly.any()) {
64  dets &= mskOnly;
65  } else {
66  dets ^= mskSkip;
67  }
68  if ((inpNames = configcontext.options().get<std::string>("ctf-input")).empty()) {
69  throw std::runtime_error("--ctf-input <file,...> is not provided");
70  }
71  }
72 
73  specs.push_back(o2::ctf::getCTFReaderSpec(dets, inpNames));
74  // add decodors for all allowed detectors.
75  if (dets[DetID::ITS]) {
76  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::ITS)));
77  }
78  if (dets[DetID::MFT]) {
79  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::MFT)));
80  }
81  if (dets[DetID::TPC]) {
82  specs.push_back(o2::tpc::getEntropyDecoderSpec());
83  }
84  if (dets[DetID::TOF]) {
85  specs.push_back(o2::tof::getEntropyDecoderSpec());
86  }
87  if (dets[DetID::FT0]) {
88  specs.push_back(o2::ft0::getEntropyDecoderSpec());
89  }
90  if (dets[DetID::FV0]) {
91  specs.push_back(o2::fv0::getEntropyDecoderSpec());
92  }
93  if (dets[DetID::FDD]) {
94  specs.push_back(o2::fdd::getEntropyDecoderSpec());
95  }
96  if (dets[DetID::MID]) {
97  specs.push_back(o2::mid::getEntropyDecoderSpec());
98  }
99  if (dets[DetID::EMC]) {
100  specs.push_back(o2::emcal::getEntropyDecoderSpec());
101  }
102  if (dets[DetID::PHS]) {
103  specs.push_back(o2::phos::getEntropyDecoderSpec());
104  }
105  if (dets[DetID::CPV]) {
106  specs.push_back(o2::cpv::getEntropyDecoderSpec());
107  }
108  if (dets[DetID::ZDC]) {
109  specs.push_back(o2::zdc::getEntropyDecoderSpec());
110  }
111 
112  return std::move(specs);
113 }
