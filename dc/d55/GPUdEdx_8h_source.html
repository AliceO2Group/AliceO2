89 GPUdi() void GPUdEdx::checkSubThresh(int roc)
90 {
91  if (roc != mLastROC) {
92  if (mNSubThresh && mCount + mNSubThresh <= MAX_NCL) {
93  for (int i = 0; i < mNSubThresh; i++) {
94  mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinTot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
95  mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinMax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
96  }

98  mNClsROCSubThresh[mLastROC] += mNSubThresh;
99  }

108 GPUdnii() void GPUdEdx::fillCluster(float qtot, float qmax, int padRow, float trackSnp, float trackTgl, const GPUParam& GPUrestrict() param, const TPCdEdxCalibrationSplines* splines, float z)
109 {
110  if (mCount >= MAX_NCL) {
111  return;
112  }
113  const int roc = param.tpcGeometry.GetROC(padRow);
114  checkSubThresh(roc);
115  float snp2 = trackSnp * trackSnp;
116  if (snp2 > GPUCA_MAX_SIN_PHI_LOW) {
117  snp2 = GPUCA_MAX_SIN_PHI_LOW;
118  }
119  const float tgl2 = trackTgl * trackTgl;
120  float factor = CAMath::Sqrt((1 - snp2) / (1 + tgl2));
121  factor /= param.tpcGeometry.PadHeight(padRow);
122  qtot *= factor;
123  qmax *= factor;
124 
125  const float sec2 = 1.f / (1.f - snp2);
126  // angleZ local dip angle: z angle - dz/dx (cm/cm)
127  float angleZ = CAMath::Sqrt(tgl2 * sec2);
128  if (angleZ > 3) {
129  angleZ = 3;
130  }
131 
132  const int region = param.tpcGeometry.GetRegion(padRow);
133  z = CAMath::Abs(z);
134 
135  // get the correction for qMax and qTot from the splines for given angle and drift length
136  const float qMaxCorr = splines->interpolateqMax(region, angleZ, z);
137  const float qTotCorr = splines->interpolateqTot(region, angleZ, z);
138  qmax /= qMaxCorr;
139  qtot /= qTotCorr;
140 
141  mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(qtot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
142  mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(qmax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
143  mNClsROC[roc]++;
144  if (qtot < mSubThreshMinTot) {
145  mSubThreshMinTot = qtot;
146  }
147  if (qmax < mSubThreshMinMax) {
148  mSubThreshMinMax = qmax;
149  }
150 }
