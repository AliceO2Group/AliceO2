44 void Calibrations::setCCDB(int calibrationobjecttype, long timestamp)
45 {
46  auto& ccdbmgr = o2::ccdb::BasicCCDBManager::instance();
47  ccdbmgr.setTimestamp(timestamp); // set which time stamp of data we want this is called per timeframe, and removes the need to call it when querying a value.
48  switch (calibrationobjecttype) {
49  case 1: //simulation
50  mChamberCalibrations = ccdbmgr.get<o2::trd::ChamberCalibrations>("TRD_test/ChamberCalibrations");
51  if (!mChamberCalibrations) {
52  LOG(fatal) << "No chamber calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
53  }
54  mLocalVDrift = ccdbmgr.get<o2::trd::LocalVDrift>("TRD_test/LocalVDrift");
55  if (!mLocalVDrift) {
56  LOG(fatal) << "No Local V Drift calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
57  }
58  mLocalT0 = ccdbmgr.get<o2::trd::LocalT0>("TRD_test/LocalT0");
59  if (!mLocalT0) {
60  LOG(fatal) << "No Local T0 calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
61  }
62  mLocalGainFactor = ccdbmgr.get<o2::trd::LocalGainFactor>("TRD_test/LocalGainFactor");
63  if (!mLocalT0) {
64  LOG(fatal) << "No Local T0 calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
65  }
66  mPadNoise = ccdbmgr.get<o2::trd::PadNoise>("TRD_test/PadNoise");
67  if (!mPadNoise) {
68  LOG(fatal) << "No Padnoise calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
69  }
70  mChamberStatus = ccdbmgr.get<o2::trd::ChamberStatus>("TRD_test/ChamberStatus");
71  if (!mChamberStatus) {
72  LOG(fatal) << "No ChamberStatus calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
73  }
74  mPadStatus = ccdbmgr.get<o2::trd::PadStatus>("TRD_test/PadStatus");
75  if (!mPadStatus) {
76  LOG(fatal) << "No Pad Status calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
77  }
78  mChamberNoise = ccdbmgr.get<o2::trd::ChamberNoise>("TRD_test/ChamberNoise");
79  if (!mChamberNoise) {
80  LOG(fatal) << "No ChamberNoise calibrations returned from CCDB for TRD calibrations, or TRD from what ever you are running.";
81  }
82  //mCalOnlineGainTables= ccdbmgr.get<o2::trd::CalOnlineGainTables>("TRD_test/OnlineGainTables");
83  //std::shared_ptr<TrapConfig> mTrapConfig;
84  //std::shared_ptr<PRFWidth> mPRDWidth
85  //std::shared_ptr<CalOnlineGainTables> mCalOnlineGainTables;
86 
87  break;
88  case 2: //reconstruction
89  LOG(fatal) << "Reconstruction calibration not implemented yet ";
90  break;
91  case 3: // calibration
92  LOG(fatal) << "Calibrating calibration not implemented yet ";
93  break;
94  default:
95  LOG(fatal) << "unknown calibration type coming into setCCDB for TRD Calibrations";
96  }
97 }
