49 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
50 {
51  // Update the (declared) parameters if changed from the command line
52  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
53  // write the configuration used for the digitizer workflow
54  o2::conf::ConfigurableParam::writeINI("o2itsrecoflow_configuration.ini");
55 
56  auto useMC = !configcontext.options().get<bool>("disable-mc");
57  auto useCAtracker = configcontext.options().get<bool>("trackerCA");
58  auto trmode = configcontext.options().get<std::string>("tracking-mode");
59  auto gpuDevice = static_cast<o2::gpu::GPUDataTypes::DeviceType>(configcontext.options().get<int>("gpuDevice"));
60  auto extDigits = configcontext.options().get<bool>("digits-from-upstream");
61  auto extClusters = configcontext.options().get<bool>("clusters-from-upstream");
62  auto disableRootOutput = configcontext.options().get<bool>("disable-root-output");
63  auto eencode = configcontext.options().get<bool>("entropy-encoding");
64 
65  std::transform(trmode.begin(), trmode.end(), trmode.begin(), [](unsigned char c) { return std::tolower(c); });
66  if (trmode != "sync" && !useCAtracker) {
67  LOG(ERROR) << "requested CookedTracker supports only sync tracking-mode, use --trackerCA";
68  throw std::runtime_error("incompatible options provided");
69  }
70  return std::move(o2::its::reco_workflow::getWorkflow(useMC, useCAtracker, trmode, gpuDevice, extDigits, extClusters, disableRootOutput, eencode));
71 }
