38 static void BM_TreeToTable(benchmark::State& state)
39 {
40 
41  // initialize a random generator
42  std::default_random_engine e1(1234567891);
43  std::uniform_real_distribution<double> rd(0, 1);
44  std::normal_distribution<float> rf(5., 2.);
45  std::discrete_distribution<ULong64_t> rl({10, 20, 30, 30, 5, 5});
46  std::discrete_distribution<int> ri({10, 20, 30, 30, 5, 5});
47 
48  // create a table and fill the columns with random numbers
49  TableBuilder builder;
50  auto rowWriter =
51  builder.persist<double, float, ULong64_t, int>({"a", "b", "c", "d"});
52  for (auto i = 0; i < state.range(0); ++i) {
53  rowWriter(0, rd(e1), rf(e1), rl(e1), ri(e1));
54  }
55  auto table = builder.finalize();
56 
57  // now convert the table to a tree
58  TFile fout("tree2table.root", "RECREATE");
59  TableToTree ta2tr(table, &fout, "tree2table");
60  ta2tr.addAllBranches();
61  ta2tr.process();
62  fout.Close();
63 
64  // read tree and convert to table again
65  TFile* f = nullptr;
66  TreeToTable* tr2ta = nullptr;
67  for (auto _ : state) {
68 
69  // Open file and create tree
70  f = new TFile("tree2table.root", "READ");
71  auto tr = (TTree*)f->Get("tree2table");
72 
73  // benchmark TreeToTable
74  if (tr) {
75  tr2ta = new TreeToTable;
76  if (tr2ta->addAllColumns(tr)) {
77  tr2ta->fill(tr);
78  auto ta = tr2ta->finalize();
79  }
80 
81  } else {
82  LOG(INFO) << "tree is empty!";
83  }
84 
85  // clean up
86  delete tr2ta;
87 
88  f->Close();
89  delete f;
90  }
91 
92  state.SetBytesProcessed(state.iterations() * state.range(0) * 24);
93 }
