61  // this is the virtual interface for context objects
62  class ContextObject
63  {
64  public:
65  ContextObject() = delete;
66  ContextObject(FairMQMessagePtr&& headerMsg, FairMQMessagePtr&& payloadMsg, const std::string& bindingChannel)
67  : mParts{}, mChannel{bindingChannel}
68  {
69  mParts.AddPart(std::move(headerMsg));
70  mParts.AddPart(std::move(payloadMsg));
71  }
72  ContextObject(FairMQMessagePtr&& headerMsg, const std::string& bindingChannel)
73  : mParts{}, mChannel{bindingChannel}
74  {
75  mParts.AddPart(std::move(headerMsg));
76  }
77  virtual ~ContextObject() = default;
78 
82  virtual FairMQParts finalize()
83  {
84  FairMQParts parts = std::move(mParts);
85  assert(parts.Size() == 2);
86  auto* header = o2::header::get<o2::header::DataHeader*>(parts.At(0)->GetData());
87  if (header == nullptr) {
88  throw std::logic_error("No valid header message found");
89  } else {
90  // o2::header::get returns const pointer, but here we can change the message
91  const_cast<o2::header::DataHeader*>(header)->payloadSize = parts.At(1)->GetSize();
92  }
93  // return value optimization returns by move
94  return parts;
95  }
96 
98  const std::string& channel() const
99  {
100  return mChannel;
101  }
102 
103  bool empty() const
104  {
105  return mParts.Size() == 0;
106  }
107 
108  o2::header::DataHeader const* header()
109  {
110  // we would expect this function to be const but the FairMQParts API does not allow this
111  if (empty() || mParts.At(0) == nullptr) {
112  return nullptr;
113  }
114  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData());
115  }
116 
117  protected:
118  FairMQParts mParts;
119  std::string const& mChannel;
120  };

222  ContainerRefObject() = delete;
224  template <typename ContextType, typename... Args>
225  ContainerRefObject(ContextType* context, FairMQMessagePtr&& headerMsg, const std::string& bindingChannel, int index, Args&&... args)
226  : ContextObject(std::forward<FairMQMessagePtr>(headerMsg), context->getChannelRef(bindingChannel)),
227  // the transport factory
228  mFactory{context->proxy().getTransport(bindingChannel, index)},
229  // the memory resource takes ownership of the message
230  mResource{mFactory ? AlignedMemoryResource(mFactory->GetMemoryResource()) : AlignedMemoryResource(nullptr)},
231  // create the vector with apropriate underlying memory resource for the message
232  mData{std::forward<Args>(args)..., pmr::polymorphic_allocator<value_type>(&mResource)}
233  {
234  // FIXME: drop this repeated check and make sure at initial setup of devices that everything is fine
235  // introduce error policy
236  if (mFactory == nullptr) {
237  throw runtime_error_f("failed to get transport factory for channel %s", bindingChannel.c_str());
238  }
239  if (mResource.isValid() == false) {
240  throw runtime_error_f("no memory resource for channel %s", bindingChannel.c_str());
241  }
242  }
243  ~ContainerRefObject() override = default;

462  void schedule(Messages::value_type&& message)
463  {
464  auto const* header = message->header();
465  if (header == nullptr) {
466  throw std::logic_error("No valid header message found");
467  }
468  mScheduledMessages.emplace_back(std::move(message));
469  if (mDispatchControl.dispatch != nullptr) {
470  // send all scheduled messages if there is no trigger callback or its result is true
471  if (mDispatchControl.trigger == nullptr || mDispatchControl.trigger(*header)) {
472  std::unordered_map<std::string const*, FairMQParts> outputs;
473  for (auto& message : mScheduledMessages) {
474  FairMQParts parts = std::move(message->finalize());
475  assert(message->empty());
476  assert(parts.Size() == 2);
477  for (auto& part : parts) {
478  outputs[&(message->channel())].AddPart(std::move(part));
479  }
480  }
481  for (auto& [channel, parts] : outputs) {
482  mDispatchControl.dispatch(std::move(parts), *channel, DefaultChannelIndex);
483  }
484  mScheduledMessages.clear();
485  }
486  }
487  }
488
