58 template <typename ListT,
59  typename PageHeaderT,
60  typename GroupT = void,
61  bool GroupHeaderPerPage = false>
62 std::pair<std::unique_ptr<uint8_t[]>, size_t> MakeBuffer(size_t pagesize,
63  PageHeaderT pageheader,
64  const ListT& dataset)
65 {
66  static_assert(std::is_void<GroupT>::value || std::is_integral<GroupT>::value,
67  "Invalid group type");
68  auto totalSize = dataset.size() * sizeof(typename ListT::value_type);
69  totalSize += o2::algorithm::pageparser::sizeofGroupHeader<GroupT>();
70  auto maxElementsPerPage = pagesize - (sizeof(pageheader) + o2::algorithm::pageparser::sizeofGroupHeader<GroupT>());
71  maxElementsPerPage /= sizeof(typename ListT::value_type);
72 
73  if (std::is_void<GroupT>::value || !GroupHeaderPerPage) {
74  unsigned nPages = 0;
75  do {
76  totalSize += sizeof(PageHeaderT);
77  ++nPages;
78  } while (nPages * pagesize < totalSize);
79  } else {
80  auto nRequiredPages = dataset.size() / maxElementsPerPage;
81  if (dataset.size() % maxElementsPerPage > 0) {
82  ++nRequiredPages;
83  }
84  totalSize = (nRequiredPages > 0 ? nRequiredPages : 1) * pagesize;
85  }
86 
87  auto buffer = std::make_unique<uint8_t[]>(totalSize);
88  memset(buffer.get(), 0, totalSize);
89 
90  unsigned position = 0;
91  auto target = buffer.get();
92  GroupT* groupHeader = nullptr;
93  size_t nElementsInCurrentGroup = 0;
94  for (auto element : dataset) {
95  if (GroupHeaderPerPage && nElementsInCurrentGroup == maxElementsPerPage) {
96  // write the number of elements in the group and forward to next
97  // page boundary
98  o2::algorithm::pageparser::set(groupHeader, nElementsInCurrentGroup);
99  nElementsInCurrentGroup = 0;
100  if (position % pagesize) {
101  target += pagesize - (position % pagesize);
102  position += pagesize - (position % pagesize);
103  }
104  }
105  auto source = reinterpret_cast<uint8_t*>(&element);
106  auto copySize = sizeof(typename ListT::value_type);
107  if ((position % pagesize) == 0) {
108  memcpy(target, &pageheader, sizeof(PageHeaderT));
109  position += sizeof(PageHeaderT);
110  target += sizeof(PageHeaderT);
111  }
112  if (!std::is_void<GroupT>::value &&
113  (position % pagesize) == sizeof(PageHeader) &&
114  (GroupHeaderPerPage || position < pagesize)) {
115  // write one GroupHeader at the beginning of the data, currently
116  // GroupHeader must be of integral type
117  groupHeader = reinterpret_cast<GroupT*>(target);
118  position += o2::algorithm::pageparser::sizeofGroupHeader<GroupT>();
119  target += o2::algorithm::pageparser::sizeofGroupHeader<GroupT>();
120  }
121  ++nElementsInCurrentGroup;
122  if ((position % pagesize) + copySize > pagesize) {
123  copySize -= ((position % pagesize) + copySize) - pagesize;
124  }
125  if (copySize > 0) {
126  memcpy(target, source, copySize);
127  position += copySize;
128  target += copySize;
129  source += copySize;
130  }
131  copySize = sizeof(typename ListT::value_type) - copySize;
132  if (copySize > 0) {
133  memcpy(target, &pageheader, sizeof(PageHeaderT));
134  position += sizeof(PageHeaderT);
135  target += sizeof(PageHeaderT);
136  memcpy(target, source, copySize);
137  }
138  position += copySize;
139  target += copySize;
140  }
141  if (!std::is_void<GroupT>::value) {
142  o2::algorithm::pageparser::set(groupHeader, nElementsInCurrentGroup);
143  }
144 
145  std::pair<std::unique_ptr<uint8_t[]>, size_t> result;
146  result.first = std::move(buffer);
147  result.second = totalSize;
148  return result;
149 }
