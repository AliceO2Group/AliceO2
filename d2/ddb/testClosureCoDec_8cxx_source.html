49 // Create a vector of SampaCluster from a string d
50 // where d is of the form ts-#-bc-#-cs-#-q-# or
51 // ts-#-bc-#-cs-#-q-#-#-# ...
52 // d is expected to be a valid SampaCluster string representation
53 // see the asString function in SampaCluster
54 std::vector<SampaCluster> getSampaClusters(const std::string& d)
55 {
56  std::vector<SampaCluster> clusters;
57 
58  std::cout << "d: " << d << std::endl;
59 
60  auto index = d.find("ts-");
61  auto ts = std::stoi(d.substr(index + 3));
62  index = d.find("bc-");
63  auto bc = std::stoi(d.substr(index + 3));
64  index = d.find("cs-");
65  auto cs = std::stoi(d.substr(index + 3));
66  index = d.find("q-");
67  auto q = d.substr(index + 2);
68  index = q.find("-");
69  if (index != std::string::npos) {
70  std::vector<uint10_t> charges;
71  std::istringstream ss(q);
72  std::string adc;
73  while (std::getline(ss, adc, '-')) {
74  charges.emplace_back(std::stoi(adc) & 0x3FF);
75  }
76  clusters.emplace_back(SampaCluster(ts, bc, charges));
77  } else {
78  clusters.emplace_back(SampaCluster(ts, bc, std::stoi(q), cs));
79  }
80  return clusters;
81 }

83 // create a raw data buffer from a list of strings
84 // where each string is of the form
85 // S#-J#-DS#-CH#-ts-#-bc-#-q-#
86 //
87 // it's a two steps process :
88 //
89 // - first create a buffer of payloads using a PayloadEncoder
90 // - then create the raw data itself (with proper RDHs) from the payload buffer
91 //
92 template <typename FORMAT, typename CHARGESUM>
93 std::vector<std::byte> createBuffer(gsl::span<std::string> data,
94  uint32_t orbit = 12345, uint16_t bc = 678)
95 {
96  auto encoder = createPayloadEncoder<FORMAT, CHARGESUM, true>();
97  encoder->startHeartbeatFrame(orbit, bc);
98  for (auto d : data) {
99  auto dsElecId = decodeDsElecId(d);
100  if (!dsElecId) {
101  std::cout << "Could not get dsElecId for " << d << "\n";
102  }
103  auto channel = decodeChannelId(d);
104  if (!channel) {
105  std::cout << "Could not get channel for " << d << "\n";
106  }
107  auto sampaClusters = getSampaClusters(d);
108  encoder->addChannelData(dsElecId.value(), channel.value(), sampaClusters);
109  }
110  std::vector<std::byte> buffer;
111  encoder->moveToBuffer(buffer);
112 
113  const o2::raw::HBFUtils& hbfutils = o2::raw::HBFUtils::Instance();
114  o2::conf::ConfigurableParam::setValue<uint32_t>("HBFUtils", "orbitFirst", orbit);
115  o2::conf::ConfigurableParam::setValue<uint16_t>("HBFUtils", "bcFirst", bc);
116  std::vector<std::byte> out = o2::mch::raw::paginate(buffer,
117  isUserLogicFormat<FORMAT>::value,
118  isChargeSumMode<CHARGESUM>::value,
119  fmt::format("mch-closure-codec-{}-{}.raw",
120  orbit, bc));
121  return out;
122 }

132 // decode the buffer and check its content against the expected vector of strings
133 bool testDecode(gsl::span<const std::byte> testBuffer, gsl::span<std::string> expected)
134 {
135  std::vector<std::string> result;
136 
137  DecodedDataHandlers handlers;
138  handlers.sampaChannelHandler = handlePacketStoreAsVec(result);
139  auto pageDecoder = createPageDecoder(testBuffer, handlers);
140 
141  auto parser = createPageParser();
142 
143  parser(testBuffer, pageDecoder);
144 
145  bool sameSize = result.size() == expected.size();
146  bool permutation = std::is_permutation(begin(result), end(result), begin(expected));
147  BOOST_CHECK_EQUAL(sameSize, true);
148  BOOST_CHECK_EQUAL(permutation, true);
149  if (!permutation || !sameSize) {
150  std::cout << "Got " << result.size() << " results:\n";
151  for (auto s : result) {
152  std::cout << s << "\n";
153  }
154  std::cout << "Expected " << expected.size() << ":\n";
155  for (auto s : expected) {
156  std::cout << s << "\n";
157  }
158  return false;
159  }
160  return true;
161 }
