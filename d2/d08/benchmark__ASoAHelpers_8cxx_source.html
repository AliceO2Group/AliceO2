41 {

43 
44  for (auto _ : state) {
45  count = 0;
46  int n = state.range(0);
47  for (int i = 0; i < n - 1; i++) {
48  for (int j = i + 1; j < n; j++) {

52  benchmark::DoNotOptimize(count);
53  }
54  state.counters["Combinations"] = count;
