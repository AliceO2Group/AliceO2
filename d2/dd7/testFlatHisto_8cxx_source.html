24 // basic Vertex tests
25 BOOST_AUTO_TEST_CASE(FlatHisto)
26 {
27  o2::dataformats::FlatHisto1D_f h1(100, -100., 100.);
28  o2::dataformats::FlatHisto2D_f h2(100, -100., 100., 50, -5., 45.);
29  for (int i = 0; i < 1000000; i++) {
30  h1.fill(gRandom->Gaus(10, 30));
31  }
32  auto h2ref = h2.createTH2F("h2ref");
33  for (int i = 0; i < 10000000; i++) {
34  auto x = gRandom->Gaus(10, 40), y = gRandom->Gaus(10, 10);
35  h2.fill(x, y);
36  h2ref.Fill(x, y);
37  }
38  auto th1f = h1.createTH1F();
39  auto res = th1f.Fit("gaus", "S");
40  BOOST_CHECK_CLOSE(res->GetParams()[1], 10, 0.2);
41 
42  printf("%e %e\n", h2.getSum(), h2ref.Integral());
43  BOOST_CHECK(h2.getSum() == h2ref.Integral());
44 
45  o2::dataformats::FlatHisto1D_f h1v(h1);
46 
47  BOOST_CHECK_CLOSE(h1.getBinStart(0), -100, 1e-5);
48  BOOST_CHECK_CLOSE(h1.getBinEnd(h1.getNBins() - 1), 100, 1e-5);
49 
50  BOOST_CHECK_CLOSE(h2.getBinStartX(0), -100, 1e-5);
51  BOOST_CHECK_CLOSE(h2.getBinEndY(h2.getNBinsY() - 1), 45, 1e-5);
52  BOOST_CHECK_CLOSE(h2.getBinStartY(h2.getNBinsY() - 1), 45 - h2.getBinSizeY(), 1e-5);
53 
54  BOOST_CHECK(h1.canFill() && h1v.canFill());
55  BOOST_CHECK(h1.getSum() == h1v.getSum());
56  {
57  TFile flout("flathisto.root", "recreate");
58  flout.WriteObjectAny(&h1, "o2::dataformats::FlatHisto1D_f", "h1");
59  flout.WriteObjectAny(&h2, "o2::dataformats::FlatHisto2D_f", "h2");
60  flout.Close();
61  }
62 
63  TFile flin("flathisto.root");
64  o2::dataformats::FlatHisto1D_f* h1r = (o2::dataformats::FlatHisto1D_f*)flin.GetObjectUnchecked("h1");
65  o2::dataformats::FlatHisto2D_f* h2r = (o2::dataformats::FlatHisto2D_f*)flin.GetObjectUnchecked("h2");
66  flin.Close();
67  h1r->init();
68  h2r->init();
69 
70  o2::dataformats::FlatHisto1D_f h1vv;
71  h1vv.adoptExternal(h1r->getView());
72  BOOST_CHECK(h1.getSum() == h1vv.getSum());
73  h1.add(h1vv);
74  h2.subtract(*h2r);
75  BOOST_CHECK(h1.getSum() == 2 * h1vv.getSum());
76  BOOST_CHECK(h2.getSum() == 0.);
77 }
