25 TopologyFastSimulation::TopologyFastSimulation(std::string fileName, unsigned seed)
26 {
27  mDictionary.readBinaryFile(fileName);
28  double tot_freq = 0.;
29  int dictSize = mDictionary.getSize();
30  mFreqArray.reserve(dictSize);
31  for (int iKey = 0; iKey < dictSize; iKey++) {
32  tot_freq += mDictionary.getFrequency(iKey);
33  mFreqArray[iKey] = tot_freq;
34  }
35  mGenerator = std::mt19937(seed);
36  mDistribution = std::uniform_real_distribution<double>(0.0, 1.0);
37 }
38 
39 int TopologyFastSimulation::getRandom()
40 {
41  double rnd = mDistribution(mGenerator);
42  auto ind = std::upper_bound(mFreqArray.begin(), mFreqArray.end(), rnd,
43  [](const double& comp1, const double& comp2) { return comp1 < comp2; });
44  return std::distance(mFreqArray.begin(), ind);
45 }
