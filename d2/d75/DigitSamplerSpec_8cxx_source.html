48 class DigitSamplerTask
49 {
50  public:
51  //_________________________________________________________________________________________________
52  void init(framework::InitContext& ic)
53  {
55  LOG(INFO) << "initializing digit sampler";
56 
57  auto inputFileName = ic.options().get<std::string>("infile");
58  mInputFile.open(inputFileName, ios::binary);
59  if (!mInputFile.is_open()) {
60  throw invalid_argument("Cannot open input file " + inputFileName);
61  }
62  if (mInputFile.peek() == EOF) {
63  throw length_error("input file is empty");
64  }
65 
66  mUseRun2DigitUID = ic.options().get<bool>("useRun2DigitUID");
67  mPrint = ic.options().get<bool>("print");
68  mNevents = ic.options().get<int>("nevents");
69  mEvent = ic.options().get<int>("event");
70  mNEventsPerTF = ic.options().get<int>("nEventsPerTF");
71  if (mNEventsPerTF < 1) {
72  throw invalid_argument("number of events per time frame must be >= 1");
73  }
74 
75  auto stop = [this]() {
76  // close the input file
77  LOG(INFO) << "stop digit sampler";
78  this->mInputFile.close();
79  };
80  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
81  }
82 
83  //_________________________________________________________________________________________________
84  void run(framework::ProcessingContext& pc)
85  {
87 
88  // skip events until the requested one, if any
89  while (mCurrentEvent < mEvent && mInputFile.peek() != EOF) {
90  LOG(INFO) << "skipping event " << mCurrentEvent;
91  skipOneEvent();
92  ++mCurrentEvent;
93  }
94 
95  // stop if requested event(s) already processed or eof reached
96  if (mNevents == 0 || (mEvent >= 0 && mCurrentEvent > mEvent) || mInputFile.peek() == EOF) {
97  pc.services().get<ControlService>().endOfStream();
98  //pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
99  return;
100  }
101 
102  // create the output messages
103  auto& rofs = pc.outputs().make<std::vector<ROFRecord>>(OutputRef{"rofs"});
104  auto& digits = pc.outputs().make<std::vector<Digit>>(OutputRef{"digits"});
105 
106  if (mCurrentEvent == mEvent) {
107 
108  // send only the requested event
109  int nDigits = readOneEvent(digits);
110  rofs.emplace_back(o2::InteractionRecord{0, static_cast<uint32_t>(mCurrentEvent)},
111  digits.size() - nDigits, nDigits);
112  ++mCurrentEvent;
113 
114  } else {
115 
116  // or loop over the requested number of events (or until eof) and send all of them
117  rofs.reserve(mNEventsPerTF);
118  for (int iEvt = 0; iEvt < mNEventsPerTF && mNevents != 0 && mInputFile.peek() != EOF; ++iEvt, --mNevents) {
119  int nDigits = readOneEvent(digits);
120  rofs.emplace_back(o2::InteractionRecord{0, static_cast<uint32_t>(mCurrentEvent)},
121  digits.size() - nDigits, nDigits);
122  ++mCurrentEvent;
123  }
124  rofs.shrink_to_fit(); // fix error "could not set used size: boost::interprocess::bad_alloc" for some sizes
125  }
126 
127  // convert the digits UID if needed
128  if (mUseRun2DigitUID) {
129  convertDigitUID2PadID(digits);
130  }
131  }
132 
133  private:
134  //_________________________________________________________________________________________________
135  void skipOneEvent()
136  {
138 
139  // get the number of digits
140  int nDigits(-1);
141  mInputFile.read(reinterpret_cast<char*>(&nDigits), sizeof(int));
142  if (mInputFile.fail() || nDigits < 0) {
143  throw length_error("invalid input");
144  }
145 
146  // skip the digits if any
147  if (nDigits > 0) {
148  mInputFile.seekg(nDigits * sizeof(Digit), std::ios::cur);
149  if (mInputFile.fail()) {
150  throw length_error("invalid input");
151  }
152  }
153  }
154 
155  //_________________________________________________________________________________________________
156  int readOneEvent(std::vector<Digit, o2::pmr::polymorphic_allocator<Digit>>& digits)
157  {
159 
160  // get the number of digits
161  int nDigits(-1);
162  mInputFile.read(reinterpret_cast<char*>(&nDigits), sizeof(int));
163  if (mInputFile.fail() || nDigits < 0) {
164  throw length_error("invalid input");
165  }
166 
167  // get the digits if any
168  if (nDigits > 0) {
169  auto digitOffset = digits.size();
170  digits.resize(digitOffset + nDigits);
171  mInputFile.read(reinterpret_cast<char*>(&digits[digitOffset]), nDigits * sizeof(Digit));
172  if (mInputFile.fail()) {
173  throw length_error("invalid input");
174  }
175  } else {
176  LOG(INFO) << "event is empty";
177  }
178 
179  return nDigits;
180  }
181 
182  //_________________________________________________________________________________________________
183  void convertDigitUID2PadID(std::vector<Digit, o2::pmr::polymorphic_allocator<Digit>>& digits)
184  {
186 
187  for (auto& digit : digits) {
188 
189  int deID = digit.getDetID();
190  int digitID = digit.getPadID();
191  int manuID = (digitID & 0xFFF000) >> 12;
192  int manuCh = (digitID & 0x3F000000) >> 24;
193 
194  int padID = mapping::segmentation(deID).findPadByFEE(manuID, manuCh);
195  if (mPrint) {
196  cout << deID << " " << digitID << " " << manuID << " " << manuCh << " " << padID << endl;
197  }
198  if (padID < 0) {
199  throw runtime_error(std::string("digitID ") + digitID + " does not exist in the mapping");
200  }
201 
202  digit.setPadID(padID);
203  }
204  }

207  bool mUseRun2DigitUID = false;
208  bool mPrint = false;
209  int mNevents = -1;
210  int mEvent = -1;
211  int mCurrentEvent = 0;
212  int mNEventsPerTF = 1;
213 };
