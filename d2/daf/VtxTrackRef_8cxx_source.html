39 // set the last +1 element index and check consistency
40 void VtxTrackRef::setEnd(int end)
41 {
42  if (end <= 0) {
43  return; // empty
44  }
45  setEntries(end - getFirstEntry());
46  for (int i = VtxTrackIndex::NSources - 1; i--;) {
47  if (getFirstEntryOfSource(i) < 0) {
48  throw std::runtime_error(fmt::format("1st entry for source {:d} was not set", i));
49  }
50  if (getEntriesOfSource(i) < 0) {
51  throw std::runtime_error(fmt::format("Source {:d} has negative number of entrie", getEntriesOfSource(i)));
52  }
53  }
54 }
