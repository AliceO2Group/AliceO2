64 bool PID::processTrack(const o2::tpc::TrackTPC& track)
65 {
66  // ===| variables required for cutting and filling |===
67  const auto p = track.getP();
68  const auto dEdxTot = track.getdEdx().dEdxTotTPC;
69  const auto dEdxMax = track.getdEdx().dEdxMaxTPC;
70  const auto phi = track.getPhi();
71  const auto tgl = track.getTgl();
72  const auto snp = track.getSnp();
73  const auto nclusters = track.getNClusterReferences();
74 
75  // ===| histogram filling |===
76  mHist1D[0].Fill(nclusters);
77  mHist1D[1].Fill(dEdxTot);
78  mHist1D[2].Fill(dEdxMax);
79  mHist1D[3].Fill(phi);
80  mHist1D[4].Fill(tgl);
81  mHist1D[5].Fill(snp);
82 
83  mHist2D[0].Fill(phi, dEdxTot);
84  mHist2D[1].Fill(tgl, dEdxTot);
85  mHist2D[2].Fill(nclusters, dEdxTot);
86  mHist2D[3].Fill(p, dEdxTot);
87 
88  return true;
89 }
