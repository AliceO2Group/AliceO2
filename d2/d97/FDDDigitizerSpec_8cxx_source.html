40 {
41  using GRP = o2::parameters::GRPObject;

43  public:
44  void initDigitizerTask(framework::InitContext& ic) override
45  {
46  LOG(INFO) << "initializing FDD digitization";
47 
48  //mDigitizer.setCCDBServer(dopt.ccdb);
49  mDigitizer.init();
50  //mROMode = mDigitizer.isContinuous() ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
51  }
52  void run(framework::ProcessingContext& pc)
53  {
54  if (mFinished) {
55  return;
56  }
57  LOG(INFO) << "Doing FDD digitization";
58 
59  // TODO: this should eventually come from the framework and depend on the TF timestamp
60  //mDigitizer.refreshCCDB();
61 
62  // read collision context from input
63  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
64  auto& irecords = context->getEventRecords();
65 
66  context->initSimChains(o2::detectors::DetID::FDD, mSimChains);
67  mDigitizer.setEventTime(context->getGRP().getTimeStart());
68  for (auto& record : irecords) {
69  LOG(INFO) << "FDD TIME RECEIVED " << record.getTimeNS();
70  }
71 
72  auto& eventParts = context->getEventParts();
73 
74  // loop over all composite collisions given from context
75  // (aka loop over all the interaction records)
76  std::vector<o2::fdd::Hit> hits;
77  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel> labels;
78 
79  for (int collID = 0; collID < irecords.size(); ++collID) {
80 
81  const auto& irec = irecords[collID];
82  mDigitizer.setInteractionRecord(irec);
83 
84  for (auto& part : eventParts[collID]) {
85 
86  // get the hits for this event and this source
87  context->retrieveHits(mSimChains, "FDDHit", part.sourceID, part.entryID, &hits);
88  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found FDD " << hits.size() << " hits ";
89 
90  mDigitizer.setEventID(part.entryID);
91  mDigitizer.setSrcID(part.sourceID);
92 
93  mDigitizer.process(hits, mDigitsBC, mDigitsCh, mDigitsTrig, labels);
94  }
95  }
96 
97  o2::InteractionTimeRecord terminateIR;
98  terminateIR.orbit = 0xffffffff; // supply IR in the infinite future to flush all cached BC
99  mDigitizer.setInteractionRecord(terminateIR);
100  mDigitizer.flush(mDigitsBC, mDigitsCh, mDigitsTrig, labels);
101 
102  // send out to next stage
103  pc.outputs().snapshot(Output{"FDD", "DIGITSBC", 0, Lifetime::Timeframe}, mDigitsBC);
104  pc.outputs().snapshot(Output{"FDD", "DIGITSCH", 0, Lifetime::Timeframe}, mDigitsCh);
105  pc.outputs().snapshot(Output{"FDD", "TRIGGERINPUT", 0, Lifetime::Timeframe}, mDigitsTrig);
106  if (pc.outputs().isAllowed({"FDD", "DIGITLBL", 0})) {
107  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::fdd::MCLabel>>(Output{"FDD", "DIGITLBL", 0, Lifetime::Timeframe});
108  labels.flatten_to(sharedlabels);
109  labels.clear_andfreememory();
110  }
111 
112  LOG(INFO) << "FDD: Sending ROMode= " << mROMode << " to GRPUpdater";
113  pc.outputs().snapshot(Output{"FDD", "ROMode", 0, Lifetime::Timeframe}, mROMode);
114 
115  // we should be only called once; tell DPL that this process is ready to exit
116  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
117  mFinished = true;
118  }
119 
120  private:
121  bool mFinished = false;
122  Digitizer mDigitizer;
123  std::vector<TChain*> mSimChains;
124  std::vector<o2::fdd::ChannelData> mDigitsCh;
125  std::vector<o2::fdd::Digit> mDigitsBC;
126  std::vector<o2::fdd::DetTrigInput> mDigitsTrig;
127 
128  // RS: at the moment using hardcoded flag for continuous readout
129  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
130 };

134  // create the full data processor spec using

136  // input description

139  std::vector<OutputSpec> outputs;
140  outputs.emplace_back("FDD", "DIGITSBC", 0, Lifetime::Timeframe);
141  outputs.emplace_back("FDD", "DIGITSCH", 0, Lifetime::Timeframe);
142  outputs.emplace_back("FDD", "TRIGGERINPUT", 0, Lifetime::Timeframe);
143  if (mctruth) {
144  outputs.emplace_back("FDD", "DIGITLBL", 0, Lifetime::Timeframe);

147 
148  return DataProcessorSpec{
149  "FDDDigitizer",
150  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
151  outputs,
152  AlgorithmSpec{adaptFromTask<FDDDPLDigitizerTask>()},
153  Options{}};
154 }
