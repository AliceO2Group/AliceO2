70 template <class T>
71 int PadParameters<T>::init(int chamberindex)
72 {
73  mPlane = Geometry::getLayer(chamberindex);
74  mChamber = Geometry::getStack(chamberindex);
75  if (mChamber == 2) {
76  mNrows = constants::NROWC0;
77  } else {
78  mNrows = constants::NROWC1;
79  }
80  // the FeeParam variables need to be unprotected, and dont want to change FeeParam in this PR.
81  mNchannels = mNrows * mNcols;
82  mData.resize(mNchannels);
83  if (mData.size() != mNchannels || mData.size() == 0) {
84  return -1;
85  }
86  return 0;
87 }

89 template <class T>
90 int PadParameters<T>::reset(int chamberindex, int cols, int rows, std::vector<T>& data)
91 {
92  mPlane = Geometry::getLayer(chamberindex);
93  mChamber = Geometry::getStack(chamberindex);
94  mNrows = rows;
95  mNcols = cols;
96  // the FeeParam variables need to be unprotected, and dont want to change FeeParam in this PR.
97  mNchannels = mNrows * mNcols;
98  if (mData.size() != mNchannels) {
99  return -2;
100  }
101  mData.resize(mNchannels);
102  if (mData.size() != mNchannels || mData.size() == 0) {
103  return -1;
104  }
105 
106  // now reset the data of the pads.
107  int counter = 0;
108  for (auto pad : mData) {
109  pad = data[counter];
110  counter++;
111  }
112  return 0;
113 }
