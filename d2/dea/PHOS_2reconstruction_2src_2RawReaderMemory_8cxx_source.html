32 o2::header::RDHAny RawReaderMemory::decodeRawHeader(const void* payloadwords)
33 {
34  auto headerversion = RDHDecoder::getVersion(payloadwords);
35  if (headerversion == 4) {
36  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(payloadwords));
37  } else if (headerversion == 5) {
38  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(payloadwords));
39  } else if (headerversion == 6) {
40  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV6*>(payloadwords));
41  }
42  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
43 }

54 void RawReaderMemory::next()
55 {
56  //Reads pages till RCU trailer found
57  //Several 8kB pages can be concatenated
58  //RCU trailer expected at the end of payload
59  //but not at the end of each page
60  mRawPayload.reset();
61  mCurrentTrailer.reset();
62  bool isDataTerminated = false;
63  do {
64  try {
65  nextPage(false);
66  } catch (RawDecodingError::ErrorType_t e) {
67  throw e;
68  }
69  if (hasNext()) {
70  auto nextheader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
71  // check continuing payload based on the bc/orbit ID
72  auto currentbc = RDHDecoder::getTriggerBC(mRawHeader),
73  nextbc = RDHDecoder::getTriggerBC(nextheader);
74  auto currentorbit = RDHDecoder::getTriggerOrbit(mRawHeader),
75  nextorbit = RDHDecoder::getTriggerOrbit(nextheader);
76  if (currentbc != nextbc || currentorbit != nextorbit) {
77  isDataTerminated = true;
78  } else {
79  auto nextpagecounter = RDHDecoder::getPageCounter(nextheader);
80  if (nextpagecounter == 0) {
81  isDataTerminated = true;
82  } else {
83  isDataTerminated = false;
84  }
85  }
86  } else {
87  isDataTerminated = true;
88  }
89  // Check if the data continues
90  } while (!isDataTerminated);
91  try {
92  mCurrentTrailer.constructFromPayloadWords(mRawBuffer.getDataWords());
93  } catch (...) {
94  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
95  }
96 }

98 void RawReaderMemory::nextPage(bool doResetPayload)
99 {
100  if (!hasNext()) {
101  throw RawDecodingError::ErrorType_t::PAGE_NOTFOUND;
102  }
103  if (doResetPayload) {
104  mRawPayload.reset();
105  }
106  mRawHeaderInitialized = false;
107  mPayloadInitialized = false;
108 
109  // Read RDH header
110  try {
111  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
112  while (RDHDecoder::getOffsetToNext(mRawHeader) == RDHDecoder::getHeaderSize(mRawHeader) &&
113  mCurrentPosition < mRawMemoryBuffer.size()) {
114  // No Payload - jump to next rawheader
115  // This will eventually move, depending on whether for events without payload in the SRU we send the RCU trailer
116  mCurrentPosition += RDHDecoder::getHeaderSize(mRawHeader);
117  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
118  }
119  mRawHeaderInitialized = true;
120  } catch (...) {
121  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
122  }
123  if (mCurrentPosition + RDHDecoder::getMemorySize(mRawHeader) > mRawMemoryBuffer.size()) {
124  // Payload incomplete
125  throw RawDecodingError::ErrorType_t::PAYLOAD_DECODING;
126  }
127 
128  mRawBuffer.readFromMemoryBuffer(gsl::span<const char>(mRawMemoryBuffer.data() + mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader),
129  RDHDecoder::getMemorySize(mRawHeader) - RDHDecoder::getHeaderSize(mRawHeader)));
130  gsl::span<const uint32_t> payloadWithoutTrailer(mRawBuffer.getDataWords().data(), mRawBuffer.getNDataWords());
131  mRawPayload.appendPayloadWords(payloadWithoutTrailer);
132  mRawPayload.increasePageCount();
133 
134  mCurrentPosition += RDHDecoder::getOffsetToNext(mRawHeader);
135 }
