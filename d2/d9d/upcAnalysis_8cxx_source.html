35  void process(soa::Join<aod::Collisions, aod::EvSels>::iterator const& col, soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::TrackSelection>> const& tracks)
36  {
37  bool checkV0 = col.bbV0A() || col.bbV0C() || col.bgV0A() || col.bgV0C();
38  if (checkV0) {
39  return;
40  }
41  bool checkFDD = col.bbFDA() || col.bbFDC() || col.bgFDA() || col.bgFDC();
42  if (checkFDD) {
43  return;
44  }
45  if (!col.alias()[kCUP9]) {
46  return;
47  }
48  if (tracks.size() != 2) {
49  return;
50  }
51  auto first = tracks.begin();
52  auto second = first + 1;
53  if (first.sign() * second.sign() >= 0) {
54  return;
55  }
56  UChar_t clustermap1 = first.itsClusterMap();
57  UChar_t clustermap2 = second.itsClusterMap();
58  bool checkClusMap = TESTBIT(clustermap1, 0) && TESTBIT(clustermap1, 1) && TESTBIT(clustermap2, 0) && TESTBIT(clustermap2, 1);
59  if (!checkClusMap) {
60  return;
61  }
62  TLorentzVector p1, p2, p;
63  p1.SetXYZM(first.px(), first.py(), first.pz(), mpion);
64  p2.SetXYZM(second.px(), second.py(), second.pz(), mpion);
65  p = p1 + p2;
66  hPt->Fill(p.Pt());
67  float signalTPC1 = first.tpcSignal();
68  float signalTPC2 = second.tpcSignal();
69  hSignalTPC1vsSignalTPC2->Fill(signalTPC1, signalTPC2);
70  if ((p.Pt() < 0.1) && (signalTPC1 + signalTPC2 < 140.)) {
71  hMass->Fill(p.M());
72  }
73  }
74 };
