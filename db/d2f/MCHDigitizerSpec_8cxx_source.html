48  void run(framework::ProcessingContext& pc)
49  {
50  static bool finished = false;
51  if (finished) {
52  return;
53  }
54  LOG(DEBUG) << "Doing MCH digitization";
55 
56  // read collision context from input
57  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
58  context->initSimChains(o2::detectors::DetID::MCH, mSimChains);
59  auto& irecords = context->getEventRecords();
60 
61  for (auto& record : irecords) {
62  LOG(DEBUG) << "MCH TIME RECEIVED " << record.getTimeNS();
63  LOG(DEBUG) << "MCH TIME RECEIVED: bc " << record.bc;
64  }
65 
66  auto& eventParts = context->getEventParts();
67  std::vector<o2::mch::Digit> digitsAccum; // accumulator for digits
68  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum;
69 
70  // loop over all composite collisions given from context
71  // (aka loop over all the interaction records)
72  for (int collID = 0; collID < irecords.size(); ++collID) {
73  // mDigitizer.setEventTime(irecords[collID].getTimeNS());
74  mDigitizer.setEventTime(irecords[collID].bc);
75  // for each collision, loop over the constituents event and source IDs
76  // (background signal merging is basically taking place here)
77  for (auto& part : eventParts[collID]) {
78  mDigitizer.setEventID(part.entryID);
79  mDigitizer.setSrcID(part.sourceID);
80 
81  // get the hits for this event and this source
82  std::vector<o2::mch::Hit> hits;
83  context->retrieveHits(mSimChains, "MCHHit", part.sourceID, part.entryID, &hits);
84  LOG(DEBUG) << "For collision " << collID << " eventID " << part.entryID << " found MCH " << hits.size() << " hits ";
85 
86  std::vector<o2::mch::Digit> digits; // digits which get filled
87  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels;
88 
89  mDigitizer.process(hits, digits, labels);
90  LOG(DEBUG) << "MCH obtained " << digits.size() << " digits ";
91  for (auto& d : digits) {
92  LOG(DEBUG) << "ADC " << d.getADC();
93  LOG(DEBUG) << "PAD " << d.getPadID();
94  LOG(DEBUG) << "TIME " << d.getTime();
95  LOG(DEBUG) << "DetID " << d.getDetID();
96  }
97  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
98  labelAccum.mergeAtBack(labels); //is this ok? check inside MCtruthContainer if this is what one wants to do.
99  LOG(DEBUG) << "labelAccum.getIndexedSize() " << labelAccum.getIndexedSize();
100  LOG(DEBUG) << "labelAccum.getNElements() " << labelAccum.getNElements();
101  LOG(DEBUG) << "Have " << digits.size() << " digits ";
102  }
103  }
104  mDigitizer.mergeDigits(digitsAccum, labelAccum); //print-out inside alos works fine
105 
106  LOG(DEBUG) << "Have " << labelAccum.getNElements() << " MCH labels "; //does not work out!
107  pc.outputs().snapshot(Output{"MCH", "DIGITS", 0, Lifetime::Timeframe}, digitsAccum);
108  if (pc.outputs().isAllowed({"MCH", "DIGITSMCTR", 0})) {
109  pc.outputs().snapshot(Output{"MCH", "DIGITSMCTR", 0, Lifetime::Timeframe}, labelAccum);
110  }
111  LOG(DEBUG) << "MCH: Sending ROMode= " << mROMode << " to GRPUpdater";
112  //ROMode: to be understood, check EMCal etc.
113  pc.outputs().snapshot(Output{"MCH", "ROMode", 0, Lifetime::Timeframe}, mROMode);
114 
115  // we should be only called once; tell DPL that this process is ready to exit
116  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
117  finished = true;
118  }
119 
120  private:
121  Digitizer mDigitizer;
122  std::vector<TChain*> mSimChains;
123  // RS: at the moment using hardcoded flag for continuos readout
124  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
125 };
