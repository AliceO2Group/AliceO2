51 void FullHistoryMerger::run(framework::ProcessingContext& ctx)
52 {
53  // we have to avoid mistaking the timer input with data inputs.
54  auto* timerHeader = ctx.inputs().get("timer-publish").header;
55 
56  for (const DataRef& ref : InputRecordWalker(ctx.inputs())) {
57  if (ref.header != timerHeader) {
58  updateCache(ref);
59  mUpdatesReceived++;
60  }
61  }
62 
63  if (ctx.inputs().isValid("timer-publish") && !mFirstObjectSerialized.first.empty()) {
64  mergeCache();
65  publish(ctx.outputs());
66  }
67 }

69 void FullHistoryMerger::updateCache(const DataRef& ref)
70 {
71  auto* dh = get<DataHeader*>(ref.header);
72  std::string sourceID = std::string(dh->dataOrigin.str) + "/" + std::string(dh->dataDescription.str) + "/" + std::to_string(dh->subSpecification);
73 
74  // I am not sure if ref.spec is always a concrete spec and not a broader matcher. Comparing it this way should be safer.
75  if (mFirstObjectSerialized.first.empty() || mFirstObjectSerialized.first == sourceID) {
76 
77  // We store one object in the serialized form, so we can take it as the first object to be merged (multiple times).
78  // If we kept it deserialized, we would need to require implementing a clone() method in MergeInterface.
79 
80  // Clang-Tidy: 'if' statement is unnecessary; deleting null pointer has no effect
81  delete mFirstObjectSerialized.second.spec;
82  delete mFirstObjectSerialized.second.header;
83  delete mFirstObjectSerialized.second.payload;
84 
85  mFirstObjectSerialized.first = sourceID;
86  mFirstObjectSerialized.second.spec = new InputSpec(*ref.spec);
87  mFirstObjectSerialized.second.header = new char[Stack::headerStackSize(reinterpret_cast<o2::byte const*>(dh))];
88  memcpy((void*)mFirstObjectSerialized.second.header, ref.header, dh->headerSize);
89  mFirstObjectSerialized.second.payload = new char[dh->payloadSize];
90  memcpy((void*)mFirstObjectSerialized.second.payload, ref.payload, dh->payloadSize);
91 
92  } else {
93  mCache[sourceID] = object_store_helpers::extractObjectFrom(ref);
94  }
95 }

97 void FullHistoryMerger::mergeCache()
98 {
99  LOG(INFO) << "Merging " << mCache.size() + 1 << " objects.";
100 
101  mMergedObject = object_store_helpers::extractObjectFrom(mFirstObjectSerialized.second);
102  assert(!std::holds_alternative<std::monostate>(mMergedObject));
103  mObjectsMerged++;
104 
105  // We expect that all the objects use the same kind of interface
106  if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
107 
108  auto target = std::get<TObjectPtr>(mMergedObject);
109  for (auto& [name, entry] : mCache) {
110  (void)name;
111  auto other = std::get<TObjectPtr>(entry);
112  algorithm::merge(target.get(), other.get());
113  mObjectsMerged++;
114  }
115 
116  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
117  auto target = std::get<MergeInterfacePtr>(mMergedObject);
118  for (auto& [name, entry] : mCache) {
119  (void)name;
120  auto other = std::get<MergeInterfacePtr>(entry);
121  target->merge(other.get());
122  mObjectsMerged++;
123  }
124  }
125 }

127 void FullHistoryMerger::publish(framework::DataAllocator& allocator)
128 {
129  // todo see if std::visit is faster here
130  if (std::holds_alternative<std::monostate>(mMergedObject)) {
131  LOG(INFO) << "Nothing to publish yet";
132  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
133  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
134  *std::get<MergeInterfacePtr>(mMergedObject));
135  } else if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
136  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
137  *std::get<TObjectPtr>(mMergedObject));
138  } else {
139  throw std::runtime_error("mMergedObject' variant has no value.");
140  }
141 
142  mTotalObjectsMerged += mObjectsMerged;
143  mTotalUpdatesReceived += mUpdatesReceived;
144  mCollector->send({mTotalObjectsMerged, "total_objects_merged"}, monitoring::DerivedMetricMode::RATE);
145  mCollector->send({mObjectsMerged, "objects_merged_since_last_publication"});
146  mCollector->send({mTotalUpdatesReceived, "total_updates_received"}, monitoring::DerivedMetricMode::RATE);
147  mCollector->send({mUpdatesReceived, "updates_received_since_last_publication"});
148  mObjectsMerged = 0;
149  mUpdatesReceived = 0;
150 }
