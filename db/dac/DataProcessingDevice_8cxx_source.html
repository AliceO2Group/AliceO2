82  : mSpec{spec},

84  mInit{spec.algorithm.onInit},
85  mStatefulProcess{nullptr},
86  mStatelessProcess{spec.algorithm.onProcess},
87  mError{spec.algorithm.onError},
88  mConfigRegistry{nullptr},
89  mAllocator{&mTimingInfo, &registry, spec.outputs},
90  mServiceRegistry{registry}
91 {
93  if (mError != nullptr) {
94  mErrorHandling = [&errorCallback = mError,
95  &serviceRegistry = mServiceRegistry](RuntimeErrorRef e, InputRecord& record) {
96  ZoneScopedN("Error handling");
97  auto& err = error_from_ref(e);
98  LOGP(ERROR, "Exception caught: {} ", err.what);
99  backtrace_symbols_fd(err.backtrace, err.maxBacktrace, STDERR_FILENO);
100  serviceRegistry.get<DataProcessingStats>().exceptionCount++;
101  ErrorContext errorContext{record, serviceRegistry, e};
102  errorCallback(errorContext);
103  };
104  } else {
105  mErrorHandling = [&errorPolicy = mErrorPolicy,
106  &serviceRegistry = mServiceRegistry](RuntimeErrorRef e, InputRecord& record) {
107  ZoneScopedN("Error handling");
108  auto& err = error_from_ref(e);
109  LOGP(ERROR, "Exception caught: {} ", err.what);
110  backtrace_symbols_fd(err.backtrace, err.maxBacktrace, STDERR_FILENO);
111  serviceRegistry.get<DataProcessingStats>().exceptionCount++;
112  switch (errorPolicy) {
113  case TerminationPolicy::QUIT:
114  throw e;
115  default:
116  break;
117  }
118  };
119  }

166  ZoneScopedN("socket prioritized");
167  LOG(debug) << "socket polled UV_PRIORITIZED";

180 void DataProcessingDevice::Init()
181 {
182  TracyAppInfo(mSpec.name.data(), mSpec.name.size());
183  ZoneScopedN("DataProcessingDevice::Init");
184  mRelayer = &mServiceRegistry.get<DataRelayer>();
185  // If available use the ConfigurationInterface, otherwise go for
186  // the command line options.
187  bool hasConfiguration = false;
188  bool hasOverrides = false;
189  if (mServiceRegistry.active<ConfigurationInterface>()) {
190  auto& cfg = mServiceRegistry.get<ConfigurationInterface>();
191  hasConfiguration = true;
192  try {
193  cfg.getRecursive(mSpec.name);
194  hasOverrides = true;
195  } catch (...) {
196  // No overrides...
197  }
198  }
199  // We only use the configuration file if we have a stanza for the given
200  // dataprocessor
201  std::vector<std::unique_ptr<ParamRetriever>> retrievers;
202  if (hasConfiguration && hasOverrides) {
203  auto& cfg = mServiceRegistry.get<ConfigurationInterface>();
204  retrievers.emplace_back(std::make_unique<ConfigurationOptionsRetriever>(&cfg, mSpec.name));
205  } else {
206  retrievers.emplace_back(std::make_unique<FairOptionsRetriever>(GetConfig()));
207  }
208  auto configStore = std::make_unique<ConfigParamStore>(mSpec.options, std::move(retrievers));
209  configStore->preload();
210  configStore->activate();
211  using boost::property_tree::ptree;
212 
214  for (auto& entry : configStore->store()) {
215  std::stringstream ss;
216  std::string str;
217  if (entry.second.empty() == false) {
218  boost::property_tree::json_parser::write_json(ss, entry.second, false);
219  str = ss.str();
220  str.pop_back(); //remove EoL
221  } else {
222  str = entry.second.get_value<std::string>();
223  }
224  std::string configString = fmt::format("[CONFIG];{}={};1;{}", entry.first, str, configStore->provenance(entry.first.c_str())).c_str();
225  mServiceRegistry.get<DriverClient>().tell(configString.c_str());
226  }
227 
228  mConfigRegistry = std::make_unique<ConfigParamRegistry>(std::move(configStore));
229 
230  mExpirationHandlers.clear();
231 
232  auto distinct = DataRelayerHelpers::createDistinctRouteIndex(mSpec.inputs);
233  int i = 0;
234  for (auto& di : distinct) {
235  auto& route = mSpec.inputs[di];
236  if (route.configurator.has_value() == false) {
237  i++;
238  continue;
239  }
240  ExpirationHandler handler{
241  RouteIndex{i++},
242  route.matcher.lifetime,
243  route.configurator->creatorConfigurator(mState, *mConfigRegistry),
244  route.configurator->danglingConfigurator(mState, *mConfigRegistry),
245  route.configurator->expirationConfigurator(mState, *mConfigRegistry)};
246  mExpirationHandlers.emplace_back(std::move(handler));
247  }
248 
249  if (mInit) {
250  InitContext initContext{*mConfigRegistry, mServiceRegistry};
251  mStatefulProcess = mInit(initContext);
252  }
253  mState.inputChannelInfos.resize(mSpec.inputChannels.size());
257  for (size_t ci = 0; ci < mSpec.inputChannels.size(); ++ci) {
258  auto& name = mSpec.inputChannels[ci].name;
259  if (name.find(mSpec.channelPrefix + "from_internal-dpl-clock") == 0) {
260  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
261  } else if (name.find(mSpec.channelPrefix + "from_internal-dpl-ccdb-backend") == 0) {
262  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
263  }
264  }
266  SubscribeToNewTransition("dpl", [loop = mState.loop](fair::mq::Transition t) {
267  if (loop) {
268  uv_async_t handle;
269  int res = uv_async_init(loop, &handle, nullptr);
270  if (res < 0) {
271  LOG(ERROR) << "Unable to initialise subscription";
272  }
273  res = uv_async_send(&handle);
274  if (res < 0) {
275  LOG(ERROR) << "Unable to notify subscription";
276  }
277  LOG(debug) << "State transition requested";
278  }
279  });
280 }

288 void DataProcessingDevice::InitTask()
289 {
290  for (auto& channel : fChannels) {
291  channel.second.at(0).Transport()->SubscribeToRegionEvents([& pendingRegionInfos = mPendingRegionInfos, &regionInfoMutex = mRegionInfoMutex](FairMQRegionInfo info) {
292  std::lock_guard<std::mutex> lock(regionInfoMutex);
293  LOG(debug) << ">>> Region info event" << info.event;
294  LOG(debug) << "id: " << info.id;
295  LOG(debug) << "ptr: " << info.ptr;
296  LOG(debug) << "size: " << info.size;
297  LOG(debug) << "flags: " << info.flags;
298  pendingRegionInfos.push_back(info);
299  });
300  }
301 
302  // Add a signal manager for SIGUSR1 so that we can force
303  // an event from the outside, making sure that the event loop can
304  // be unblocked (e.g. by a quitting DPL driver) even when there
305  // is no data pending to be processed.
306  uv_signal_t* sigusr1Handle = (uv_signal_t*)malloc(sizeof(uv_signal_t));
307  uv_signal_init(mState.loop, sigusr1Handle);
308  uv_signal_start(sigusr1Handle, on_signal_callback, SIGUSR1);
309  // Handle SIGWINCH by simply forcing the event loop to continue.
310  // This will hopefully hide it from FairMQ on linux.
311  uv_signal_t* sigwinchHandle = (uv_signal_t*)malloc(sizeof(uv_signal_t));
312  uv_signal_init(mState.loop, sigwinchHandle);
313  uv_signal_start(sigwinchHandle, on_signal_callback, SIGWINCH);
314 
315  // We add a timer only in case a channel poller is not there.
316  if ((mStatefulProcess != nullptr) || (mStatelessProcess != nullptr)) {
317  for (auto& x : fChannels) {
318  if ((x.first.rfind("from_internal-dpl", 0) == 0) && (x.first.rfind("from_internal-dpl-aod", 0) != 0) && (x.first.rfind("from_internal-dpl-injected", 0)) != 0) {
319  LOG(debug) << x.first << " is an internal channel. Skipping as no input will come from there." << std::endl;
320  continue;
321  }
322  // We only watch receiving sockets.
323  if (x.first.rfind("from_" + mSpec.name + "_", 0) == 0) {
324  LOG(debug) << x.first << " is to send data. Not polling." << std::endl;
325  continue;
326  }
327  // We assume there is always a ZeroMQ socket behind.
328  int zmq_fd = 0;
329  size_t zmq_fd_len = sizeof(zmq_fd);
330  // FIXME: I should probably save those somewhere... ;-)
331  uv_poll_t* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
332  x.second[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
333  if (zmq_fd == 0) {
334  LOG(error) << "Cannot get file descriptor for channel." << x.first;
335  continue;
336  }
337  LOG(debug) << "Polling socket for " << x.second[0].GetName();
338  // FIXME: leak
339  PollerContext* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
340  pCtx->name = strdup(x.first.c_str());
341  pCtx->loop = mState.loop;
342  pCtx->device = this;
343  pCtx->fd = zmq_fd;
344  poller->data = pCtx;
345  uv_poll_init(mState.loop, poller, zmq_fd);
346  uv_poll_start(poller, UV_READABLE | UV_DISCONNECT, &on_socket_polled);
347  mState.activeInputPollers.push_back(poller);
348  }
349  // In case we do not have any input channel and we do not have
350  // any timers or signal watchers we still wake up whenever we can send data to downstream
351  // devices to allow for enumerations.
352  if (mState.activeInputPollers.empty() && mState.activeTimers.empty() && mState.activeSignals.empty()) {
353  for (auto& x : fChannels) {
354  if (x.first.rfind(mSpec.channelPrefix + "from_internal-dpl", 0) == 0) {
355  LOG(debug) << x.first << " is an internal channel. Not polling." << std::endl;
356  continue;
357  }
358  assert(x.first.rfind(mSpec.channelPrefix + "from_" + mSpec.name + "_", 0) == 0);
359  // We assume there is always a ZeroMQ socket behind.
360  int zmq_fd = 0;
361  size_t zmq_fd_len = sizeof(zmq_fd);
362  // FIXME: I should probably save those somewhere... ;-)
363  uv_poll_t* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
364  x.second[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
365  if (zmq_fd == 0) {
366  LOG(error) << "Cannot get file descriptor for channel." << x.first;
367  continue;
368  }
369  LOG(debug) << "Polling socket for " << x.second[0].GetName();
370  // FIXME: leak
371  PollerContext* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
372  pCtx->name = strdup(x.first.c_str());
373  pCtx->loop = mState.loop;
374  pCtx->device = this;
375  pCtx->fd = zmq_fd;
376  poller->data = pCtx;
377  uv_poll_init(mState.loop, poller, zmq_fd);
378  uv_poll_start(poller, UV_WRITABLE, &on_socket_polled);
379  mState.activeOutputPollers.push_back(poller);
380  }
381  }
382  } else {
383  // This is a fake device, so we can request to exit immediately
384  mServiceRegistry.get<ControlService>().readyToQuit(QuitRequest::Me);
385  // A two second timer to stop internal devices which do not want to
386  uv_timer_t* timer = (uv_timer_t*)malloc(sizeof(uv_timer_t));
387  uv_timer_init(mState.loop, timer);
388  uv_timer_start(timer, idle_timer, 2000, 2000);
389  mState.activeTimers.push_back(timer);
390  }
391 
392  // Whenever we InitTask, we consider as if the previous iteration
393  // was successful, so that even if there is no timer or receiving
394  // channel, we can still start an enumeration.
395  mWasActive = true;
396 
397  // We should be ready to run here. Therefore we copy all the
398  // required parts in the DataProcessorContext. Eventually we should
399  // do so on a per thread basis, with fine grained locks.
400  mDataProcessorContexes.resize(1);
401  this->fillContext(mDataProcessorContexes.at(0));
402 }

439 bool DataProcessingDevice::ConditionalRun()
440 {
441  // This will block for the correct delay (or until we get data
442  // on a socket). We also do not block on the first iteration
443  // so that devices which do not have a timer can still start an
444  // enumeration.
445  if (mState.loop) {
446  ZoneScopedN("uv idle");
447  TracyPlot("past activity", (int64_t)mWasActive);
448  mServiceRegistry.get<DriverClient>().flushPending();
449  auto shouldNotWait = (mWasActive &&
450  (mDataProcessorContexes.at(0).state->streaming != StreamingState::Idle) && (mState.activeSignals.empty())) ||
451  (mDataProcessorContexes.at(0).state->streaming == StreamingState::EndOfStreaming);
452  if (NewStatePending()) {
453  shouldNotWait = true;
454  }
455  uv_run(mState.loop, shouldNotWait ? UV_RUN_NOWAIT : UV_RUN_ONCE);
456 
457  // A new state was requested, we exit.
458  if (NewStatePending()) {
459  return false;
460  }
461  }
462 
463  // Notify on the main thread the new region callbacks, making sure
464  // no callback is issued if there is something still processing.
465  {
466  std::lock_guard<std::mutex> lock(mRegionInfoMutex);
467  if (mPendingRegionInfos.empty() == false) {
468  std::vector<FairMQRegionInfo> toBeNotified;
469  toBeNotified.swap(mPendingRegionInfos); // avoid any MT issue.
470  for (auto const& info : toBeNotified) {
471  mServiceRegistry.get<CallbackService>()(CallbackService::Id::RegionInfoCallback, info);
472  }
473  }
474  }
475  // Synchronous execution of the callbacks. This will be moved in the
476  // moved in the on_socket_polled once we have threading in place.
477  uv_work_t handle;
478  handle.data = &mDataProcessorContexes;
479  run_callback(&handle);
480  run_completion(&handle, 0);
481  FrameMark;
482  return true;
483 }

487 void DataProcessingDevice::doPrepare(DataProcessorContext& context)
488 {
489  ZoneScopedN("DataProcessingDevice::doPrepare");
490  context.registry->get<DataProcessingStats>().beginIterationTimestamp = uv_hrtime() / 1000000;
491 
492  *context.wasActive = false;
493  {
494  ZoneScopedN("CallbackService::Id::ClockTick");
495  context.registry->get<CallbackService>()(CallbackService::Id::ClockTick);
496  }
497  // Whether or not we had something to do.
498 
499  // Notice that fake input channels (InputChannelState::Pull) cannot possibly
500  // expect to receive an EndOfStream signal. Thus we do not wait for these
501  // to be completed. In the case of data source devices, as they do not have
502  // real data input channels, they have to signal EndOfStream themselves.
503  context.allDone = std::any_of(context.state->inputChannelInfos.begin(), context.state->inputChannelInfos.end(), [](const auto& info) {
504  return info.state != InputChannelState::Pull;
505  });
506  // Whether or not all the channels are completed
507  for (size_t ci = 0; ci < context.spec->inputChannels.size(); ++ci) {
508  auto& channel = context.spec->inputChannels[ci];
509  auto& info = context.state->inputChannelInfos[ci];
510 
511  if (info.state != InputChannelState::Completed && info.state != InputChannelState::Pull) {
512  context.allDone = false;
513  }
514  if (info.state != InputChannelState::Running) {
515  continue;
516  }
517  int64_t result = -2;
518  if (info.channel == nullptr) {
519  info.channel = &context.device->GetChannel(channel.name, 0);
520  }
521  auto& socket = info.channel->GetSocket();
522  // If we have pending events from a previous iteration,
523  // we do receive in any case.
524  // Otherwise we check if there is any pending event and skip
525  // this channel in case there is none.
526  if (info.hasPendingEvents == 0) {
527  socket.Events(&info.hasPendingEvents);
528  // If we do not read, we can continue.
529  if ((info.hasPendingEvents & 1) == 0) {
530  continue;
531  }
532  }
533  // Notice that there seems to be a difference between the documentation
534  // of zeromq and the observed behavior. The fact that ZMQ_POLLIN
535  // is raised does not mean that a message is immediately available to
536  // read, just that it will be available soon, so the receive can
537  // still return -2. To avoid this we keep receiving on the socket until
538  // we get a message. In order not to overflow the DPL queue we process
539  // one message at the time and we keep track of wether there were more
540  // to process.
541  while (true) {
542  FairMQParts parts;
543  result = info.channel->Receive(parts, 0);
544  if (result >= 0) {
545  DataProcessingDevice::handleData(context, parts, info);
546  // Receiving data counts as activity now, so that
547  // We can make sure we process all the pending
548  // messages without hanging on the uv_run.
549  break;
550  }
551  }
552  // We check once again for pending events, keeping track if this was the
553  // case so that we can immediately repeat this loop and avoid remaining
554  // stuck in uv_run. This is because we will not get notified on the socket
555  // if more events are pending due to zeromq level triggered approach.
556  socket.Events(&info.hasPendingEvents);
557  if (info.hasPendingEvents) {
558  *context.wasActive |= true;
559  }
560  }
561 }

563 void DataProcessingDevice::doRun(DataProcessorContext& context)
564 {
565  auto switchState = [& registry = context.registry,
566  &state = context.state](StreamingState newState) {
567  LOG(debug) << "New state " << (int)newState << " old state " << (int)state->streaming;
568  state->streaming = newState;
569  registry->get<ControlService>().notifyStreamingState(state->streaming);
570  };
571 
572  context.completed->clear();
573  context.completed->reserve(16);
574  *context.wasActive |= DataProcessingDevice::tryDispatchComputation(context, *context.completed);
575  DanglingContext danglingContext{*context.registry};
576 
577  context.registry->preDanglingCallbacks(danglingContext);
578  if (*context.wasActive == false) {
579  context.registry->get<CallbackService>()(CallbackService::Id::Idle);
580  }
581  auto activity = context.relayer->processDanglingInputs(*context.expirationHandlers, *context.registry);
582  *context.wasActive |= activity.expiredSlots > 0;
583 
584  context.completed->clear();
585  *context.wasActive |= DataProcessingDevice::tryDispatchComputation(context, *context.completed);
586 
587  context.registry->postDanglingCallbacks(danglingContext);
588 
589  // If we got notified that all the sources are done, we call the EndOfStream
590  // callback and return false. Notice that what happens next is actually
591  // dependent on the callback, not something which is controlled by the
592  // framework itself.
593  if (context.allDone == true && context.state->streaming == StreamingState::Streaming) {
594  switchState(StreamingState::EndOfStreaming);
595  *context.wasActive = true;
596  }
597 
598  if (context.state->streaming == StreamingState::EndOfStreaming) {
599  // We keep processing data until we are Idle.
600  // FIXME: not sure this is the correct way to drain the queues, but
601  // I guess we will see.
602  while (DataProcessingDevice::tryDispatchComputation(context, *context.completed)) {
603  context.relayer->processDanglingInputs(*context.expirationHandlers, *context.registry);
604  }
605  EndOfStreamContext eosContext{*context.registry, *context.allocator};
606 
607  context.registry->preEOSCallbacks(eosContext);
608  context.registry->get<CallbackService>()(CallbackService::Id::EndOfStream, eosContext);
609  context.registry->postEOSCallbacks(eosContext);
610 
611  for (auto& channel : context.spec->outputChannels) {
612  DataProcessingHelpers::sendEndOfStream(*context.device, channel);
613  }
614  // This is needed because the transport is deleted before the device.
615  context.relayer->clear();
616  switchState(StreamingState::Idle);
617  *context.wasActive = true;
618  return;
619  }

621  return;

633 void DataProcessingDevice::handleData(DataProcessorContext& context, FairMQParts& parts, InputChannelInfo& info)
634 {
635  ZoneScopedN("DataProcessingDevice::handleData");
636  assert(context.spec->inputChannels.empty() == false);
637  assert(parts.Size() > 0);
638 
639  // Initial part. Let's hide all the unnecessary and have
640  // simple lambdas for each of the steps I am planning to have.
641  assert(!context.spec->inputs.empty());
642 
643  enum struct InputType {
644  Invalid,
645  Data,
646  SourceInfo
647  };
648 
649  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
650  // and we do a few stats. We bind parts as a lambda captured variable, rather
651  // than an input, because we do not want the outer loop actually be exposed
652  // to the implementation details of the messaging layer.
653  auto getInputTypes = [&stats = context.registry->get<DataProcessingStats>(),
654  &parts, &info, &context]() -> std::optional<std::vector<InputType>> {
655  stats.inputParts = parts.Size();
656 
657  TracyPlot("messages received", (int64_t)parts.Size());
658  if (parts.Size() % 2) {
659  return std::nullopt;
660  }
661  std::vector<InputType> results(parts.Size() / 2, InputType::Invalid);
662 
663  for (size_t hi = 0; hi < parts.Size() / 2; ++hi) {
664  auto pi = hi * 2;
665  auto sih = o2::header::get<SourceInfoHeader*>(parts.At(pi)->GetData());
666  if (sih) {
667  info.state = sih->state;
668  results[hi] = InputType::SourceInfo;
669  *context.wasActive = true;
670  continue;
671  }
672  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
673  if (!dh) {
674  results[hi] = InputType::Invalid;
675  LOGP(error, "Header is not a DataHeader?");
676  continue;
677  }
678  if (dh->payloadSize != parts.At(pi + 1)->GetSize()) {
679  results[hi] = InputType::Invalid;
680  LOGP(error, "DataHeader payloadSize mismatch");
681  continue;
682  }
683  TracyPlot("payload size", (int64_t)dh->payloadSize);
684  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
685  TracyAlloc(parts.At(pi + 1)->GetData(), parts.At(pi + 1)->GetSize());
686  if (!dph) {
687  results[hi] = InputType::Invalid;
688  LOGP(error, "Header stack does not contain DataProcessingHeader");
689  continue;
690  }
691  results[hi] = InputType::Data;
692  }
693  return results;
694  };
695 
696  auto reportError = [&registry = *context.registry, &context](const char* message) {
697  registry.get<DataProcessingStats>().errorCount++;
698  };
699 
700  auto handleValidMessages = [&parts, &context = context, &relayer = *context.relayer, &reportError](std::vector<InputType> const& types) {
701  // We relay execution to make sure we have a complete set of parts
702  // available.
703  for (size_t pi = 0; pi < (parts.Size() / 2); ++pi) {
704  switch (types[pi]) {
705  case InputType::Data: {
706  auto headerIndex = 2 * pi;
707  auto payloadIndex = 2 * pi + 1;
708  assert(payloadIndex < parts.Size());
709  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
710  std::move(parts.At(payloadIndex)));
711  if (relayed == DataRelayer::WillNotRelay) {
712  reportError("Unable to relay part.");
713  }
714  } break;
715  case InputType::SourceInfo: {
716  *context.wasActive = true;
717 
718  } break;
719  case InputType::Invalid: {
720  reportError("Invalid part found.");
721  } break;
722  }
723  }
724  };
725 
726  // Second part. This is the actual outer loop we want to obtain, with
727  // implementation details which can be read. Notice how most of the state
728  // is actually hidden. For example we do not expose what "input" is. This
729  // will allow us to keep the same toplevel logic even if the actual meaning
730  // of input is changed (for example we might move away from multipart
731  // messages). Notice also that we need to act diffently depending on the
732  // actual CompletionOp we want to perform. In particular forwarding inputs
733  // also gets rid of them from the cache.
734  auto inputTypes = getInputTypes();
735  if (bool(inputTypes) == false) {
736  reportError("Parts should come in couples. Dropping it.");
737  return;
738  }
739  handleValidMessages(*inputTypes);
740  return;
741 }

784 bool DataProcessingDevice::tryDispatchComputation(DataProcessorContext& context, std::vector<DataRelayer::RecordAction>& completed)
785 {
786  ZoneScopedN("DataProcessingDevice::tryDispatchComputation");
787  // This is the actual hidden state for the outer loop. In case we decide we
788  // want to support multithreaded dispatching of operations, I can simply
789  // move these to some thread local store and the rest of the lambdas
790  // should work just fine.
791  std::vector<MessageSet> currentSetOfInputs;
792 
793  auto reportError = [&registry = *context.registry, &context](const char* message) {
794  registry.get<DataProcessingStats>().errorCount++;
795  };
796 
797  // For the moment we have a simple "immediately dispatch" policy for stuff
798  // in the cache. This could be controlled from the outside e.g. by waiting
799  // for a few sets of inputs to arrive before we actually dispatch the
800  // computation, however this can be defined at a later stage.
801  auto canDispatchSomeComputation = [&completed,
802  &relayer = context.relayer]() -> bool {
803  relayer->getReadyToProcess(completed);
804  return completed.empty() == false;
805  };
806 
807  // We use this to get a list with the actual indexes in the cache which
808  // indicate a complete set of inputs. Notice how I fill the completed
809  // vector and return it, so that I can have a nice for loop iteration later
810  // on.
811  auto getReadyActions = [& relayer = context.relayer,
812  &completed,
813  &stats = context.registry->get<DataProcessingStats>()]() -> std::vector<DataRelayer::RecordAction> {
814  stats.pendingInputs = (int)relayer->getParallelTimeslices() - completed.size();
815  stats.incomplete = completed.empty() ? 1 : 0;
816  return completed;
817  };
818 
819  // This is needed to convert from a pair of pointers to an actual DataRef
820  // and to make sure the ownership is moved from the cache in the relayer to
821  // the execution.
822  auto fillInputs = [& relayer = context.relayer,
823  &spec = context.spec,
824  &currentSetOfInputs](TimesliceSlot slot) -> InputRecord {
825  currentSetOfInputs = std::move(relayer->getInputsForTimeslice(slot));
826  auto getter = [&currentSetOfInputs](size_t i, size_t partindex) -> DataRef {
827  if (currentSetOfInputs[i].size() > partindex) {
828  return DataRef{nullptr,
829  static_cast<char const*>(currentSetOfInputs[i].at(partindex).header->GetData()),
830  static_cast<char const*>(currentSetOfInputs[i].at(partindex).payload->GetData())};
831  }
832  return DataRef{nullptr, nullptr, nullptr};
833  };
834  auto nofPartsGetter = [&currentSetOfInputs](size_t i) -> size_t {
835  return currentSetOfInputs[i].size();
836  };
837  InputSpan span{getter, nofPartsGetter, currentSetOfInputs.size()};
838  return InputRecord{spec->inputs, std::move(span)};
839  };
840 
841  // I need a preparation step which gets the current timeslice id and
842  // propagates it to the various contextes (i.e. the actual entities which
843  // create messages) because the messages need to have the timeslice id into
844  // it.
845  auto prepareAllocatorForCurrentTimeSlice = [& timingInfo = context.timingInfo,
846  &relayer = context.relayer](TimesliceSlot i) {
847  ZoneScopedN("DataProcessingDevice::prepareForCurrentTimeslice");
848  auto timeslice = relayer->getTimesliceForSlot(i);
849  timingInfo->timeslice = timeslice.value;
850  timingInfo->tfCounter = relayer->getFirstTFCounterForSlot(i);
851  timingInfo->firstTFOrbit = relayer->getFirstTFOrbitForSlot(i);
852  };
853 
854  // When processing them, timers will have to be cleaned up
855  // to avoid double counting them.
856  // This was actually the easiest solution we could find for
857  // O2-646.
858  auto cleanTimers = [&currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
859  assert(record.size() == currentSetOfInputs.size());
860  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
861  DataRef input = record.getByPos(ii);
862  if (input.spec->lifetime != Lifetime::Timer) {
863  continue;
864  }
865  if (input.header == nullptr) {
866  continue;
867  }
868  // This will hopefully delete the message.
869  currentSetOfInputs[ii].clear();
870  }
871  };
872 
873  // Function to cleanup record. For the moment we
874  // simply use it to keep track of input messages
875  // which are not needed, to display them in the GUI.
876  auto cleanupRecord = [](InputRecord& record) {
877  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
878  DataRef input = record.getByPos(ii);
879  if (input.header == nullptr) {
880  continue;
881  }
882  auto sih = o2::header::get<SourceInfoHeader*>(input.header);
883  if (sih) {
884  continue;
885  }
886 
887  auto dh = o2::header::get<DataHeader*>(input.header);
888  if (!dh) {
889  continue;
890  }
891  TracyFree(input.payload);
892  }
893  };
894 
895  // This is how we do the forwarding, i.e. we push
896  // the inputs which are shared between this device and others
897  // to the next one in the daisy chain.
898  // FIXME: do it in a smarter way than O(N^2)
899  auto forwardInputs = [&reportError,
900  &spec = context.spec,
901  &device = context.device, &currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
902  ZoneScopedN("forward inputs");
903  assert(record.size() == currentSetOfInputs.size());
904  // we collect all messages per forward in a map and send them together
905  std::unordered_map<std::string, FairMQParts> forwardedParts;
906  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
907  DataRef input = record.getByPos(ii);
908 
909  // If is now possible that the record is not complete when
910  // we forward it, because of a custom completion policy.
911  // this means that we need to skip the empty entries in the
912  // record for being forwarded.
913  if (input.header == nullptr) {
914  continue;
915  }
916  auto sih = o2::header::get<SourceInfoHeader*>(input.header);
917  if (sih) {
918  continue;
919  }
920 
921  auto dh = o2::header::get<DataHeader*>(input.header);
922  if (!dh) {
923  reportError("Header is not a DataHeader?");
924  continue;
925  }
926  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
927  if (!dph) {
928  reportError("Header stack does not contain DataProcessingHeader");
929  continue;
930  }
931 
932  for (auto& part : currentSetOfInputs[ii]) {
933  for (auto const& forward : spec->forwards) {
934  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification) == false || (dph->startTime % forward.maxTimeslices) != forward.timeslice) {
935  continue;
936  }
937  auto& header = part.header;
938  auto& payload = part.payload;
939 
940  if (header.get() == nullptr) {
941  // FIXME: this should not happen, however it's actually harmless and
942  // we can simply discard it for the moment.
943  // LOG(ERROR) << "Missing header! " << dh->dataDescription.as<std::string>();
944  continue;
945  }
946  auto fdph = o2::header::get<DataProcessingHeader*>(header.get()->GetData());
947  if (fdph == nullptr) {
948  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
949  continue;
950  }
951  auto fdh = o2::header::get<DataHeader*>(header.get()->GetData());
952  if (fdh == nullptr) {
953  LOG(ERROR) << "Forwarded data does not have a DataHeader";
954  continue;
955  }
956 
957  forwardedParts[forward.channel].AddPart(std::move(header));
958  forwardedParts[forward.channel].AddPart(std::move(payload));
959  }
960  }
961  }
962  for (auto& [channelName, channelParts] : forwardedParts) {
963  if (channelParts.Size() == 0) {
964  continue;
965  }
966  assert(channelParts.Size() % 2 == 0);
967  assert(o2::header::get<DataProcessingHeader*>(channelParts.At(0)->GetData()));
968  // in DPL we are using subchannel 0 only
969  device->Send(channelParts, channelName, 0);
970  }
971  };
972 
973  auto switchState = [& control = context.registry->get<ControlService>(),
974  &state = context.state](StreamingState newState) {
975  state->streaming = newState;
976  control.notifyStreamingState(state->streaming);
977  };
978 
979  if (canDispatchSomeComputation() == false) {
980  return false;
981  }

983  auto postUpdateStats = [& stats = context.registry->get<DataProcessingStats>()](DataRelayer::RecordAction const& action, InputRecord const& record, uint64_t tStart) {
984  std::atomic_thread_fence(std::memory_order_release);
985  for (size_t ai = 0; ai != record.size(); ai++) {
986  auto cacheId = action.slot.index * record.size() + ai;
987  auto state = record.isValid(ai) ? 3 : 0;
988  update_maximum(stats.statesSize, cacheId + 1);
989  assert(cacheId < DataProcessingStats::MAX_RELAYER_STATES);
990  stats.relayerState[cacheId].store(state);
991  }
992  uint64_t tEnd = uv_hrtime();
993  stats.lastElapsedTimeMs = tEnd - tStart;
994  stats.lastTotalProcessedSize = calculateTotalInputRecordSize(record);
995  stats.lastLatency = calculateInputRecordLatency(record, tStart);
996  };
997 
998  auto preUpdateStats = [& stats = context.registry->get<DataProcessingStats>()](DataRelayer::RecordAction const& action, InputRecord const& record, uint64_t tStart) {
999  std::atomic_thread_fence(std::memory_order_release);
1000  for (size_t ai = 0; ai != record.size(); ai++) {
1001  auto cacheId = action.slot.index * record.size() + ai;
1002  auto state = record.isValid(ai) ? 2 : 0;
1003  update_maximum(stats.statesSize, cacheId + 1);
1004  assert(cacheId < DataProcessingStats::MAX_RELAYER_STATES);
1005  stats.relayerState[cacheId].store(state);
1006  }
1007  };
1008 
1009  for (auto action : getReadyActions()) {
1010  if (action.op == CompletionPolicy::CompletionOp::Wait) {
1011  continue;
1012  }
1013 
1014  prepareAllocatorForCurrentTimeSlice(TimesliceSlot{action.slot});
1015  InputRecord record = fillInputs(action.slot);
1016  ProcessingContext processContext{record, *context.registry, *context.allocator};
1017  {
1018  ZoneScopedN("service pre processing");
1019  context.registry->preProcessingCallbacks(processContext);
1020  }
1021  if (action.op == CompletionPolicy::CompletionOp::Discard) {
1022  context.registry->postDispatchingCallbacks(processContext);
1023  if (context.spec->forwards.empty() == false) {
1024  forwardInputs(action.slot, record);
1025  continue;
1026  }
1027  }
1028 
1029  uint64_t tStart = uv_hrtime();
1030  preUpdateStats(action, record, tStart);
1031  try {
1032  if (context.state->quitRequested == false) {
1033 
1034  if (*context.statefulProcess) {
1035  ZoneScopedN("statefull process");
1036  (*context.statefulProcess)(processContext);
1037  }
1038  if (*context.statelessProcess) {
1039  ZoneScopedN("stateless process");
1040  (*context.statelessProcess)(processContext);
1041  }
1042 
1043  {
1044  ZoneScopedN("service post processing");
1045  context.registry->postProcessingCallbacks(processContext);
1046  }
1047  }
1048  } catch (std::exception& ex) {
1049  ZoneScopedN("error handling");
1053  auto e = runtime_error(ex.what());
1054  (*context.errorHandling)(e, record);
1055  } catch (o2::framework::RuntimeErrorRef e) {
1056  ZoneScopedN("error handling");
1057  (*context.errorHandling)(e, record);
1058  }
1059 
1060  postUpdateStats(action, record, tStart);
1061  // We forward inputs only when we consume them. If we simply Process them,
1062  // we keep them for next message arriving.
1063  if (action.op == CompletionPolicy::CompletionOp::Consume) {
1064  context.registry->postDispatchingCallbacks(processContext);
1065  if (context.spec->forwards.empty() == false) {
1066  forwardInputs(action.slot, record);
1067  }
1068 #ifdef TRACY_ENABLE
1069  cleanupRecord(record);
1070 #endif
1071  } else if (action.op == CompletionPolicy::CompletionOp::Process) {
1072  cleanTimers(action.slot, record);
1073  }
1074  }
1075  // We now broadcast the end of stream if it was requested
1076  if (context.state->streaming == StreamingState::EndOfStreaming) {
1077  for (auto& channel : context.spec->outputChannels) {
1078  DataProcessingHelpers::sendEndOfStream(*context.device, channel);
1079  }
1080  switchState(StreamingState::Idle);
1081  }
1082 
1083  return true;
1084 }
