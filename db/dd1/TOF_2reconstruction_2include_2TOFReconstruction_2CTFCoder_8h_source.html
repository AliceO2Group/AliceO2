67 template <typename VEC>
68 void CTFCoder::encode(VEC& buff, const gsl::span<const ReadoutWindowData>& rofRecVec, const gsl::span<const Digit>& cdigVec, const gsl::span<const uint8_t>& pattVec)
69 {
70  using MD = o2::ctf::Metadata::OptStore;
71  // what to do which each field: see o2::ctd::Metadata explanation
72  constexpr MD optField[CTF::getNBlocks()] = {
73  MD::EENCODE, //BLCbcIncROF
74  MD::EENCODE, //BLCorbitIncROF
75  MD::EENCODE, //BLCndigROF
76  MD::EENCODE, //BLCndiaROF
77  MD::EENCODE, //BLCndiaCrate
78  MD::EENCODE, //BLCtimeFrameInc
79  MD::EENCODE, //BLCtimeTDCInc
80  MD::EENCODE, //BLCstripID
81  MD::EENCODE, //BLCchanInStrip
82  MD::EENCODE, //BLCtot
83  MD::EENCODE, //BLCpattMap
84  };
85  CompressedInfos cc;
86  compress(cc, rofRecVec, cdigVec, pattVec);
87  // book output size with some margin
88  auto szIni = estimateCompressedSize(cc);
89  buff.resize(szIni);
90 
91  auto ec = CTF::create(buff);
92  using ECB = CTF::base;
93 
94  ec->setHeader(cc.header);
95  ec->getANSHeader().majorVersion = 0;
96  ec->getANSHeader().minorVersion = 1;
97  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
98 #define ENCODETOF(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
99  // clang-format off
100  ENCODETOF(cc.bcIncROF, CTF::BLCbcIncROF, 0);
101  ENCODETOF(cc.orbitIncROF, CTF::BLCorbitIncROF, 0);
102  ENCODETOF(cc.ndigROF, CTF::BLCndigROF, 0);
103  ENCODETOF(cc.ndiaROF, CTF::BLCndiaROF, 0);
104  ENCODETOF(cc.ndiaCrate, CTF::BLCndiaCrate, 0);
105  ENCODETOF(cc.timeFrameInc, CTF::BLCtimeFrameInc, 0);
106  ENCODETOF(cc.timeTDCInc, CTF::BLCtimeTDCInc, 0);
107  ENCODETOF(cc.stripID, CTF::BLCstripID, 0);
108  ENCODETOF(cc.chanInStrip, CTF::BLCchanInStrip, 0);
109  ENCODETOF(cc.tot, CTF::BLCtot, 0);
110  ENCODETOF(cc.pattMap, CTF::BLCpattMap, 0);
111  // clang-format on
112  CTF::get(buff.data())->print(getPrefix());
113 }

118 void CTFCoder::decode(const CTF::base& ec, VROF& rofRecVec, VDIG& cdigVec, VPAT& pattVec)

143 template <typename VROF, typename VDIG, typename VPAT>
144 void CTFCoder::decompress(const CompressedInfos& cc, VROF& rofRecVec, VDIG& cdigVec, VPAT& pattVec)
145 {
146  rofRecVec.resize(cc.header.nROFs);
147  cdigVec.resize(cc.header.nDigits);
148  pattVec.resize(cc.header.nPatternBytes);
149  std::vector<Digit> digCopy;
150 
151  o2::InteractionRecord prevIR(cc.header.firstBC, cc.header.firstOrbit);
152  uint32_t firstEntry = 0, digCount = 0, stripCount = 0, ndiagnostic = 0;
153  for (uint32_t irof = 0; irof < cc.header.nROFs; irof++) {
154  // restore ROFRecord
155  auto& rofRec = rofRecVec[irof];
156  if (cc.orbitIncROF[irof]) { // new orbit
157  prevIR.bc = cc.bcIncROF[irof]; // bcInc has absolute meaning
158  prevIR.orbit += cc.orbitIncROF[irof];
159  } else {
160  prevIR.bc += cc.bcIncROF[irof];
161  }
162  rofRec.setBCData(prevIR);
163  rofRec.setFirstEntry(firstEntry);
164  rofRec.setNEntries(cc.ndigROF[irof]);
165  rofRec.setFirstEntryDia(ndiagnostic);
166  rofRec.setNEntriesDia(cc.ndiaROF[irof]);
167  for (int icrate = 0; icrate < 72; icrate++) {
168  rofRec.setDiagnosticInCrate(icrate, cc.ndiaCrate[irof * 72 + icrate] - 1); // -1 because number were traslated since (-1 means crate not available)
169  }
170  firstEntry += cc.ndigROF[irof];
171  ndiagnostic += cc.ndiaROF[irof];
172 
173  if (!cc.ndigROF[irof]) {
174  continue;
175  }
176 
177  // restore hit data
178  uint ctimeframe = 0;
179  uint ctdc = 0;
180 
181  int firstDig = digCount;
182 
183  int64_t BCrow = prevIR.toLong();
184 
185  digCopy.resize(cc.ndigROF[irof]);
186  for (uint32_t idig = 0; idig < cc.ndigROF[irof]; idig++) {
187  auto& digit = digCopy[idig]; //cdigVec[digCount];
188  LOGF(DEBUG, "%d) TF=%d, TDC=%d, STRIP=%d, CH=%d", idig, cc.timeFrameInc[digCount], cc.timeTDCInc[digCount], cc.stripID[digCount], cc.chanInStrip[digCount]);
189  if (cc.timeFrameInc[digCount]) { // new time frame
190  ctdc = cc.timeTDCInc[digCount];
191  ctimeframe += cc.timeFrameInc[digCount];
192  } else {
193  ctdc += cc.timeTDCInc[digCount];
194  }
195  LOGF(DEBUG, "BC=%ld, TDC=%d, TOT=%d, CH=%d", uint32_t(ctimeframe) * 64 + ctdc / 1024 + BCrow, ctdc % 1024, cc.tot[digCount], uint32_t(cc.stripID[digCount]) * 96 + cc.chanInStrip[digCount]);
196 
197  digit.setBC(uint32_t(ctimeframe) * 64 + ctdc / 1024 + BCrow);
198  digit.setTDC(ctdc % 1024);
199  digit.setTOT(cc.tot[digCount]);
200  digit.setChannel(uint32_t(cc.stripID[digCount]) * 96 + cc.chanInStrip[digCount]);
201 
202  digCount++;
203  }
204 
205  // sort digits according to strip number within the ROF
206  std::sort(digCopy.begin(), digCopy.end(),
207  [](o2::tof::Digit a, o2::tof::Digit b) {
208  int str1 = a.getChannel() / Geo::NPADS;
209  int str2 = b.getChannel() / Geo::NPADS;
210  if (str1 == str2) {
211  return (a.getOrderingKey() < b.getOrderingKey());
212  }
213  return (str1 < str2);
214  });
215 
216  // fill digits, once sorted, of rof in digit vector
217  for (uint32_t idig = 0; idig < digCopy.size(); idig++) {
218  cdigVec[firstDig + idig] = digCopy[idig];
219  }
220 
221  digCopy.clear();
222  }
223  // explicit patterns
224  memcpy(pattVec.data(), cc.pattMap.data(), cc.header.nPatternBytes); // RSTODO use swap?
225  assert(digCount == cc.header.nDigits);
226 
227  if (digCount != cc.header.nDigits) {
228  LOG(ERROR) << "expected " << cc.header.nDigits << " but counted " << digCount << " in ROFRecords";
229  throw std::runtime_error("mismatch between expected and counter number of digits");
230  }
231 }
