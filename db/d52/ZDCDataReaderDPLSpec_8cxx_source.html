32 void ZDCDataReaderDPLSpec::run(ProcessingContext& pc)
33 {
34  DPLRawParser parser(pc.inputs());
35  mRawReader.clear();
36 
37  //>> update Time-dependent CCDB stuff, at the moment set the moduleconfig only once
38  if (!mRawReader.getModuleConfig()) {
39  long timeStamp = 0;
40  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
41  mgr.setTimestamp(timeStamp);
42  auto moduleConfig = mgr.get<o2::zdc::ModuleConfig>(o2::zdc::CCDBPathConfigModule);
43  if (!moduleConfig) {
44  LOG(FATAL) << "Cannot module configuratio for timestamp " << timeStamp;
45  return;
46  } else {
47  LOG(INFO) << "Loaded module configuration for timestamp " << timeStamp;
48  }
49  mRawReader.setModuleConfig(moduleConfig);
50  mRawReader.setTriggerMask();
51  mRawReader.mVerifyTrigger = mVerifyTrigger;
52  LOG(INFO) << "Check of trigger condition during conversion is " << (mVerifyTrigger ? "ON" : "OFF");
53  }
54  uint64_t count = 0;
55  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
56  //Proccessing each page
57  count++;
58  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
59  gsl::span<const uint8_t> payload(it.data(), it.size());
60  mRawReader.processBinaryData(payload, rdhPtr->linkID);
61  }
62  LOG(INFO) << "Pages: " << count;
63  mRawReader.accumulateDigits();
64  mRawReader.makeSnapshot(pc);
65 }
