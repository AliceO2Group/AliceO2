50 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)

53  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
54  // write the configuration used for the workflow
55  o2::conf::ConfigurableParam::writeINI("o2matchtpcits-workflow_configuration.ini");
56 
57  auto useFT0 = configcontext.options().get<bool>("use-ft0");
58  auto useMC = !configcontext.options().get<bool>("disable-mc");
59  auto disableRootInp = configcontext.options().get<bool>("disable-root-input");
60  auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
61  auto calib = configcontext.options().get<bool>("produce-calibration-data");
62  return std::move(o2::globaltracking::getMatchTPCITSWorkflow(useFT0, useMC, disableRootInp, disableRootOut, calib));
63 }
