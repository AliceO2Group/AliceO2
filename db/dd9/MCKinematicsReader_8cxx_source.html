33 void MCKinematicsReader::initIndexedTrackRefs(std::vector<o2::TrackReference>& refs, o2::dataformats::MCTruthContainer<o2::TrackReference>& indexedrefs) const
34 {
35  // sort trackrefs according to track index then according to track length
36  std::sort(refs.begin(), refs.end(), [](const o2::TrackReference& a, const o2::TrackReference& b) {
37  if (a.getTrackID() == b.getTrackID()) {
38  return a.getLength() < b.getLength();
39  }
40  return a.getTrackID() < b.getTrackID();
41  });
42 
43  // make final indexed container for track references
44  indexedrefs.clear();
45  for (auto& ref : refs) {
46  if (ref.getTrackID() >= 0) {
47  indexedrefs.addElement(ref.getTrackID(), ref);
48  }
49  }
50 }

109 void MCKinematicsReader::loadTrackRefsForSource(int source) const
110 {
111  auto chain = mInputChains[source];
112  if (chain) {
113  // todo: get name from NameConfig
114  auto br = chain->GetBranch("TrackRefs");
115  if (br) {
116  std::vector<o2::TrackReference>* refs = nullptr;
117  br->SetAddress(&refs);
118  mIndexedTrackRefs[source].resize(br->GetEntries());
119  for (int event = 0; event < br->GetEntries(); ++event) {
120  br->GetEntry(event);
121  if (refs) {
122  // we convert the original flat vector into an indexed structure
123  initIndexedTrackRefs(*refs, mIndexedTrackRefs[source][event]);
124  delete refs;
125  refs = nullptr;
126  }
127  }
128  } else {
129  LOG(WARN) << "TrackRefs branch not found";
130  }
131  }
132 }
