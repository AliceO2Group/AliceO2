36 class MCHChannelCalibDevice : public o2::framework::Task
37 {
38  public:
39  explicit MCHChannelCalibDevice() = default;
40 
41  void init(o2::framework::InitContext& ic) final
42  {
43  float pedThreshold = ic.options().get<float>("pedestal-threshold");
44  float noiseThreshold = ic.options().get<float>("noise-threshold");
45  mCalibrator = std::make_unique<o2::mch::calibration::MCHChannelCalibrator>(pedThreshold, noiseThreshold);
46 
47  int slotL = ic.options().get<int>("tf-per-slot");
48  int delay = ic.options().get<int>("max-delay");
49  mCalibrator->setSlotLength(slotL);
50  mCalibrator->setMaxSlotsDelay(delay);
51  mCalibrator->setUpdateAtTheEndOfRunOnly();
52  }
53 
54  void run(o2::framework::ProcessingContext& pc) final
55  {
56  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
57 
58  auto data = pc.inputs().get<gsl::span<o2::mch::calibration::PedestalDigit>>("input");
59  mCalibrator->process(tfcounter, data);
60  }
61 
62  void endOfStream(o2::framework::EndOfStreamContext& ec) final
63  {
64  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
65  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
66  mCalibrator->endOfStream();
67  LOG(INFO) << "End of stream reached, sending output to CCDB";
68  sendOutput(ec.outputs());
69  }

71  private:
72  std::unique_ptr<o2::mch::calibration::MCHChannelCalibrator> mCalibrator;

74  //________________________________________________________________
75  void sendOutput(DataAllocator& output)
76  {
77  auto createBuffer = [&](auto& vec, size_t& size) {
78  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
79  char* buf = nullptr;
80  if (size > 0) {
81  buf = (char*)malloc(size);
82  if (buf) {
83  char* p = buf;
84  size_t sizeofElement = sizeof(*(vec.begin()));
85  for (auto& element : vec) {
86  memcpy(p, &element, sizeofElement);
87  p += sizeofElement;
88  }
89  }
90  }
91  return buf;
92  };
93 
94  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
95  // TODO in principle, this routine is generic, can be moved to Utils.h
96  using clbUtils = o2::calibration::Utils;
97  const auto& payload = mCalibrator->getBadChannelsVector();
98  auto& info = mCalibrator->getBadChannelsInfo(); // use non-const version as we update it
99  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
100  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
101  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
102  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
103  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info); // root-serialized
104 
105  size_t pedestalsSize;
106  char* pedestalsBuffer = createBuffer(mCalibrator->getPedestalsVector(), pedestalsSize);
107  auto freefct = [](void* data, void*) { free(data); };
108  output.adoptChunk(Output{"MCH", "PEDESTALS", 0}, pedestalsBuffer, pedestalsSize, freefct, nullptr);
109 
110  mCalibrator->initOutput(); // reset the outputs once they are already sent
111  }
112 };

120 DataProcessorSpec getMCHChannelCalibDeviceSpec()
121 {
122  constexpr int64_t INFINITE_TF = 0xffffffffffffffff;
123  using device = o2::mch::calibration::MCHChannelCalibDevice;
124  using clbUtils = o2::calibration::Utils;
125 
126  std::vector<OutputSpec> outputs;
127  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload});
128  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo});
129  outputs.emplace_back(OutputSpec{"MCH", "PEDESTALS", 0, Lifetime::Timeframe});
130 
131  std::vector<InputSpec> inputs;
132  inputs.emplace_back("input", "MCH", "PDIGITS");
133 
134  return DataProcessorSpec{
135  "calib-mchchannel-calibration",
136  inputs,
137  outputs,
138  AlgorithmSpec{adaptFromTask<device>()},
139  Options{
140  {"tf-per-slot", VariantType::Int64, INFINITE_TF - 10, {"number of TFs per calibration time slot"}},
141  {"max-delay", VariantType::Int, 1, {"number of slots in past to consider"}},
142  {"pedestal-threshold", VariantType::Float, 200.0f, {"maximum allowed pedestal value"}},
143  {"noise-threshold", VariantType::Float, 2.0f, {"maximum allowed noise value"}},
144  {"ccdb-path", VariantType::String, "http://ccdb-test.cern.ch:8080", {"Path to CCDB"}}}};
145 }
