70 void HmpidCoder2::setRDHFields(int eq)
71 {
72  int st, en;
73  uint32_t wr = (mBusyTime << 9) | ((mHmpidErrorFlag & 0x01F) << 4) | (mHmpidFrwVersion & 0x0F);
74  st = (eq < 0 || eq >= Geo::MAXEQUIPMENTS) ? 0 : eq;
75  en = (eq < 0 || eq >= Geo::MAXEQUIPMENTS) ? Geo::MAXEQUIPMENTS : eq + 1;
76  for (int l = st; l < en; l++) {
77  o2::raw::RawFileWriter::LinkData& link = mWriter.getLinkWithSubSpec(mTheRFWLinks[l]);
78  RDHAny* RDHptr = link.getLastRDH();
79  if (RDHptr != nullptr) {
80  o2::raw::RDHUtils::setDetectorField(RDHptr, wr);
81  }
82  }
83  return;
84 }

105 void HmpidCoder2::fillTheOutputBuffer(uint32_t* padMap)
106 {
107  uint32_t rowMarker, segMarker, eoeMarker, padWord;
108  uint32_t rowSize;
109  uint32_t ptr = 0;
110  int pads[Geo::MAXEQUIPMENTS];
111  int padsCount;
112  int segSize;
113 
114  for (int i = 0; i < Geo::MAXEQUIPMENTS; i++) {
115  mEventSizePerEquipment[i] = 0;
116  }
117 
118  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
119  int startPtr = ptr;
120  padsCount = 0;
121  for (int s = 1; s <= Geo::N_SEGMENTS; s++) {
122  segSize = 0;
123  for (int c = 1; c <= Geo::N_COLXSEGMENT; c++) {
124  // ---- Pre-calculate the size of each column
125  for (int j = 0; j < Geo::N_DILOGICS; j++) {
126  pads[j] = 0;
127  }
128  rowSize = 0;
129  for (int j = 0; j < Geo::N_DILOGICS; j++) {
130  for (int k = 0; k < Geo::N_CHANNELS; k++) {
131  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), j, k);
132  if (padMap[idx] > 0) {
133  pads[j]++;
134  rowSize++;
135  padsCount++;
136  }
137  }
138  }
139  rowSize += Geo::N_DILOGICS;
140  segSize += (rowSize + 1);
141  rowMarker = 0x000036A8 | ((rowSize << 16) & 0x03ff0000);
142 
143  // ---- fills the Payload Buffer
144  mPayloadBufferPtr[ptr++] = rowMarker;
145  int col = (s - 1) * Geo::N_COLXSEGMENT + c;
146  for (int d = 1; d <= Geo::N_DILOGICS; d++) {
147  for (int p = 0; p < Geo::N_CHANNELS; p++) {
148  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), (d - 1), p);
149  if (padMap[idx] > 0) {
150  padWord = ((col << 22) & 0x07c00000) | ((d << 18) & 0x003C0000) | ((p << 12) & 0x0003F000) | (padMap[idx] & 0x00000FFF);
151  mPayloadBufferPtr[ptr++] = padWord;
152  }
153  }
154  eoeMarker = 0x08000080 | ((col << 22) & 0x07c00000) | (d << 18 & 0x003C0000) | (pads[d - 1] & 0x0000007F);
155  mPayloadBufferPtr[ptr++] = eoeMarker;
156  }
157  }
158  segSize += 1;
159  segMarker = 0xAB000000 | ((segSize << 8) & 0x000fff00) | (s & 0x0000000F);
160  mPayloadBufferPtr[ptr++] = segMarker;
161  }
162  mPadsCoded += padsCount;
163  mEventPadsPerEquipment[eq] = padsCount;
164  mEventSizePerEquipment[eq] = ptr - startPtr;
165  }
166  return;
167 }

175 void HmpidCoder2::writePaginatedEvent(uint32_t orbit, uint16_t bc)
176 {
177  uint32_t* ptrStartEquipment = mPayloadBufferPtr;
178  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
179  int EventSize = mEventSizePerEquipment[eq];
180  LOG(DEBUG) << "writePaginatedEvent() Eq=" << eq << " Size:" << EventSize << " Pads:" << mEventPadsPerEquipment[eq] << " Orbit:" << orbit << " BC:" << bc;
181  if (mEventPadsPerEquipment[eq] > 0 || !mSkipEmptyEvents) { // Skips the Events with 0 Pads
182  mWriter.addData(ReadOut::FeeId(eq), ReadOut::CruId(eq), ReadOut::LnkId(eq), 0, {bc, orbit}, gsl::span<char>(reinterpret_cast<char*>(ptrStartEquipment), EventSize * sizeof(uint32_t)));
183  // We fill the fields !
184  // TODO: we can fill the detector field with Simulated Data
185  setDetectorSpecificFields(0.000001 * EventSize);
186  setRDHFields(eq);
187  }
188  ptrStartEquipment += EventSize;
189  }
190  return;
191 }

217 void HmpidCoder2::codeEventChunkDigits(std::vector<Digit>& digits, Trigger ir)
218 {
219  int eq, col, dil, cha, mo, x, y, idx;
220  uint32_t orbit = ir.getOrbit();
221  uint16_t bc = ir.getBc();
222 
223  int padsCount = 0;
224  LOG(INFO) << "Manage chunk Orbit :" << orbit << " BC:" << bc << " Digits size:" << digits.size();
225  for (o2::hmpid::Digit d : digits) {
226  Digit::Pad2Equipment(d.getPadID(), &eq, &col, &dil, &cha); // From Digit to Hardware coords
227  eq = ReadOut::FeeId(eq); // converts the Equipment Id in Cru/Link position ref
228  idx = getEquipmentPadIndex(eq, col, dil, cha); // finally to the unique padmap index
229  if (mPadMap[idx] != 0) { // We already have the pad set
230  LOG(WARNING) << "Duplicated DIGIT =" << d << " (" << eq << "," << col << "," << dil << "," << cha << ")";
231  } else {
232  mPadMap[idx] = d.getCharge();
233  padsCount++;
234  }
235  }
236  fillTheOutputBuffer(mPadMap); // Fill the Buffer for all Equipments per Event
237  writePaginatedEvent(orbit, bc);
238  std::memset(mPadMap, 0, sizeof(uint32_t) * Geo::N_HMPIDTOTALPADS); // Update for the new event
239  return;
240 }

248 void HmpidCoder2::openOutputStream(const std::string& outputFileName, const std::string& fileFor)
249 {
250  RAWDataHeader rdh; // by default, v6 is used currently.
251  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
252  rdh.feeId = ReadOut::FeeId(eq);
253  rdh.cruID = ReadOut::CruId(eq);
254  rdh.linkID = ReadOut::LnkId(eq);
255  rdh.endPointID = 0;
256  std::string outfname;
257  if (fileFor == "link") {
258  outfname = fmt::format("{}_L{}.raw", outputFileName, ReadOut::FeeId(eq));
259  } else if (fileFor == "flp") {
260  outfname = fmt::format("{}_flp{}.raw", outputFileName, ReadOut::FlpId(eq));
261  } else if (fileFor == "all") {
262  outfname = fmt::format("{}.raw", outputFileName);
263  } else {
264  throw std::runtime_error(fmt::format("unknown raw file grouping option {}", fileFor));
265  }
266 
267  mWriter.registerLink(rdh, outfname); // register the link
268  LinkSubSpec_t ap = RDHUtils::getSubSpec(ReadOut::CruId(eq), ReadOut::LnkId(eq), 0, ReadOut::FeeId(eq));
269  mTheRFWLinks[eq] = ap; // Store the RawFileWriter Link ID
270  }
271  return;
272 }
