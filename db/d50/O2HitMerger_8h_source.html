152  template <typename T, typename V>
153  V insertAdd(std::map<T, V>& m, T const& key, V value)
154  {
155  const auto iter = m.find(key);
156  V accum{0};
157  if (iter != m.end()) {
158  iter->second += value;
159  accum = iter->second;
160  } else {
161  m.insert(std::make_pair(key, value));
162  accum = value;
163  }
164  return accum;
165  }

193  template <typename T>
194  void fillBranch(int eventID, std::string const& name, T* ptr)
195  {
196  // fetch tree into which to fill
197  const std::lock_guard<std::mutex> lock(mMapsMtx);
198  auto iter = mEventToTTreeMap.find(eventID);
199  if (iter == mEventToTTreeMap.end()) {
200  {
201  std::stringstream str;
202  str << "memfile" << eventID;
203  mEventToTMemFileMap[eventID] = new TMemFile(str.str().c_str(), "RECREATE");
204  }
205  {
206  std::stringstream str;
207  str << "o2sim" << eventID;
208  mEventToTTreeMap[eventID] = new TTree(str.str().c_str(), str.str().c_str());
209  mEventToTTreeMap[eventID]->SetDirectory(mEventToTMemFileMap[eventID]);
210  }
211  }
212  TTree* tree = mEventToTTreeMap[eventID];
213 
214  auto br = o2::base::getOrMakeBranch(*tree, name.c_str(), &ptr);
215  br->SetAddress(&ptr);
216  br->Fill();
217  br->ResetAddress();
218  }

253  bool handleSimData(FairMQParts& data, int /*index*/)
254  {
255  bool expectmore = true;
256  int index = 0;
257  auto infoptr = o2::base::decodeTMessage<o2::data::SubEventInfo*>(data, index++);
258  o2::data::SubEventInfo& info = *infoptr;
259  auto accum = insertAdd<uint32_t, uint32_t>(mPartsCheckSum, info.eventID, (uint32_t)info.part);
260 
261  LOG(INFO) << "SIMDATA channel got " << data.Size() << " parts for event " << info.eventID << " part " << info.part << " out of " << info.nparts;
262 
263  fillSubEventInfoEntry(info);
264  consumeData<std::vector<o2::MCTrack>>(info.eventID, "MCTrack", data, index);
265  consumeData<std::vector<o2::TrackReference>>(info.eventID, "TrackRefs", data, index);
266  while (index < data.Size()) {
267  consumeHits(info.eventID, data, index);
268  }
269  // set the number of entries in the tree
270  auto tree = mEventToTTreeMap[info.eventID];
271  auto memfile = mEventToTMemFileMap[info.eventID];
272  tree->SetEntries(tree->GetEntries() + 1);
273  LOG(INFO) << "tree has file " << tree->GetDirectory()->GetFile()->GetName();
274  mEntries++;
275 
276  if (isDataComplete<uint32_t>(accum, info.nparts)) {
277  LOG(INFO) << "EVERYTHING IS HERE FOR EVENT " << info.eventID << "\n";
278 
279  // check if previous flush finished
280  if (mMergerIOThread.joinable()) {
281  mMergerIOThread.join();
282  }
283 
284  // start hit merging and flushing in a separate thread in order not to block
285  mMergerIOThread = std::thread([info, this]() { mergeAndFlushData(info.eventID); });
286 
287  mEventChecksum += info.eventID;
288  // we also need to check if we have all events
289  if (isDataComplete<uint32_t>(mEventChecksum, info.maxEvents)) {
290  LOG(INFO) << "ALL EVENTS HERE; CHECKSUM " << mEventChecksum;
291 
292  // flush remaining data and close file
293  if (mMergerIOThread.joinable()) {
294  mMergerIOThread.join();
295  }
296 
297  expectmore = false;
298  }
299 
300  if (mPipeToDriver != -1) {
301  if (write(mPipeToDriver, &info.eventID, sizeof(info.eventID)) == -1) {
302  LOG(ERROR) << "FAILED WRITING TO PIPE";
303  };
304  }
305  }
306  if (!expectmore) {
307  // somehow FairMQ has difficulties shutting down; helping manually
308  raise(SIGINT);
309  }
310  return expectmore;
311  }

329  void reorderAndMergeMCTRacks(TTree& origin, TTree& target, const std::vector<int>& nprimaries, const std::vector<int>& nsubevents)
330  {
331  std::vector<MCTrack>* incomingdata = nullptr;
332  auto targetdata = new std::vector<MCTrack>;
333  auto originbr = origin.GetBranch("MCTrack");
334  originbr->SetAddress(&incomingdata);
335  const auto entries = origin.GetEntries();
336  //
337  // loop over subevents to store the primary events
338  //
339  Int_t nprimTot = 0;
340  for (auto entry = entries - 1; entry >= 0; --entry) {
341  int index = nsubevents[entry];
342  nprimTot += nprimaries[index];
343  printf("merge %lld %5d %5d %5d \n", entry, index, nsubevents[entry], nsubevents[index]);
344  originbr->GetEntry(index);
345  for (Int_t i = 0; i < nprimaries[index]; i++) {
346  auto& track = incomingdata->at(i);
347  if (track.isTransported()) { // reset daughters only if track was transported, it will be fixed below
348  track.SetFirstDaughterTrackId(-1);
349  track.SetLastDaughterTrackId(-1);
350  }
351  targetdata->push_back(track);
352  }
353  incomingdata->clear();
354  delete incomingdata;
355  incomingdata = nullptr;
356  }
357  //
358  // loop a second time to store the secondaries and fix the mother track IDs
359  //
360  Int_t idelta1 = nprimTot;
361  Int_t idelta0 = 0;
362  for (auto entry = entries - 1; entry >= 0; --entry) {
363  int index = nsubevents[entry];
364 
365  originbr->GetEntry(index);
366 
367  Int_t npart = (int)(incomingdata->size());
368  Int_t nprim = nprimaries[index];
369  idelta1 -= nprim;
370 
371  for (Int_t i = nprim; i < npart; i++) {
372  auto& track = incomingdata->at(i);
373  Int_t cId = track.getMotherTrackId();
374  if (cId >= nprim) {
375  cId += idelta1;
376  } else {
377  cId += idelta0;
378  }
379  track.SetMotherTrackId(cId);
380  track.SetFirstDaughterTrackId(-1);
381 
382  Int_t hwm = (int)(targetdata->size());
383  auto& mother = targetdata->at(cId);
384  if (mother.getFirstDaughterTrackId() == -1) {
385  mother.SetFirstDaughterTrackId(hwm);
386  }
387  mother.SetLastDaughterTrackId(hwm);
388 
389  targetdata->push_back(track);
390  }
391  idelta0 += nprim;
392  idelta1 += npart;
393  incomingdata->clear();
394  delete incomingdata;
395  incomingdata = nullptr;
396  }
397 
398  //
399  // write to output
400  auto targetbr = o2::base::getOrMakeBranch(target, "MCTrack", &targetdata);
401  targetbr->SetAddress(&targetdata);
402  targetbr->Fill();
403  targetbr->ResetAddress();
404  targetdata->clear();
405  }

407  template <typename T>
408  void remapTrackIdsAndMerge(std::string brname, TTree& origin, TTree& target,
409  const std::vector<int>& trackoffsets, const std::vector<int>& nprimaries, const std::vector<int>& subevOrdered)
410  {
411  //
412  // Remap the mother track IDs by adding an offset.
413  // The offset calculated as the sum of the number of entries in the particle list of the previous subevents.
414  // This method is called by O2HitMerger::mergeAndFlushData(int)
415  //
416  T* incomingdata = nullptr;
417  auto targetdata = new T;
418  auto originbr = origin.GetBranch(brname.c_str());
419  originbr->SetAddress(&incomingdata);
420  const auto entries = origin.GetEntries();
421 
422  if (entries == 1) {
423  // nothing to do in case there is only one entry
424  originbr->GetEntry(0);
425  targetdata = incomingdata;
426  } else {
427  // loop over subevents
428  Int_t nprimTot = 0;
429  for (auto entry = 0; entry < entries; entry++) {
430  nprimTot += nprimaries[entry];
431  }
432  Int_t idelta0 = 0;
433  Int_t idelta1 = nprimTot;
434  for (auto entry = entries - 1; entry >= 0; --entry) {
435  Int_t index = subevOrdered[entry];
436  Int_t nprim = nprimaries[index];
437  originbr->GetEntry(index);
438  idelta1 -= nprim;
439  for (auto& data : *incomingdata) {
440  updateTrackIdWithOffset(data, nprim, idelta0, idelta1);
441  targetdata->push_back(data);
442  }
443  incomingdata->clear();
444  idelta0 += nprim;
445  idelta1 += trackoffsets[index];
446  delete incomingdata;
447  incomingdata = nullptr;
448  }
449  }
450  auto targetbr = o2::base::getOrMakeBranch(target, brname.c_str(), &targetdata);
451  targetbr->SetAddress(&targetdata);
452  targetbr->Fill();
453  targetbr->ResetAddress();
454  targetdata->clear();
455  }

457  void updateTrackIdWithOffset(MCTrack& track, Int_t nprim, Int_t idelta0, Int_t idelta1)
458  {
459  Int_t cId = track.getMotherTrackId();
460  Int_t ioffset = (cId < nprim) ? idelta0 : idelta1;
461  if (cId != -1) {
462  track.SetMotherTrackId(cId + ioffset);
463  }
464  }

466  void updateTrackIdWithOffset(TrackReference& ref, Int_t nprim, Int_t idelta0, Int_t idelta1)
467  {
468  Int_t cId = ref.getTrackID();
469  Int_t ioffset = (cId < nprim) ? idelta0 : idelta1;
470  ref.setTrackID(cId + ioffset);
471  }

473  // this merges all entries from the TBranch brname from the origin TTree (containing one event only)
474  // into a single entry in a target TTree / same branch
475  // (assuming T is typically a vector; merging is simply done by appending)
476  template <typename T>
477  void merge(std::string brname, TTree& origin, TTree& target)
478  {
479  auto originbr = origin.GetBranch(brname.c_str());
480  auto targetdata = new T;
481  T* incomingdata = nullptr;
482  originbr->SetAddress(&incomingdata);
483 
484  const auto entries = origin.GetEntries();
485 
486  T* filladdress = nullptr;
487  if (entries == 1) {
488  // this avoids useless copy in case there was no sub-event splitting; we just use the original data
489  originbr->GetEntry(0);
490  filladdress = incomingdata;
491  } else {
492  filladdress = targetdata;
493  for (auto entry = 0; entry < entries; ++entry) {
494  originbr->GetEntry(entry);
495  backInsert(*incomingdata, *targetdata);
496  delete incomingdata;
497  incomingdata = nullptr;
498  }
499  }
500 
501  // fill target for this event
502  auto targetbr = o2::base::getOrMakeBranch(target, brname.c_str(), &filladdress);
503  targetbr->SetAddress(&filladdress);
504  targetbr->Fill();
505  targetbr->ResetAddress();
506  targetdata->clear();
507  if (incomingdata) {
508  delete incomingdata;
509  incomingdata = nullptr;
510  }
511 
512  delete targetdata;
513  }

528  // This method goes over the tree containing data for a given event; potentially merges
529  // it and flushes it into the actual output file.
530  // The method can be called asynchronously to data collection
531  bool mergeAndFlushData(int eventID)
532  {
533  LOG(INFO) << "ENTERING MERGING/FLUSHING HITS STAGE FOR EVENT " << eventID;
534 
535  auto tree = mEventToTTreeMap[eventID];
536  if (!tree) {
537  LOG(INFO) << "NO TTREE FOUND FOR EVENT " << eventID;
538  return false;
539  }
540 
541  if (tree->GetEntries() == 0 || mNExpectedEvents == 0) {
542  LOG(INFO) << "NO ENTRY IN TTREE FOUND FOR EVENT " << eventID;
543  return false;
544  }
545 
546  TStopwatch timer;
547  timer.Start();
548 
549  // calculate trackoffsets
550  auto infobr = tree->GetBranch("SubEventInfo");
551 
552  auto& confref = o2::conf::SimConfig::Instance();
553 
554  std::vector<int> trackoffsets; // collecting trackoffsets to be applied to correct
555  std::vector<int> nprimaries; // collecting primary particles in each subevent
556  std::vector<int> nsubevents; // collecting of subevent numbers
557 
558  o2::dataformats::MCEventHeader* eventheader = nullptr; // The event header
559 
560  // the MC labels (trackID) for hits
561  o2::data::SubEventInfo* info = nullptr;
562  infobr->SetAddress(&info);
563  for (int i = 0; i < infobr->GetEntries(); ++i) {
564  infobr->GetEntry(i);
565  assert(info->npersistenttracks >= 0);
566  trackoffsets.emplace_back(info->npersistenttracks);
567  nprimaries.emplace_back(info->nprimarytracks);
568  nsubevents.emplace_back(info->part);
569  info->mMCEventHeader.printInfo();
570  if (eventheader == nullptr) {
571  eventheader = &info->mMCEventHeader;
572  } else {
573  eventheader->getMCEventStats().add(info->mMCEventHeader.getMCEventStats());
574  }
575  }
576 
577  // now see which events can be discarded in any case due to no hits
578  if (confref.isFilterOutNoHitEvents()) {
579  if (eventheader && eventheader->getMCEventStats().getNHits() == 0) {
580  LOG(INFO) << " Taking out event " << eventID << " due to no hits ";
581  cleanEvent(eventID);
582  return false;
583  }
584  }
585 
586  // put the event headers into the new TTree
587  eventheader->printInfo();
588  auto headerbr = o2::base::getOrMakeBranch(*mOutTree, "MCEventHeader.", &eventheader);
589  headerbr->SetAddress(&eventheader);
590  headerbr->Fill();
591  headerbr->ResetAddress();
592 
593  // attention: We need to make sure that we write everything in the same event order
594  // but iteration over keys of a standard map in C++ is ordered
595 
596  // b) merge the general data
597  //
598  // for MCTrack remap the motherIds and merge at the same go
599  const auto entries = tree->GetEntries();
600  std::vector<int> subevOrdered((int)(nsubevents.size()));
601  for (auto entry = entries - 1; entry >= 0; --entry) {
602  subevOrdered[nsubevents[entry] - 1] = entry;
603  printf("HitMerger entry: %lld nprimry: %5d trackoffset: %5d \n", entry, nprimaries[entry], trackoffsets[entry]);
604  }
605 
606  reorderAndMergeMCTRacks(*tree, *mOutTree, nprimaries, subevOrdered);
607  Int_t ioffset = 0;
608  remapTrackIdsAndMerge<std::vector<o2::TrackReference>>("TrackRefs", *tree, *mOutTree, trackoffsets, nprimaries, subevOrdered);
609 
610  // c) do the merge procedure for all hits ... delegate this to detector specific functions
611  // since they know about types; number of branches; etc.
612  // this will also fix the trackIDs inside the hits
613  for (int id = 0; id < mDetectorInstances.size(); ++id) {
614  auto& det = mDetectorInstances[id];
615  if (det) {
616  auto hittree = mDetectorToTTreeMap[id];
617  det->mergeHitEntries(*tree, *hittree, trackoffsets, nprimaries, subevOrdered);
618  hittree->SetEntries(hittree->GetEntries() + 1);
619  LOG(INFO) << "flushing tree to file " << hittree->GetDirectory()->GetFile()->GetName();
620  mDetectorOutFiles[id]->Write("", TObject::kOverwrite);
621  }
622  }
623 
624  // increase the entry count in the tree
625  mOutTree->SetEntries(mOutTree->GetEntries() + 1);
626  LOG(INFO) << "outtree has file " << mOutTree->GetDirectory()->GetFile()->GetName();
