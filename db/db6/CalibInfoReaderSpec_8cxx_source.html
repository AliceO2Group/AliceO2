46 void CalibInfoReader::run(ProcessingContext& pc)
47 {
48  if (mState != 1) {
49  return;
50  }
51 
52  char filename[100];
53 
54  if ((mTree && mCurrentEntry < mTree->GetEntries()) || fscanf(mFile, "%s", filename) == 1) {
55  if (!mTree || mCurrentEntry >= mTree->GetEntries()) {
56  TFile* fin = TFile::Open(filename);
57  mTree = (TTree*)fin->Get("calibTOF");
58  mCurrentEntry = 0;
59  mTree->SetBranchAddress("TOFCalibInfo", &mPvect);
60  }
61  if ((mGlobalEntry % mNinstances) == mInstance) {
62  mTree->GetEvent(mCurrentEntry);
63  LOG(INFO) << "Send " << mVect.size() << " calib infos";
64  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, mTOFTPC ? ddCalib_tpc : ddCalib, 0, Lifetime::Timeframe}, mVect);
65  usleep(10000);
66  }
67  mGlobalEntry++;
68  mCurrentEntry++;
69  } else {
70  mState = 2;
71  pc.services().get<ControlService>().endOfStream();
72  }
73  return;
74 }
