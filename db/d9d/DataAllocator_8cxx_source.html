45 std::string const& DataAllocator::matchDataHeader(const Output& spec, size_t timeslice)
46 {
47  // FIXME: we should take timeframeId into account as well.
48  for (auto& output : mAllowedOutputRoutes) {
49  if (DataSpecUtils::match(output.matcher, spec.origin, spec.description, spec.subSpec) && ((timeslice % output.maxTimeslices) == output.timeslice)) {
50  return output.channel;
51  }
52  }
53  throw runtime_error_f(
54  "Worker is not authorised to create message with "
55  "origin(%s) description(%s) subSpec(%d)",
56  spec.origin.as<std::string>().c_str(),
57  spec.description.as<std::string>().c_str(),
58  spec.subSpec);
59 }

90 FairMQMessagePtr DataAllocator::headerMessageFromOutput(Output const& spec, //
91  std::string const& channel, //
92  o2::header::SerializationMethod method, //
93  size_t payloadSize) //
94 {
95  DataHeader dh;
96  dh.dataOrigin = spec.origin;
97  dh.dataDescription = spec.description;
98  dh.subSpecification = spec.subSpec;
99  dh.payloadSize = payloadSize;
100  dh.payloadSerializationMethod = method;
101  dh.tfCounter = mTimingInfo->tfCounter;
102  dh.firstTForbit = mTimingInfo->firstTFOrbit;
103 
104  DataProcessingHeader dph{mTimingInfo->timeslice, 1};
105  auto& context = mRegistry->get<MessageContext>();
106 
107  auto channelAlloc = o2::pmr::getTransportAllocator(context.proxy().getTransport(channel, 0));
108  return o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, dph, spec.metaHeader});

111 void DataAllocator::addPartToContext(FairMQMessagePtr&& payloadMessage, const Output& spec,
112  o2::header::SerializationMethod serializationMethod)
113 {
114  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
115  auto headerMessage = headerMessageFromOutput(spec, channel, serializationMethod, 0);
116 
117  // FIXME: this is kind of ugly, we know that we can change the content of the
118  // header message because we have just created it, but the API declares it const
119  const DataHeader* cdh = o2::header::get<DataHeader*>(headerMessage->GetData());
120  DataHeader* dh = const_cast<DataHeader*>(cdh);
121  dh->payloadSize = payloadMessage->GetSize();
122  auto& context = mRegistry->get<MessageContext>();
123  // make_scoped creates the context object inside of a scope handler, since it goes out of
124  // scope immediately, the created object is scheduled and can be directly sent if the context
125  // is configured with the dispatcher callback
126  context.make_scoped<MessageContext::TrivialObject>(std::move(headerMessage), std::move(payloadMessage), channel);
127 }

129 void DataAllocator::adopt(const Output& spec, std::string* ptr)
130 {
131  std::unique_ptr<std::string> payload(ptr);
132  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
133  // the correct payload size is set later when sending the
134  // StringContext, see DataProcessor::doSend
135  auto header = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, 0);
136  mRegistry->get<StringContext>().addString(std::move(header), std::move(payload), channel);
137  assert(payload.get() == nullptr);
138 }

140 void DataAllocator::adopt(const Output& spec, TableBuilder* tb)
141 {
142  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
143  auto header = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodArrow, 0);
144  auto& context = mRegistry->get<ArrowContext>();
145 
146  auto creator = [device = context.proxy().getDevice()](size_t s) -> std::unique_ptr<FairMQMessage> { return device->NewMessage(s); };
147  auto buffer = std::make_shared<FairMQResizableBuffer>(creator);
148 
152  std::shared_ptr<TableBuilder> p(tb);
153  auto finalizer = [payload = p](std::shared_ptr<FairMQResizableBuffer> b) -> void {
154  auto table = payload->finalize();
155  if (O2_BUILTIN_UNLIKELY(table->num_rows() == 0)) {
156  LOG(DEBUG) << "Empty table was produced: " << table->ToString();
157  }
158 
159  auto stream = std::make_shared<arrow::io::BufferOutputStream>(b);
160  auto outBatch = arrow::ipc::NewStreamWriter(stream.get(), table->schema());
161  if (outBatch.ok() == true) {
162  auto outStatus = outBatch.ValueOrDie()->WriteTable(*table);
163  if (outStatus.ok() == false) {
164  throw std::runtime_error("Unable to Write table");
165  }
166  } else {
167  throw ::std::runtime_error("Unable to create batch writer");
168  }
169  };
170 
171  context.addBuffer(std::move(header), buffer, std::move(finalizer), channel);
172 }

245 Output DataAllocator::getOutputByBind(OutputRef&& ref)
246 {
247  if (ref.label.empty()) {
248  throw runtime_error("Invalid (empty) OutputRef provided.");
249  }
250  for (auto ri = 0ul, re = mAllowedOutputRoutes.size(); ri != re; ++ri) {
251  if (mAllowedOutputRoutes[ri].matcher.binding.value == ref.label) {
252  auto spec = mAllowedOutputRoutes[ri].matcher;
253  auto dataType = DataSpecUtils::asConcreteDataTypeMatcher(spec);
254  return Output{dataType.origin, dataType.description, ref.subSpec, spec.lifetime, std::move(ref.headerStack)};
255  }
256  }
257  throw runtime_error_f("Unable to find OutputSpec with label %s", ref.label.c_str());
258  O2_BUILTIN_UNREACHABLE();
259 }
