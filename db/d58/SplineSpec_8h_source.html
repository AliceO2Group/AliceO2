197 GPUdi() void SplineContainer<DataT>::getKnotU(int iKnot, int u[/* mXdim */]) const
198 {
200  for (int dim = 0; dim < mXdim; dim++) {
201  int n = mGrid[dim].getNumberOfKnots();
202  u[dim] = mGrid[dim].getKnot(iKnot % n).getU();
203  iKnot /= n;
204  }
205 }

207 template <typename DataT>
208 GPUdi() int SplineContainer<DataT>::getKnotIndex(const int iKnot[/* mXdim */]) const
209 {
211  int ind = iKnot[0];
212  int n = 1;
213  for (int dim = 1; dim < mXdim; dim++) {
214  n *= mGrid[dim - 1].getNumberOfKnots();
215  ind += n * iKnot[dim];
216  }
217  return ind;
218 }

271  GPUd() void interpolate(const DataT x[/*mXdim*/], GPUgeneric() DataT S[/*mYdim*/]) const
272  {
273  const auto nXdimTmp = SplineUtil::getNdim<XdimT>(mXdim);
274  const auto nXdim = nXdimTmp.get();
275  const auto maxXdimTmp = SplineUtil::getMaxNdim<XdimT>(mXdim);
276  DataT u[maxXdimTmp.get()];
277  for (int i = 0; i < nXdim; i++) {
278  u[i] = mGrid[i].convXtoU(x[i]);
279  }
280  interpolateU<SafetyLevel::kSafe>(mXdim, mYdim, mParameters, u, S);
281  }

284  template <SafetyLevel SafeT = SafetyLevel::kSafe>
285  GPUd() void interpolateU(int inpXdim, int inpYdim, GPUgeneric() const DataT Parameters[],
286  const DataT u[/*inpXdim*/], GPUgeneric() DataT S[/*inpYdim*/]) const
287  {
288  const auto nXdimTmp = SplineUtil::getNdim<XdimT>(mXdim);
289  const auto nXdim = nXdimTmp.get();
290  const auto maxXdimTmp = SplineUtil::getMaxNdim<XdimT>(mXdim);
291  const auto maxXdim = maxXdimTmp.get();
292  const auto nYdimTmp = SplineUtil::getNdim<YdimT>(mYdim);
293  const auto nYdim = nYdimTmp.get();
294  const auto maxYdimTmp = SplineUtil::getMaxNdim<XdimT>(mYdim);
295  const auto maxYdim = maxYdimTmp.get();
296 
297  const auto nParameters = 1 << (2 * nXdim); //total Nr of Parameters necessary for one interpolation
298  const auto nKnotParametersPerY = 1 << nXdim; // Nr of Parameters per Knot per Y dimension
299  const auto nKnotParameters = (1 << nXdim) * nYdim; // Nr of Parameters per Knot
300 
301  DataT iParameters[(1 << (2 * maxXdim)) * maxYdim]; // Array for all parameters
302 
303  //get the indices of the "most left" Knot:
304 
305  int indices[maxXdim]; //indices of the 'most left' knot
306  for (int i = 0; i < nXdim; i++) {
307  indices[i] = mGrid[i].getLeftKnotIndexForU(u[i]);
308  }
309  // get all the needed parameters into one array iParameters[nParameters]:
310  int indicestmp[maxXdim];
311  for (int i = 0; i < nKnotParametersPerY; i++) { // for every necessary Knot
312  for (int k = 0; k < nXdim; k++) {
313  indicestmp[k] = indices[k] + (i / (1 << k)) % 2; //get the knot-indices in every dimension (mirrored order binary counting)
314  }
315  int index = TBase::getKnotIndex(indicestmp); //get index of the current Knot
316 
317  for (int j = 0; j < nKnotParameters; j++) { //and fill the iparameter array with according parameters
318  iParameters[i * nKnotParameters + j] = Parameters[index * nKnotParameters + j];
319  }
320  }
321  //now start with the interpolation loop:
322 
323  auto maxInterpolations = (1 << (2 * maxXdim - 2)) * maxYdim;
324 
325  DataT S0[maxInterpolations];
326  DataT D0[maxInterpolations];
327  DataT S1[maxInterpolations];
328  DataT D1[maxInterpolations];
329 
330  int nInterpolations = (1 << (2 * nXdim - 2)) * nYdim;
331  int nKnots = 1 << (nXdim);
332 
333  for (int d = 0; d < nXdim; d++) { //for every dimension
334  DataT* pointer[4] = {S0, D0, S1, D1}; // pointers for interpolation arrays S0, D0, S1, D1 point to Arraystart
335  for (int i = 0; i < nKnots; i++) { //for every knot
336  for (int j = 0; j < nKnots; j++) { // for every parametertype
337  int pointernr = 2 * (i % 2) + (j % 2); //to which array should it be delivered
338  for (int k = 0; k < nYdim; k++) {
339  pointer[pointernr][0] = iParameters[(i * nKnots + j) * nYdim + k];
340  pointer[pointernr]++;
341  }
342  } // end for j (every parametertype)
343  } // end for i (every knot)
344 
345  const typename Spline1D<DataT>::Knot& knotL = mGrid[d].getKnot(indices[d]);
346  DataT coordinate = u[d];
347  typedef Spline1DSpec<DataT, 0, 0> TGridX;
348  const TGridX& gridX = *((const TGridX*)&(mGrid[d]));
349  gridX.interpolateU(nInterpolations, knotL, S0, D0, S1, D1, coordinate, iParameters);
350  nInterpolations /= 4;
351  nKnots /= 2;
352  } //end d (every dimension)
353 
354  for (int i = 0; i < nYdim; i++) {
355  S[i] = iParameters[i]; // write into result-array
356  //std::cout<<iParameters[i] <<", ";
357  }
358  } // end interpolateU
359 
360  protected:
361  using TBase::mGrid;
362  using TBase::mParameters;
363  using TBase::mXdim;
364  using TBase::mYdim;
365  using TBase::TBase; // inherit constructors and hide them
366 };
