35 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
36 {
37  WorkflowSpec specs;
38  auto useCCDB = configcontext.options().get<bool>("use-ccdb");
39  auto forceUpdate = configcontext.options().get<bool>("forceupdate");
40  auto doPedestals = configcontext.options().get<bool>("pedestals");
41  auto doHgLgRatio = configcontext.options().get<bool>("hglgratio");
42  auto doEtree = configcontext.options().get<int>("etree");
43  auto doBadMap = configcontext.options().get<bool>("badmap");
44  auto path = configcontext.options().get<std::string>("path");
45  if (doPedestals && doHgLgRatio) {
46  LOG(FATAL) << "Can not run pedestal and HG/LG calibration simulteneously";
47  }
48 
49  LOG(INFO) << "PHOS Calibration workflow: options";
50  LOG(INFO) << "useCCDB = " << useCCDB;
51  if (doPedestals) {
52  LOG(INFO) << "pedestals ";
53  specs.emplace_back(o2::phos::getPedestalCalibSpec(useCCDB, forceUpdate, path));
54  } else {
55  if (doHgLgRatio) {
56  LOG(INFO) << "hglgratio ";
57  specs.emplace_back(o2::phos::getHGLGRatioCalibSpec(useCCDB, forceUpdate, path));
58  }
59  }
60  if (doBadMap) {
61  LOG(INFO) << "bad map calculation ";
62  short m = 0;
63  // specs.emplace_back(o2::phos::getBadMapCalibSpec(useCCDB,forceUpdate,path,m));
64  }
65  if (doEtree >= 0) {
66  LOG(INFO) << "Filling tree for energy and time calibration ";
67  specs.emplace_back(o2::phos::getPHOSCalibCollectorDeviceSpec(doEtree));
68  }
69  return specs;
70 }
