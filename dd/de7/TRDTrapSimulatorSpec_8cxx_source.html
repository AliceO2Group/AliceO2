40 TrapConfig* TRDDPLTrapSimulatorTask::getTrapConfig()
41 {
42  // return an existing TRAPconfig or load it from the CCDB
43  // in case of failure, a default TRAPconfig is created
44  LOG(debug) << "start of gettrapconfig";
45  if (mTrapConfig) {
46  LOG(debug) << "mTrapConfig is valid : 0x" << std::hex << mTrapConfig << std::dec;
47  return mTrapConfig;
48  } else {
49  LOG(debug) << "mTrapConfig is invalid : 0x" << std::hex << mTrapConfig << std::dec;
50 
52  // try to load the requested configuration
53  loadTrapConfig();
54  //calib.
55  if (mTrapConfig->getConfigName() == "" && mTrapConfig->getConfigVersion() == "") {
56  //some trap configs dont have config name and version set, in those cases, just show the file name used.
57  LOG(info) << "using TRAPconfig :\"" << mTrapConfigName;
58  } else {
59  LOG(info) << "using TRAPconfig :\"" << mTrapConfig->getConfigName().c_str() << "\".\"" << mTrapConfig->getConfigVersion().c_str() << "\"";
60  }
61  // we still have to load the gain tables
62  // if the gain filter is active
63  return mTrapConfig;
64  } // end of else from if mTrapConfig

102 void TRDDPLTrapSimulatorTask::setOnlineGainTables()
103 {
104  //check FGBY from trapconfig.
105  //check the input parameter of trd-onlinegaincorrection.
106  //warn if you have chosen a trapconfig with gaincorrections but chosen not to use them.
107  if (mEnableOnlineGainCorrection) {
108  if (mTrapConfig->getTrapReg(TrapConfig::kFGBY) == 0) {
109  LOG(warn) << "you have asked to do online gain calibrations but the selected trap config does not have FGBY enabled, so modifying trapconfig to conform to your command line request. OnlineGains will be 1, i.e. no effect.";
110  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
111  mTrapConfig->setTrapReg(TrapConfig::kFGBY, 1, iDet);
112  }
113  }
114  mCalib->setOnlineGainTables(mOnlineGainTableName);
115  //TODO add some error checking inhere.
116  // gain factors are per MCM
117  // allocate the registers accordingly
118  for (int ch = 0; ch < NADCMCM; ++ch) {
119  TrapConfig::TrapReg_t regFGAN = (TrapConfig::TrapReg_t)(TrapConfig::kFGA0 + ch);
120  TrapConfig::TrapReg_t regFGFN = (TrapConfig::TrapReg_t)(TrapConfig::kFGF0 + ch);
121  mTrapConfig->setTrapRegAlloc(regFGAN, TrapConfig::kAllocByMCM);
122  mTrapConfig->setTrapRegAlloc(regFGFN, TrapConfig::kAllocByMCM);
123  }
124 
125  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
126  const int nRobs = Geometry::getStack(iDet) == 2 ? Geometry::ROBmaxC0() : Geometry::ROBmaxC1();
127  for (int rob = 0; rob < nRobs; ++rob) {
128  for (int mcm = 0; mcm < NMCMROB; ++mcm) {
129  // set ADC reference voltage
130  mTrapConfig->setTrapReg(TrapConfig::kADCDAC, mCalib->getOnlineGainAdcdac(iDet, rob, mcm), iDet, rob, mcm);
131  // set constants channel-wise
132  for (int ch = 0; ch < NADCMCM; ++ch) {
133  TrapConfig::TrapReg_t regFGAN = (TrapConfig::TrapReg_t)(TrapConfig::kFGA0 + ch);
134  TrapConfig::TrapReg_t regFGFN = (TrapConfig::TrapReg_t)(TrapConfig::kFGF0 + ch);
135  mTrapConfig->setTrapReg(regFGAN, mCalib->getOnlineGainFGAN(iDet, rob, mcm, ch), iDet, rob, mcm);
136  mTrapConfig->setTrapReg(regFGFN, mCalib->getOnlineGainFGFN(iDet, rob, mcm, ch), iDet, rob, mcm);
137  }
138  }
139  }
140  }
141  } else if (mTrapConfig->getTrapReg(TrapConfig::kFGBY) == 1) {
142  LOG(warn) << "you have asked to not use online gain calibrations but the selected trap config does have FGBY enabled. Gain calibrations will be 1, i.e. no effect";
143  }

146 void TRDDPLTrapSimulatorTask::processTRAPchips(int currDetector, int& nTrackletsInTrigRec, std::vector<Tracklet64>& trapTrackletsAccum, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& lblTracklets, const o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>* lblDigits)
147 {
148  // Loop over all TRAP chips of detector number currDetector.
149  // TRAP chips without input data are skipped
150  int currStack = (currDetector % NCHAMBERPERSEC) / NLAYER;
151  int nTrapsMax = (currStack == 2) ? NROBC0 * NMCMROB : NROBC1 * NMCMROB;
152  for (int iTrap = 0; iTrap < nTrapsMax; ++iTrap) {
153  if (!mTrapSimulator[iTrap].isDataSet()) {
154  continue;
155  }
156  auto timeTrapProcessingStart = std::chrono::high_resolution_clock::now();
157  mTrapSimulator[iTrap].filter();
158  mTrapSimulator[iTrap].tracklet();
159  mTrapSimTime += std::chrono::high_resolution_clock::now() - timeTrapProcessingStart;
160  auto trackletsOut = mTrapSimulator[iTrap].getTrackletArray64();
161  int nTrackletsOut = trackletsOut.size();
162  nTrackletsInTrigRec += nTrackletsOut;
163  if (mUseMC) {
164  auto digitCountOut = mTrapSimulator[iTrap].getTrackletDigitCount(); // number of digits contributing to each tracklet
165  auto digitIndicesOut = mTrapSimulator[iTrap].getTrackletDigitIndices(); // global indices of the digits composing the tracklets
166  int currDigitIndex = 0; // count the total number of digits which are associated to tracklets for this TRAP
167  int trkltIdxStart = trapTrackletsAccum.size();
168  for (int iTrklt = 0; iTrklt < nTrackletsOut; ++iTrklt) {
169  // for each tracklet of this TRAP check the MC labels of the digits which contribute to the tracklet
170  int tmp = currDigitIndex;
171  for (int iDigitIndex = tmp; iDigitIndex < tmp + digitCountOut[iTrklt]; ++iDigitIndex) {
172  if (iDigitIndex == tmp) {
173  // for the first digit composing the tracklet we don't need to check for duplicate labels
174  lblTracklets.addElements(trkltIdxStart + iTrklt, lblDigits->getLabels(digitIndicesOut[iDigitIndex]));
175  } else {
176  // in case more than one digit composes the tracklet we add only the labels
177  // from the additional digit(s) which are not already contained in the previous
178  // digit(s)
179  auto currentLabels = lblTracklets.getLabels(trkltIdxStart + iTrklt);
180  auto newLabels = lblDigits->getLabels(digitIndicesOut[iDigitIndex]);
181  for (const auto& newLabel : newLabels) {
182  bool isAlreadyIn = false;
183  for (const auto& currLabel : currentLabels) {
184  if (currLabel.compare(newLabel)) {
185  isAlreadyIn = true;
186  }
187  }
188  if (!isAlreadyIn) {
189  lblTracklets.addElement(trkltIdxStart + iTrklt, newLabel);
190  }
191  }
192  }
193  ++currDigitIndex;
194  }
195  }
196  }
197  trapTrackletsAccum.insert(trapTrackletsAccum.end(), trackletsOut.begin(), trackletsOut.end());
198  mTrapSimulator[iTrap].reset();
199  }

220 void TRDDPLTrapSimulatorTask::run(o2::framework::ProcessingContext& pc)
221 {
222  // this method steeres the processing of the TRAP simulation
223  LOG(info) << "TRD Trap Simulator Device running over incoming message";
224 
225  // input
226  auto inputDigits = pc.inputs().get<gsl::span<o2::trd::Digit>>("digitinput"); // block of TRD digits
227  auto inputTriggerRecords = pc.inputs().get<std::vector<o2::trd::TriggerRecord>>("triggerrecords"); // time and number of digits for each collision
228  // the above is changed to a vector from a span as the span is constant and cant modify elements.
229  if (inputDigits.size() == 0 || inputTriggerRecords.size() == 0) {
230  LOG(warn) << "Did not receive any digits, trigger records, or neither one nor the other. Aborting.";
231  return;
232  }
233  LOG(debug) << "Read in " << inputDigits.size() << " digits";
234 
235  const o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>* lblDigitsPtr = nullptr;
236  using lblType = std::decay_t<decltype(pc.inputs().get<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(""))>;
237  std::optional<lblType> lblDigits;
238 
239  if (mUseMC) {
240  lblDigits.emplace(pc.inputs().get<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>("labelinput")); // MC labels associated to the input digits
241  lblDigitsPtr = &lblDigits.value();
242  LOG(debug) << "Labels contain " << lblDigitsPtr->getNElements() << " elements with and indexed size of " << lblDigitsPtr->getIndexedSize();
243  if (lblDigitsPtr->getIndexedSize() != inputDigits.size()) {
244  LOG(warn) << "Digits and Labels coming into TrapSimulator are of differing sizes, labels will be jibberish. " << lblDigitsPtr->getIndexedSize() << "!=" << inputDigits.size();
245  }
246  }
247  LOG(debug) << "Trigger records are available for " << inputTriggerRecords.size() << " collisions";
248 
249  // output
250  std::vector<Tracklet64> trapTrackletsAccum; // calculated tracklets
251 
252  o2::dataformats::MCTruthContainer<o2::MCCompLabel> lblTracklets; // MC labels for the tracklets, taken from the digits which make up the tracklet (duplicates are removed)
253 
254  // sort digits by chamber ID for each collision and keep track in index vector
255  auto sortStart = std::chrono::high_resolution_clock::now();
256  std::vector<unsigned int> digitIndices(inputDigits.size()); // digit indices sorted by chamber ID for each time frame
257  std::iota(digitIndices.begin(), digitIndices.end(), 0);
258  for (auto& trig : inputTriggerRecords) {
259  std::stable_sort(std::begin(digitIndices) + trig.getFirstDigit(), std::begin(digitIndices) + trig.getNumberOfDigits() + trig.getFirstDigit(),
260  [&inputDigits](unsigned int i, unsigned int j) { return inputDigits[i].getDetector() < inputDigits[j].getDetector(); });
261  }
262  std::chrono::duration<double> sortTime = std::chrono::high_resolution_clock::now() - sortStart;
263 
264  // initialize timers
265  auto timeDigitLoopStart = std::chrono::high_resolution_clock::now();
266  std::chrono::duration<double> trapSimAccumulatedTime{};
267  mTrapSimTime = std::chrono::duration<double>::zero();
268 
269  for (int iTrig = 0; iTrig < inputTriggerRecords.size(); ++iTrig) {
270  int nTrackletsInTrigRec = 0;
271  int currDetector = -1;
272  for (int iDigit = inputTriggerRecords[iTrig].getFirstDigit(); iDigit < (inputTriggerRecords[iTrig].getFirstDigit() + inputTriggerRecords[iTrig].getNumberOfDigits()); ++iDigit) {
273  const auto& digit = &inputDigits[digitIndices[iDigit]];
274  if (currDetector < 0) {
275  currDetector = digit->getDetector();
276  }
277  if (currDetector != digit->getDetector()) {
278  // we switch to a new chamber, process all TRAPs of the previous chamber which contain data
279  processTRAPchips(currDetector, nTrackletsInTrigRec, trapTrackletsAccum, lblTracklets, lblDigitsPtr);
280  currDetector = digit->getDetector();
281  }
282  // fill the digit data into the corresponding TRAP chip
283  int trapIdx = digit->getROB() * NMCMROB + digit->getMCM();
284  if (!mTrapSimulator[trapIdx].isDataSet()) {
285  mTrapSimulator[trapIdx].init(mTrapConfig, digit->getDetector(), digit->getROB(), digit->getMCM());
286  }
287  if (digit->isSharedDigit() && mShareDigitsManually) {
288  LOG(error) << "Digit duplication requested, but found shared digit in input stream. Digits will be duplicated twice.";
289  }
290  if (mShareDigitsManually) {
291  if ((digit->getChannel() == 2) && !((digit->getROB() % 2 != 0) && (digit->getMCM() % NMCMROBINCOL == 3))) {
292  // shared left, if not leftmost MCM of left ROB of chamber
293  int robShared = (digit->getMCM() % NMCMROBINCOL == 3) ? digit->getROB() + 1 : digit->getROB(); // for the leftmost MCM on a ROB the shared digit is added to the neighbouring ROB
294  int mcmShared = (robShared == digit->getROB()) ? digit->getMCM() + 1 : digit->getMCM() - 3;
295  int trapIdxLeft = robShared * NMCMROB + mcmShared;
296  if (!mTrapSimulator[trapIdxLeft].isDataSet()) {
297  mTrapSimulator[trapIdxLeft].init(mTrapConfig, digit->getDetector(), robShared, mcmShared);
298  }
299  mTrapSimulator[trapIdxLeft].setData(NADCMCM - 1, digit->getADC(), digitIndices[iDigit]);
300  }
301  if ((digit->getChannel() == 18 || digit->getChannel() == 19) && !((digit->getROB() % 2 == 0) && (digit->getMCM() % NMCMROBINCOL == 0))) {
302  // shared right, if not rightmost MCM of right ROB of chamber
303  int robShared = (digit->getMCM() % NMCMROBINCOL == 0) ? digit->getROB() - 1 : digit->getROB(); // for the rightmost MCM on a ROB the shared digit is added to the neighbouring ROB
304  int mcmShared = (robShared == digit->getROB()) ? digit->getMCM() - 1 : digit->getMCM() + 3;
305  int trapIdxRight = robShared * NMCMROB + mcmShared;
306  if (!mTrapSimulator[trapIdxRight].isDataSet()) {
307  mTrapSimulator[trapIdxRight].init(mTrapConfig, digit->getDetector(), robShared, mcmShared);
308  }
309  mTrapSimulator[trapIdxRight].setData(digit->getChannel() - NCOLMCM, digit->getADC(), digitIndices[iDigit]);
310  }
311  }
312  mTrapSimulator[trapIdx].setData(digit->getChannel(), digit->getADC(), digitIndices[iDigit]);
313  }
314  // take care of the TRAPs for the last chamber
315  processTRAPchips(currDetector, nTrackletsInTrigRec, trapTrackletsAccum, lblTracklets, lblDigitsPtr);
316  inputTriggerRecords[iTrig].setTrackletRange(trapTrackletsAccum.size() - nTrackletsInTrigRec, nTrackletsInTrigRec);
317  }
318 
319  LOG(info) << "Trap simulator found " << trapTrackletsAccum.size() << " tracklets from " << inputDigits.size() << " Digits.";
320  if (mUseMC) {
321  LOG(info) << "In total " << lblTracklets.getNElements() << " MC labels are associated to the tracklets";
322  }
323  if (mShowTrackletStats > 0) {
324  std::chrono::duration<double> digitLoopTime = std::chrono::high_resolution_clock::now() - timeDigitLoopStart;
325  LOG(info) << "Trap Simulator done ";
326  LOG(info) << "Digit Sorting took " << sortTime.count() << "s";
327  LOG(info) << "Digit loop took : " << digitLoopTime.count() << "s";
328  LOG(info) << "TRAP processing TrapSimulator::filter() + TrapSimulator::tracklet() took : " << trapSimAccumulatedTime.count() << "s";
329  }
330  pc.outputs().snapshot(Output{"TRD", "TRACKLETS", 0, Lifetime::Timeframe}, trapTrackletsAccum);
331  pc.outputs().snapshot(Output{"TRD", "TRKTRGRD", 0, Lifetime::Timeframe}, inputTriggerRecords);
332  if (mUseMC) {
333  pc.outputs().snapshot(Output{"TRD", "TRKLABELS", 0, Lifetime::Timeframe}, lblTracklets);
334  }
335 
336  LOG(debug) << "TRD Trap Simulator Device exiting";
337 }

339 o2::framework::DataProcessorSpec getTRDTrapSimulatorSpec(bool useMC)
340 {
341  std::vector<InputSpec> inputs;
342  std::vector<OutputSpec> outputs;
343 
344  inputs.emplace_back("digitinput", "TRD", "DIGITS", 0);
345  inputs.emplace_back("triggerrecords", "TRD", "TRGRDIG", 0);
346 
347  outputs.emplace_back("TRD", "TRACKLETS", 0, Lifetime::Timeframe);
348  outputs.emplace_back("TRD", "TRKTRGRD", 0, Lifetime::Timeframe);
349 
350  if (useMC) {
351  inputs.emplace_back("labelinput", "TRD", "LABELS", 0);
352  outputs.emplace_back("TRD", "TRKLABELS", 0, Lifetime::Timeframe);
353  }
354 
355  return DataProcessorSpec{"TRAP",
356  inputs,
357  outputs,
358  AlgorithmSpec{adaptFromTask<TRDDPLTrapSimulatorTask>(useMC)},
359  Options{
360  {"show-trd-trackletstats", VariantType::Int, 1, {"Display the processing time of the tracklet processing in the TRAPs"}},
361  {"trd-trapconfig", VariantType::String, "cf_pg-fpnp32_zs-s16-deh_tb30_trkl-b5n-fs1e24-ht200-qs0e24s24e23-pidlinear-pt100_ptrg.r5549", {"Name of the trap config from the CCDB default:cf_pg-fpnp32_zs-s16-deh_tb30_trkl-b5n-fs1e24-ht200-qs0e24s24e23-pidlinear-pt100_ptrg.r5549"}},
362  {"trd-onlinegaincorrection", VariantType::Bool, false, {"Apply online gain calibrations, mostly for back checking to run2 by setting FGBY to 0"}},
363  {"trd-onlinegaintable", VariantType::String, "Krypton_2015-02", {"Online gain table to be use, names found in CCDB, obviously trd-onlinegaincorrection must be set as well."}},
364  {"trd-dumptrapconfig", VariantType::Bool, false, {"Dump the selected trap configuration at loading time, to text file"}},
365  {"trd-share-digits-manually", VariantType::Bool, false, {"Duplicate digits connected to shared pads if the digitizer did not already do so."}},
366  {"trd-runnum", VariantType::Int, 297595, {"Run number to use to anchor simulation to, defaults to 297595"}}}};
367 };
