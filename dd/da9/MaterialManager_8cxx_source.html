94 void MaterialManager::Mixture(const char* modname, Int_t imat, const char* name, Float_t* a, Float_t* z, Float_t dens,
95  Int_t nlmat, Float_t* wmat)
96 {
97  TString uniquename = modname;
98  uniquename.Append("_");
99  uniquename.Append(name);
100 
101  if (TVirtualMC::GetMC()) {
102  // Check this!!!
103  int kmat = -1;
104  TVirtualMC::GetMC()->Mixture(kmat, uniquename.Data(), a, z, dens * mDensityFactor, nlmat, wmat);
105  mMaterialMap[modname][imat] = kmat;
106  insertMaterialName(uniquename.Data(), kmat);
107 
108  } else {
109  auto uid = gGeoManager->GetListOfMaterials()->GetSize();
110  if (nlmat < 0) {
111  nlmat = -nlmat;
112  Double_t amol = 0;
113  Int_t i;
114  for (i = 0; i < nlmat; i++) {
115  amol += a[i] * wmat[i];
116  }
117  for (i = 0; i < nlmat; i++) {
118  wmat[i] *= a[i] / amol;
119  }
120  }
121  auto mix = gGeoManager->Mixture(uniquename.Data(), a, z, dens * mDensityFactor, nlmat, wmat, uid);
122  mMaterialMap[modname][imat] = uid;
123  insertMaterialName(uniquename.Data(), uid);
124  }

171 {
172  // this check is needed, in principal only for G3, otherwise SegFault
173  if (val < 0.) {
174  return;
175  }
176  auto it = mCutIDToName.find(cut);
177  if (it == mCutIDToName.end()) {
178  return;
179  }
180  if (special == ESpecial::kFALSE) {
181  mDefaultCutMap[cut] = val;
183  TVirtualMC::GetMC()->SetCut(it->second, val);
184  } else if (mApplySpecialCuts) {
185  mMediumCutMap[globalindex][cut] = val;
186  TVirtualMC::GetMC()->Gstpar(globalindex, it->second, val);
187  }

329 void MaterialManager::loadCutsAndProcessesFromFile(const char* modname, const char* filename)
330 {
331  // Implementation of a method to set cuts and processes as done in AliRoot.
332  // The file is expected to contain columns of the form
333  // MODNAME LOCALMEDIUMID CUT0 ... CUT9 FLAG0 ... FLAG11
334  // where cuts and flags correspond to keys denoted in cutnames and procnames below.
335 
336  const int NCUTS = 10; // number of cut columns expected in file
337  const int NFLAGS = 12; // number of process flag columns expected in file
338 
340  using namespace o2::base;
341  ECut cutnames[NCUTS] = {ECut::kCUTGAM,
342  ECut::kCUTELE,
343  ECut::kCUTNEU,
344  ECut::kCUTHAD,
345  ECut::kCUTMUO,
346  ECut::kBCUTE,
347  ECut::kBCUTM,
348  ECut::kDCUTE,
349  ECut::kDCUTM,
350  ECut::kPPCUTM};
351 
353  // missing STRA for the moment
354  EProc procnames[NFLAGS - 1] = {EProc::kANNI,
355  EProc::kBREM,
356  EProc::kCOMP,
357  EProc::kDCAY,
358  EProc::kDRAY,
359  EProc::kHADR,
360  EProc::kLOSS,
361  EProc::kMULS,
362  EProc::kPAIR,
363  EProc::kPHOT,
364  EProc::kRAYL};
365 
366  std::ifstream cutfile(filename);
367 
368  if (!cutfile.is_open()) {
369  LOG(WARN) << "File " << filename << " does not exist; Cannot apply cuts";
370  return;
371  }
372 
373  // reading from file
374  float cut[NCUTS]; // to store cut values
375  int flag[NFLAGS]; // to store flags
376  int itmed, iret;
377  char line[256];
378  char detName[7];
379 
380  while (cutfile.getline(line, 256)) {
381  // Initialise cuts and flags for this line
382  for (int i = 0; i < NCUTS; i++) {
383  cut[i] = -99;
384  }
385  for (int i = 0; i < NFLAGS; i++) {
386  flag[i] = -99;
387  }
388  if (strlen(line) == 0) {
389  continue;
390  }
391  // ignore comments marked by *
392  if (line[0] == '*') {
393  continue;
394  }
395  // Read the numbers
396  iret = sscanf(line, "%6s %d %f %f %f %f %f %f %f %f %f %f %d %d %d %d %d %d %d %d %d %d %d %d",
397  detName, &itmed, &cut[0], &cut[1], &cut[2], &cut[3], &cut[4], &cut[5], &cut[6], &cut[7], &cut[8],
398  &cut[9], &flag[0], &flag[1], &flag[2], &flag[3], &flag[4], &flag[5], &flag[6], &flag[7],
399  &flag[8], &flag[9], &flag[10], &flag[11]);
400  if (!iret) {
401  // nothing read
402  continue;
403  }
404 
405  // apply cuts via material manager interface
406  for (int i = 0; i < NCUTS; ++i) {
407  if (cut[i] >= 0.) {
408  SpecialCut(detName, itmed, cutnames[i], cut[i]);
409  }
410  }
411 
412  // apply process flags
413  for (int i = 0; i < NFLAGS - 1; ++i) {
414  if (flag[i] >= 0) {
415  SpecialProcess(detName, itmed, procnames[i], flag[i]);
416  }
417  }
418  } // end loop over lines
419 }

485 {

494  // get the biggest mapped value (maps are sorted in keys)

499  for (auto& p : lookup) {
