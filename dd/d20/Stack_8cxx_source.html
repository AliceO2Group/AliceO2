45 template <typename T, typename I>
46 void insertInVector(std::vector<T>& v, I index, T e)
47 {
48  auto currentsize = v.size();
49  if (index >= currentsize) {
50  const auto newsize = std::max(index + 1, (I)(1 + currentsize * 1.2));
51  v.resize(newsize, T(-1));
52  }
53  // new size must at least be as large as index
54  // assert(index < v.size());
55  v[index] = e;
56 }

190 void Stack::PushTrack(Int_t toBeDone, Int_t parentId, Int_t pdgCode, Double_t px, Double_t py, Double_t pz, Double_t e,
191  Double_t vx, Double_t vy, Double_t vz, Double_t time, Double_t polx, Double_t poly, Double_t polz,
192  TMCProcess proc, Int_t& ntr, Double_t weight, Int_t is, Int_t secondparentId, Int_t daughter1Id, Int_t daughter2Id,
193  TMCProcess proc2)
194 {
195  // printf("Pushing %s toBeDone %5d parentId %5d pdgCode %5d is %5d entries %5d \n",
196  // proc == kPPrimary ? "Primary: " : "Secondary: ",
197  // toBeDone, parentId, pdgCode, is, mNumberOfEntriesInParticles);
198 
199  //
200  // This method is called
201  //
202  // - during serial simulation directly from the event generator
203  // - during parallel simulation to fill the stack of the primary generator device
204  // - in all cases to push a secondary particle
205  //
206  //
207  // Create new TParticle and add it to the TParticle array
208 
209  Int_t trackId = mNumberOfEntriesInParticles;
210  // Set track variable
211  ntr = trackId;
212  // Int_t daughter1Id = -1;
213  // Int_t daughter2Id = -1;
214  Int_t iStatus = (proc == kPPrimary) ? is : trackId;
215  TParticle p(pdgCode, iStatus, parentId, secondparentId, daughter1Id, daughter2Id, px, py, pz, e, vx, vy, vz, time);
216  p.SetPolarisation(polx, poly, polz);
217  p.SetWeight(weight);
218  p.SetUniqueID(proc); // using the unique ID to transfer process ID
219  p.SetBit(ParticleStatus::kPrimary, proc == kPPrimary ? 1 : 0); // set primary bit
220  p.SetBit(ParticleStatus::kToBeDone, toBeDone == 1 ? 1 : 0); // set to be done bit
221  mNumberOfEntriesInParticles++;
222 
223  insertInVector(mTrackIDtoParticlesEntry, trackId, (int)(mParticles.size()));
224 
225  handleTransportPrimary(p); // handle selective transport of primary particles
226 
227  // Push particle on the stack if toBeDone is set
228  if (p.TestBit(ParticleStatus::kPrimary)) {
229  // This is a particle from the primary particle generator
230  //
231  // SetBit is used to pass information about the primary particle to the stack during transport.
232  // Sime particles have already decayed or are partons from a shower. They are needed for the
233  // event history in the stack, but not for transport.
234  //
235 
236  // primary particles might have been pushed with a second creation process
237  // in case we pushed a secondary track of a previous simulation to be continued.
238  // We save therefore in the UniqueID the correct process
239  // while the particle will still be treated as a primary given its bit settings
240  p.SetUniqueID(proc2);
241 
242  mIndexMap[trackId] = trackId;
243  p.SetBit(ParticleStatus::kKeep, 1);
244  if (p.TestBit(ParticleStatus::kToBeDone)) {
245  mNumberOfPrimariesforTracking++;
246  }
247  mNumberOfPrimaryParticles++;
248  mPrimaryParticles.push_back(p);
249  mTracks->emplace_back(p);
250  } else {
251  mParticles.emplace_back(p);
252  mCurrentParticle0 = p;
253  }
254  mStack.push(p);
255 }

257 void Stack::handleTransportPrimary(TParticle& p)
258 {
259  // this function tests whether we really want to transport
260  // this particle and sets the relevant bits accordingly
261 
262  if (!p.TestBit(ParticleStatus::kToBeDone) || !p.TestBit(ParticleStatus::kPrimary)) {
263  return;
264  }
265 
266  if (!mTransportPrimary(p, mPrimaryParticles)) {
267  p.SetBit(ParticleStatus::kToBeDone, 0);
268  p.SetBit(ParticleStatus::kInhibited, 1);
269  }

308 // calculates a hash based on particle properties
309 // hash may serve as seed for this track
310 ULong_t getHash(TParticle const& p)
311 {
312  auto asLong = [](double x) {
313  return (ULong_t) * (reinterpret_cast<ULong_t*>(&x));
314  };
315 
316  ULong_t hash;
317  o2::MCTrackT<double> track(p);
318 
319  hash = asLong(track.GetStartVertexCoordinatesX());
320  hash ^= asLong(track.GetStartVertexCoordinatesY());
321  hash ^= asLong(track.GetStartVertexCoordinatesZ());
322  hash ^= asLong(track.GetStartVertexCoordinatesT());
323  hash ^= asLong(track.GetStartVertexMomentumX());
324  hash ^= asLong(track.GetStartVertexMomentumY());
325  hash ^= asLong(track.GetStartVertexMomentumZ());
326  hash += (ULong_t)track.GetPdgCode();
327  return hash;
328 }

330 TParticle* Stack::PopNextTrack(Int_t& iTrack)
331 {
332 
333  // This functions is mainly used by Geant3?
334  Int_t nprod = (int)(mParticles.size());
335 
336  // If end of stack: Return empty pointer
337  if (mStack.empty()) {
338  iTrack = -1;
339  return nullptr;
340  }
341  Bool_t found = kFALSE;
342 
343  TParticle* nextParticle = nullptr;
344  while (!found && !mStack.empty()) {
345  // get next particle from stack
346  mCurrentParticle = mStack.top();
347  // remove particle from the top
348  mStack.pop();
349  // test if primary to be transported
350  if (mCurrentParticle.TestBit(ParticleStatus::kToBeDone)) {
351  if (mCurrentParticle.TestBit(ParticleStatus::kPrimary)) {
352  // particle is primary and needs to be tracked -> indicates that previous particle finished
353  mNumberOfPrimariesPopped++;
354  mIndexOfCurrentPrimary = mStack.size();
355  mIndexOfCurrentTrack = mIndexOfCurrentPrimary;
356  } else {
357  mIndexOfCurrentTrack = mCurrentParticle.GetStatusCode();
358  }
359  iTrack = mIndexOfCurrentTrack;
360  if (o2::conf::SimCutParams::Instance().trackSeed) {
361  auto hash = getHash(mCurrentParticle);
362  // LOG(INFO) << "SEEDING NEW TRACK USING HASH" << hash;
363  // init seed per track
364  gRandom->SetSeed(hash);
365  // NOTE: THE BETTER PLACE WOULD BE IN PRETRACK HOOK BUT THIS DOES NOT SEEM TO WORK
366  // WORKS ONLY WITH G3 SINCE G4 DOES NOT CALL THIS FUNCTION
367  } // .trackSeed ?
368  nextParticle = &mCurrentParticle;
369  found = kTRUE;
370  } else {
371  iTrack = -1;
372  nextParticle = nullptr;
373  }
374  } // while
375  return nextParticle;
376 }

416 void Stack::FinishPrimary()
417 {
418  // Here transport of a primary and all its secondaries is finished
419  // we can do some cleanup of the memory structures
420  mPrimariesDone++;
421  LOG(DEBUG) << "Finish primary hook " << mPrimariesDone;
422  // preserve particles and theire ancestors that produced hits
423 
424  auto selected = selectTracks();
425 
426  // loop over current particle buffer
427  // - build index map indicesKept
428  // - update mother index information
429  int indexOld = 0;
430  int indexNew = 0;
431  int indexoffset = mTracks->size();
432  int neglected = 0;
433  std::vector<int> indicesKept((int)(mParticles.size()));
434  std::vector<MCTrack> tmpTracks;
435  Int_t ic = 0;
436 
437  // mTrackIDtoParticlesEntry
438  // trackID to mTrack -> index in mParticles
439  //
440  // indicesKept
441  // index in mParticles -> index in mTrack - highWaterMark
442  //
443  // mReorderIndices
444  // old (mTrack-highWaterMark) -> new (mTrack-highWaterMark)
445  //
446 
447  for (auto& particle : mParticles) {
448  if (particle.getStore() || !mPruneKinematics) {
449  // map the global track index to the new persistent index
450  auto imother = particle.getMotherTrackId();
451  // here mother is relative to the mParticles array
452  if (imother >= 0) {
453  // daughter of a secondary: obtain index from lookup table
454  imother = indicesKept[imother];
455  particle.SetMotherTrackId(imother);
456  }
457  // at this point we have the correct mother index in mParticles or
458  // a negative one which is a pointer to a primary
459  tmpTracks.emplace_back(particle);
460  indicesKept[indexOld] = indexNew;
461  indexNew++;
462  } else {
463  indicesKept[indexOld] = -1;
464  neglected++;
465  }
466  indexOld++;
467  mTracksDone++;
468  }
469  Int_t ntr = (int)(tmpTracks.size());
470  std::vector<int> reOrderedIndices(ntr);
471  std::vector<int> invreOrderedIndices(ntr);
472  for (Int_t i = 0; i < ntr; i++) {
473  invreOrderedIndices[i] = i;
474  reOrderedIndices[i] = i;
475  }
476 
477  if (mIsG4Like) {
478  ReorderKine(tmpTracks, reOrderedIndices);
479  for (Int_t i = 0; i < ntr; i++) {
480  Int_t index = reOrderedIndices[i];
481  invreOrderedIndices[index] = i;
482  }
483  }
484  for (Int_t i = 0; i < ntr; i++) {
485  Int_t index = reOrderedIndices[i];
486  auto& particle = tmpTracks[index];
487  Int_t imo = particle.getMotherTrackId();
488  Int_t imo0 = imo;
489  if (imo >= 0) {
490  imo = invreOrderedIndices[imo];
491  }
492  imo += indexoffset;
493  particle.SetMotherTrackId(imo);
494  mTracks->emplace_back(particle);
495  auto& mother = mTracks->at(imo);
496  if (mother.getFirstDaughterTrackId() == -1) {
497  mother.SetFirstDaughterTrackId((int)(mTracks->size()) - 1);
498  }
499  mother.SetLastDaughterTrackId((int)(mTracks->size()) - 1);
500  }
501 
502  //
503  // Update index map
504  //
505  Int_t imax = mNumberOfEntriesInParticles;
506  Int_t imin = imax - mParticles.size();
507  for (Int_t idTrack = imin; idTrack < imax; idTrack++) {
508  Int_t index1 = mTrackIDtoParticlesEntry[idTrack];
509  Int_t index2 = indicesKept[index1];
510  if (index2 == -1) {
511  continue;
512  }
513  Int_t index3 = (mIsG4Like) ? invreOrderedIndices[index2] : index2;
514  mIndexMap[idTrack] = index3 + indexoffset;
515  }
516 
517  // we can now clear the particles buffer!
518  reOrderedIndices.clear();
519  invreOrderedIndices.clear();
520  mParticles.clear();
521  tmpTracks.clear();
522  mTransportedIDs.clear();
523  mTrackIDtoParticlesEntry.clear();
524  mIndexOfPrimaries.clear();
525 }

527 void Stack::UpdateTrackIndex(TRefArray* detList)
528 {
529  // we can avoid any updating in case no tracks have been filtered out
530  // check this like this
531  if (mIndexMap.size() == 0) {
532  LOG(INFO) << "No TrackIndex update necessary\n";
533  return;
534  }
535 
536  // we are getting the detectorlist from FairRoot as TRefArray
537  // at each call, but this list never changes so we cache it here
538  // as the right type to avoid repeated dynamic casts
539  if (mActiveDetectors.size() == 0) {
540  if (detList == nullptr) {
541  LOG(FATAL) << "No detList passed to Stack";
542  }
543  auto iter = detList->MakeIterator();
544  while (auto det = iter->Next()) {
545  auto o2det = dynamic_cast<o2::base::Detector*>(det);
546  if (o2det) {
547  mActiveDetectors.emplace_back(o2det);
548  } else {
549  LOG(INFO) << "Found nonconforming detector " << det->GetName();
550  }
551  }
552  }
553 
554  LOG(DEBUG) << "Stack::UpdateTrackIndex: Stack: Updating track indices...";
555  Int_t nColl = 0;
556 
557  // update track references
558  // use some caching since repeated trackIDs
559  for (auto& ref : *mTrackRefs) {
560  const auto id = ref.getTrackID();
561  auto iter = mIndexMap.find(id);
562  if (iter == mIndexMap.end()) {
563  LOG(INFO) << "Invalid trackref ... needs to be rmoved \n";
564  ref.setTrackID(-1);
565  } else {
566  ref.setTrackID(iter->second);
567  }
568  }
569 
570  // sort trackrefs according to new track index
571  // then according to track length
572  std::sort(mTrackRefs->begin(), mTrackRefs->end(), [](const o2::TrackReference& a, const o2::TrackReference& b) {
573  if (a.getTrackID() == b.getTrackID()) {
574  return a.getLength() < b.getLength();
575  }
576  return a.getTrackID() < b.getTrackID();
577  });
578 
579  for (auto det : mActiveDetectors) {
580  // update the track indices by delegating to specialized detector functions
581  det->updateHitTrackIndices(mIndexMap);
582  } // List of active detectors
583 
584  LOG(DEBUG) << "Stack::UpdateTrackIndex: ...stack and " << nColl << " collections updated.";
585 }

587 void Stack::Reset()
588 {
589 
590  mIndexOfCurrentTrack = -1;
591  mNumberOfPrimaryParticles = mNumberOfEntriesInParticles = mNumberOfEntriesInTracks = 0;
592  while (!mStack.empty()) {
593  mStack.pop();
594  }
595  mParticles.clear();
596  mTracks->clear();
597  if (!mIsExternalMode && (mPrimariesDone != mNumberOfPrimariesforTracking)) {
598  LOG(FATAL) << "Inconsistency in primary particles treated " << mPrimariesDone << " vs expected "
599  << mNumberOfPrimariesforTracking << "\n(This points to a flaw in the stack logic)";
600  }
601  mPrimariesDone = 0;
602  mNumberOfPrimariesforTracking = 0;
603  mNumberOfPrimariesPopped = 0;
604  mPrimaryParticles.clear();
605  mTrackRefs->clear();
606  mTrackIDtoParticlesEntry.clear();
607  mHitCounter = 0;
608 }

617 {

667 bool Stack::selectTracks()
668 {
669  bool tracksdiscarded = false;
670  // Check particles in the fParticle array
671  int prim = -1; // counter how many primaries seen (mainly to constrain search in motherindex remapping)
672  LOG(DEBUG) << "Stack: Entering track selection on " << mParticles.size() << " tracks";
673  for (auto& thisPart : mParticles) {
674  Bool_t store = kTRUE;
675  // Get track parameters
676  Bool_t isPrimary = (thisPart.getProcess() == 0);
677  Int_t iMother = thisPart.getMotherTrackId();
678  auto& mother = mParticles[iMother];
679  Bool_t motherIsPrimary = (iMother < mNumberOfPrimaryParticles);
680 
681  if (isPrimary) {
682  prim++;
683  // for primaries we are done quickly
684  // in fact the primary has already been stored
685  // so this should not happen
686  store = kTRUE;
687  } else {
688  // for other particles we potentially need to correct the mother indices
689  if (!motherIsPrimary) {
690  // the mapping is from the index in the stack to the index in mParticles
691  thisPart.SetMotherTrackId(mTrackIDtoParticlesEntry[iMother]);
692  } else {
693  // for a secondary which is a direct decendant of a primary use a negative index which will be restored later
694  thisPart.SetMotherTrackId(iMother - (int)(mTracks->size()));
695  }
696  // no secondaries; also done
697  if (!mStoreSecondaries) {
698  store = kFALSE;
699  tracksdiscarded = true;
700  } else {
701  // Calculate number of hits created by this track
702  // Note: we only distinguish between no hit and more than 0 hits
703  int nHits = thisPart.hasHits();
704  // Check for cuts (store primaries in any case)
705  if (nHits < mMinHits) {
706  store = kFALSE;
707  tracksdiscarded = true;
708  }
709  // only if we have non-trival energy cut
710  if (mEnergyCut > 0.) {
711  Double_t energy = thisPart.GetEnergy();
712  Double_t mass = thisPart.GetMass();
713  Double_t eKin = energy - mass;
714  if (eKin < mEnergyCut) {
715  store = kFALSE;
716  tracksdiscarded = true;
717  }
718  }
719  if (keepPhysics(thisPart)) {
720  store = kTRUE;
721  tracksdiscarded = false;
722  }
723  }
724  }
725 
726  store = store || thisPart.getStore();
727  thisPart.setStore(store);
728  }
729 
730  // If flag is set, flag recursively mothers of selected tracks
731  //
732  if (mStoreMothers) {
733  for (auto& particle : mParticles) {
734  if (particle.getStore()) {
735  Int_t iMother = particle.getMotherTrackId();
736  while (iMother >= 0) {
737  auto& mother = mParticles[iMother];
738  mother.setStore(true);
739  iMother = mother.getMotherTrackId();
740  } // while mother
741  } // store ?
742  } // particle loop
743  }
744 
745  return !tracksdiscarded;
746 }

856 void Stack::ReorderKine(std::vector<MCTrack>& particles, std::vector<int>& reOrderedIndices)
857 {
858  //
859  // Particles are ordered in a way that descendants of a particle appear next to each other.
860  // This has the advantage that their position in the stack can be identified by two number,
861  // for example the index of the first and last descentant.
862  // The result of the ordering is returned via the look-up table reOrderedIndices
863  //
864 
865  Int_t ntr = (int)(particles.size());
866  std::vector<bool> done(ntr, false);
867 
868  int indexoffset = mTracks->size();
869  Int_t index = 0;
870  Int_t imoOld = 0;
871  for (Int_t i = 0; i < ntr; i++) {
872  reOrderedIndices[i] = i;
873  }
874 
875  for (Int_t i = -1; i < ntr; i++) {
876  if (i != -1) {
877  // secondaries
878  if (!done[i]) {
879  reOrderedIndices[index] = i;
880  index++;
881  done[i] = true;
882  }
883  imoOld = i;
884  } else {
885  // current primary
886  imoOld = mIndexOfCurrentPrimary - indexoffset;
887  }
888  for (Int_t j = i + 1; j < ntr; j++) {
889  if (!done[j]) {
890  if ((particles[j]).getMotherTrackId() == imoOld) {
891  reOrderedIndices[index] = j;
892  index++;
893  done[j] = true;
894  } // child found
895  } // done
896  } // j
897  } // i
898 }
