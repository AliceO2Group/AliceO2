115 struct ATask {
116  void process(aod::Eins const& unos, aod::Zwei const& dues, aod::Drei const& tres)
117  {
118  int cnt = 0;
119  for (auto& uno : unos) {
120  auto eta = uno.eta();
121  auto mom = uno.mom();
122 
123  cnt++;
124  LOGF(INFO, "Eins (%i): (%f, %f)", cnt, eta, mom);
125  }
126  LOGF(INFO, "ATask Processed %i data points from Eins", cnt);
127 
128  cnt = 0;
129  for (auto& due : dues) {
130  auto ok = due.ok();
131  auto phi = due.phi();
132  auto pt = due.pt();
133 
134  cnt++;
135  LOGF(INFO, "Zwei (%i): (%i, %f, %f)", cnt, ok, phi, pt);
136  }
137  LOGF(INFO, "ATask Processed %i data points from Zwei", cnt);
138 
139  cnt = 0;
140  for (auto& tre : tres) {
141  auto eta = tre.eta();
142  auto phi = tre.phi();
143  auto mom = tre.mom();
144 
145  cnt++;
146  LOGF(INFO, "Drei (%i): (%f, %f, %f)", cnt, eta, phi, mom);
147  }
148  LOGF(INFO, "ATask Processed %i data points from Drei", cnt);
149  }
150 };
