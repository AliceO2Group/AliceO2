56 void CosmicsMatchingSpec::init(InitContext& ic)
57 {
58  mTimer.Stop();
59  mTimer.Reset();
60  //-------- init geometry and field --------//
61  o2::base::GeometryManager::loadGeometry();
62  o2::base::Propagator::initFieldFromGRP(o2::base::NameConf::getGRPFileName());
63  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom(o2::base::NameConf::getGRPFileName())};
64  const auto& alpParams = o2::itsmft::DPLAlpideParam<DetID::ITS>::Instance();
65  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
66  mMatching.setITSROFrameLengthMUS(alpParams.roFrameLengthTrig / 1.e3); // ITS ROFrame duration in \mus
67  } else {
68  mMatching.setITSROFrameLengthMUS(alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3); // ITS ROFrame duration in \mus
69  }
70  //
71  std::string dictPath = ic.options().get<std::string>("its-dictionary-path");
72  std::string dictFile = o2::base::NameConf::getDictionaryFileName(DetID::ITS, dictPath, ".bin");
73  auto itsDict = std::make_unique<o2::itsmft::TopologyDictionary>();
74  if (o2::base::NameConf::pathExists(dictFile)) {
75  itsDict->readBinaryFile(dictFile);
76  LOG(INFO) << "Matching is running with a provided ITS dictionary: " << dictFile;
77  } else {
78  LOG(INFO) << "Dictionary " << dictFile << " is absent, Matching expects ITS cluster patterns";
79  }
80  o2::its::GeometryTGeo::Instance()->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2GRot) | o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L));
81  mMatching.setITSDict(itsDict);
82 
83  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
84  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
85  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
86  if (o2::base::NameConf::pathExists(matLUTFile)) {
87  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
88  o2::base::Propagator::Instance()->setMatLUT(lut);
89  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
90  } else {
91  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
92  }
93  mMatching.setUseMC(mUseMC);
94  mMatching.init();
95  //
96 }

98 void CosmicsMatchingSpec::run(ProcessingContext& pc)
99 {
100  mTimer.Start(false);
101 
102  RecoContainer recoData;
103  recoData.collectData(pc, dataRequestCosm);
104  mMatching.process(recoData);
105  pc.outputs().snapshot(Output{"GLO", "COSMICTRC", 0, Lifetime::Timeframe}, mMatching.getCosmicTracks());
106  if (mUseMC) {
107  pc.outputs().snapshot(Output{"GLO", "COSMICTRC_MC", 0, Lifetime::Timeframe}, mMatching.getCosmicTracksLbl());
108  }
109  mTimer.Stop();
110 }

118 DataProcessorSpec getCosmicsMatchingSpec(GTrackID::mask_t src, bool useMC)
119 {
120 
121  std::vector<InputSpec> inputs;
122  std::vector<OutputSpec> outputs;
123 
124  dataRequestCosm.requestTracks(src, useMC);
125  dataRequestCosm.requestClusters(src, false); // no MC labels for clusters needed for refit only
126 
127  outputs.emplace_back("GLO", "COSMICTRC", 0, Lifetime::Timeframe);
128  if (useMC) {
129  outputs.emplace_back("GLO", "COSMICTRC_MC", 0, Lifetime::Timeframe);
130  }
131 
132  return DataProcessorSpec{
133  "cosmics-matcher",
134  dataRequestCosm.inputs,
135  outputs,
136  AlgorithmSpec{adaptFromTask<CosmicsMatchingSpec>(useMC)},
137  Options{
138  {"its-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
139  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
140 }
