37 int ioutils::loadROFrameData(const o2::itsmft::ROFRecord& rof, ROframe& event, gsl::span<const itsmft::CompClusterExt> clusters, gsl::span<const unsigned char>::iterator& pattIt, const itsmft::TopologyDictionary& dict,
38  const dataformats::MCTruthContainer<MCCompLabel>* mcLabels, const o2::mft::Tracker* tracker)
39 {
40  event.clear();
41  GeometryTGeo* geom = GeometryTGeo::Instance();
42  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
43  int clusterId{0};
44  auto first = rof.getFirstEntry();
45  auto clusters_in_frame = rof.getROFData(clusters);
46  for (auto& c : clusters_in_frame) {
47  int layer = geom->getLayer(c.getSensorID());
48  auto pattID = c.getPatternID();
49  o2::math_utils::Point3D<float> locXYZ;
50  float sigmaX2 = ioutils::DefClusError2Row, sigmaY2 = ioutils::DefClusError2Col; //Dummy COG errors (about half pixel size)
51  if (pattID != itsmft::CompCluster::InvalidPatternID) {
52  //sigmaX2 = dict.getErr2X(pattID); // ALPIDE local X coordinate => MFT global X coordinate (ALPIDE rows)
53  //sigmaY2 = dict.getErr2Z(pattID); // ALPIDE local Z coordinate => MFT global Y coordinate (ALPIDE columns)
54  // temporary, until ITS bug fix
55  sigmaX2 = dict.getErrX(pattID) * dict.getErrX(pattID);
56  sigmaY2 = dict.getErrZ(pattID) * dict.getErrZ(pattID);
57  if (!dict.isGroup(pattID)) {
58  locXYZ = dict.getClusterCoordinates(c);
59  } else {
60  o2::itsmft::ClusterPattern patt(pattIt);
61  locXYZ = dict.getClusterCoordinates(c, patt, false);
62  }
63  } else {
64  o2::itsmft::ClusterPattern patt(pattIt);
65  locXYZ = dict.getClusterCoordinates(c, patt);
66  }
67  auto sensorID = c.getSensorID();
68  // Transformation to the local --> global
69  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
70 
71  auto clsPoint2D = math_utils::Point2D<Float_t>(gloXYZ.x(), gloXYZ.y());
72  Float_t rCoord = clsPoint2D.R();
73  Float_t phiCoord = clsPoint2D.Phi();
74  o2::math_utils::bringTo02PiGen(phiCoord);
75  int rBinIndex = tracker->getRBinIndex(rCoord);
76  int phiBinIndex = tracker->getPhiBinIndex(phiCoord);
77  int binIndex = tracker->getBinIndex(rBinIndex, phiBinIndex);
78  // TODO: Check consistency of sigmaX2 and sigmaY2
79  event.addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), phiCoord, rCoord, event.getClustersInLayer(layer).size(), binIndex, sigmaX2, sigmaY2, sensorID);
80  if (mcLabels) {
81  event.addClusterLabelToLayer(layer, *(mcLabels->getLabels(first + clusterId).begin()));
82  }
83  event.addClusterExternalIndexToLayer(layer, first + clusterId);
84  clusterId++;
85  }
86  return clusters_in_frame.size();
87 }
