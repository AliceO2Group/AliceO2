31 template <class T>
32 TCanvas* painter::draw(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, TCanvas* outputCanvas)
33 {
34  using DetType = CalDet<T>;
35  using CalType = CalArray<T>;
36 
37  static const Mapper& mapper = Mapper::instance();
38 
39  // ===| name and title |======================================================
40  const auto title = calDet.getName().c_str();
41  std::string name = calDet.getName();
42  std::replace(name.begin(), name.end(), ' ', '_');
43 
44  // ===| define histograms |===================================================
45  // TODO: auto scaling of ranges based on mean and variance?
46  // for the moment use roots auto scaling
47 
48  // set buffer size such that autoscaling uses the full range. This is about 2MB per histogram!
49  const int bufferSize = TH1::GetDefaultBufferSize();
50  TH1::SetDefaultBufferSize(Sector::MAXSECTOR * mapper.getPadsInSector());
51 
52  auto hAside1D = new TH1F(Form("h_Aside_1D_%s", name.c_str()), Form("%s (A-Side)", title),
53  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
54 
55  auto hCside1D = new TH1F(Form("h_Cside_1D_%s", name.c_str()), Form("%s (C-Side)", title),
56  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
57 
58  auto hAside2D = new TH2F(Form("h_Aside_2D_%s", name.c_str()), Form("%s (A-Side);x (cm);y (cm)", title),
59  300, -300, 300, 300, -300, 300);
60 
61  auto hCside2D = new TH2F(Form("h_Cside_2D_%s", name.c_str()), Form("%s (C-Side);x (cm);y (cm)", title),
62  300, -300, 300, 300, -300, 300);
63 
64  for (ROC roc; !roc.looped(); ++roc) {
65 
66  auto hist2D = hAside2D;
67  auto hist1D = hAside1D;
68  if (roc.side() == Side::C) {
69  hist2D = hCside2D;
70  hist1D = hCside1D;
71  }
72 
73  const int nrows = mapper.getNumberOfRowsROC(roc);
74  for (int irow = 0; irow < nrows; ++irow) {
75  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
76  for (int ipad = 0; ipad < npads; ++ipad) {
77  const auto val = calDet.getValue(roc, irow, ipad);
78  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
79  const int bin = hist2D->FindBin(pos.X(), pos.Y());
80  if (!hist2D->GetBinContent(bin)) {
81  hist2D->SetBinContent(bin, val);
82  }
83  hist1D->Fill(val);
84  }
85  }
86  }
87 
88  if (xMax1D > xMin1D) {
89  hAside2D->SetMinimum(xMin1D);
90  hAside2D->SetMaximum(xMax1D);
91  hCside2D->SetMinimum(xMin1D);
92  hCside2D->SetMaximum(xMax1D);
93  }
94 
95  // ===| Draw histograms |=====================================================
96  auto c = outputCanvas;
97  if (!c) {
98  c = new TCanvas(Form("c_%s", name.c_str()), title, 1000, 1000);
99  }
100  c->Clear();
101  c->Divide(2, 2);
102 
103  c->cd(1);
104  hAside2D->Draw("colz");
105 
106  c->cd(2);
107  hCside2D->Draw("colz");
108 
109  c->cd(3);
110  hAside1D->Draw();
111 
112  c->cd(4);
113  hCside1D->Draw();
114 
115  // reset the buffer size
116  TH1::SetDefaultBufferSize(bufferSize);
117 
118  return c;
119 }

137 template <class T>
138 void painter::fillHistogram2D(TH2& h2D, const CalDet<T>& calDet, Side side)
139 {
140  static const Mapper& mapper = Mapper::instance();
141 
142  for (ROC roc; !roc.looped(); ++roc) {
143  if (roc.side() != side) {
144  continue;
145  }
146 
147  const int nrows = mapper.getNumberOfRowsROC(roc);
148  for (int irow = 0; irow < nrows; ++irow) {
149  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
150  for (int ipad = 0; ipad < npads; ++ipad) {
151  const auto val = calDet.getValue(roc, irow, ipad);
152  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
153  const int bin = h2D.FindBin(pos.X(), pos.Y());
154  if (!h2D.GetBinContent(bin)) {
155  h2D.SetBinContent(bin, val);
156  }
157  }
158  }
159  }
160 }

163 template <class T>
164 void painter::fillHistogram2D(TH2& h2D, const CalArray<T>& calArray)
165 {
166  static const Mapper& mapper = Mapper::instance();
167 
168  const size_t position = calArray.getPadSubsetNumber();
169  const PadSubset padSubset = calArray.getPadSubset();
170  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
171 
172  // ===| fill hist |===========================================================
173  for (int irow = 0; irow < nrows; ++irow) {
174  const int padsInRow = mapper.getNumberOfPadsInRow(padSubset, position, irow);
175  for (int ipad = 0; ipad < padsInRow; ++ipad) {
176  const GlobalPadNumber pad = mapper.getPadNumber(padSubset, position, irow, ipad);
177  const auto val = calArray.getValue(pad);
178  const int cpad = ipad - padsInRow / 2;
179  h2D.Fill(irow, cpad, val);
180  }
181  }
182 }

185 template <class T>
186 TH2* painter::getHistogram2D(const CalDet<T>& calDet, Side side)
187 {
188  const auto title = calDet.getName().c_str();
189  std::string name = calDet.getName();
190  std::replace(name.begin(), name.end(), ' ', '_');
191  const char side_name = (side == Side::A) ? 'A' : 'C';
192 
193  auto h2D = new TH2F(Form("h_%cside_2D_%s", side_name, name.c_str()),
194  Form("%s (%c-Side);x (cm);y (cm)", title, side_name),
195  300, -300, 300, 300, -300, 300);
196 
197  fillHistogram2D(*h2D, calDet, side);
198 
199  return h2D;
200 }

203 template <class T>
204 TH2* painter::getHistogram2D(const CalArray<T>& calArray)
205 {
206  static const Mapper& mapper = Mapper::instance();
207 
208  const size_t position = calArray.getPadSubsetNumber();
209  const PadSubset padSubset = calArray.getPadSubset();
210 
211  // ===| maximum number of rows and pads |=====================================
212  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
213  const int npads = mapper.getNumberOfPadsInRow(padSubset, position, nrows - 1) + 6;
214 
215  // ===| create histogram |====================================================
216  const auto title = calArray.getName().c_str();
217  std::string name = calArray.getName();
218  std::replace(name.begin(), name.end(), ' ', '_');
219  auto hist = new TH2F(Form("h_%s", name.c_str()),
220  Form("%s;pad row;pad", title),
221  nrows, 0., nrows,
222  npads, -npads / 2, npads / 2);
223 
224  fillHistogram2D(*hist, calArray);
225 
226  return hist;
227 }

241 template <class T>
242 std::vector<TCanvas*> painter::makeSummaryCanvases(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, bool onlyFilled, std::vector<TCanvas*>* outputCanvases)
243 {
244 
245  std::vector<TCanvas*> vecCanvases;
246 
247  auto nROCs = calDet.getData().size();
248 
249  if (onlyFilled) {
250  nROCs = 0;
251  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
252  const auto& roc = calDet.getCalArray(iroc);
253 
254  if (hasData(roc)) {
255  ++nROCs;
256  }
257  }
258 
259  if (!nROCs) {
260  return vecCanvases;
261  }
262  }
263 
264  // ===| set up canvases |===
265  TCanvas* cSides = nullptr;
266  TCanvas* cROCs1D = nullptr;
267  TCanvas* cROCs2D = nullptr;
268  const std::string_view calName = calDet.getName();
269 
270  if (outputCanvases) {
271  if (outputCanvases->size() < 3) {
272  LOGP(error, "At least 3 canvases are needed to fill the output, only {} given", outputCanvases->size());
273  return vecCanvases;
274  }
275 
276  cSides = outputCanvases->at(0);
277  cROCs1D = outputCanvases->at(1);
278  cROCs2D = outputCanvases->at(2);
279  cSides->Clear();
280  cROCs1D->Clear();
281  cROCs2D->Clear();
282  } else {
283 
284  cROCs1D = new TCanvas(fmt::format("c_ROCs_{}_1D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
285  cROCs2D = new TCanvas(fmt::format("c_ROCs_{}_2D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
286  }
287  vecCanvases.emplace_back(cSides);
288  vecCanvases.emplace_back(cROCs1D);
289  vecCanvases.emplace_back(cROCs2D);
290 
291  cSides = draw(calDet, nbins1D, xMin1D, xMax1D, cSides);
292  cROCs1D->DivideSquare(nROCs);
293  cROCs2D->DivideSquare(nROCs);
294 
295  // ===| produce plots for each ROC |===
296  size_t pad = 1;
297  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
298  const auto& roc = calDet.getCalArray(iroc);
299 
300  if (onlyFilled && !hasData(roc)) {
301  continue;
302  }
303 
304  // ===| 1D histogram |===
305  auto h1D = new TH1F(fmt::format("h_{}_{:02d}", calName, iroc).data(), fmt::format("{} distribution ROC {:02d};ADC value", calName, iroc).data(), nbins1D, xMin1D, xMax1D);
306  for (const auto& val : roc.getData()) {
307  h1D->Fill(val);
308  }
309 
310  // ===| 2D histogram |===
311  auto h2D = painter::getHistogram2D(roc);
312  h2D->SetStats(0);
313  if (xMax1D > xMin1D) {
314  h2D->SetMinimum(xMin1D);
315  h2D->SetMaximum(xMax1D);
316  }
317  h2D->SetUniqueID(iroc);
318 
319  cROCs1D->cd(pad);
320  h1D->Draw();
321 
322  cROCs2D->cd(pad);
323  h2D->Draw("colz");
324 
325  ++pad;
326  }
327 
328  return vecCanvases;
329 }

332 std::vector<TCanvas*> painter::makeSummaryCanvases(const std::string_view fileName, const std::string_view calPadNames, int nbins1D, float xMin1D, float xMax1D, bool onlyFilled)
333 {
