30 void TrackParFwd::propagateParamToZlinear(double zEnd)
31 {
32  // Track parameters linearly extrapolated to the plane at "zEnd".
33 
34  if (getZ() == zEnd) {
35  return; // nothing to be done if same z
36  }
37 
38  // Compute track parameters
39  auto dZ = (zEnd - getZ());
40  auto phi0 = getPhi();
41  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
42  auto invtanl0 = 1.0 / getTanl();
43  auto n = dZ * invtanl0;
44  mParameters(0) += n * cosphi0;
45  mParameters(1) += n * sinphi0;
46  mZ = zEnd;
47 }

50 void TrackParCovFwd::propagateToZlinear(double zEnd)
51 {
52  // Track parameters and their covariances linearly extrapolated to the plane at "zEnd".
53 
54  // Calculate the jacobian related to the track parameters extrapolated to "zEnd"
55  auto dZ = (zEnd - getZ());
56  auto phi0 = getPhi();
57  auto tanl0 = getTanl();
58  auto invtanl0 = 1.0 / tanl0;
59  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
60  auto n = dZ * invtanl0;
61  auto m = n * invtanl0;
62 
63  // Extrapolate track parameters to "zEnd"
64  mParameters(0) += n * cosphi0;
65  mParameters(1) += n * sinphi0;
66  setZ(zEnd);
67 
68  // Calculate Jacobian
69  SMatrix55 jacob = ROOT::Math::SMatrixIdentity();
70  jacob(0, 2) = -n * sinphi0;
71  jacob(0, 3) = -m * cosphi0;
72  jacob(1, 2) = n * cosphi0;
73  jacob(1, 3) = -m * sinphi0;
74 
75  // Extrapolate track parameter covariances to "zEnd"
76  setCovariances(ROOT::Math::Similarity(jacob, mCovariances));
77 }

80 void TrackParFwd::propagateParamToZquadratic(double zEnd, double zField)
81 {
82  // Track parameters extrapolated to the plane at "zEnd" considering a helix
83 
84  if (getZ() == zEnd) {
85  return; // nothing to be done if same z
86  }
87 
88  // Compute track parameters
89  auto dZ = (zEnd - getZ());
90  auto phi0 = getPhi();
91  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
92  auto invtanl0 = 1.0 / getTanl();
93  auto invqpt0 = getInvQPt();
94  auto Hz = std::copysign(1, zField);
95  auto k = TMath::Abs(o2::constants::math::B2C * zField);
96  auto n = dZ * invtanl0;
97  auto theta = -invqpt0 * dZ * k * invtanl0;
98 
99  mParameters(0) += n * cosphi0 - 0.5 * n * theta * Hz * sinphi0;
100  mParameters(1) += n * sinphi0 + 0.5 * n * theta * Hz * cosphi0;
101  mParameters(2) += Hz * theta;
102  setZ(zEnd);
103 }

106 void TrackParCovFwd::propagateToZquadratic(double zEnd, double zField)
107 {
108  // Extrapolate track parameters and covariances matrix to "zEnd"
109  // using quadratic track model
110 
111  if (getZ() == zEnd) {
112  return; // nothing to be done if same z
113  }
114 
115  // Compute track parameters
116  auto dZ = (zEnd - getZ());
117  auto phi0 = getPhi();
118  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
119  auto invtanl0 = 1.0 / getTanl();
120  auto invqpt0 = getInvQPt();
121  auto Hz = std::copysign(1, zField);
122  auto k = TMath::Abs(o2::constants::math::B2C * zField);
123  auto n = dZ * invtanl0;
124  auto m = n * invtanl0;
125  auto theta = -invqpt0 * dZ * k * invtanl0;
126 
127  // Extrapolate track parameters to "zEnd"
128  mParameters(0) += n * cosphi0 - 0.5 * n * theta * Hz * sinphi0;
129  mParameters(1) += n * sinphi0 + 0.5 * n * theta * Hz * cosphi0;
130  mParameters(2) += Hz * theta;
131  mZ = zEnd;
132 
133  // Calculate Jacobian
134  SMatrix55 jacob = ROOT::Math::SMatrixIdentity();
135  jacob(0, 2) = -n * theta * 0.5 * Hz * cosphi0 - n * sinphi0;
136  jacob(0, 3) = Hz * m * theta * sinphi0 - m * cosphi0;
137  jacob(0, 4) = k * m * 0.5 * Hz * dZ * sinphi0;
138  jacob(1, 2) = -n * theta * 0.5 * Hz * sinphi0 + n * cosphi0;
139  jacob(1, 3) = -Hz * m * theta * cosphi0 - m * sinphi0;
140  jacob(1, 4) = -k * m * 0.5 * Hz * dZ * cosphi0;
141  jacob(2, 3) = -Hz * theta * invtanl0;
142  jacob(2, 4) = -Hz * k * n;
143 
144  // Extrapolate track parameter covariances to "zEnd"
145  setCovariances(ROOT::Math::Similarity(jacob, mCovariances));
146 }

149 void TrackParFwd::propagateParamToZhelix(double zEnd, double zField)
150 {
151  // Track parameters extrapolated to the plane at "zEnd"
152  // using helix track model
153 
154  if (getZ() == zEnd) {
155  return; // nothing to be done if same z
156  }
157 
158  // Compute track parameters
159  auto dZ = (zEnd - getZ());
160  auto phi0 = getPhi();
161  auto tanl0 = getTanl();
162  auto invtanl0 = 1.0 / tanl0;
163  auto invqpt0 = getInvQPt();
164  auto qpt0 = 1.0 / invqpt0;
165  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
166 
167  auto k = TMath::Abs(o2::constants::math::B2C * zField);
168  auto invk = 1.0 / k;
169  auto theta = -invqpt0 * dZ * k * invtanl0;
170  auto [sintheta, costheta] = o2::math_utils::sincosd(theta);
171  auto Hz = std::copysign(1, zField);
172  auto Y = sinphi0 * qpt0 * invk;
173  auto X = cosphi0 * qpt0 * invk;
174  auto YC = Y * costheta;
175  auto YS = Y * sintheta;
176  auto XC = X * costheta;
177  auto XS = X * sintheta;
178 
179  // Extrapolate track parameters to "zEnd"
180  mParameters(0) += Hz * (Y - YC) - XS;
181  mParameters(1) += Hz * (-X + XC) - YS;
182  mParameters(2) += Hz * theta;
183  mZ = zEnd;
184 }

187 void TrackParCovFwd::propagateToZhelix(double zEnd, double zField)
188 {
189  // Extrapolate track parameters and covariances matrix to "zEnd"
190  // using helix track model
191 
192  auto dZ = (zEnd - getZ());
193  auto phi0 = getPhi();
194  auto tanl0 = getTanl();
195  auto invtanl0 = 1.0 / tanl0;
196  auto invqpt0 = getInvQPt();
197  auto qpt0 = 1.0 / invqpt0;
198  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
199  auto k = TMath::Abs(o2::constants::math::B2C * zField);
200  auto invk = 1.0 / k;
201  auto theta = -invqpt0 * dZ * k * invtanl0;
202  auto [sintheta, costheta] = o2::math_utils::sincosd(theta);
203  auto Hz = std::copysign(1, zField);
204  auto L = qpt0 * qpt0 * invk;
205  auto N = dZ * invtanl0 * qpt0;
206  auto O = sintheta * cosphi0;
207  auto P = sinphi0 * costheta;
208  auto R = sinphi0 * sintheta;
209  auto S = cosphi0 * costheta;
210  auto Y = sinphi0 * qpt0 * invk;
211  auto X = cosphi0 * qpt0 * invk;
212  auto YC = Y * costheta;
213  auto YS = Y * sintheta;
214  auto XC = X * costheta;
215  auto XS = X * sintheta;
216  auto T = qpt0 * costheta;
217  auto U = qpt0 * sintheta;
218  auto V = qpt0;
219  auto n = dZ * invtanl0;
220  auto m = n * invtanl0;
221 
222  // Extrapolate track parameters to "zEnd"
223  mParameters(0) += Hz * (Y - YC) - XS;
224  mParameters(1) += Hz * (-X + XC) - YS;
225  mParameters(2) += Hz * theta;
226  mZ = zEnd;
227 
228  // Calculate Jacobian
229  SMatrix55 jacob = ROOT::Math::SMatrixIdentity();
230  jacob(0, 2) = Hz * X - Hz * XC + YS;
231  jacob(0, 3) = Hz * R * m - S * m;
232  jacob(0, 4) = -Hz * N * R + Hz * T * Y - Hz * V * Y + N * S + U * X;
233  jacob(1, 2) = Hz * Y - Hz * YC - XS;
234  jacob(1, 3) = -Hz * O * m - P * m;
235  jacob(1, 4) = Hz * N * O - Hz * T * X + Hz * V * X + N * P + U * Y;
236  jacob(2, 3) = -Hz * theta * invtanl0;
237  jacob(2, 4) = -Hz * k * n;
238 
239  // Extrapolate track parameter covariances to "zEnd"
240  setCovariances(ROOT::Math::Similarity(jacob, mCovariances));
241 }

244 bool TrackParCovFwd::update(const std::array<float, 2>& p, const std::array<float, 2>& cov)
245 {
248 
249  using SVector2 = ROOT::Math::SVector<double, 2>;
250  using SMatrix22 = ROOT::Math::SMatrix<double, 2>;
251  using SMatrix25 = ROOT::Math::SMatrix<double, 2, 5>;
252  using SMatrix52 = ROOT::Math::SMatrix<double, 5, 2>;
253  using SMatrix55Std = ROOT::Math::SMatrix<double, 5>;
254 
255  SMatrix55 I = ROOT::Math::SMatrixIdentity();
256  SMatrix25 H_k;
257  SMatrix22 V_k;
258  SVector2 m_k(p[0], p[1]), r_k_kminus1;
259  V_k(0, 0) = cov[0];
260  V_k(1, 1) = cov[1];
261  H_k(0, 0) = 1.0;
262  H_k(1, 1) = 1.0;
263 
264  // Covariance of residuals
265  SMatrix22 invResCov = (V_k + ROOT::Math::Similarity(H_k, mCovariances));
266  invResCov.Invert();
267 
268  // Kalman Gain Matrix
269  SMatrix52 K_k = mCovariances * ROOT::Math::Transpose(H_k) * invResCov;
270 
271  // Update Parameters
272  r_k_kminus1 = m_k - H_k * mParameters; // Residuals of prediction
273  mParameters = mParameters + K_k * r_k_kminus1;
274 
275  // Update covariances Matrix
276  SMatrix55Std updatedCov = (I - K_k * H_k) * mCovariances;
277  mCovariances(0, 0) = updatedCov(0, 0);
278  mCovariances(0, 1) = updatedCov(0, 1);
279  mCovariances(0, 2) = updatedCov(0, 2);
280  mCovariances(0, 3) = updatedCov(0, 3);
281  mCovariances(0, 4) = updatedCov(0, 4);
282  mCovariances(1, 1) = updatedCov(1, 1);
283  mCovariances(1, 2) = updatedCov(1, 2);
284  mCovariances(1, 3) = updatedCov(1, 3);
285  mCovariances(1, 4) = updatedCov(1, 4);
286  mCovariances(2, 2) = updatedCov(2, 2);
287  mCovariances(2, 3) = updatedCov(2, 3);
288  mCovariances(2, 4) = updatedCov(2, 4);
289  mCovariances(3, 3) = updatedCov(3, 3);
290  mCovariances(3, 4) = updatedCov(3, 4);
291  mCovariances(4, 4) = updatedCov(4, 4);
292 
293  auto addChi2Track = ROOT::Math::Similarity(r_k_kminus1, invResCov);
294  mTrackChi2 += addChi2Track;
295 
296  return true;
297 }

300 void TrackParCovFwd::addMCSEffect(double dZ, double x_over_X0)
301 {
308 
309  auto phi0 = getPhi();
310  auto tanl0 = getTanl();
311  auto invtanl0 = 1.0 / tanl0;
312  auto invqpt0 = getInvQPt();
313 
314  auto [sinphi0, cosphi0] = o2::math_utils::sincosd(phi0);
315 
316  auto csclambda = TMath::Abs(TMath::Sqrt(1 + tanl0 * tanl0) * invtanl0);
317  auto pathLengthOverX0 = x_over_X0 * csclambda; //
318 
319  // Angular dispersion square of the track (variance) in a plane perpendicular to the trajectory
320  auto sigmathetasq = 0.0136 * getInverseMomentum();
321  sigmathetasq *= sigmathetasq * pathLengthOverX0;
322 
323  // Get covariance matrix
324  SMatrix55 newParamCov(getCovariances());
325 
326  if (dZ > 0) {
327  auto A = tanl0 * tanl0 + 1;
328  auto B = dZ * cosphi0 * invtanl0;
329  auto C = dZ * sinphi0 * invtanl0;
330  auto D = A * B * invtanl0;
331  auto E = -A * C * invtanl0;
332  auto F = -C - D;
333  auto G = B + E;
334  auto H = -invqpt0 * tanl0;
335 
336  newParamCov(0, 0) += sigmathetasq * F * F;
337 
338  newParamCov(0, 1) += sigmathetasq * F * G;
339 
340  newParamCov(1, 1) += sigmathetasq * G * G;
341 
342  newParamCov(2, 0) += sigmathetasq * F;
343 
344  newParamCov(2, 1) += sigmathetasq * G;
345 
346  newParamCov(2, 2) += sigmathetasq;
347 
348  newParamCov(3, 0) += sigmathetasq * A * F;
349 
350  newParamCov(3, 1) += sigmathetasq * A * G;
351 
352  newParamCov(3, 2) += sigmathetasq * A;
353 
354  newParamCov(3, 3) += sigmathetasq * A * A;
355 
356  newParamCov(4, 0) += sigmathetasq * F * H;
357 
358  newParamCov(4, 1) += sigmathetasq * G * H;
359 
360  newParamCov(4, 2) += sigmathetasq * H;
361 
362  newParamCov(4, 3) += sigmathetasq * A * H;
363 
364  newParamCov(4, 4) += sigmathetasq * tanl0 * tanl0 * invqpt0 * invqpt0;
365  } else {
366 
367  auto A = tanl0 * tanl0 + 1;
368  auto H = -invqpt0 * tanl0;
369 
370  newParamCov(2, 2) += sigmathetasq;
371 
372  newParamCov(3, 2) += sigmathetasq * A;
373 
374  newParamCov(3, 3) += sigmathetasq * A * A;
375 
376  newParamCov(4, 2) += sigmathetasq * H;
377 
378  newParamCov(4, 3) += sigmathetasq * A * H;
379 
380  newParamCov(4, 4) += sigmathetasq * tanl0 * tanl0 * invqpt0 * invqpt0;
381  }
382 
383  // Set new covariances
384  setCovariances(newParamCov);
385 }
