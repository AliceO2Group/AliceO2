45 void* ShmManager::tryAttach(bool& success)
46 {
47  success = false;
48  if (auto id = getenv(SHMIDNAME)) {
49  mShmID = atoi(getenv(SHMIDNAME));
50  } else {
51  mShmID = -1;
52  return nullptr;
53  }
54  LOG(INFO) << "FOUND ID TO ATTACH " << mShmID;
55 
56  // if the segment was not created in the first place ...
57  if (mShmID == -1) {
58  return nullptr;
59  }
60 
61  void* addr_wanted;
62  if (auto addrstr = getenv(SHMADDRNAME)) {
63  addr_wanted = (void*)atoll(addrstr);
64  } else {
65  mShmID = -1;
66  return nullptr;
67  }
68  LOG(INFO) << "TRYING ADDRESS " << addr_wanted;
69  auto addr = shmat(mShmID, addr_wanted, 0);
70  if (addr != (void*)-1) {
71  // this means success
72  assert(addr == addr_wanted);
73  success = true;
74  mSegPtr = addr;
75  mSegInfoPtr = static_cast<o2::utils::ShmMetaInfo*>(mSegPtr);
76  return addr;
77  }
78 
79  // trying without constraints
80  addr = shmat(mShmID, nullptr, 0);
81  if (addr == (void*)(-1)) {
82  LOG(FATAL) << "SHOULD NOT HAPPEN";
83  }
84  // signal failure
85  auto info = static_cast<o2::utils::ShmMetaInfo*>(addr);
86  info->failures.fetch_add(1);
87  shmdt(addr);
88  mShmID = -1;
89  mSegInfoPtr = nullptr;
90  mSegPtr = nullptr;
91  // otherwise for now this means that we could not map the segment
92  // at the right place ...
93  return nullptr;
94 }

106 bool ShmManager::createGlobalSegment(int nsegments)
107 {
108  mIsMaster = true;
109  // first of all take a look if we really start from a clean state
110  if (mShmID != -1) {
111  LOG(WARN) << "A SEGMENT IS ALREADY INITIALIZED";
112  return false;
113  }
114  // make sure no one else has created the shm pool
115  // TODO: this can be relaxed / generalized
116  if (getenv(SHMIDNAME) || getenv(SHMADDRNAME)) {
117  LOG(WARN) << "A SEGMET IS ALREADY PRESENT ON THE SYSTEM";
118  return false;
119  }
120 
121  LOG(INFO) << "CREATING SIM SHARED MEM SEGMENT FOR " << nsegments << " WORKERS";
122 #ifdef USESHM
123  // LOG(INFO) << "SIZEOF ShmMetaInfo " << sizeof(ShmMetaInfo);
124  const auto totalsize = sizeof(ShmMetaInfo) + SHMPOOLSIZE * nsegments;
125  if ((mShmID = shmget(IPC_PRIVATE, totalsize, IPC_CREAT | 0666)) == -1) {
126  perror("shmget: shmget failed");
127  } else {
128  // We are attaching once to determine a common virtual address under which everyone else should attach.
129  // In this case (if it succeeds) we can also share objects with shm pointers
130  auto addr = shmat(mShmID, nullptr, 0);
131  mSegPtr = addr;
132  LOG(DEBUG) << "COMMON ADDRESS " << addr << " AS NUMBER " << (unsigned long long)addr;
133 
134  // initialize the meta information (counter)
135  o2::utils::ShmMetaInfo info;
136  std::memcpy(addr, &info, sizeof(info));
137  mSegInfoPtr = static_cast<o2::utils::ShmMetaInfo*>(mSegPtr);
138  mSegInfoPtr->allocedbytes = totalsize;
139 
140  // communicating information about this segment via
141  // an environment variable
142  // TODO: consider using named posix shared memory segments to avoid this
143  setenv(SHMIDNAME, std::to_string(mShmID).c_str(), 1);
144  setenv(SHMADDRNAME, std::to_string((unsigned long long)(addr)).c_str(), 1);
145  return true;
146  }
147  LOG(INFO) << "SHARED MEM INITIALIZED AT ID " << mShmID;
148  if (mShmID == -1) {
149  LOG(WARN) << "COULD NOT CREATE SHARED MEMORY ... FALLING BACK TO SIMPLE MODE";
150  setenv(SHMIDNAME, std::to_string(mShmID).c_str(), 1);
151  setenv(SHMADDRNAME, std::to_string(0).c_str(), 1);
152  }
153 #endif
154  return false;
155 }

158 {
159  LOG(INFO) << "OCCUPYING A SEGMENT IN A SHARED REGION";
160  // get information about the global shared mem in which to occupy a region

176 void ShmManager::occupySegment()
177 {
178  LOG(INFO) << "OCCUPYING A SEGMENT IN A SHARED REGION";
179  // get information about the global shared mem in which to occupy a region
180  if (!(getenv(SHMIDNAME) && getenv(SHMADDRNAME))) {
181  LOG(WARN) << "NO INFORMATION ABOUT SHARED SEGMENT FOUND";
182  return;
183  }
184 
185  if (mShmID != -1) {
186  LOG(WARN) << "REGION ALREADY OCCUPIED OR CREATED";
187  return;
188  }
189 
190  bool b;
191  auto addr = tryAttach(b);
192  if (b) {
193  // read meta information in the segment to determine the id of this segment
194  auto info = static_cast<o2::utils::ShmMetaInfo*>(addr);
195  const int segmentcounter = info->counter.fetch_add(1);
196  LOG(INFO) << "SEGMENTCOUNT " << segmentcounter;
197 
198  const auto offset_in_bytes = sizeof(ShmMetaInfo) + segmentcounter * SHMPOOLSIZE;
199  mBufferPtr = (void*)(((char*)addr) + offset_in_bytes);
200 
201  assert((unsigned long long)((char*)mBufferPtr - (char*)addr) + SHMPOOLSIZE <= info->allocedbytes);
202 
203  boostmanagedbuffer = new boost::interprocess::wmanaged_external_buffer(create_only, mBufferPtr, SHMPOOLSIZE);
204  boostallocator = new boost::interprocess::allocator<char, wmanaged_external_buffer::segment_manager>(
205  boostmanagedbuffer->get_segment_manager());
206 
207  LOG(INFO) << "SHARED MEM OCCUPIED AT ID " << mShmID << " AND SEGMENT COUNTER " << segmentcounter;
208  } else {
209  LOG(INFO) << "ATTACH NOT SUCCESSFUL";
210  }
211 }
