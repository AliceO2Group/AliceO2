43 void EntropyDecoderSpec::run(ProcessingContext& pc)
44 {
45  auto cput = mTimer.CpuTime();
46  mTimer.Start(false);
47 
48  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
49 
50  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(OutputRef{"ROframes"});
51  auto& compcl = pc.outputs().make<std::vector<o2::itsmft::CompClusterExt>>(OutputRef{"compClusters"});
52  auto& patterns = pc.outputs().make<std::vector<unsigned char>>(OutputRef{"patterns"});
53 
54  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
55  const auto ctfImage = o2::itsmft::CTF::getImage(buff.data());
56  mCTFCoder.decode(ctfImage, rofs, compcl, patterns);
57 
58  mTimer.Stop();
59  LOG(INFO) << "Decoded " << compcl.size() << " clusters in " << rofs.size() << " RO frames in " << mTimer.CpuTime() - cput << " s";
60 }

68 DataProcessorSpec getEntropyDecoderSpec(o2::header::DataOrigin orig)
69 {
70  std::vector<OutputSpec> outputs{
71  OutputSpec{{"compClusters"}, orig, "COMPCLUSTERS", 0, Lifetime::Timeframe},
72  OutputSpec{{"patterns"}, orig, "PATTERNS", 0, Lifetime::Timeframe},
73  OutputSpec{{"ROframes"}, orig, "CLUSTERSROF", 0, Lifetime::Timeframe}};
74 
75  std::string dictOptName = (orig == o2::header::gDataOriginITS) ? "its-ctf-dictionary" : "mft-ctf-dictionary";
76 
77  return DataProcessorSpec{
78  orig == o2::header::gDataOriginITS ? "its-entropy-decoder" : "mft-entropy-decoder",
79  Inputs{InputSpec{"ctf", orig, "CTFDATA", 0, Lifetime::Timeframe}},
80  outputs,
81  AlgorithmSpec{adaptFromTask<EntropyDecoderSpec>(orig)},
82  Options{{dictOptName, VariantType::String, "ctf_dictionary.root", {"File of CTF decoding dictionary"}}}};
83 }
