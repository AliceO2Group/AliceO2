79 bool RawFileReader::readHB(bool sendCompleteHBs)
80 {
82  if (mState != 0) {
83  return false;
84  }
85  bool isHBClosed = false;
86  while (!isHBClosed) {
87  // Read header
88  size_t headerIndex = mBytes.size();
89  read(sHeaderSize);
90 
91  // The check on the eof needs to be placed here and not at the beginning of the function.
92  // The reason is that the eof flag is set if we try to read after the eof
93  // But, since we know the size, we read up to the last character.
94  // So we turn on the eof flag only if we try to read past the last data.
95  // Of course, we resized the mBytes before trying to read.
96  // Since we read 0, we need to remove the last bytes
97  if (mFile.eof()) {
98  mBytes.resize(headerIndex);
99  if (mReadContinuous) {
100  mFile.clear();
101  mFile.seekg(0, std::ios::beg);
102  read(sHeaderSize);
103  } else {
104  mState = 1;
105  return false;
106  }
107  }
108  replaceRDH(headerIndex);
109  // We use the buffer only to correctly initialize the RDH
110  o2::framework::RawParser parser(mBytes.data(), mBytes.size());
111  auto lastIt = parser.begin();
112  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
113  lastIt = it;
114  }
115  auto const* rdhPtr = reinterpret_cast<const o2::header::RDHAny*>(lastIt.raw());
116  isHBClosed = o2::raw::RDHUtils::getStop(rdhPtr);
117  auto offsetNext = o2::raw::RDHUtils::getOffsetToNext(rdhPtr);
118  if (offsetNext > sHeaderSize) {
119  read(offsetNext - sHeaderSize);
120  }
121  if (!sendCompleteHBs) {
122  break;
123  }
124  }
125  return true;
126 }
