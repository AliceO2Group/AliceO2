63 void doTest(const std::map<o2::InteractionRecord, std::vector<o2::mid::ColumnData>>& inData, const std::vector<o2::mid::ROFRecord>& rofRecords, const std::vector<o2::mid::ColumnData>& data, const o2::mid::EventType inEventType = o2::mid::EventType::Standard)
64 {
65  BOOST_REQUIRE(rofRecords.size() == inData.size());
66  auto inItMap = inData.begin();
67  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
68  BOOST_TEST(static_cast<int>(rofIt->eventType) == static_cast<int>(inEventType));
69  BOOST_TEST(rofIt->interactionRecord == inItMap->first);
70  BOOST_TEST(rofIt->nEntries == inItMap->second.size());
71  auto sortedIn = sortData(inItMap->second, 0, inItMap->second.size());
72  auto sortedOut = sortData(data, rofIt->firstEntry, rofIt->firstEntry + rofIt->nEntries);
73  BOOST_REQUIRE(sortedOut.size() == sortedIn.size());
74  for (size_t icol = 0; icol < sortedOut.size(); ++icol) {
75  BOOST_TEST(sortedOut[icol].deId == sortedIn[icol].deId);
76  BOOST_TEST(sortedOut[icol].columnId == sortedIn[icol].columnId);
77  BOOST_TEST(sortedOut[icol].getNonBendPattern() == sortedIn[icol].getNonBendPattern());
78  for (int iline = 0; iline < 4; ++iline) {
79  BOOST_TEST(sortedOut[icol].getBendPattern(iline) == sortedIn[icol].getBendPattern(iline));
80  }
81  }
82  ++inItMap;
83  }
84 }

87 {
88  auto severity = fair::Logger::GetConsoleSeverity();

90  std::string tmpFilename0 = "tmp_mid_raw";
91  std::string tmpFilename = tmpFilename0 + ".raw";
92  o2::mid::Encoder encoder;
93  encoder.init(tmpFilename0.c_str());
94  std::string tmpConfigFilename = "tmp_MIDConfig.cfg";
95  encoder.getWriter().writeConfFile("MID", "RAWDATA", tmpConfigFilename.c_str(), false);

101  o2::raw::RawFileReader rawReader(tmpConfigFilename.c_str());
102  rawReader.init();
103  std::vector<char> buffer;
104  for (size_t itf = 0; itf < rawReader.getNTimeFrames(); ++itf) {
105  rawReader.setNextTFToRead(itf);
106  for (size_t ilink = 0; ilink < rawReader.getNLinks(); ++ilink) {
107  auto& link = rawReader.getLink(ilink);
108  auto tfsz = link.getNextTFSize();
109  if (!tfsz) {
110  continue;
111  }
112  std::vector<char> linkBuffer(tfsz);
113  link.readNextTF(linkBuffer.data());
114  buffer.insert(buffer.end(), linkBuffer.begin(), linkBuffer.end());
115  }
116  }

118 
119  std::remove(tmpFilename.c_str());
120  std::remove(tmpConfigFilename.c_str());
121 
122  o2::mid::Decoder decoder;
123  gsl::span<const uint8_t> data(reinterpret_cast<uint8_t*>(buffer.data()), buffer.size());
124  decoder.process(data);
125 
126  o2::mid::DecodedDataAggregator aggregator;
127  aggregator.process(decoder.getData(), decoder.getROFRecords());
128 
129  return std::make_tuple(aggregator.getData(), aggregator.getROFRecords());
130 }

175  o2::mid::LocalBoardRO loc;
176  // Crate 5 link 0
177  loc.statusWord = o2::mid::raw::sSTARTBIT | o2::mid::raw::sCARDTYPE;
178  loc.triggerWord = 0;
179  loc.boardId = 2;
180  loc.firedChambers = 0x1;
181  loc.patternsBP[0] = 0xF;
182  loc.patternsNBP[0] = 0xF;
183  inData[bc].emplace_back(loc);
184 
185  bc = 200;
186  loc.patternsBP.fill(0);
187  loc.patternsNBP.fill(0);
188  loc.boardId = 5;
189  loc.firedChambers = 0x4;
190  loc.patternsBP[2] = 0xF0;
191  loc.patternsNBP[2] = 0x5;
192  inData[bc].emplace_back(loc);
193 
194  uint8_t crateId = 5;
195  uint8_t linkInCrate = 0;
196  uint16_t feeId = o2::mid::crateparams::makeROId(crateId, linkInCrate);
197  o2::mid::GBTUserLogicEncoder encoder;
198  encoder.setFeeId(feeId);
199  for (auto& item : inData) {
200  encoder.process(item.second, o2::InteractionRecord(item.first, 0));
201  }
202  std::vector<char> buf;
203  encoder.flush(buf, o2::InteractionRecord());
204  o2::header::RAWDataHeader rdh;
205  auto memSize = buf.size() + 64;
206  rdh.word1 |= (memSize | (memSize << 16));
207  // Sets the feeId
208  rdh.word0 |= ((5 * 2) << 16);
209  auto decoder = o2::mid::createGBTDecoder(feeId);
210  std::vector<o2::mid::LocalBoardRO> data;
211  std::vector<o2::mid::ROFRecord> rofs;
212  std::vector<uint8_t> convertedBuffer(buf.size());
213  memcpy(convertedBuffer.data(), buf.data(), buf.size());
214  decoder->process(convertedBuffer, rdh, data, rofs);
215  BOOST_REQUIRE(rofs.size() == inData.size());
216  auto inItMap = inData.begin();

218  BOOST_TEST(rofIt->interactionRecord.bc == inItMap->first);
219  BOOST_TEST(rofIt->nEntries == inItMap->second.size());
220  auto outLoc = data.begin() + rofIt->firstEntry;
221  for (auto inLoc = inItMap->second.begin(); inLoc != inItMap->second.end(); ++inLoc) {
222  BOOST_TEST(inLoc->statusWord == outLoc->statusWord);
223  BOOST_TEST(inLoc->triggerWord == outLoc->triggerWord);
224  BOOST_TEST(o2::mid::crateparams::makeUniqueLocID(crateId, inLoc->boardId) == outLoc->boardId);
225  BOOST_TEST(inLoc->firedChambers == outLoc->firedChambers);
226  for (int ich = 0; ich < 4; ++ich) {
227  BOOST_TEST(inLoc->patternsBP[ich] == outLoc->patternsBP[ich]);
228  BOOST_TEST(inLoc->patternsNBP[ich] == outLoc->patternsNBP[ich]);
229  }
230  ++outLoc;
231  }
232  ++inItMap;
233  }
234 }
