21 void Reconstructor::process(const o2::fdd::Digit& digitBC, gsl::span<const o2::fdd::ChannelData> digitCh, std::vector<o2::fdd::RecPoint>& recPoints) const
22 {
23  //Compute charge weighted average time
24  Double_t timeFDA = 0, timeFDC = 0;
25  Double_t weightFDA = 0.0, weightFDC = 0.0;
26 
27  for (const auto& channel : digitCh) {
28  Float_t adc = channel.mChargeADC;
29  Float_t time = channel.mTime;
30  //LOG(INFO) <<adc <<" "<<time;
31  if (time == o2::InteractionRecord::DummyTime) {
32  continue;
33  }
34  Float_t timeErr = 1;
35  if (adc > 1) {
36  timeErr = 1 / adc;
37  }
38  if (channel.mPMNumber < 8) {
39  timeFDC += time / (timeErr * timeErr);
40  weightFDC += 1. / (timeErr * timeErr);
41  } else {
42  timeFDA += time / (timeErr * timeErr);
43  weightFDA += 1. / (timeErr * timeErr);
44  }
45  }
46  timeFDA = (weightFDA > 1) ? timeFDA / weightFDA : o2::InteractionRecord::DummyTime;
47  timeFDC = (weightFDC > 1) ? timeFDC / weightFDC : o2::InteractionRecord::DummyTime;
48 
49  recPoints.emplace_back(timeFDA, timeFDC, digitBC.getIntRecord());
50 }
