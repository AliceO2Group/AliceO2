177  InputSpec spec2{"tracks", "TPC", "TRACKS"};
178 
179  std::vector<InputRoute> inputs = {

181  InputRoute{spec2, 1, "Fake2", 0}};
182 
183  std::vector<ForwardRoute> forwards;

185 
186  auto policy = CompletionPolicyHelpers::consumeWhenAny();
187  DataRelayer relayer(policy, inputs, metrics, index);
188  relayer.setPipelineLength(4);
189 
190  // Let's create a dummy O2 Message with two headers in the stack:
191  // - DataHeader matching the one provided in the input
192  DataHeader dh1;
193  dh1.dataDescription = "CLUSTERS";
194  dh1.dataOrigin = "TPC";
195  dh1.subSpecification = 0;
196 
197  DataHeader dh2;
198  dh2.dataDescription = "TRACKS";
199  dh2.dataOrigin = "TPC";
200  dh2.subSpecification = 0;
201 
202  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
203  size_t timeslice = 0;
204 
205  for (auto _ : state) {
206  // FIXME: Understand why pausing the timer makes it slower..
207  //state.PauseTiming();
208 
209  DataProcessingHeader dph1{timeslice, 1};
210  Stack stack1{dh1, dph1};
211 
212  FairMQMessagePtr header1 = transport->CreateMessage(stack1.size());
213  FairMQMessagePtr payload1 = transport->CreateMessage(1000);
214 
215  memcpy(header1->GetData(), stack1.data(), stack1.size());
216 
217  DataProcessingHeader dph2{timeslice, 1};
218  Stack stack2{dh2, dph2};
219 
220  FairMQMessagePtr header2 = transport->CreateMessage(stack2.size());
221  FairMQMessagePtr payload2 = transport->CreateMessage(1000);
222 
223  memcpy(header2->GetData(), stack2.data(), stack2.size());
224  //state.ResumeTiming();
225 
226  relayer.relay(std::move(header1), std::move(payload1));
227  std::vector<RecordAction> ready;
228  relayer.getReadyToProcess(ready);
229  assert(ready.size() == 1);
230  assert(ready[0].op == CompletionPolicy::CompletionOp::Consume);
