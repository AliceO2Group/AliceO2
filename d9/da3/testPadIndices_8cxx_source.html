32 int CheckPadIndexIsAsExpected(std::string filepath)
33 {
34  InputWrapper data(filepath.c_str());
35 
36  rapidjson::Value& padIndices = data.document()["channels"];
37 
38  int nbad{0};
39  int ntested{0};
40 
41  for (auto& tp : padIndices.GetArray()) {
42  ntested++;
43  int detElemId = tp["de"].GetInt();
44  bool isBendingPlane = (tp["bending"].GetString() == std::string("true"));
45  int refPadIndex = tp["padindex"].GetInt();
46  const auto& seg = o2::mch::mapping::segmentation(detElemId);
47  int bpad;
48  int nbpad;
49  seg.findPadPairByPosition(tp["x"].GetDouble(), tp["y"].GetDouble(), bpad, nbpad);
50  int padIndex = (isBendingPlane ? bpad : nbpad);
51  if (padIndex != refPadIndex) {
52  nbad++;
53  std::cout << fmt::format(">>> {} : Expected index = {} but got {}\n", filepath.c_str(), refPadIndex, padIndex);
54  std::cout << "expected=" << seg.padAsString(refPadIndex) << "\n";
55  std::cout << "got =" << seg.padAsString(padIndex) << "\n";
56  std::cout << "b =" << seg.padAsString(bpad) << "\n";
57  std::cout << "nb =" << seg.padAsString(nbpad) << "\n";
58  }
59  }
60 
61  if (nbad) {
62  std::cout << fmt::format("\n{} : {} pad indices error(s) over {} tested pads\n\n", filepath.c_str(), nbad, ntested);
63  } else {
64  std::cout << fmt::format("\n{} : the indices of all tested {} pads are OK\n\n", filepath.c_str(), ntested);
65  }
66  return nbad;
67 }
