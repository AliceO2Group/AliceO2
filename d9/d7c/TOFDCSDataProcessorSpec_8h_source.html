52 class TOFDCSDataProcessor : public o2::framework::Task
53 {
54  public:
55  void init(o2::framework::InitContext& ic) final
56  {
57 
58  std::vector<DPID> vect;
59  mDPsUpdateInterval = ic.options().get<int64_t>("DPs-update-interval");
60  if (mDPsUpdateInterval == 0) {
61  LOG(ERROR) << "TOF DPs update interval set to zero seconds --> changed to 60";
62  mDPsUpdateInterval = 60;
63  }
64  bool useCCDBtoConfigure = ic.options().get<bool>("use-ccdb-to-configure");
65  if (useCCDBtoConfigure) {
66  LOG(INFO) << "Configuring via CCDB";
67  std::string ccdbpath = ic.options().get<std::string>("ccdb-path");
68  auto& mgr = CcdbManager::instance();
69  mgr.setURL(ccdbpath);
70  CcdbApi api;
71  api.init(mgr.getURL());
72  long ts = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
73  std::unordered_map<DPID, std::string>* dpid2DataDesc = mgr.getForTimeStamp<std::unordered_map<DPID, std::string>>("TOF/DCSconfig", ts);
74  for (auto& i : *dpid2DataDesc) {
75  vect.push_back(i.first);
76  }
77  } else {
78  LOG(INFO) << "Configuring via hardcoded strings";
79  std::vector<std::string> aliases = {"tof_hv_vp_[00..89]", "tof_hv_vn_[00..89]", "tof_hv_ip_[00..89]", "tof_hv_in_[00..89]"};
80  std::vector<std::string> aliasesInt = {"TOF_FEACSTATUS_[00..71]"};
81  std::vector<std::string> expaliases = o2::dcs::expandAliases(aliases);
82  std::vector<std::string> expaliasesInt = o2::dcs::expandAliases(aliasesInt);
83  for (const auto& i : expaliases) {
84  vect.emplace_back(i, o2::dcs::RAW_DOUBLE);
85  }
86  for (const auto& i : expaliasesInt) {
87  vect.emplace_back(i, o2::dcs::RAW_INT);
88  }
89  }
90 
91  LOG(INFO) << "Listing Data Points for TOF:";
92  for (auto& i : vect) {
93  LOG(INFO) << i;
94  }
95 
96  mProcessor = std::make_unique<o2::tof::TOFDCSProcessor>();
97  bool useVerboseMode = ic.options().get<bool>("use-verbose-mode");
98  LOG(INFO) << " ************************* Verbose?" << useVerboseMode;
99  if (useVerboseMode) {
100  mProcessor->useVerboseMode();
101  }
102  mProcessor->init(vect);
103  mTimer = HighResClock::now();
104  }
105 
106  void run(o2::framework::ProcessingContext& pc) final
107  {
108  auto tfid = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
109  auto dps = pc.inputs().get<gsl::span<DPCOM>>("input");
110  mProcessor->setTF(tfid);
111  mProcessor->process(dps);
112  auto timeNow = HighResClock::now();
113  Duration elapsedTime = timeNow - mTimer; // in seconds
114  if (elapsedTime.count() >= mDPsUpdateInterval) {
115  sendDPsoutput(pc.outputs());
116  mTimer = timeNow;
117  }
118  sendLVandHVoutput(pc.outputs());
119  }
120 
121  void endOfStream(o2::framework::EndOfStreamContext& ec) final
122  {
123  sendDPsoutput(ec.outputs());
124  sendLVandHVoutput(ec.outputs());
125  }
126 
127  private:
128  std::unique_ptr<TOFDCSProcessor> mProcessor;
129  HighResClock::time_point mTimer;
130  int64_t mDPsUpdateInterval;
131 
132  //________________________________________________________________
133  void sendDPsoutput(DataAllocator& output)
134  {
135  // extract CCDB infos and calibration object for DPs
136  mProcessor->updateDPsCCDB();
137  const auto& payload = mProcessor->getTOFDPsInfo();
138  auto& info = mProcessor->getccdbDPsInfo();
139  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
140  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
141  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
142  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
143  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info);
144  mProcessor->clearDPsinfo();
145  }
146 
147  //________________________________________________________________
148  void sendLVandHVoutput(DataAllocator& output)
149  {
150  // extract CCDB infos and calibration objects for LV and HV, convert it to TMemFile and send them to the output
151 
152  if (mProcessor->isLVUpdated()) {
153  const auto& payload = mProcessor->getLVStatus();
154  auto& info = mProcessor->getccdbLVInfo();
155  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
156  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
157  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
158  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
159  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info);
160  }
161  if (mProcessor->isHVUpdated()) {
162  const auto& payload = mProcessor->getHVStatus();
163  auto& info = mProcessor->getccdbHVInfo();
164  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
165  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
166  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
167  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
168  output.snapshot(Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info);
169  }
170  }

172 }; // end class

194  {"use-verbose-mode", VariantType::Bool, false, {"Use verbose mode"}},
195  {"DPs-update-interval", VariantType::Int64, 600ll, {"Interval (in s) after which to update the DPs CCDB entry"}}}};
196 }
