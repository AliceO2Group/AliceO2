40 void NoiseCalibratorSpec::run(ProcessingContext& pc)
41 {
42  const auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
43  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
44  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
45 
46  if (mCalibrator->processTimeFrame(compClusters, patterns, rofs)) {
47  LOG(INFO) << "Minimum number of noise counts has been reached !";
48  sendOutput(pc.outputs());
49  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
50  }
51 }

53 void NoiseCalibratorSpec::sendOutput(DataAllocator& output)
54 {
55  mCalibrator->finalize();
56 
57  long tstart = 0, tend = 9999999;
58 #ifdef TIME_SLOT_CALIBRATION
59  const auto& payload = mCalibrator->getNoiseMap(tstart, tend);
60 #else
61  const auto& payload = mCalibrator->getNoiseMap();
62 #endif
63  std::map<std::string, std::string> md;
64  o2::ccdb::CcdbObjectInfo info("ITS/Noise", "NoiseMap", "noise.root", md, tstart, tend);
65 
66  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
67  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName()
68  << " of size " << image->size()
69  << " bytes, valid for " << info.getStartValidityTimestamp()
70  << " : " << info.getEndValidityTimestamp();
71 
72  using clbUtils = o2::calibration::Utils;
73  output.snapshot(
74  Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload, 0}, *image.get());
75  output.snapshot(
76  Output{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo, 0}, info);
77 }

84 DataProcessorSpec getNoiseCalibratorSpec()
85 {
86  std::vector<InputSpec> inputs;
87  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
88  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
89  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
90 
91  using clbUtils = o2::calibration::Utils;
92  std::vector<OutputSpec> outputs;
93  outputs.emplace_back(
94  ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBPayload});
95  outputs.emplace_back(
96  ConcreteDataTypeMatcher{clbUtils::gDataOriginCLB, clbUtils::gDataDescriptionCLBInfo});
97 
98  return DataProcessorSpec{
99  "its-noise-calibrator",
100  inputs,
101  outputs,
102  AlgorithmSpec{adaptFromTask<NoiseCalibratorSpec>()},
103  Options{
104  {"1pix-only", VariantType::Bool, false, {"Fast 1-pixel calibration only"}},
105  {"prob-threshold", VariantType::Float, 3.e-6f, {"Probability threshold for noisy pixels"}}}};
106 }
