40 DataProcessorSpec getDigitsReaderSpec(bool propagateMC)
41 {
42 
43  auto initFunction = [propagateMC](InitContext& ic) {
44  // get the option from the init context
45  auto filename = ic.options().get<std::string>("infile");
46  auto treename = ic.options().get<std::string>("treename");
47  auto nofEvents = ic.options().get<int>("nevents");
48  auto publishingMode = nofEvents == -1 ? RootTreeReader::PublishingMode::Single : RootTreeReader::PublishingMode::Loop;
49 
50  auto processAttributes = std::make_shared<ProcessAttributes>();
51  {
52  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
53  processAttributes->finished = false;
54  processAttributes->datatype = "CPVDigit";
55  constexpr auto persistency = Lifetime::Timeframe;
56  o2::header::DataHeader::SubSpecificationType subSpec = 0;
57  if (propagateMC) {
58  processAttributes->reader = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
59  filename.c_str(), // input file name
60  nofEvents, // number of entries to publish
61  publishingMode,
62  Output{"CPV", "DIGITS", subSpec, persistency},
63  "CPVDigit", // name of data branch
64  Output{"CPV", "DIGITTRIGREC", subSpec, persistency},
65  "CPVDigitTrigRecords", // name of data triggerrecords branch
66  Output{"CPV", "DIGITSMCTR", subSpec, persistency},
67  "CPVDigitMCTruth"); // name of mc label branch
68  } else {
69  processAttributes->reader = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
70  filename.c_str(), // input file name
71  nofEvents, // number of entries to publish
72  publishingMode,
73  Output{"CPV", "DIGITS", subSpec, persistency},
74  "CPVDigit", // name of data branch
75  Output{"CPV", "DIGITTRIGREC", subSpec, persistency},
76  "CPVDigitTrigRecords"); // name of data triggerrecords branch
77  }
78  }
79 
80  auto processFunction = [processAttributes, propagateMC](ProcessingContext& pc) {
81  if (processAttributes->finished) {
82  return;
83  }
84 
85  auto publish = [&processAttributes, &pc, propagateMC]() {
86  o2::cpv::CPVBlockHeader cpvheader(true);
87  if (processAttributes->reader->next()) {
88  (*processAttributes->reader)(pc, cpvheader);
89  } else {
90  processAttributes->reader.reset();
91  return false;
92  }
93  return true;
94  };
95 
96  bool active(true);
97  if (!publish()) {
98  active = false;
99  // Send dummy header with no payload option
100  o2::cpv::CPVBlockHeader dummyheader(false);
101  pc.outputs().snapshot(OutputRef{"output", 0, {dummyheader}}, 0);
102  pc.outputs().snapshot(OutputRef{"outputTR", 0, {dummyheader}}, 0);
103  if (propagateMC) {
104  pc.outputs().snapshot(OutputRef{"outputMC", 0, {dummyheader}}, 0);
105  }
106  }
107  if ((processAttributes->finished = (active == false)) && processAttributes->terminateOnEod) {
108  pc.services().get<ControlService>().endOfStream();
109  pc.services().get<ControlService>().readyToQuit(framework::QuitRequest::Me);
110  }
111  };
112  return processFunction;
113  };
114 
115  std::vector<OutputSpec> outputSpecs;
116  outputSpecs.emplace_back(OutputSpec{{"output"}, "CPV", "DIGITS", 0, Lifetime::Timeframe});
117  outputSpecs.emplace_back(OutputSpec{{"outputTR"}, "CPV", "DIGITTRIGREC", 0, Lifetime::Timeframe});
118  if (propagateMC) {
119  outputSpecs.emplace_back(OutputSpec{{"outputMC"}, "CPV", "DIGITSMCTR", 0, Lifetime::Timeframe});
120  }
121 
122  return DataProcessorSpec{
123  "cpv-digit-reader",
124  Inputs{}, // no inputs
125  outputSpecs,
126  AlgorithmSpec(initFunction),
127  Options{
128  {"infile", VariantType::String, "cpvdigits.root", {"Name of the input file"}},
129  {"treename", VariantType::String, "o2sim", {"Name of input tree"}},
130  {"nevents", VariantType::Int, -1, {"number of events to run, -1: inf loop"}},
131  {"terminate-on-eod", VariantType::Bool, true, {"terminate on end-of-data"}},
132  }};
133 }
