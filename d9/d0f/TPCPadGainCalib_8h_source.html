99  GPUd() static T pack(float f)
100  {
101  f = CAMath::Clamp(f, MinCorrectionFactor, MaxCorrectionFactor);
102  f -= MinCorrectionFactor;
103  f *= float(NumOfSteps);
104  f /= (MaxCorrectionFactor - MinCorrectionFactor);
105  return CAMath::Nint(f);
106  }
107 
108  GPUd() static float unpack(T c)
109  {
110  return MinCorrectionFactor + (MaxCorrectionFactor - MinCorrectionFactor) * float(c) / float(NumOfSteps);
111  }
