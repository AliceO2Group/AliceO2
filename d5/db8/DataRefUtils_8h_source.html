32  template <typename T>
33  static auto as(DataRef const& ref)
34  {
35  // SFINAE makes this available only for the case we are using
36  // trivially copyable type, this is to distinguish it from the
37  // alternative below, which works for TObject (which are serialised).
38  if constexpr (is_messageable<T>::value == true) {
39  using DataHeader = o2::header::DataHeader;
40  auto header = o2::header::get<const DataHeader*>(ref.header);
41  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
42  throw runtime_error("Attempt to extract a POD from a wrong message kind");
43  }
44  if ((header->payloadSize % sizeof(T)) != 0) {
45  throw runtime_error("Cannot extract POD from message as size do not match");
46  }
47  //FIXME: provide a const collection
48  return gsl::span<T>(reinterpret_cast<T*>(const_cast<char*>(ref.payload)), header->payloadSize / sizeof(T));
49  } else if constexpr (has_root_dictionary<T>::value == true &&
50  is_messageable<T>::value == false) {
51  std::unique_ptr<T> result;
52  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
53  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
54  // Classes using the ROOT ClassDef/ClassImp macros can be detected automatically
55  // and the serialization method can be deduced. If the class is also
56  // messageable, gSerializationMethodNone is used by default. This substitution
57  // does not apply for such types, the serialization method needs to be specified
58  // explicitely using type wrapper @a ROOTSerialized.
59  using RSS = std::decay_t<decltype(*p)>;
60  using DataHeader = o2::header::DataHeader;
61  auto header = o2::header::get<const DataHeader*>(ref.header);
62  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
63  throw runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
64  }
65 
66  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
67  auto* storedClass = ftm.GetClass();
68  auto* requestedClass = RSS::TClass::GetClass(typeid(T));
69  // should always have the class description if has_root_dictionary is true
70  assert(requestedClass != nullptr);
71 
72  auto* object = ftm.ReadObjectAny(storedClass);
73  if (object == nullptr) {
74  throw runtime_error_f("Failed to read object with name %s from message using ROOT serialization.",
75  (storedClass != nullptr ? storedClass->GetName() : "<unknown>"));
76  }
77 
78  if constexpr (std::is_base_of<typename RSS::TObject, T>::value) { // compile time switch
79  // if the type to be extracted inherits from TObject, the class descriptions
80  // do not need to match exactly, only the dynamic_cast has to work
81  // FIXME: could probably try to extend this to arbitrary types T being a base
82  // to the stored type, but this would require to cast the extracted object to
83  // the actual type. This requires this information to be available as a type
84  // in the ROOT dictionary, check if this is the case or if TClass::DynamicCast
85  // can be used for the test. Right now, this case was and is not covered and
86  // not yet checked in the unit test either.
87  auto* r = dynamic_cast<T*>(static_cast<typename RSS::TObject*>(object));
88  if (r) {
89  result.reset(r);
90  }
91  // This check includes the case: storedClass == requestedClass
92  } else if (storedClass->InheritsFrom(requestedClass)) {
93  result.reset(static_cast<T*>(object));
94  }
95  if (result == nullptr) {
96  // did not manage to cast the pointer to result
97  // delete object via the class info if available, not the case for all types,
98  // e.g. for standard containers of ROOT objects apparently this is not always
99  // the case
100  auto* delfunc = storedClass->GetDelete();
101  if (delfunc) {
102  (*delfunc)(object);
103  }
104 
105  throw runtime_error_f("Attempting to extract a %s but a %s is actually stored which cannot be casted to the requested one.",
106  (requestedClass != nullptr ? requestedClass->GetName() : "<unknown>"),
107  (storedClass != nullptr ? storedClass->GetName() : "<unknown>"));
108  }
109  // collections in ROOT can be non-owning or owning and the proper cleanup depends on
110  // this flag. Be it a bug or a feature in ROOT, but the owning flag of the extracted
111  // object only depends on the state at serialization of the original object. However,
112  // all objects created during deserialization are new and must be owned by the collection
113  // to avoid memory leak. So we call SetOwner if it is available for the type.
114  if constexpr (has_root_setowner<T>::value) {
115  result->SetOwner(true);
116  }
117  });
118 
119  return std::move(result);
120  } else if constexpr (is_specialization<T, ROOTSerialized>::value == true) {
121  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
122  // with a somewhat uniform API. ROOT serialization method is enforced by using
123  // type wrapper @a ROOTSerialized
124  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
125  using wrapped = typename T::wrapped_type;
126  using DataHeader = o2::header::DataHeader;
127  std::unique_ptr<wrapped> result;
128 
129  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
130  using RSS = std::decay_t<decltype(*p)>;
131 
132  auto header = o2::header::get<const DataHeader*>(ref.header);
133  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
134  throw runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
135  }
136  auto* cl = RSS::TClass::GetClass(typeid(wrapped));
137  if (has_root_dictionary<wrapped>::value == false && cl == nullptr) {
138  throw runtime_error("ROOT serialization not supported, dictionary not found for data type");
139  }
140 
141  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
142  result.reset(static_cast<wrapped*>(ftm.ReadObjectAny(cl)));
143  if (result.get() == nullptr) {
144  throw runtime_error_f("Unable to extract class %s", cl == nullptr ? "<name not available>" : cl->GetName());
145  }
146  // workaround for ROOT feature, see above
147  if constexpr (has_root_setowner<T>::value) {
148  result->SetOwner(true);
149  }
150  });
151  return std::move(result);
152  }
153  }

155  static o2::header::DataHeader::PayloadSizeType getPayloadSize(const DataRef& ref)
156  {
157  using DataHeader = o2::header::DataHeader;
158  auto* header = o2::header::get<const DataHeader*>(ref.header);

165  template <typename T>
166  static auto getHeader(const DataRef& ref)
167  {
168  using HeaderT = typename std::remove_pointer<T>::type;
169  static_assert(std::is_pointer<T>::value && std::is_base_of<o2::header::BaseHeader, HeaderT>::value,
170  "pointer to BaseHeader-derived type required");
171  return o2::header::get<T>(ref.header);
172  }
