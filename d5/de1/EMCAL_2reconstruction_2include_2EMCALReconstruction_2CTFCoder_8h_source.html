57 template <typename VEC>
58 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Cell>& cellData)
59 {
60  using MD = o2::ctf::Metadata::OptStore;
61  // what to do which each field: see o2::ctd::Metadata explanation
62  constexpr MD optField[CTF::getNBlocks()] = {
63  MD::EENCODE, // BLC_bcIncTrig
64  MD::EENCODE, // BLC_orbitIncTrig
65  MD::EENCODE, // BLC_entriesTrig
66  MD::EENCODE, // BLC_towerID
67  MD::EENCODE, // BLC_time
68  MD::EENCODE, // BLC_energy
69  MD::EENCODE // BLC_status
70  };
71 
72  CTFHelper helper(trigData, cellData);
73 
74  // book output size with some margin
75  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
76  buff.resize(szIni);
77 
78  auto ec = CTF::create(buff);
79  using ECB = CTF::base;
80 
81  ec->setHeader(helper.createHeader());
82  ec->getANSHeader().majorVersion = 0;
83  ec->getANSHeader().minorVersion = 1;
84  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
85 #define ENCODEEMC(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
86  // clang-format off
87  ENCODEEMC(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
88  ENCODEEMC(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
89  ENCODEEMC(helper.begin_entriesTrig(), helper.end_entriesTrig(), CTF::BLC_entriesTrig, 0);
90 
91  ENCODEEMC(helper.begin_towerID(), helper.end_towerID(), CTF::BLC_towerID, 0);
92  ENCODEEMC(helper.begin_time(), helper.end_time(), CTF::BLC_time, 0);
93  ENCODEEMC(helper.begin_energy(), helper.end_energy(), CTF::BLC_energy, 0);
94  ENCODEEMC(helper.begin_status(), helper.end_status(), CTF::BLC_status, 0);
95  // clang-format on
96  CTF::get(buff.data())->print(getPrefix());
97 }

100 template <typename VTRG, typename VCELL>
101 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCELL& cellVec)
102 {
103  auto header = ec.getHeader();
104  ec.print(getPrefix());
105  std::vector<uint16_t> bcInc, entries, energy, cellTime, tower;
106  std::vector<uint32_t> orbitInc;
107  std::vector<uint8_t> status;
108 
109 #define DECODEEMCAL(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
110  // clang-format off
111  DECODEEMCAL(bcInc, CTF::BLC_bcIncTrig);
112  DECODEEMCAL(orbitInc, CTF::BLC_orbitIncTrig);
113  DECODEEMCAL(entries, CTF::BLC_entriesTrig);
114  DECODEEMCAL(tower, CTF::BLC_towerID);
115 
116  DECODEEMCAL(cellTime, CTF::BLC_time);
117  DECODEEMCAL(energy, CTF::BLC_energy);
118  DECODEEMCAL(status, CTF::BLC_status);
119  // clang-format on
120  //
121  trigVec.clear();
122  cellVec.clear();
123  trigVec.reserve(header.nTriggers);
124  status.reserve(header.nCells);
125 
126  uint32_t firstEntry = 0, cellCount = 0;
127  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
128 
129  Cell cell;
130  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
131  // restore TrigRecord
132  if (orbitInc[itrig]) { // non-0 increment => new orbit
133  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
134  ir.orbit += orbitInc[itrig];
135  } else {
136  ir.bc += bcInc[itrig];
137  }
138 
139  firstEntry = cellVec.size();
140  for (uint16_t ic = 0; ic < entries[itrig]; ic++) {
141  cell.setPacked(tower[cellCount], cellTime[cellCount], energy[cellCount], status[cellCount]);
142  cellVec.emplace_back(cell);
143  cellCount++;
144  }
145  trigVec.emplace_back(ir, firstEntry, entries[itrig]);
146  }
147  assert(cellCount == header.nCells);
148 }
