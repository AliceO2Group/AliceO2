92  std::unique_ptr<TFile> digitfile(TFile::Open(digitfilename.data(), "READ"));
93  auto treereader = std::make_unique<TTreeReader>(static_cast<TTree*>(digitfile->Get("o2sim")));
94  TTreeReaderValue<std::vector<o2::emcal::Digit>> digitbranch(*treereader, "EMCALDigit");
95  TTreeReaderValue<std::vector<o2::emcal::TriggerRecord>> triggerbranch(*treereader, "EMCALDigitTRGR");
96 
97  o2::emcal::RawWriter::FileFor_t granularity = o2::emcal::RawWriter::FileFor_t::kFullDet;
98  if (filefor == "all") {
99  granularity = o2::emcal::RawWriter::FileFor_t::kFullDet;
100  } else if (filefor == "subdet") {
101  granularity = o2::emcal::RawWriter::FileFor_t::kSubDet;
102  } else if (filefor == "link") {
103  granularity = o2::emcal::RawWriter::FileFor_t::kLink;
104  }
105 
106  o2::emcal::RawWriter rawwriter;
107  rawwriter.setOutputLocation(outputdir.data());
108  rawwriter.setFileFor(granularity);
109  rawwriter.setGeometry(o2::emcal::Geometry::GetInstanceFromRunNumber(300000));
110  rawwriter.setNumberOfADCSamples(15); // @TODO Needs to come from CCDB
111  rawwriter.setPedestal(3); // @TODO Needs to come from CCDB
112  rawwriter.init();
113 
114  // Loop over all entries in the tree, where each tree entry corresponds to a time frame
115  for (auto en : *treereader) {
116  rawwriter.digitsToRaw(*digitbranch, *triggerbranch);
117  }
118  rawwriter.getWriter().writeConfFile("EMC", "RAWDATA", o2::utils::concat_string(outputdir, "/EMCraw.cfg"));
