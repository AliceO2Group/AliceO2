29 void CosmicProcessor::process(DigitDataReader& reader, bool fill)
30 {
31  TStopwatch timerProcess;
32  timerProcess.Start();
33 
34  reader.init();
35 
36  auto array = reader.getDigitArray();
37  int ndig = array->size();
38  int ndig2 = ndig * fill;
39 
40  int bcdist = 200;
41  float thr = 5000000; // in ps
42 
43  int volID1[5], volID2[5];
44  float pos1[3], pos2[3];
45 
46  for (int i = 0; i < ndig; i++) {
47  auto& dig1 = (*array)[i];
48  int ch1 = dig1.getChannel();
49  mCounters[ch1]++;
50  if (mCounters[ch1] > 3) {
51  continue;
52  }
53  int64_t bc1 = int64_t(dig1.getBC());
54  int tdc1 = int(dig1.getTDC());
55  float tot1 = dig1.getTOT() * 48.8E-3;
56  Geo::getVolumeIndices(ch1, volID1);
57  Geo::getPos(volID1, pos1);
58 
59  for (int j = i + 1; j < ndig2; j++) {
60  auto& dig2 = (*array)[j];
61  int64_t bc2 = int64_t(dig2.getBC()) - bc1;
62  if (abs(bc2) > bcdist) {
63  continue;
64  }
65 
66  int ch2 = dig2.getChannel();
67  if (mCounters[ch2] > 3) {
68  continue;
69  }
70  int tdc2 = int(dig2.getTDC()) - tdc1;
71  float tot2 = dig2.getTOT() * 48.8E-3;
72  Geo::getVolumeIndices(ch2, volID2);
73  Geo::getPos(volID2, pos2);
74  pos2[0] -= pos1[0];
75  pos2[1] -= pos1[1];
76  pos2[2] -= pos1[2];
77 
78  float dtime = (bc2 * 1024 + tdc2) * Geo::TDCBIN; // in ps
79 
80  float l = sqrt(pos2[0] * pos2[0] + pos2[1] * pos2[1] + pos2[2] * pos2[2]);
81 
82  if (l < 500) {
83  continue;
84  }
85  if (pos2[1] > 0) {
86  l = -l;
87  }
88 
89  dtime -= l * 33.356409; // corrected for pad distance assuiming muonn downward
90 
91  if (abs(dtime) > thr) {
92  continue;
93  }
94 
95  mCosmicInfo.emplace_back(ch1, ch2, dtime, tot1, tot2, l);
96  }
97  }
98 
99  LOG(DEBUG) << "We had " << ndig << " digits in this event";
100  timerProcess.Stop();
101 }
