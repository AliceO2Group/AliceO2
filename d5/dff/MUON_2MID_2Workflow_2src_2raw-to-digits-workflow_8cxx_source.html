39 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
40 {
41  auto feeIdConfigFilename = cfgc.options().get<std::string>("feeId-config-file");
42  o2::mid::FEEIdConfig feeIdConfig;
43  if (!feeIdConfigFilename.empty()) {
44  feeIdConfig = o2::mid::FEEIdConfig(feeIdConfigFilename.c_str());
45  }
46  auto crateMasksFilename = cfgc.options().get<std::string>("crate-masks-file");
47  o2::mid::CrateMasks crateMasks;
48  if (!crateMasksFilename.empty()) {
49  crateMasks = o2::mid::CrateMasks(crateMasksFilename.c_str());
50  }
51  auto electronicsDelayFilename = cfgc.options().get<std::string>("electronics-delay-file");
52  o2::mid::ElectronicsDelay electronicsDelay;
53  if (!electronicsDelayFilename.empty()) {
54  electronicsDelay = o2::mid::readElectronicsDelay(electronicsDelayFilename.c_str());
55  }
56 
57  bool decodeOnly = cfgc.options().get<bool>("decode-only");
58 
59  o2::framework::WorkflowSpec specs;
60  specs.emplace_back(o2::mid::getRawDecoderSpec(false, feeIdConfig, crateMasks, electronicsDelay));
61  if (!decodeOnly) {
62  specs.emplace_back(o2::mid::getRawAggregatorSpec());
63  }
64  return specs;
65 }
