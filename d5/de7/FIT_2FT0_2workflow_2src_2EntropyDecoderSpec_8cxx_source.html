40 void EntropyDecoderSpec::run(ProcessingContext& pc)
41 {
42  auto cput = mTimer.CpuTime();
43  mTimer.Start(false);
44 
45  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
46 
47  auto& digits = pc.outputs().make<std::vector<o2::ft0::Digit>>(OutputRef{"digits"});
48  auto& channels = pc.outputs().make<std::vector<o2::ft0::ChannelData>>(OutputRef{"channels"});
49 
50  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
51  const auto ctfImage = o2::ft0::CTF::getImage(buff.data());
52  mCTFCoder.decode(ctfImage, digits, channels);
53 
54  mTimer.Stop();
55  LOG(INFO) << "Decoded " << channels.size() << " FT0 channels in " << digits.size() << " digits in " << mTimer.CpuTime() - cput << " s";
56 }
