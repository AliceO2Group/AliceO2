142 template <typename T>
143 T* CCDBManagerInstance::getForTimeStamp(std::string const& path, long timestamp)
144 {
145  if (!isCachingEnabled()) {
146  return mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp, nullptr, "",
147  mCreatedNotAfter ? std::to_string(mCreatedNotAfter) : "",
148  mCreatedNotBefore ? std::to_string(mCreatedNotBefore) : "");
149  }
150  auto& cached = mCache[path];
151  if (mCheckObjValidityEnabled && cached.isValid(timestamp)) {
152  return reinterpret_cast<T*>(cached.objPtr.get());
153  }
154 
155  T* ptr = mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp, &mHeaders, cached.uuid,
156  mCreatedNotAfter ? std::to_string(mCreatedNotAfter) : "",
157  mCreatedNotBefore ? std::to_string(mCreatedNotBefore) : "");
158  if (ptr) { // new object was shipped, old one (if any) is not valid anymore
159  cached.objPtr.reset(ptr);
160  cached.uuid = mHeaders["ETag"];
161  cached.startvalidity = std::stol(mHeaders["Valid-From"]);
162  cached.endvalidity = std::stol(mHeaders["Valid-Until"]);
163  } else if (mHeaders.count("Error")) { // in case of errors the pointer is 0 and headers["Error"] should be set
164  clearCache(path); // in case of any error clear cache for this object
165  } else { // the old object is valid
166  ptr = reinterpret_cast<T*>(cached.objPtr.get());
167  }
168  mHeaders.clear();
169  return ptr;
170 }
