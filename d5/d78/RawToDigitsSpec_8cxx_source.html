44  void init(o2::framework::InitContext& ic) final
45  {
46  // set up ADC value filling
47  mRawReader.createReader("");
48  mDigitDump.init();
49  mDigitDump.setInMemoryOnly();
50  const auto pedestalFile = ic.options().get<std::string>("pedestal-file");
51  if (pedestalFile.length()) {
52  LOGP(info, "Setting pedestal file: {}", pedestalFile);
53  mDigitDump.setPedestalAndNoiseFile(pedestalFile);
54  }

56  mRawReader.setADCDataCallback([this](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> int {
57  const int timeBins = mDigitDump.update(padROCPos, cru, data);
58  mDigitDump.setNumberOfProcessedTimeBins(std::max(mDigitDump.getNumberOfProcessedTimeBins(), size_t(timeBins)));
59  return timeBins;
60  });
61 
62  mRawReader.setLinkZSCallback([this](int cru, int rowInSector, int padInRow, int timeBin, float adcValue) -> bool {
63  CRU cruID(cru);
64  const PadRegionInfo& regionInfo = Mapper::instance().getPadRegionInfo(cruID.region());
65  mDigitDump.updateCRU(cruID, rowInSector - regionInfo.getGlobalRowOffset(), padInRow, timeBin, adcValue);
66  return true;
67  });
68 
69  mMaxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
70  mUseOldSubspec = ic.options().get<bool>("use-old-subspec");
71  mForceQuit = ic.options().get<bool>("force-quit");
72  if (mUseOldSubspec) {
73  LOGP(info, "Using old subspecification (CruId << 16) | ((LinkId + 1) << (CruEndPoint == 1 ? 8 : 0))");
74  }

77  void run(o2::framework::ProcessingContext& pc) final
78  {
79  // in case the maximum number of events was reached don't do further processing
80  if (mReadyToQuit) {
81  return;
82  }
83 
84  auto& reader = mRawReader.getReaders()[0];
85  mActiveSectors = calib_processing_helper::processRawData(pc.inputs(), reader, mUseOldSubspec);
86 
87  mDigitDump.incrementNEvents();
88  LOGP(info, "Number of processed events: {} ({})", mDigitDump.getNumberOfProcessedEvents(), mMaxEvents);
89 
90  snapshotDigits(pc.outputs());
91 
92  if ((mDigitDump.getNumberOfProcessedEvents() >= mMaxEvents)) {
93  LOGP(info, "Maximm number of events reached ({}), no more processing will be done", mMaxEvents);
94  mReadyToQuit = true;
95  if (mForceQuit) {
96  pc.services().get<ControlService>().endOfStream();
97  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
98  } else {
99  //pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
100  }
101  }
102  }
103 
104  void endOfStream(o2::framework::EndOfStreamContext& ec) final
105  {
106  LOGP(info, "endOfStream");
107  if (mActiveSectors) {
108  snapshotDigits(ec.outputs());
109  }
110  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
111  }

124  //____________________________________________________________________________
125  void snapshotDigits(DataAllocator& output)
126  {
127  mDigitDump.sortDigits();
128  for (auto isector : mSectors) {
129  o2::tpc::TPCSectorHeader header{isector};
130  header.activeSectors = mActiveSectors;
131  // digit for now are transported per sector, not per lane
132  output.snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe, header},
133  mDigitDump.getDigits(isector));
134  }
135  mDigitDump.clearDigits();
136  mActiveSectors = 0;
137  }
138 };

140 DataProcessorSpec getRawToDigitsSpec(int channel, const std::string inputSpec, std::vector<int> const& tpcSectors)
141 {
142  using device = o2::tpc::TPCDigitDumpDevice;
143 
144  std::vector<OutputSpec> outputs;
145  for (auto isector : tpcSectors) {
146  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
147  }
148 
149  return DataProcessorSpec{
150  fmt::format("TPCDigitizer{}", channel),
151  select(inputSpec.data()),
152  outputs,
153  AlgorithmSpec{adaptFromTask<device>(tpcSectors)},
154  Options{
155  {"max-events", VariantType::Int, 100, {"maximum number of events to process"}},
156  {"use-old-subspec", VariantType::Bool, false, {"use old subsecifiation definition"}},
157  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
158  {"pedestal-file", VariantType::String, "", {"file with pedestals and noise for zero suppression"}},
159  } // end Options
160  }; // end DataProcessorSpec
