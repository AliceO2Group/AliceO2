36 void ReconstructionDPL::run(ProcessingContext& pc)
37 {
38  mTimer.Start(false);
39  mRecPoints.clear();
40  auto digits = pc.inputs().get<gsl::span<o2::ft0::Digit>>("digits");
41  //auto digits = pc.inputs().get<const std::vector<o2::ft0::Digit>>("digits");
42  auto digch = pc.inputs().get<gsl::span<o2::ft0::ChannelData>>("digch");
43  // RS: if we need to process MC truth, uncomment lines below
44  //std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>> labels;
45  //const o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>* lblPtr = nullptr;
46  if (mUseMC) {
47  // labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>*>("labels");
48  // lblPtr = labels.get();
49  LOG(INFO) << "Ignoring MC info";
50  }
51  int nDig = digits.size();
52  LOG(DEBUG) << " nDig " << nDig;
53  mRecPoints.reserve(nDig);
54  mRecChData.resize(digch.size());
55  for (int id = 0; id < nDig; id++) {
56  const auto& digit = digits[id];
57  LOG(DEBUG) << " ndig " << id << " bc " << digit.getBC() << " orbit " << digit.getOrbit();
58  auto channels = digit.getBunchChannelData(digch);
59  gsl::span<o2::ft0::ChannelDataFloat> out_ch(mRecChData);
60  out_ch = out_ch.subspan(digit.ref.getFirstEntry(), digit.ref.getEntries());
61  mRecPoints.emplace_back(mReco.process(digit, channels, out_ch));
62  }
63  // do we ignore MC in this task?
64 
65  LOG(DEBUG) << "FT0 reconstruction pushes " << mRecPoints.size() << " RecPoints";
66  pc.outputs().snapshot(Output{mOrigin, "RECPOINTS", 0, Lifetime::Timeframe}, mRecPoints);
67  pc.outputs().snapshot(Output{mOrigin, "RECCHDATA", 0, Lifetime::Timeframe}, mRecChData);
68 
69  mTimer.Stop();
70 }

78 DataProcessorSpec getReconstructionSpec(bool useMC)
79 {
80  std::vector<InputSpec> inputSpec;
81  std::vector<OutputSpec> outputSpec;
82  inputSpec.emplace_back("digits", o2::header::gDataOriginFT0, "DIGITSBC", 0, Lifetime::Timeframe);
83  inputSpec.emplace_back("digch", o2::header::gDataOriginFT0, "DIGITSCH", 0, Lifetime::Timeframe);
84  if (useMC) {
85  LOG(INFO) << "Currently Reconstruction does not consume and provide MC truth";
86  inputSpec.emplace_back("labels", o2::header::gDataOriginFT0, "DIGITSMCTR", 0, Lifetime::Timeframe);
87  }
88  outputSpec.emplace_back(o2::header::gDataOriginFT0, "RECPOINTS", 0, Lifetime::Timeframe);
89  outputSpec.emplace_back(o2::header::gDataOriginFT0, "RECCHDATA", 0, Lifetime::Timeframe);
90 
91  return DataProcessorSpec{
92  "ft0-reconstructor",
93  inputSpec,
94  outputSpec,
95  AlgorithmSpec{adaptFromTask<ReconstructionDPL>(useMC)},
96  Options{}};
97 }
