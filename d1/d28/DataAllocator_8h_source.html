100  template <typename T, typename... Args>
101  decltype(auto) make(const Output& spec, Args... args)
102  {
103  if constexpr (is_specialization<T, std::vector>::value && has_messageable_value_type<T>::value) {
104  // this catches all std::vector objects with messageable value type before checking if is also
105  // has a root dictionary, so non-serialized transmission is preferred
106  using ValueType = typename T::value_type;
107  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
108  auto& context = mRegistry->get<MessageContext>();
109 
110  // Note: initial payload size is 0 and will be set by the context before sending
111  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, 0);
112  return context.add<MessageContext::VectorObject<ValueType>>(std::move(headerMessage), channel, 0, std::forward<Args>(args)...).get();
113  } else if constexpr (has_root_dictionary<T>::value == true && is_messageable<T>::value == false) {
114  // Extended support for types implementing the Root ClassDef interface, both TObject
115  // derived types and others
116  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
117  auto& context = mRegistry->get<MessageContext>();
118 
119  // Note: initial payload size is 0 and will be set by the context before sending
120  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodROOT, 0);
121  return context.add<MessageContext::RootSerializedObject<T>>(std::move(headerMessage), channel, std::forward<Args>(args)...).get();
122  } else if constexpr (std::is_base_of_v<std::string, T>) {
123  std::string* s = new std::string(args...);
124  adopt(spec, s);
125  return *s;
126  } else if constexpr (std::is_base_of_v<struct TableBuilder, T>) {
127  TableBuilder* tb = nullptr;
128  call_if_defined<struct TableBuilder>([&](auto* p) {
129  tb = new std::decay_t<decltype(*p)>(args...);
130  adopt(spec, tb);
131  });
132  return *tb;
133  } else if constexpr (std::is_base_of_v<struct TreeToTable, T>) {
134  void* t2tr = nullptr;
135  call_if_defined<struct TreeToTable>([&](auto* p) {
136  auto t2t = new std::decay_t<decltype(*p)>(args...);
137  adopt(spec, t2t);
138  t2tr = t2t;
139  });
140  return *reinterpret_cast<TreeToTable*>(t2tr);
141  } else if constexpr (sizeof...(Args) == 0) {
142  constexpr bool isBoostSerializable = framework::is_boost_serializable<T>::value;
143  if constexpr (is_messageable<T>::value == true) {
144  return *reinterpret_cast<T*>(newChunk(spec, sizeof(T)).data());
145  } else if constexpr (is_specialization<T, BoostSerialized>::value == true) {
146  return make_boost<typename T::wrapped_type>(std::move(spec));
147  } else if constexpr (is_specialization<T, BoostSerialized>::value == false && isBoostSerializable == true && std::is_base_of<std::string, T>::value == false) {
148  return make_boost<T>(std::move(spec));
149  } else {
150  static_assert(always_static_assert_v<T>, ERROR_STRING);
151  }
152  } else if constexpr (sizeof...(Args) == 1) {
153  using FirstArg = typename std::tuple_element<0, std::tuple<Args...>>::type;
154  if constexpr (std::is_integral_v<FirstArg>) {
155  if constexpr (is_messageable<T>::value == true) {
156  auto [nElements] = std::make_tuple(args...);
157  auto size = nElements * sizeof(T);
158  std::string const& channel = matchDataHeader(spec, mTimingInfo->timeslice);
159  auto& context = mRegistry->get<MessageContext>();
160 
161  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, size);
162  return context.add<MessageContext::SpanObject<T>>(std::move(headerMessage), channel, 0, nElements).get();
163  }
164  } else if constexpr (std::is_same_v<FirstArg, std::shared_ptr<arrow::Schema>>) {
165  if constexpr (std::is_base_of_v<arrow::ipc::RecordBatchWriter, T>) {
166  auto [schema] = std::make_tuple(args...);
167  std::shared_ptr<arrow::ipc::RecordBatchWriter> writer;
168  create(spec, &writer, schema);
169  return writer;
170  }
171  } else if constexpr (is_specialization<T, BoostSerialized>::value) {
172  return make_boost<FirstArg>(std::move(spec));
173  } else {
174  static_assert(always_static_assert_v<T>, ERROR_STRING);
175  }
176  } else {
177  static_assert(always_static_assert_v<T>, ERROR_STRING);
178  }

260  template <typename T>
261  void snapshot(const Output& spec, T const& object)
262  {
263  auto proxy = mRegistry->get<MessageContext>().proxy();
264  FairMQMessagePtr payloadMessage;
265  auto serializationType = o2::header::gSerializationMethodNone;
266  if constexpr (is_messageable<T>::value == true) {
267  // Serialize a snapshot of a trivially copyable, non-polymorphic object,
268  payloadMessage = proxy.createMessage(sizeof(T));
269  memcpy(payloadMessage->GetData(), &object, sizeof(T));
270 
271  serializationType = o2::header::gSerializationMethodNone;
272  } else if constexpr (is_specialization<T, std::vector>::value == true ||
273  (gsl::details::is_span<T>::value && has_messageable_value_type<T>::value)) {
274  using ElementType = typename std::remove_pointer<typename T::value_type>::type;
275  if constexpr (is_messageable<ElementType>::value) {
276  // Serialize a snapshot of a std::vector of trivially copyable, non-polymorphic elements
277  // Note: in most cases it is better to use the `make` function und work with the provided
278  // reference object
279  constexpr auto elementSizeInBytes = sizeof(ElementType);
280  auto sizeInBytes = elementSizeInBytes * object.size();
281  payloadMessage = proxy.createMessage(sizeInBytes);
282 
283  if constexpr (std::is_pointer<typename T::value_type>::value == false) {
284  // vector of elements
285  memcpy(payloadMessage->GetData(), object.data(), sizeInBytes);
286  } else {
287  // serialize vector of pointers to elements
288  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
289  for (auto const& pointer : object) {
290  memcpy(target, pointer, elementSizeInBytes);
291  target += elementSizeInBytes;
292  }
293  }
294 
295  serializationType = o2::header::gSerializationMethodNone;
296  } else if constexpr (has_root_dictionary<ElementType>::value) {
297  return snapshot(spec, ROOTSerialized<T const>(object));
298  } else {
299  static_assert(always_static_assert_v<T>,
300  "value type of std::vector not supported by API, supported types:"
301  "\n - messageable tyeps (trivially copyable, non-polymorphic structures)"
302  "\n - pointers to those"
303  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
304  }
305  } else if constexpr (has_root_dictionary<T>::value == true || is_specialization<T, ROOTSerialized>::value == true) {
306  // Serialize a snapshot of an object with root dictionary
307  payloadMessage = proxy.createMessage();
308  if constexpr (is_specialization<T, ROOTSerialized>::value == true) {
309  // Explicitely ROOT serialize a snapshot of object.
310  // An object wrapped into type `ROOTSerialized` is explicitely marked to be ROOT serialized
311  // and is expected to have a ROOT dictionary. Availability can not be checked at compile time
312  // for all cases.
313  using WrappedType = typename T::wrapped_type;
314  static_assert(std::is_same<typename T::hint_type, const char>::value ||
315  std::is_same<typename T::hint_type, TClass>::value ||
316  std::is_void<typename T::hint_type>::value,
317  "class hint must be of type TClass or const char");
318 
319  const TClass* cl = nullptr;
320  if (object.getHint() == nullptr) {
321  // get TClass info by wrapped type
322  cl = TClass::GetClass(typeid(WrappedType));
323  } else if (std::is_same<typename T::hint_type, TClass>::value) {
324  // the class info has been passed directly
325  cl = reinterpret_cast<const TClass*>(object.getHint());
326  } else if (std::is_same<typename T::hint_type, const char>::value) {
327  // get TClass info by optional name
328  cl = TClass::GetClass(reinterpret_cast<const char*>(object.getHint()));
329  }
330  if (has_root_dictionary<WrappedType>::value == false && cl == nullptr) {
331  if (std::is_same<typename T::hint_type, const char>::value) {
332  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
333  reinterpret_cast<const char*>(object.getHint()));
334  } else {
335  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
336  typeid(WrappedType).name());
337  }
338  }
339  TMessageSerializer().Serialize(*payloadMessage, &object(), cl);
340  } else {
341  TMessageSerializer().Serialize(*payloadMessage, &object, TClass::GetClass(typeid(T)));
342  }
343  serializationType = o2::header::gSerializationMethodROOT;
344  } else {
345  static_assert(always_static_assert_v<T>,
346  "data type T not supported by API, \n specializations available for"
347  "\n - trivially copyable, non-polymorphic structures"
348  "\n - std::vector of messageable structures or pointers to those"
349  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
350  }
351  addPartToContext(std::move(payloadMessage), spec, serializationType);
352  }
