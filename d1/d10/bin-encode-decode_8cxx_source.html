33 template <typename T>
34 void readFile(const std::string& filename, std::vector<T>* tokens)
35 {
36  std::ifstream is(filename, std::ios_base::binary | std::ios_base::in);
37  if (is) {
38  // get length of file:
39  is.seekg(0, is.end);
40  size_t length = is.tellg();
41  is.seekg(0, is.beg);
42 
43  // reserve size of tokens
44  if (!tokens) {
45  throw std::runtime_error("Cannot read file into nonexistent vector");
46  }
47 
48  if (length % sizeof(T)) {
49  throw std::runtime_error("Filesize is not a multiple of datatype.");
50  }
51  // size the vector appropriately
52  size_t num_elems = length / sizeof(T);
53  tokens->resize(num_elems);
54 
55  // read data as a block:
56  is.read(reinterpret_cast<char*>(tokens->data()), length);
57  is.close();
58  }
59 }
