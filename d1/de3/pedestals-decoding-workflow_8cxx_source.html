77 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
78 {
79  static int mNrdhs = 0;
80  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
81  mNrdhs++;
82 
83  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
84  auto flags = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF00;
85  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);
86  uint32_t feeId = cruId * 2 + endpoint + flags;
87  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);
88 
89  if (verbose) {
90  std::cout << mNrdhs << "--\n";
91  o2::raw::RDHUtils::printRDH(rdhAny);
92  }
93 };

162  void decodePage(gsl::span<const std::byte> page)
163  {
164  size_t ndigits{0};
165 
166  uint32_t orbit;
167 
168  auto tStart = std::chrono::high_resolution_clock::now();
169 
170  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
171  auto solarId = dsElecId.solarId();
172  auto dsId = dsElecId.elinkId();
173 
174  mDigits.emplace_back(o2::mch::calibration::PedestalDigit(solarId, dsId, channel, sc.bunchCrossing, 0, sc.samples));
175  ++ndigits;
176  };
177 
178  auto errorHandler = [&](DsElecId dsElecId, int8_t chip, uint32_t error) {
179  auto solarId = dsElecId.solarId();
180  auto dsId = dsElecId.elinkId();
181 
182  mErrors.emplace_back(o2::mch::DecoderError(solarId, dsId, chip, error));
183  };
184 
185  if (mPatchRDH) {
186  patchPage(page, mDebug);
187  }
188 
189  if (!mDecoder) {
190  DecodedDataHandlers handlers;
191  handlers.sampaChannelHandler = channelHandler;
192  handlers.sampaErrorHandler = errorHandler;
193  mDecoder = mFee2Solar ? o2::mch::raw::createPageDecoder(page, handlers, mFee2Solar)
194  : o2::mch::raw::createPageDecoder(page, handlers);
195  }
196  try {
197  mDecoder(page);
198  } catch (std::exception& e) {
199  std::cout << e.what() << '\n';
200  }
201  }

203  //_________________________________________________________________________________________________
204  void decodeBuffer(gsl::span<const std::byte> buf)
205  {
206  if (mDebug) {
207  std::cout << "\n\n============================\nStart of new buffer\n";
208  }
209  size_t bufSize = buf.size();
210  size_t pageStart = 0;
211  while (bufSize > pageStart) {
212  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
213  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
214  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
215  if (rdhHeaderSize != 64) {
216  break;
217  }
218  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
219 
220  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
221  decodePage(page);
222 
223  pageStart += pageSize;
224  }
225  }

228  // the decodeTF() function processes the messages generated by the (sub)TimeFrame builder
229  void decodeTF(framework::ProcessingContext& pc)
230  {
231  // get the input buffer
232  auto& inputs = pc.inputs();
233  DPLRawParser parser(inputs, o2::framework::select(mInputSpec.c_str()));
234 
235  auto tStart = std::chrono::high_resolution_clock::now();
236  size_t totPayloadSize = 0;
237  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
238  auto const* raw = it.raw();
239  if (!raw) {
240  continue;
241  }
242  size_t payloadSize = it.size();
243  totPayloadSize += payloadSize;
244 
245  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), sizeof(RDH) + payloadSize);
246  decodeBuffer(buffer);
247  }
248  auto tEnd = std::chrono::high_resolution_clock::now();
249 
250  if (totPayloadSize > 0) {
251  std::chrono::duration<double, std::milli> elapsed = tEnd - tStart;
252  mTimeDecoder += elapsed;
253  if (!mTimeDecoderMin || (elapsed < mTimeDecoderMin)) {
254  mTimeDecoderMin = elapsed;
255  }
256  if (!mTimeDecoderMax || (elapsed > mTimeDecoderMax)) {
257  mTimeDecoderMax = elapsed;
258  }
259  mTFcount += 1;
260  }
261  }

280  void run(framework::ProcessingContext& pc)
281  {
282  auto createBuffer = [&](auto& vec, size_t& size) {
283  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
284  char* buf = nullptr;
285  if (size > 0) {
286  buf = (char*)malloc(size);
287  if (buf) {
288  char* p = buf;
289  size_t sizeofElement = sizeof(*(vec.begin()));
290  for (auto& element : vec) {
291  memcpy(p, &element, sizeofElement);
292  p += sizeofElement;
293  }
294  }
295  }
296  return buf;
297  };
298 
299  reset();
300  for (auto&& input : pc.inputs()) {
301  if (input.spec->binding == "TF") {
302  decodeTF(pc);
303  }
304  if (input.spec->binding == "readout") {
305  decodeReadout(input);
306  }
307  }
308 
309  if (mDebug) {
310  usleep(100000);
311  }
312 
313  size_t digitsSize;
314  char* digitsBuffer = createBuffer(mDigits, digitsSize);
315 
316  size_t errorsSize;
317  char* errorsBuffer = createBuffer(mErrors, errorsSize);
318 
319  // create the output message
320  auto freefct = [](void* data, void*) { free(data); };
321  pc.outputs().adoptChunk(Output{"MCH", "PDIGITS", 0}, digitsBuffer, digitsSize, freefct, nullptr);
322  pc.outputs().adoptChunk(Output{"MCH", "ERRORS", 0}, errorsBuffer, errorsSize, freefct, nullptr);
323  }
