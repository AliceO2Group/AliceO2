45 void TrackerDPL::init(InitContext& ic)
46 {
47  mTimer.Stop();
48  mTimer.Reset();
49  auto filename = ic.options().get<std::string>("grp-file");
50  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
51  if (grp) {
52  mGRP.reset(grp);
53  o2::base::Propagator::initFieldFromGRP(grp);
54  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
55 
56  o2::base::GeometryManager::loadGeometry();
57  o2::mft::GeometryTGeo* geom = o2::mft::GeometryTGeo::Instance();
58  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::T2GRot,
59  o2::math_utils::TransformType::T2G));
60 
61  // tracking configuration parameters
62  auto& mftTrackingParam = MFTTrackingParam::Instance();
63  // create the tracker: set the B-field, the configuration and initialize
64  mTracker = std::make_unique<o2::mft::Tracker>(mUseMC);
65  double centerMFT[3] = {0, 0, -61.4}; // Field at center of MFT
66  mTracker->setBz(field->getBz(centerMFT));
67  mTracker->initConfig(mftTrackingParam);
68  mTracker->initialize();
69  } else {
70  throw std::runtime_error(o2::utils::concat_string("Cannot retrieve GRP from the ", filename));
71  }
72 
73  std::string dictPath = ic.options().get<std::string>("mft-dictionary-path");
74  std::string dictFile = o2::base::NameConf::getDictionaryFileName(o2::detectors::DetID::MFT, dictPath, ".bin");
75  if (o2::base::NameConf::pathExists(dictFile)) {
76  mDict.readBinaryFile(dictFile);
77  LOG(INFO) << "Tracker running with a provided dictionary: " << dictFile;
78  } else {
79  LOG(INFO) << "Dictionary " << dictFile << " is absent, Tracker expects cluster patterns";
80  }
81 }

83 void TrackerDPL::run(ProcessingContext& pc)
84 {
85  mTimer.Start(false);
86  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
87  auto compClusters = pc.inputs().get<const std::vector<o2::itsmft::CompClusterExt>>("compClusters");
88  auto nTracksLTF = 0;
89  auto nTracksCA = 0;
90 
91  // code further down does assignment to the rofs and the altered object is used for output
92  // we therefore need a copy of the vector rather than an object created directly on the input data,
93  // the output vector however is created directly inside the message memory thus avoiding copy by
94  // snapshot
95  auto rofsinput = pc.inputs().get<const std::vector<o2::itsmft::ROFRecord>>("ROframes");
96  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"MFT", "TRACKSROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
97 
98  LOG(INFO) << "MFTTracker pulled " << compClusters.size() << " compressed clusters in "
99  << rofsinput.size() << " RO frames";
100 
101  const dataformats::MCTruthContainer<MCCompLabel>* labels = mUseMC ? pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("labels").release() : nullptr;
102  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
103  if (mUseMC) {
104  // get the array as read-only span, a snapshot of the object is sent forward
105  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
106  LOG(INFO) << labels->getIndexedSize() << " MC label objects , in "
107  << mc2rofs.size() << " MC events";
108  }
109 
110  //std::vector<o2::mft::TrackMFTExt> tracks;
111  auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{"MFT", "TRACKCLSID", 0, Lifetime::Timeframe});
112  o2::dataformats::MCTruthContainer<o2::MCCompLabel> trackLabels;
113  o2::dataformats::MCTruthContainer<o2::MCCompLabel> allTrackLabels;
114  std::vector<o2::mft::TrackLTF> tracksLTF;
115  std::vector<o2::mft::TrackCA> tracksCA;
116  auto& allTracksMFT = pc.outputs().make<std::vector<o2::mft::TrackMFT>>(Output{"MFT", "TRACKS", 0, Lifetime::Timeframe});
117 
118  std::uint32_t roFrame = 0;
119  o2::mft::ROframe event(0);
120 
121  Bool_t continuous = mGRP->isDetContinuousReadOut("MFT");
122  LOG(INFO) << "MFTTracker RO: continuous=" << continuous;
123 
124  // snippet to convert found tracks to final output tracks with separate cluster indices
125  auto copyTracks = [&event](auto& tracks, auto& allTracks, auto& allClusIdx) {
126  for (auto& trc : tracks) {
127  trc.setExternalClusterIndexOffset(allClusIdx.size());
128  int ncl = trc.getNumberOfPoints();
129  for (int ic = 0; ic < ncl; ic++) {
130  auto externalClusterID = trc.getExternalClusterIndex(ic);
131  allClusIdx.push_back(externalClusterID);
132  }
133  allTracks.emplace_back(trc);
134  }
135  };
136 
137  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
138  if (continuous) {
139  for (auto& rof : rofs) {
140  int nclUsed = ioutils::loadROFrameData(rof, event, compClusters, pattIt, mDict, labels, mTracker.get());
141  if (nclUsed) {
142  event.setROFrameId(roFrame);
143  event.initialize();
144  LOG(INFO) << "ROframe: " << roFrame << ", clusters loaded : " << nclUsed;
145  mTracker->setROFrame(roFrame);
146  mTracker->clustersToTracks(event);
147  tracksLTF.swap(event.getTracksLTF());
148  tracksCA.swap(event.getTracksCA());
149  nTracksLTF += tracksLTF.size();
150  nTracksCA += tracksCA.size();
151 
152  if (mUseMC) {
153  mTracker->computeTracksMClabels(tracksLTF);
154  mTracker->computeTracksMClabels(tracksCA);
155  trackLabels = mTracker->getTrackLabels();
156  allTrackLabels.mergeAtBack(trackLabels);
157  }
158 
159  LOG(INFO) << "Found tracks LTF: " << tracksLTF.size();
160  LOG(INFO) << "Found tracks CA: " << tracksCA.size();
161  int first = allTracksMFT.size();
162  int number = tracksLTF.size() + tracksCA.size();
163  rof.setFirstEntry(first);
164  rof.setNEntries(number);
165  copyTracks(tracksLTF, allTracksMFT, allClusIdx);
166  copyTracks(tracksCA, allTracksMFT, allClusIdx);
167  }
168  roFrame++;
169  }
170  }
171 
172  LOG(INFO) << "MFTTracker found " << nTracksLTF << " tracks LTF";
173  LOG(INFO) << "MFTTracker found " << nTracksCA << " tracks CA";
174  LOG(INFO) << "MFTTracker pushed " << allTracksMFT.size() << " tracks";
175 
176  if (mUseMC) {
177  pc.outputs().snapshot(Output{"MFT", "TRACKSMCTR", 0, Lifetime::Timeframe}, allTrackLabels);
178  pc.outputs().snapshot(Output{"MFT", "TRACKSMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
179  }
180  mTimer.Stop();
181 }

189 DataProcessorSpec getTrackerSpec(bool useMC)
190 {
191  std::vector<InputSpec> inputs;
192  inputs.emplace_back("compClusters", "MFT", "COMPCLUSTERS", 0, Lifetime::Timeframe);
193  inputs.emplace_back("patterns", "MFT", "PATTERNS", 0, Lifetime::Timeframe);
194  inputs.emplace_back("ROframes", "MFT", "CLUSTERSROF", 0, Lifetime::Timeframe);
195 
196  std::vector<OutputSpec> outputs;
197  outputs.emplace_back("MFT", "TRACKS", 0, Lifetime::Timeframe);
198  outputs.emplace_back("MFT", "TRACKSROF", 0, Lifetime::Timeframe);
199  outputs.emplace_back("MFT", "TRACKCLSID", 0, Lifetime::Timeframe);
200 
201  if (useMC) {
202  inputs.emplace_back("labels", "MFT", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
203  inputs.emplace_back("MC2ROframes", "MFT", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
204  outputs.emplace_back("MFT", "TRACKSMCTR", 0, Lifetime::Timeframe);
205  outputs.emplace_back("MFT", "TRACKSMC2ROF", 0, Lifetime::Timeframe);
206  }
207 
208  return DataProcessorSpec{
209  "mft-tracker",
210  inputs,
211  outputs,
212  AlgorithmSpec{adaptFromTask<TrackerDPL>(useMC)},
213  Options{
214  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the output file"}},
215  {"mft-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}}}};
216 }
