258 {
259  using namespace o2::framework;
260  TableBuilder builder1;

271  }

273  BOOST_REQUIRE_EQUAL(table2->num_columns(), 2);
274  BOOST_REQUIRE_EQUAL(table2->num_rows(), 8);
275  for (size_t i = 0; i < 8; ++i) {
276  auto p2 = std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int32Type>>(table2->column(0)->chunk(0));
277  BOOST_CHECK_EQUAL(p2->Value(i), i);
278  }

280  auto source1 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
281  auto source2 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
282  auto cross = ROOT::RDF::MakeCrossProductDataFrame(std::move(source1), std::move(source2));
283  auto source3 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
284  auto source4 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
285  auto indexed = ROOT::RDF::MakeColumnIndexedDataFrame(std::move(source3), std::move(source4), "x");
286  auto source5 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
287  auto source6 = std::make_unique<ROOT::RDF::RArrowDS>(table2, std::vector<std::string>{});
288  auto unionDS = ROOT::RDF::MakeFriendDataFrame(std::move(source5), std::move(source6));
289  auto source7 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{});
290  auto source8 = std::make_unique<ROOT::RDF::RArrowDS>(table1, std::vector<std::string>{}); // Notice the table needs to be the same

292 
293  ROOT::RDataFrame finalDF{std::move(cross)};

297 
298  BOOST_CHECK_EQUAL(*finalDF.Count(), 64); // Full cross product of 8x8 rows, 64 entries
