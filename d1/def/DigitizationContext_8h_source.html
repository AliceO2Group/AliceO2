27 // a structure describing EventPart
28 // (an elementary constituent of a collision)
29 
30 constexpr static int QEDSOURCEID = 99;
31 
32 struct EventPart {
33  EventPart() = default;
34  EventPart(int s, int e) : sourceID(s), entryID(e) {}
35  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)
36  // the sourceID should correspond to the chain ID
37  int entryID = 0; // the event/entry ID inside the chain corresponding to sourceID
38 
39  static bool isSignal(EventPart e) { return e.sourceID > 1 && e.sourceID != QEDSOURCEID; }
40  static bool isBackGround(EventPart e) { return !isSignal(e); }
41  static bool isQED(EventPart e) { return e.sourceID == QEDSOURCEID; }
42  ClassDefNV(EventPart, 1);
43 };

136  const char* brname,

138  int entryID,
139  std::vector<T>* hits) const

145  }
