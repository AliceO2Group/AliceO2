37 // create histogram from specification and insert it into the registry
38 void HistogramRegistry::insert(const HistogramSpec& histSpec)
39 {
40  validateHistName(histSpec.name.data(), histSpec.hash);
41  const uint32_t idx = imask(histSpec.hash);
42  for (auto i = 0u; i < MAX_REGISTRY_SIZE; ++i) {
43  TObject* rawPtr = nullptr;
44  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mRegistryValue[imask(idx + i)]);
45  if (!rawPtr) {
46  registerName(histSpec.name);
47  mRegistryKey[imask(idx + i)] = histSpec.hash;
48  mRegistryValue[imask(idx + i)] = HistFactory::createHistVariant(histSpec);
49  lookup += i;
50  return;
51  }
52  }
53  LOGF(FATAL, R"(Internal array of HistogramRegistry "%s" is full.)", mName);
54 }

57 void HistogramRegistry::validateHistName(const char* name, const uint32_t hash)
58 {
59  // validate that hash is unique
60  auto it = std::find(mRegistryKey.begin(), mRegistryKey.end(), hash);
61  if (it != mRegistryKey.end()) {
62  auto idx = it - mRegistryKey.begin();
63  std::string collidingName{};

91 // store a copy of an existing histogram (or group of histograms) under a different name
92 void HistogramRegistry::addClone(const std::string& source, const std::string& target)
93 {
94  auto doInsertClone = [&](const auto& sharedPtr) {
95  if (!sharedPtr.get()) {
96  return;
97  }
98  std::string sourceName{((TNamed*)sharedPtr.get())->GetName()};
99  // search for histograms starting with source_ substring
100  if (sourceName.rfind(source, 0) == 0) {
101  // when cloning groups of histograms source_ and target_ must end with "/"
102  if (sourceName.size() != source.size() && (source.back() != '/' || target.back() != '/')) {
103  return;
104  }
105  // when cloning a single histogram the specified target_ must not be a group name
106  if (sourceName.size() == source.size() && target.back() == '/') {
107  LOGF(FATAL, "Cannot turn histogram into folder!");
108  }
109  std::string targetName{target};
110  targetName += sourceName.substr(sourceName.find(source) + source.size());
111  insertClone(targetName.data(), sharedPtr);
112  }
113  };
114 
115  for (auto& histVariant : mRegistryValue) {
116  std::visit(doInsertClone, histVariant);
117  }
118 }

120 // function to query if name is already in use
121 bool HistogramRegistry::contains(const HistName& histName)
122 {
123  // check for all occurances of the hash
124  auto iter = mRegistryKey.begin();
125  while ((iter = std::find(iter, mRegistryKey.end(), histName.hash)) != mRegistryKey.end()) {
126  const char* curName = nullptr;
127  std::visit([&](auto&& hist) { if(hist) { curName = hist->GetName(); } }, mRegistryValue[iter - mRegistryKey.begin()]);
128  // if hash is the same, make sure that name is indeed the same
129  if (strcmp(curName, histName.str) == 0) {
130  return true;
131  }
132  }
133  return false;
134 }

154 // print some useful meta-info about the stored histograms
155 void HistogramRegistry::print(bool showAxisDetails)
156 {
157  std::vector<double> fillFractions{0.1, 0.25, 0.5};
158  std::vector<double> totalSizes(fillFractions.size());
159 
160  uint32_t nHistos{};
161  bool containsSparseHist{};
162  auto printHistInfo = [&](auto&& hist) {
163  if (hist) {
164  using T = std::decay_t<decltype(*hist)>;
165  bool isSparse{};
166  if (hist->InheritsFrom(THnSparse::Class())) {
167  isSparse = true;
168  containsSparseHist = true;
169  }
170  ++nHistos;
171  std::vector<double> sizes;
172  std::string sizeInfo{};
173  if (isSparse) {
174  std::transform(std::begin(fillFractions), std::end(fillFractions), std::back_inserter(sizes), [&hist](auto& fraction) { return HistFiller::getSize(hist, fraction); });
175  for (int i = 0; i < fillFractions.size(); ++i) {
176  sizeInfo += fmt::format("{:.2f} kB ({:.0f} %)", sizes[i] * 1024, fillFractions[i] * 100);
177  if (i != fillFractions.size() - 1) {
178  sizeInfo += ", ";
179  }
180  }
181  } else {
182  double size = HistFiller::getSize(hist);
183  sizes.resize(fillFractions.size(), size);
184  sizeInfo = fmt::format("{:.2f} kB", sizes[0] * 1024);
185  }
186  std::transform(totalSizes.begin(), totalSizes.end(), sizes.begin(), totalSizes.begin(), std::plus<double>());
187  LOGF(INFO, "Hist %03d: %-35s %-19s [%s]", nHistos, hist->GetName(), hist->IsA()->GetName(), sizeInfo);
188 
189  if (showAxisDetails) {
190  int nDim = 0;
191  if constexpr (std::is_base_of_v<THnBase, T>) {
192  nDim = hist->GetNdimensions();
193  } else if constexpr (std::is_base_of_v<TH1, T>) {
194  nDim = hist->GetDimension();
195  }
196  for (int d = 0; d < nDim; ++d) {
197  TAxis* axis = HistFactory::getAxis(d, hist);
198  LOGF(INFO, "- Axis %d: %-20s (%d bins)", d, axis->GetTitle(), axis->GetNbins());
199  }
200  }
201  }
202  };
203 
204  std::string titleString{"======================== HistogramRegistry ========================"};
205  LOGF(INFO, "");
206  LOGF(INFO, "%s", titleString);
207  LOGF(INFO, "%s\"%s\"", std::string((int)(0.5 * titleString.size() - (1 + 0.5 * mName.size())), ' '), mName);
208  std::sort(mRegisteredNames.begin(), mRegisteredNames.end());
209  for (auto& curHistName : mRegisteredNames) {
210  std::visit(printHistInfo, mRegistryValue[getHistIndex(HistName{curHistName.data()})]);
211  }
212  std::string totalSizeInfo{};
213  if (containsSparseHist) {
214  for (int i = 0; i < totalSizes.size(); ++i) {
215  totalSizeInfo += fmt::format("{:.2f} MB ({:.0f} %)", totalSizes[i], fillFractions[i] * 100);
216  if (i != totalSizes.size() - 1) {
217  totalSizeInfo += ", ";
218  }
219  }
220  } else {
221  totalSizeInfo = fmt::format("{:.2f} MB", totalSizes[0]);
222  }
223  LOGF(INFO, "%s", std::string(titleString.size(), '='), titleString);
224  LOGF(INFO, "Total: %d histograms, ca. %s", nHistos, totalSizeInfo);
225  if (lookup) {
226  LOGF(INFO, "Due to index collisions, histograms were shifted by %d registry slots in total.", lookup);
227  }
228  LOGF(INFO, "%s", std::string(titleString.size(), '='), titleString);
229  LOGF(INFO, "");
230 }

232 // create output structure will be propagated to file-sink
233 TList* HistogramRegistry::operator*()
234 {
235  TList* list = new TList();
236  list->SetName(mName.data());
237 
238  for (auto i = 0u; i < MAX_REGISTRY_SIZE; ++i) {
239  TNamed* rawPtr = nullptr;
240  std::visit([&](const auto& sharedPtr) { rawPtr = (TNamed*)sharedPtr.get(); }, mRegistryValue[i]);
241  if (rawPtr) {
242  std::deque<std::string> path = splitPath(rawPtr->GetName());
243  std::string name = path.back();
244  path.pop_back();
245  TList* targetList{getSubList(list, path)};
246  if (targetList) {
247  rawPtr->SetName(name.data());
248  targetList->Add(rawPtr);
249  } else {
250  LOGF(FATAL, "Specified subfolder could not be created.");
251  }
252  }
253  }
254 
255  // sort histograms in output file alphabetically
256  if (mSortHistos) {
257  std::function<void(TList*)> sortList;
258  sortList = [&](TList* list) {
259  list->Sort();
260  TIter next(list);
261  TNamed* subList = nullptr;
262  std::vector<TObject*> subLists;
263  while ((subList = (TNamed*)next())) {
264  if (subList->InheritsFrom(TList::Class())) {
265  subLists.push_back(subList);
266  sortList((TList*)subList);
267  }
268  }
269  // place lists always at the top
270  std::reverse(subLists.begin(), subLists.end());
271  for (auto curList : subLists) {
272  list->Remove(curList);
273  list->AddFirst(curList);
274  }
275  };
276  sortList(list);
277  }
278 
279  // create dedicated directory containing all of the registrys histograms
280  if (mCreateRegistryDir) {
281  // propagate this to the writer by adding a 'flag' to the output list
282  list->AddLast(new TNamed("createFolder", ""));
283  }
284  return list;
285 }

287 // helper function to create resp. find the subList defined by path
288 TList* HistogramRegistry::getSubList(TList* list, std::deque<std::string>& path)
289 {
290  if (path.empty()) {
291  return list;
292  }
293  TList* targetList{nullptr};
294  std::string nextList = path[0];
295  path.pop_front();
296  if (auto subList = (TList*)list->FindObject(nextList.data())) {
297  if (subList->InheritsFrom(TList::Class())) {
298  targetList = getSubList((TList*)subList, path);
299  } else {
300  return nullptr;
301  }
302  } else {
303  subList = new TList();
304  subList->SetName(nextList.data());
305  list->Add(subList);
306  targetList = getSubList(subList, path);
307  }
308  return targetList;
309 }

323 // helper function that checks if name of histogram is reasonable and keeps track of names already in use
324 void HistogramRegistry::registerName(const std::string& name)
325 {
326  if (name.empty() || name.back() == '/') {
327  LOGF(FATAL, "Invalid name for a histogram.");
328  }
329  std::deque<std::string> path = splitPath(name);
330  std::string cumulativeName{};
331  int depth = path.size();
332  for (auto& step : path) {
333  if (step.empty()) {
334  LOGF(FATAL, R"(Found empty group name in path for histogram "%s".)", name);
335  }
336  cumulativeName += step;
337  for (auto& curName : mRegisteredNames) {
338  // there is already a histogram where we want to put a folder or histogram
339  if (cumulativeName == curName) {
340  LOGF(FATAL, R"(Histogram name "%s" is not compatible with existing names.)", name);
341  }
342  // for the full new histogram name we need to check that none of the existing histograms already uses this as a group name
343  if (depth == 1) {
344  if (curName.rfind(cumulativeName, 0) == 0 && curName.size() > cumulativeName.size() && curName.at(cumulativeName.size()) == '/') {
345  LOGF(FATAL, R"(Histogram name "%s" is not compatible with existing names.)", name);
346  }
347  }
348  }
349  --depth;
350  cumulativeName += "/";
351  }
352  mRegisteredNames.push_back(name);
353 }
