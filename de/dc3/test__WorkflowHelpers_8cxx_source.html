28 BOOST_AUTO_TEST_CASE(TestVerifyWorkflow)
29 {
30  using namespace o2::framework;
31  auto checkIncompleteInput = [](WorkflowSpec const& workflow) {
32  // Empty workflows should be invalid.
33  BOOST_CHECK_THROW((void)WorkflowHelpers::verifyWorkflow(workflow), std::runtime_error);
34  };
35 
36  auto checkOk = [](WorkflowSpec const& workflow) {
37  // Empty workflows should be invalid.
38  BOOST_CHECK_NO_THROW((void)WorkflowHelpers::verifyWorkflow(workflow));
39  };
40 
41  auto checkNotOk = [](WorkflowSpec const& workflow) {
42  // Empty workflows should be invalid.
43  BOOST_CHECK_THROW((void)WorkflowHelpers::verifyWorkflow(workflow), std::runtime_error);
44  };
45 
46  // A non fully specified input is an error, given the result is ambiguous.
47  // Completely ambiguous.
48  checkIncompleteInput(WorkflowSpec{{"A", {InputSpec{"", "", ""}}}});
49  // missing origin and description
50  checkIncompleteInput(WorkflowSpec{{"A", {InputSpec{"x", "", ""}}}});
51  // missing description
52  checkIncompleteInput(WorkflowSpec{{"A", {InputSpec{"x", "TST", ""}}}});
53  // This is fine, since by default both subSpec == 0 and
54  // Timeframe are assumed.
55  checkOk(WorkflowSpec{{"A", {InputSpec{"x", "TST", "A"}}}});
56  // Check for duplicate DataProcessorSpecs names
57  checkNotOk(WorkflowSpec{{"A"}, {"A"}});
58 }
