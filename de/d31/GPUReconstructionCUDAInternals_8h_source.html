41 static int GPUFailedMsgAI(const long long int error, const char* file, int line)
42 {
43  // Check for CUDA Error and in the case of an error display the corresponding error string
44  if (error == cudaSuccess) {
45  return (0);
46  }
47  GPUError("CUDA Error: %lld / %s (%s:%d)", error, cudaGetErrorString((cudaError_t)error), file, line);
48  return 1;
49 }

51 static void GPUFailedMsgA(const long long int error, const char* file, int line)
52 {
53  if (GPUFailedMsgAI(error, file, line)) {
54  throw std::runtime_error("CUDA Failure");

56 }
