59 bool Encoder::open(const std::string& name, const std::string& path, const std::string& fileFor)
60 {
61  bool status = false;
62 
63  // register links
64  o2::header::RAWDataHeader rdh;
65  mFileWriter.useRDHVersion(RDHUtils::getVersion<o2::header::RAWDataHeader>());
66  for (int feeid = 0; feeid < 72; feeid++) {
67  // cru=0 --> FLP 0, endpoint 0 --> 18 links -> fees 0-17
68  // cru=1 --> FLP 0, endpoint 1 --> 18 links -> fees 18-35
69  // cru=2 --> FLP 1, endpoint 0 --> 18 links -> fees 36-53
70  // cru=3 --> FLP 1, endpoint 1 --> 18 links -> fees 54-71
71  RDHUtils::setFEEID(rdh, feeid);
72  RDHUtils::setCRUID(rdh, feeid / NLINKSPERCRU);
73  RDHUtils::setLinkID(rdh, feeid % NLINKSPERCRU);
74  RDHUtils::setEndPointID(rdh, RDHUtils::getCRUID(rdh) % 2);
75  // currently storing each CRU in a separate file
76  std::string outFileLink;
77  if (mCrateOn[feeid]) {
78  if (fileFor == "all") { // single file for all links
79  outFileLink = o2::utils::concat_string(path, "/TOF.raw");
80  } else if (fileFor == "cru") {
81  outFileLink = o2::utils::concat_string(path, "/", "TOF_cru", std::to_string(RDHUtils::getCRUID(rdh)), ".raw");
82  } else if (fileFor == "link") {
83  outFileLink = o2::utils::concat_string(path, "/", "TOF_cru", std::to_string(RDHUtils::getCRUID(rdh)), "_link",
84  std::to_string(RDHUtils::getLinkID(rdh)), "_ep", std::to_string(RDHUtils::getEndPointID(rdh)), ".raw");
85  } else {
86  throw std::runtime_error("invalid option provided for file grouping");
87  }
88  mFileWriter.registerLink(rdh, outFileLink);
89  }
90  }
91 
92  std::string inputGRP = o2::base::NameConf::getGRPFileName();
93  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
94  mFileWriter.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::TOF)); // must be set explicitly
95 
96  return status;
97 }

100 {
101  int nbyte = getSize(mStart[icrate], mUnion[icrate]);
102  int cru = icrate / NLINKSPERCRU;
103  if (nbyte) {
104  if (mCrateOn[icrate]) {
105  //printf("flush crate %d -- byte = %d -- orbit = %d, bc = %d\n",icrate, nbyte, mIR.orbit, mIR.bc);
106  mFileWriter.addData(icrate, cru, icrate % NLINKSPERCRU, cru % 2, mIR, gsl::span(mBuffer[icrate], nbyte));
107  }
108  mIntegratedAllBytes += nbyte;
109  }
110  mUnion[icrate] = mStart[icrate];
111  return false;

120 bool Encoder::alloc(long size)

143 void Encoder::encodeTRM(const std::vector<Digit>& summary, Int_t icrate, Int_t itrm, int& istart) // encode one TRM assuming digit vector sorted by electronic index
144 // return next TRM index (-1 if not in the same crate)
145 // start to convert digiti from istart --> then update istart to the starting position of the new TRM
146 {
147  if (mVerbose) {
148  printf("Crate %d: encode TRM %d \n", icrate, itrm);
149  }
150 
151  // TRM HEADER
152  Union_t* trmheader = mUnion[icrate];
153  mUnion[icrate]->trmDataHeader.slotId = itrm;
154  mUnion[icrate]->trmDataHeader.eventWords = 0; // to be filled at the end
155  mUnion[icrate]->trmDataHeader.eventCnt = mEventCounter;
156  mUnion[icrate]->trmDataHeader.emptyBit = 0;
157  mUnion[icrate]->trmDataHeader.dataId = 4;
158  nextWord(icrate);
159 
160  // LOOP OVER CHAINS
161  for (int ichain = 0; ichain < 2; ichain++) {
162  // CHAIN HEADER
163  mUnion[icrate]->trmChainHeader.slotId = itrm;
164  mUnion[icrate]->trmChainHeader.bunchCnt = mIR.bc;
165  mUnion[icrate]->trmChainHeader.mbz = 0;
166  mUnion[icrate]->trmChainHeader.dataId = 2 * ichain;
167  nextWord(icrate);
168 
169  while (istart < summary.size()) { // fill hits
171  int whatChain = summary[istart].getElChainIndex();
172  if (whatChain != ichain) {
173  break;
174  }
175  int whatTRM = summary[istart].getElTRMIndex();
176  if (whatTRM != itrm) {
177  break;
178  }
179  int whatCrate = summary[istart].getElCrateIndex();
180  if (whatCrate != icrate) {
181  break;
182  }
183 
184  int hittimeTDC = (summary[istart].getBC() - mEventCounter * Geo::BC_IN_WINDOW) * 1024 + summary[istart].getTDC(); // time in TDC bin within the TOF WINDOW
185 
186  if (hittimeTDC < 0) {
187  LOG(ERROR) << "Negative hit encoded " << hittimeTDC << ", something went wrong in filling readout window";
188  printf("%llu %d %d\n", (unsigned long long)summary[istart].getBC(), mEventCounter * Geo::BC_IN_WINDOW, summary[istart].getTDC());
189  }
190  // leading time
191  mUnion[icrate]->trmDataHit.time = hittimeTDC;
192  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
193  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
194  mUnion[icrate]->trmDataHit.dataId = 0xa;
195  nextWord(icrate);
196 
197  // trailing time
198  mUnion[icrate]->trmDataHit.time = hittimeTDC + summary[istart].getTOT() * Geo::RATIO_TOT_TDC_BIN;
199  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
200  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
201  mUnion[icrate]->trmDataHit.dataId = 0xc;
202  nextWord(icrate);
203 
204  istart++;
205  }
206 
207  // CHAIN TRAILER
208  mUnion[icrate]->trmChainTrailer.status = 0;
209  mUnion[icrate]->trmChainTrailer.mbz = 0;
210  mUnion[icrate]->trmChainTrailer.eventCnt = mEventCounter;
211  mUnion[icrate]->trmChainTrailer.dataId = 1 + 2 * ichain;
212  nextWord(icrate);
213  }
214 
215  // set TRM data size
216  int neventwords = getSize(trmheader, mUnion[icrate]) / 4 + 1;
217  neventwords -= neventwords / 4 * 2;
218  trmheader->trmDataHeader.eventWords = neventwords;
219 
220  // TRM TRAILER
221  mUnion[icrate]->trmDataTrailer.trailerMark = 3;
222  mUnion[icrate]->trmDataTrailer.eventCRC = 0; // to be implemented
223  mUnion[icrate]->trmDataTrailer.tempValue = 0;
224  mUnion[icrate]->trmDataTrailer.tempAddress = 0;
225  mUnion[icrate]->trmDataTrailer.tempChain = 0;
226  mUnion[icrate]->trmDataTrailer.tempAck = 0;
227  mUnion[icrate]->trmDataTrailer.lutErrorBit = 0;
228  mUnion[icrate]->trmDataTrailer.dataId = 5;
229  nextWord(icrate);
230 }

232 bool Encoder::encode(std::vector<std::vector<o2::tof::Digit>> digitWindow, int tofwindow) // pass a vector of digits in a TOF readout window, tof window is the entry in the vector-of-vector of digits needed to extract bunch id and orbit
233 {
234  if (digitWindow.size() != Geo::NWINDOW_IN_ORBIT) {
235  printf("Something went wrong in encoding: digitWindow=%lu different from %d\n", digitWindow.size(), Geo::NWINDOW_IN_ORBIT);
236  return 999;
237  }
238 
239  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
240  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
241  // caching electronic indexes in digit array
242  for (auto dig = summary.begin(); dig != summary.end(); dig++) {
243  int digitchannel = dig->getChannel();
244  dig->setElectronicIndex(Geo::getECHFromCH(digitchannel));
245  }
246 
247  // sorting by electroni indexes
248  std::sort(summary.begin(), summary.end(),
249  [](Digit a, Digit b) { return a.getElectronicIndex() < b.getElectronicIndex(); });
250  }
251 
252 #ifdef VERBOSE
253  if (mVerbose) {
254  std::cout << "-------- START ENCODE EVENT ----------------------------------------" << std::endl;
255  }
256 #endif
257  auto start = std::chrono::high_resolution_clock::now();
258 
259  mEventCounter = tofwindow; // tof window index
260  mIR.orbit = mEventCounter / Geo::NWINDOW_IN_ORBIT;
261 
262  for (int i = 0; i < 72; i++) {
263  mNextWordStatus[i] = false;
264  }
265 
266  int bcFirstWin;
267 
268  // encode all windows
269  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
270  mEventCounter = tofwindow + iwin; // tof window index
271 
272  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
273 
274  mIR.bc = ((mEventCounter % Geo::NWINDOW_IN_ORBIT) * Geo::BC_IN_ORBIT) / Geo::NWINDOW_IN_ORBIT + mFirstBC; // bunch crossing in the current orbit at the beginning of the window.
275 
276  if (iwin == 0) {
277  bcFirstWin = mIR.bc;
278  }
279 
280  int icurrentdigit = 0;
281  // TOF data header
282  for (int i = 0; i < 72; i++) {
283  mTOFDataHeader[i] = reinterpret_cast<TOFDataHeader_t*>(mUnion[i]);
284  mTOFDataHeader[i]->bytePayload = 0; // event length in byte (to be filled later)
285  mTOFDataHeader[i]->mbz = 0;
286  mTOFDataHeader[i]->dataId = 4;
287  nextWord(i);
288 
289  mUnion[i]->tofOrbit.orbit = mIR.orbit;
290  nextWord(i);
291 
292  mDRMDataHeader[i] = reinterpret_cast<DRMDataHeader_t*>(mUnion[i]);
293  mDRMDataHeader[i]->slotId = 1;
294  mDRMDataHeader[i]->eventWords = 0; // event length in word (to be filled later) --> word = byte/4
295  mDRMDataHeader[i]->drmId = i;
296  mDRMDataHeader[i]->dataId = 4;
297  nextWord(i);
298 
299  mUnion[i]->drmHeadW1.slotId = 1;
300  mUnion[i]->drmHeadW1.partSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
301  mUnion[i]->drmHeadW1.clockStatus = 2;
302  mUnion[i]->drmHeadW1.drmhVersion = 0x12;
303  mUnion[i]->drmHeadW1.drmHSize = 5;
304  mUnion[i]->drmHeadW1.mbza = 0;
305  mUnion[i]->drmHeadW1.mbzb = 0;
306  mUnion[i]->drmHeadW1.dataId = 4;
307  nextWord(i);
308 
309  mUnion[i]->drmHeadW2.slotId = 1;
310  mUnion[i]->drmHeadW2.enaSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
311  mUnion[i]->drmHeadW2.mbz = 0;
312  mUnion[i]->drmHeadW2.faultSlotMask = 0;
313  mUnion[i]->drmHeadW2.readoutTimeOut = 0;
314  mUnion[i]->drmHeadW2.dataId = 4;
315  nextWord(i);
316 
317  mUnion[i]->drmHeadW3.slotId = 1;
318  mUnion[i]->drmHeadW3.gbtBunchCnt = mIR.bc;
319  mUnion[i]->drmHeadW3.locBunchCnt = 0;
320  mUnion[i]->drmHeadW3.dataId = 4;
321  nextWord(i);
322 
323  mUnion[i]->drmHeadW4.slotId = 1;
324  mUnion[i]->drmHeadW4.tempValue = 0;
325  mUnion[i]->drmHeadW4.mbza = 0;
326  mUnion[i]->drmHeadW4.tempAddress = 0;
327  mUnion[i]->drmHeadW4.mbzb = 0;
328  mUnion[i]->drmHeadW4.dataId = 4;
329  nextWord(i);
330 
331  mUnion[i]->drmHeadW5.slotId = 1;
332  mUnion[i]->drmHeadW5.eventCRC = 0;
333  mUnion[i]->drmHeadW5.irq = 0;
334  mUnion[i]->drmHeadW5.mbz = 0;
335  mUnion[i]->drmHeadW5.dataId = 4;
336  nextWord(i);
337 
338  int trm0 = 4 - (i % 2);
339  for (int itrm = trm0; itrm < 13; itrm++) {
340  encodeTRM(summary, i, itrm, icurrentdigit);
341  }
342 
343  mUnion[i]->drmDataTrailer.slotId = 1;
344  mUnion[i]->drmDataTrailer.locEvCnt = mEventCounter;
345  mUnion[i]->drmDataTrailer.mbz = 0;
346  mUnion[i]->drmDataTrailer.dataId = 5;
347  int neventwords = getSize(mDRMDataHeader[i], mUnion[i]) / 4 + 1;
348  neventwords -= neventwords / 4 * 2 + 6;
349  mDRMDataHeader[i]->eventWords = neventwords;
350  nextWord(i);
351  mUnion[i]->data = 0x70000000;
352  nextWord(i);
353 
354  mTOFDataHeader[i]->bytePayload = getSize(mTOFDataHeader[i], mUnion[i]);
355  }
356 
357  // check that all digits were used
358  if (icurrentdigit < summary.size()) {
359  LOG(ERROR) << "Not all digits are been used : only " << icurrentdigit << " of " << summary.size();
360  }
361  }
362 
363  mIR.bc = bcFirstWin;
364 
365  for (int i = 0; i < 72; i++) {
366  flush(i);
367  }
368 
369  mStartRun = false;
370 
371  auto finish = std::chrono::high_resolution_clock::now();
372  std::chrono::duration<double> elapsed = finish - start;
373  mIntegratedTime = elapsed.count();
374 
375  return false;
376 }
