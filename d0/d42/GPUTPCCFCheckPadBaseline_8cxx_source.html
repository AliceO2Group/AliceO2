34  Array2D<PackedCharge> chargeMap(reinterpret_cast<PackedCharge*>(clusterer.mPchargeMap));
35 
36  int basePad = iBlock * PadsPerCacheline;
37  ChargePos basePos = padToChargePos(basePad, clusterer);
38 
39  if (not basePos.valid()) {
40  return;
41  }
42 
43 #ifdef GPUCA_GPUCODE
44  static_assert(TPC_MAX_FRAGMENT_LEN % NumOfCachedTimebins == 0);
45 
46  int totalCharges = 0;
47  int consecCharges = 0;
48  int maxConsecCharges = 0;
49 
50  short localPadId = iThread / NumOfCachedTimebins;
51  short localTimeBin = iThread % NumOfCachedTimebins;
52  bool handlePad = localTimeBin == 0;
53 
54  for (tpccf::TPCFragmentTime t = fragment.firstNonOverlapTimeBin(); t < fragment.lastNonOverlapTimeBin(); t += NumOfCachedTimebins) {
55  ChargePos pos = basePos.delta({localPadId, short(t + localTimeBin)});
56  smem.charges[localPadId][localTimeBin] = (pos.valid()) ? chargeMap[pos].unpack() : 0;
57  GPUbarrier();
58  if (handlePad) {
59  for (int i = 0; i < NumOfCachedTimebins; i++) {
60  Charge q = smem.charges[localPadId][i];
61  totalCharges += (q > 0);
62  consecCharges = (q > 0) ? consecCharges + 1 : 0;
63  maxConsecCharges = CAMath::Max(consecCharges, maxConsecCharges);
64  }
65  }
66  GPUbarrier();
67  }
68 
69  GPUbarrier();
70 
71  if (handlePad) {
72  updatePadBaseline(basePad + localPadId, clusterer, totalCharges, maxConsecCharges);
73  }
74 
75 #else // CPU CODE
76

78

81 
82  UShort8 totalCharges{Vc::Zero};

98

100

102  totalCharges(isCharge)++;
103  consecCharges += 1;
104  consecCharges(not isCharge) = 0;
105  maxConsecCharges = Vc::max(consecCharges, maxConsecCharges);
106  } else {
107  consecCharges = 0;
108  }
109 #else // Vc not available

111  bool isCharge = charge[PadsPerCacheline * localtime + localpad] != 0;
112  if (isCharge) {

120 #endif
121  }

128  }
129 #endif

132 GPUd() ChargePos GPUTPCCFCheckPadBaseline::padToChargePos(int& pad, const GPUTPCClusterFinder& clusterer)
133 {
134  const GPUTPCGeometry& geo = clusterer.Param().tpcGeometry;
135 
136  int padOffset = 0;
137  for (Row r = 0; r < TPC_NUM_OF_ROWS; r++) {
138  int npads = geo.NPads(r);
139  int padInRow = pad - padOffset;
140  if (0 <= padInRow && padInRow < CAMath::nextMultipleOf<PadsPerCacheline, int>(npads)) {
141  int cachelineOffset = padInRow % PadsPerCacheline;
142  pad -= cachelineOffset;
143  return ChargePos{r, Pad(padInRow - cachelineOffset), 0};
144  }
145  padOffset += npads;
146  }
147 
148  return ChargePos{0, 0, INVALID_TIME_BIN};
149 }

151 GPUd() void GPUTPCCFCheckPadBaseline::updatePadBaseline(int pad, const GPUTPCClusterFinder& clusterer, int totalCharges, int consecCharges)
152 {
153  const CfFragment& fragment = clusterer.mPmemory->fragment;
154  int totalChargesBaseline = clusterer.Param().rec.maxTimeBinAboveThresholdIn1000Bin * fragment.lengthWithoutOverlap() / 1000;
155  int consecChargesBaseline = clusterer.Param().rec.maxConsecTimeBinAboveThreshold;
156  bool hasLostBaseline = (totalChargesBaseline > 0 && totalCharges >= totalChargesBaseline) || (consecChargesBaseline > 0 && consecCharges >= consecChargesBaseline);
157  clusterer.mPpadHasLostBaseline[pad] |= hasLostBaseline;
158 }
