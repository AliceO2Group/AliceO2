65 WorkflowSpec defineDataProcessing(ConfigContext const& config)
66 {
67 
68  using namespace o2::tpc;
69 
70  // set up configuration
71  o2::conf::ConfigurableParam::updateFromFile(config.options().get<std::string>("configFile"));
72  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
73  o2::conf::ConfigurableParam::writeINI("o2tpccalibration_configuration.ini");
74 
75  const std::string inputSpec = config.options().get<std::string>("input-spec");
76  const std::string outputFile = config.options().get<std::string>("outputFile");
77 
78  const auto tpcSectors = o2::RangeTokenizer::tokenize<int>(config.options().get<std::string>("sectors"));
79  const auto nSectors = (int)tpcSectors.size();
80  const auto nLanes = std::min(config.options().get<int>("lanes"), nSectors);
81  const auto sectorsPerLane = nSectors / nLanes + ((nSectors % nLanes) != 0);
82 
83  WorkflowSpec workflow;
84 
85  if (nLanes <= 0) {
86  return workflow;
87  }
88 
89  for (int ilane = 0; ilane < nLanes; ++ilane) {
90  auto first = tpcSectors.begin() + ilane * sectorsPerLane;
91  if (first >= tpcSectors.end()) {
92  break;
93  }
94  auto last = std::min(tpcSectors.end(), first + sectorsPerLane);
95  std::vector<int> range(first, last);
96  workflow.emplace_back(getKryptonClustererSpec(inputSpec, ilane, range));
97  }
98 
99  std::vector<int> laneConfiguration = tpcSectors; // Currently just a copy of the tpcSectors, why?
100 
102  //
103  // generation of processor specs for various types of outputs
104  // based on generic RootTreeWriter and MakeRootTreeWriterSpec generator
105  //
106  // -------------------------------------------------------------------------------------------
107  // the callbacks for the RootTreeWriter
108  //
109  // The generic writer needs a way to associate incoming data with the individual branches for
110  // the TPC sectors. The sector number is transmitted as part of the sector header, the callback
111  // finds the corresponding index in the vector of configured sectors
112  auto getIndex = [tpcSectors](o2::framework::DataRef const& ref) {
113  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
114  if (!tpcSectorHeader) {
115  throw std::runtime_error("TPC sector header missing in header stack");
116  }
117  if (tpcSectorHeader->sector() < 0) {
118  // special data sets, don't write
119  return ~(size_t)0;
120  }
121  size_t index = 0;
122  for (auto const& sector : tpcSectors) {
123  if (sector == tpcSectorHeader->sector()) {
124  return index;
125  }
126  ++index;
127  }
128  throw std::runtime_error("sector " + std::to_string(tpcSectorHeader->sector()) + " not configured for writing");
129  };
130  auto getName = [tpcSectors](std::string base, size_t index) {
131  return base + "_" + std::to_string(tpcSectors.at(index));
132  };
133 
134  auto makeWriterSpec = [tpcSectors, laneConfiguration, getIndex, getName](const char* processName,
135  const char* defaultFileName,
136  const char* defaultTreeName,
137  auto&& databranch,
138  bool singleBranch = false) {
139  if (tpcSectors.size() == 0) {
140  throw std::invalid_argument(std::string("writer process configuration needs list of TPC sectors"));
141  }
142 
143  auto amendInput = [tpcSectors, laneConfiguration](InputSpec& input, size_t index) {
144  input.binding += std::to_string(laneConfiguration[index]);
145  DataSpecUtils::updateMatchingSubspec(input, laneConfiguration[index]);
146  };
147  auto amendBranchDef = [laneConfiguration, amendInput, tpcSectors, getIndex, getName, singleBranch](auto&& def, bool enableMC = true) {
148  if (!singleBranch) {
149  def.keys = mergeInputs(def.keys, laneConfiguration.size(), amendInput);
150  // the branch is disabled if set to 0
151  def.nofBranches = enableMC ? tpcSectors.size() : 0;
152  def.getIndex = getIndex;
153  def.getName = getName;
154  } else {
155  // instead of the separate sector branches only one is going to be written
156  def.nofBranches = enableMC ? 1 : 0;
157  }
158  return std::move(def);
159  };
160 
161  return std::move(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
162  std::move(amendBranchDef(databranch)))());
163  };
164 
166  //
167  // a writer process for digits
168  //
169  // selected by output type 'difits'
170  using KrClusterOutputType = std::vector<o2::tpc::KrCluster>;
171  workflow.push_back(makeWriterSpec("tpc-krcluster-writer",
172  outputFile.data(),
173  "Clusters",
174  BranchDefinition<KrClusterOutputType>{InputSpec{"data", "TPC", "KRCLUSTERS", 0},
175  "TPCBoxCluster",
176  "boxcluster-branch-name"}));
177 
178  return workflow;
179 }
