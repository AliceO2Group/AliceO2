177 template <class Mapping>
178 GBTLink::CollectedDataStatus GBTLink::collectROFCableData(const Mapping& chmap)
179 {
180  int nw = 0;
181  status = None;
182  auto* currRawPiece = rawData.currentPiece();
183  GBTLink::ErrorType errRes = GBTLink::NoError;
184  while (currRawPiece) { // we may loop over multiple CRU page
185  if (dataOffset >= currRawPiece->size) {
186  dataOffset = 0; // start of the RDH
187  if (!(currRawPiece = rawData.nextPiece())) { // fetch next CRU page
188  break; // Data chunk (TF?) is done
189  }
190  }
191  if (!dataOffset) { // here we always start with the RDH
192  const auto* rdh = reinterpret_cast<const RDH*>(&currRawPiece->data[dataOffset]);
193  if (verbosity >= VerboseHeaders) {
194  RDHUtils::printRDH(rdh);
195  }
196 
197  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsRDH(*rdh)); // make sure we are dealing with RDH
198  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsRDHStop(*rdh)); // if new HB starts, the lastRDH must have stop
199  // GBTLINK_DECODE_ERRORCHECK(checkErrorsRDHStopPageEmpty(*rdh)); // end of HBF should be an empty page with stop
200  lastRDH = rdh;
201  statistics.nPackets++;
202 
203  dataOffset += sizeof(RDH);
204  auto psz = RDHUtils::getMemorySize(*rdh);
205  if (psz == sizeof(RDH)) {
206  continue; // filter out empty page
207  }
208  if (format == NewFormat && RDHUtils::getStop(*rdh)) { // only diagnostic word can be present after the stop
209  auto gbtDiag = reinterpret_cast<const GBTDiagnostic*>(&currRawPiece->data[dataOffset]);
210  if (verbosity >= VerboseHeaders) {
211  printDiagnostic(gbtDiag);
212  }
213  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsDiagnosticWord(gbtDiag));
214  dataOffset += RDHUtils::getOffsetToNext(*rdh) - sizeof(RDH);
215  continue;
216  }
217 
218  // data must start with the GBTHeader
219  auto gbtH = reinterpret_cast<const GBTDataHeader*>(&currRawPiece->data[dataOffset]); // process GBT header
220  dataOffset += GBTPaddedWordLength;
221  if (verbosity >= VerboseHeaders) {
222  printHeader(gbtH);
223  }
224  if (format == OldFormat) {
225  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsHeaderWord(reinterpret_cast<const GBTDataHeaderL*>(gbtH)));
226  lanesActive = reinterpret_cast<const GBTDataHeaderL*>(gbtH)->activeLanesL; // TODO do we need to update this for every page?
227  } else {
228  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsHeaderWord(gbtH));
229  lanesActive = gbtH->activeLanes; // TODO do we need to update this for every page?
230  }
231 
232  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsActiveLanes(chmap.getCablesOnRUType(ruPtr->ruInfo->ruType)));
233  if (format == OldFormat && reinterpret_cast<const GBTDataHeaderL*>(gbtH)->packetIdx == 0) { // reset flags in case of 1st page of new ROF (old format: judge by RDH)
234  lanesStop = 0;
235  lanesWithData = 0;
236  }
237 
238  continue;
239  }
240 
241  ruPtr->nCables = ruPtr->ruInfo->nCables; // RSTODO is this needed? TOREMOVE
242 
243  // then we expect GBT trigger word (unless we work with old format)
244  const GBTTrigger* gbtTrg = nullptr;
245  if (format == NewFormat) {
246  gbtTrg = reinterpret_cast<const GBTTrigger*>(&currRawPiece->data[dataOffset]); // process GBT trigger
247  dataOffset += GBTPaddedWordLength;
248  if (verbosity >= VerboseHeaders) {
249  printTrigger(gbtTrg);
250  }
251  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsTriggerWord(gbtTrg));
252  statistics.nTriggers++;
253  if (gbtTrg->noData) { // emtpy trigger
254  return status;
255  }
256  lanesStop = 0;
257  lanesWithData = 0;
258  }
259  if (format == NewFormat) { // at the moment just check if calibration word is there
260  auto gbtC = reinterpret_cast<const o2::itsmft::GBTCalibration*>(&currRawPiece->data[dataOffset]);
261  if (gbtC->isCalibrationWord()) {
262  if (verbosity >= VerboseHeaders) {
263  printCalibrationWord(gbtC);
264  }
265  dataOffset += GBTPaddedWordLength;
266  }
267  }
268  auto gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
269 
270  while (!gbtD->isDataTrailer()) { // start reading real payload
271  nw++;
272  if (verbosity >= VerboseData) {
273  gbtD->printX();
274  }
275  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsGBTDataID(gbtD));
276  if (errRes != GBTLink::Skip) {
277  int cableHW = gbtD->getCableID(), cableSW = chmap.cableHW2SW(ruPtr->ruInfo->ruType, cableHW);
278  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsGBTData(chmap.cableHW2Pos(ruPtr->ruInfo->ruType, cableHW)));
279  ruPtr->cableData[cableSW].add(gbtD->getW8(), 9);
280  ruPtr->cableHWID[cableSW] = cableHW;
281  ruPtr->cableLinkID[cableSW] = idInRU;
282  ruPtr->cableLinkPtr[cableSW] = this;
283  }
284  dataOffset += GBTPaddedWordLength;
285  gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
286  } // we are at the trailer, packet is over, check if there are more data on the next page
287 
288  auto gbtT = reinterpret_cast<const o2::itsmft::GBTDataTrailer*>(&currRawPiece->data[dataOffset]); // process GBT trailer
289  dataOffset += GBTPaddedWordLength;
290  if (verbosity >= VerboseHeaders) {
291  printTrailer(gbtT);
292  }
293  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsTrailerWord(gbtT));
294  // we finished the GBT page, but there might be continuation on the next CRU page
295  if (!gbtT->packetDone) {
296  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsPacketDoneMissing(gbtT, dataOffset < currRawPiece->size));
297  continue; // keep reading next CRU page
298  }
299  if (format == OldFormat) {
300  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsLanesStops());
301  }
302  // accumulate packet states
303  statistics.packetStates[gbtT->getPacketState()]++;
304  // before quitting, store the trigger and IR
305  if (format == NewFormat) {
306  ir.bc = gbtTrg->bc;
307  ir.orbit = gbtTrg->orbit;
308  trigger = gbtTrg->triggerType;
309  } else {
310  ir = RDHUtils::getTriggerIR(*lastRDH);
311  trigger = RDHUtils::getTriggerType(*lastRDH);
312  }
313 
314  return (status = DataSeen);
315  }
316 
317  return (status = StoppedOnEndOfData);
318 }
