51 int base64_encode(char* dest, int size, unsigned char* src, int slen)
52 {
53  int dlen, i, j;
54  uint32_t a, b, c, triple;
55 
56  dlen = base64_encsize(slen);
57 
58  // Sanity checks
59  if (src == nullptr || dest == nullptr) {
60  return -1;
61  }
62  if (dlen + 1 > size) {
63  return -1;
64  }
65  if (slen == 0) {
66  if (size > 0) {
67  dest[0] = 0;
68  return 0;
69  }
70  return -1;
71  }
72 
73  for (i = 0, j = 0; i < slen;) {
74  a = src[i++];
75 
76  // b and c may be off limit
77  b = i < slen ? src[i++] : 0;
78  c = i < slen ? src[i++] : 0;
79 
80  triple = (a << 16) + (b << 8) + c;
81 
82  dest[j++] = encoder[(triple >> 18) & 0x3F];
83  dest[j++] = encoder[(triple >> 12) & 0x3F];
84  dest[j++] = encoder[(triple >> 6) & 0x3F];
85  dest[j++] = encoder[triple & 0x3F];
86  }
87 
88  // Pad zeroes at the end
89  switch (slen % 3) {
90  case 1:
91  dest[j - 2] = '=';
92  case 2:
93  dest[j - 1] = '=';
94  }
95 
96  // Always add \0
97  dest[j] = 0;
98 
99  return dlen;
100 }

143 int base64_decode(unsigned char* dest, int size, char* src)
144 {
145  int slen, dlen, padlen, i, j;
146  uint32_t a, b, c, d, triple;
147 
148  // Initialize decoder
149  base64_init_decoder();
150 
151  // Sanity checks
152  if (src == nullptr || dest == nullptr) {
153  return -1;
154  }
155 
156  slen = strlen(src);
157  if (slen == 0) {
158  if (size > 0) {
159  dest[0] = 0;
160  return 0;
161  }
162  return -1;
163  }
164 
165  // Remove trailing pad characters.
166  for (padlen = 0; src[slen - padlen - 1] == '='; ++padlen) {
167  }
168  if (padlen > 2) {
169  slen -= padlen - 2;
170  }
171  if (slen % 4) {
172  return -1;
173  }
174 
175  dlen = base64_decsize(src);
176 
177  // Check buffer size
178  if (dlen + 1 > size) {
179  return -1;
180  }
181 
182  for (i = 0, j = 0; i < slen;) {
183  a = decoder[(int)src[i++]];
184  b = decoder[(int)src[i++]];
185  c = decoder[(int)src[i++]];
186  d = decoder[(int)src[i++]];
187 
188  // Sextet 3 and 4 may be zero at the end
189  if (i == slen) {
190  if (src[slen - 1] == '=') {
191  d = 0;
192  if (src[slen - 2] == '=') {
193  c = 0;
194  }
195  }
196  }
197 
198  // Non-Base64 char
199  if (a == -1 || b == -1 || c == -1 || d == -1) {
200  return -1;
201  }
202 
203  triple = (a << 18) + (b << 12) + (c << 6) + d;
204 
205  dest[j++] = (triple >> 16) & 0xFF;
206  dest[j++] = (triple >> 8) & 0xFF;
207  dest[j++] = triple & 0xFF;
208  }
209 
210  // Always add \0
211  dest[j] = 0;
212 
213  return dlen;
214 }
