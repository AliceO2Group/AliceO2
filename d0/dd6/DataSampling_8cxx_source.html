56 void DataSampling::DoGenerateInfrastructure(Dispatcher& dispatcher, WorkflowSpec& workflow, const boost::property_tree::ptree& policiesTree, size_t threads)
57 {
58  LOG(DEBUG) << "Generating Data Sampling infrastructure...";
59 
60  for (auto&& policyConfig : policiesTree) {
61 
62  std::unique_ptr<DataSamplingPolicy> policy;
63 
64  // We don't want the Dispatcher to exit due to one faulty Policy
65  try {
66  dispatcher.registerPolicy(std::make_unique<DataSamplingPolicy>(DataSamplingPolicy::fromConfiguration(policyConfig.second)));
67  } catch (const std::exception& ex) {
68  LOG(WARN) << "Could not load the Data Sampling Policy '"
69  << policyConfig.second.get_optional<std::string>("id").value_or("") << "', because: " << ex.what();
70  continue;
71  } catch (...) {
72  LOG(WARN) << "Could not load the Data Sampling Policy '"
73  << policyConfig.second.get_optional<std::string>("id").value_or("") << "'";
74  continue;
75  }
76  }
77 
78  if (dispatcher.numberOfPolicies() > 0) {
79  DataProcessorSpec spec;
80  spec.name = dispatcher.getName();
81  spec.inputs = dispatcher.getInputSpecs();
82  spec.outputs = dispatcher.getOutputSpecs();
83  spec.maxInputTimeslices = threads;
84  spec.labels = {{"DataSampling"}, {"Dispatcher"}};
85  spec.options = dispatcher.getOptions();
86  spec.algorithm = adaptFromTask<Dispatcher>(std::move(dispatcher));
87 
88  workflow.emplace_back(std::move(spec));
89  } else {
90  LOG(DEBUG) << "No input to this dispatcher, it won't be added to the workflow.";
91  }
92 }

112 std::vector<InputSpec> DataSampling::InputSpecsForPolicy(ConfigurationInterface* const config, const std::string& policyName)
113 {
114  std::vector<InputSpec> inputs;
115  auto policiesTree = config->getRecursive("dataSamplingPolicies");
116 
117  for (auto&& policyConfig : policiesTree) {
118  if (policyConfig.second.get<std::string>("id") == policyName) {
119  auto policy = DataSamplingPolicy::fromConfiguration(policyConfig.second);
120  for (const auto& path : policy.getPathMap()) {
121  InputSpec input = DataSpecUtils::matchingInput(path.second);
122  inputs.push_back(input);
123  }
124  break;
125  }
126  }
127  return inputs;
128 }
