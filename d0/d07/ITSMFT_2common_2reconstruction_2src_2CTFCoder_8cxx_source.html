39 void CTFCoder::compress(CompressedClusters& cc,
40  const gsl::span<const ROFRecord>& rofRecVec,
41  const gsl::span<const CompClusterExt>& cclusVec,
42  const gsl::span<const unsigned char>& pattVec)
43 {
44  // store in the header the orbit of 1st ROF
45  cc.clear();
46  if (!rofRecVec.size()) {
47  return;
48  }
49  const auto& rofRec0 = rofRecVec[0];
50  cc.header.nROFs = rofRecVec.size();
51  cc.header.firstOrbit = rofRec0.getBCData().orbit;
52  cc.header.firstBC = rofRec0.getBCData().bc;
53  cc.header.nPatternBytes = pattVec.size();
54  cc.header.nClusters = cclusVec.size();
55  cc.header.nChips = 0; // this is the version with chipInc stored once per new chip
56 
57  cc.firstChipROF.resize(cc.header.nROFs);
58  cc.bcIncROF.resize(cc.header.nROFs);
59  cc.orbitIncROF.resize(cc.header.nROFs);
60  cc.nclusROF.resize(cc.header.nROFs);
61  //
62  cc.row.resize(cc.header.nClusters);
63  cc.colInc.resize(cc.header.nClusters);
64  // cc.chipInc.resize(cc.header.nClusters); // this is the version with chipInc stored for every pixel
65  cc.chipInc.reserve(1000); // this is the version with chipInc stored once per new chip
66  cc.chipMul.reserve(1000); // this is the version with chipInc stored once per new chip
67  cc.pattID.resize(cc.header.nClusters);
68  cc.pattMap.resize(cc.header.nPatternBytes);
69 
70  uint16_t prevBC = cc.header.firstBC;
71  uint32_t prevOrbit = cc.header.firstOrbit;
72 
73  for (uint32_t irof = 0; irof < cc.header.nROFs; irof++) {
74  const auto& rofRec = rofRecVec[irof];
75  const auto& intRec = rofRec.getBCData();
76  if (intRec.orbit == prevOrbit) {
77  cc.orbitIncROF[irof] = 0;
78  cc.bcIncROF[irof] = intRec.bc - prevBC; // store increment of BC if in the same orbit
79  } else {
80  cc.orbitIncROF[irof] = intRec.orbit - prevOrbit;
81  cc.bcIncROF[irof] = intRec.bc; // otherwise, store absolute bc
82  prevOrbit = intRec.orbit;
83  }
84  prevBC = intRec.bc;
85  auto ncl = rofRec.getNEntries();
86  cc.nclusROF[irof] = ncl;
87  if (!ncl) { // no hits data for this ROF
88  cc.firstChipROF[irof] = 0;
89  continue;
90  }
91  cc.firstChipROF[irof] = cclusVec[rofRec.getFirstEntry()].getChipID();
92  int icl = rofRec.getFirstEntry(), iclMax = icl + ncl;
93 
94  uint16_t prevChip = cc.firstChipROF[irof], prevCol = 0;
95  if (icl != iclMax) { // there still clusters to store
96  cc.chipMul.push_back(0);
97  cc.chipInc.push_back(0);
98  }
99  for (; icl < iclMax; icl++) { // clusters within a chip are stored in increasing column number
100  const auto& cl = cclusVec[icl];
101  cc.row[icl] = cl.getRow(); // row is practically random, store it as it is
102  cc.pattID[icl] = cl.getPatternID();
103  if (cl.getChipID() == prevChip) { // for the same chip store column increment
104  // cc.chipInc[icl] = 0; // this is the version with chipInc stored for every pixel
105  cc.chipMul.back()++; // this is the version with chipInc stored once per new chip
106  cc.colInc[icl] = cl.getCol() - prevCol;
107  prevCol = cl.getCol();
108  } else { // for new chips store chipID increment and abs. column
109  // cc.chipInc[icl] = cl.getChipID() - prevChip; // this is the version with chipInc stored for every pixel
110  cc.chipInc.push_back(cl.getChipID() - prevChip); // this is the version with chipInc stored once per new chip
111  cc.chipMul.push_back(1); // this is the version with chipInc stored once per new chip
112  prevCol = cc.colInc[icl] = cl.getCol();
113  prevChip = cl.getChipID();
114  }
115  }
116  }
117  cc.header.nChips = cc.chipMul.size();
118  // store explicit patters as they are
119  memcpy(cc.pattMap.data(), pattVec.data(), cc.header.nPatternBytes); // RSTODO: do we need this?
120 }

123 void CTFCoder::createCoders(const std::string& dictPath, o2::ctf::CTFCoderBase::OpType op)
124 {
125  bool mayFail = true; // RS FIXME if the dictionary file is not there, do not produce exception
126  auto buff = readDictionaryFromFile<CTF>(dictPath, mayFail);
127  if (!buff.size()) {
128  if (mayFail) {
129  return;
130  }
131  throw std::runtime_error("Failed to create CTF dictionaty");
132  }
133  const auto* ctf = CTF::get(buff.data());
134 
135  auto getFreq = [ctf](CTF::Slots slot) -> o2::rans::FrequencyTable {
136  o2::rans::FrequencyTable ft;
137  auto bl = ctf->getBlock(slot);
138  auto md = ctf->getMetadata(slot);
139  ft.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min, md.max);
140  return std::move(ft);
141  };
142  auto getProbBits = [ctf](CTF::Slots slot) -> int {
143  return ctf->getMetadata(slot).probabilityBits;
144  };
145 
146  CompressedClusters cc; // just to get member types
147 #define MAKECODER(part, slot) createCoder<decltype(part)::value_type>(op, getFreq(slot), getProbBits(slot), int(slot))
148  // clang-format off
149  MAKECODER(cc.firstChipROF, CTF::BLCfirstChipROF);
150  MAKECODER(cc.bcIncROF, CTF::BLCbcIncROF );
151  MAKECODER(cc.orbitIncROF, CTF::BLCorbitIncROF );
152  MAKECODER(cc.nclusROF, CTF::BLCnclusROF );
153  //
154  MAKECODER(cc.chipInc, CTF::BLCchipInc );
155  MAKECODER(cc.chipMul, CTF::BLCchipMul );
156  MAKECODER(cc.row, CTF::BLCrow );
157  MAKECODER(cc.colInc, CTF::BLCcolInc );
158  MAKECODER(cc.pattID, CTF::BLCpattID );
159  MAKECODER(cc.pattMap, CTF::BLCpattMap );
160  // clang-format on
161 }

164 size_t CTFCoder::estimateCompressedSize(const CompressedClusters& cc)
165 {
166  size_t sz = 0;
167  // clang-format off
168  // RS FIXME this is very crude estimate, instead, an empirical values should be used
169 #define VTP(vec) typename std::remove_reference<decltype(vec)>::type::value_type
170 #define ESTSIZE(vec, slot) mCoders[int(slot)] ? \
171  rans::calculateMaxBufferSize(vec.size(), reinterpret_cast<const o2::rans::LiteralEncoder64<VTP(vec)>*>(mCoders[int(slot)].get())->getAlphabetRangeBits(), sizeof(VTP(vec)) ) : vec.size()*sizeof(VTP(vec))
172  sz += ESTSIZE(cc.firstChipROF, CTF::BLCfirstChipROF);
173  sz += ESTSIZE(cc.bcIncROF, CTF::BLCbcIncROF );
174  sz += ESTSIZE(cc.orbitIncROF, CTF::BLCorbitIncROF );
175  sz += ESTSIZE(cc.nclusROF, CTF::BLCnclusROF );
176  //
177  sz += ESTSIZE(cc.chipInc, CTF::BLCchipInc );
178  sz += ESTSIZE(cc.chipMul, CTF::BLCchipMul );
179  sz += ESTSIZE(cc.row, CTF::BLCrow );
180  sz += ESTSIZE(cc.colInc, CTF::BLCcolInc );
181  sz += ESTSIZE(cc.pattID, CTF::BLCpattID );
182  sz += ESTSIZE(cc.pattMap, CTF::BLCpattMap );
183 
184  // clang-format on
185  sz *= 2. / 3; // if needed, will be autoexpanded
186  LOG(INFO) << "Estimated output size is " << sz << " bytes";
187  return sz;
188 }
