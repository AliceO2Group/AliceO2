34 bool GBTRawDataChecker::checkLocalBoardSize(const LocalBoardRO& board)
35 {
37 
38  // This test only make sense when we have a self-trigger,
39  // since in this case we expect to have a variable number of non-zero pattern
40  // as indicated by the corresponding word.
41  for (int ich = 0; ich < 4; ++ich) {
42  bool isExpectedNull = (((board.firedChambers >> ich) & 0x1) == 0);
43  bool isNull = (board.patternsBP[ich] == 0 && board.patternsNBP[ich] == 0);
44  if (isExpectedNull != isNull) {
45  std::stringstream ss;
46  ss << "wrong size for local board:\n";
47  ss << board << "\n";
48  mEventDebugMsg += ss.str();
49  return false;
50  }
51  }
52  return true;
53 }

66 bool GBTRawDataChecker::checkConsistency(const LocalBoardRO& board)
67 {
69 
70  bool isSoxOrReset = board.triggerWord & (raw::sSOX | raw::sEOX | raw::sRESET);
71  bool isCalib = raw::isCalibration(board.triggerWord);
72  bool isPhys = board.triggerWord & raw::sPHY;
73 
74  if (isPhys) {
75  if (isCalib) {
76  mEventDebugMsg += "inconsistent trigger: calibration and physics trigger cannot be fired together\n";
77  return false;
78  }
79  if (raw::isLoc(board.statusWord)) {
80  if (board.firedChambers) {
81  mEventDebugMsg += "inconsistent trigger: fired chambers should be 0\n";
82  return false;
83  }
84  }
85  }
86  if (isSoxOrReset && (isCalib || isPhys)) {
87  mEventDebugMsg += "inconsistent trigger: cannot be SOX and calibration\n";
88  return false;
89  }
90 
91  return true;
92 }

108 bool GBTRawDataChecker::checkMasks(const std::vector<LocalBoardRO>& locs)
109 {
111  for (auto loc : locs) {
112  // The board patterns coincide with the masks ("overwritten" mode)
113  if (loc.statusWord & raw::sOVERWRITTEN) {
114  auto maskItem = mMasks.find(loc.boardId);
115  for (int ich = 0; ich < 4; ++ich) {
116  uint16_t maskBP = 0;
117  uint16_t maskNBP = 0;
118  if (maskItem != mMasks.end()) {
119  maskBP = maskItem->second.patternsBP[ich];
120  maskNBP = maskItem->second.patternsNBP[ich];
121  }
122  if (maskBP != loc.patternsBP[ich] || maskNBP != loc.patternsNBP[ich]) {
123  std::stringstream ss;
124  ss << "Pattern is not compatible with mask for:\n";
125  ss << loc << "\n";
126  mEventDebugMsg += ss.str();
127  return false;
128  }
129  }
130  }
131  }
132  return true;

135 bool GBTRawDataChecker::checkRegLocConsistency(const std::vector<LocalBoardRO>& regs, const std::vector<LocalBoardRO>& locs)
136 {
138  uint8_t regFired{0};
139  for (auto& reg : regs) {
140  uint8_t ireg = crateparams::getLocId(reg.boardId) % 2;
141  auto busyItem = mBusyFlagSelfTrig.find(8 + ireg);
142  if (reg.triggerWord == 0) {
143  // Self-triggered event: check the decision
144  regFired |= (reg.firedChambers << (4 * ireg));
145  } else {
146  // Triggered event: all active boards must answer
147  regFired |= (mCrateMask & (0xF << (4 * ireg)));
148  }
149  }
150  uint8_t locFired{0}, locBusy{0};
151  for (auto& loc : locs) {
152  auto linkId = getElinkId(loc);
153  uint8_t mask = (1 << linkId);
154  if (loc.triggerWord == 0) {
155  // Self-triggered event: check the decision
156  if (loc.firedChambers) {
157  locFired |= mask;
158  }
159  } else {
160  // Triggered event: all active boards must answer
161  locFired |= mask;
162  }
163  }
164 
165  // The XOR returns 1 in case of a difference
166  uint8_t problems = (regFired ^ locFired);
167 
168  if (problems) {
169  // It can be that a busy signal was raised by one of the board in previous events
170  // If the board is still busy it will not answer.
171  uint8_t busy{0};
172  for (uint8_t iboard = 0; iboard < crateparams::sNELinksPerGBT; ++iboard) {
173  auto busyItem = mBusyFlagSelfTrig.find(iboard);
174  if (busyItem != mBusyFlagSelfTrig.end() && busyItem->second) {
175  busy |= (iboard < crateparams::sMaxNBoardsInLink) ? (1 << iboard) : (0xF << (4 * (iboard % 2)));
176  }
177  }
178 
179  if (problems & ~busy) {
180  std::stringstream ss;
181  ss << fmt::format("loc-reg inconsistency: fired locals ({:08b}) != expected from reg ({:08b});\n", locFired, regFired);
182  ss << printBoards(regs);
183  ss << printBoards(locs);
184  mEventDebugMsg += ss.str();
185  return false;
186  }
187  }
188  return true;
189 }

202 {
204  mEventDebugMsg.clear();
205  if (!checkRegLocConsistency(regs, locs)) {
206  return false;
207  }
208 
209  if (!checkConsistency(regs) || !checkConsistency(locs)) {
210  return false;
211  }
212 
213  if (!isTriggered) {
214  if (!checkLocalBoardSize(locs)) {
215  return false;
216  }
217 
218  if (!checkMasks(locs)) {
219  return false;
220  }
221  }
222 
223  return true;

235 void GBTRawDataChecker::clearChecked(bool isTriggered, bool clearTrigEvents)
236 {
238 
239  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
240  auto& lastIndexes = isTriggered ? mLastIndexTrig : mLastIndexSelfTrig;
241  // Create a new board map with the checked events stripped
242  std::unordered_map<uint8_t, std::vector<BoardInfo>> newBoards{};
243  for (auto& lastIdxItem : lastIndexes) {
244  auto firstIdx = lastIdxItem.second + 1;
245  auto& boardVec = boards[lastIdxItem.first];
246  if (firstIdx < boardVec.size()) {
247  auto& newVec = newBoards[lastIdxItem.first];
248  newVec.insert(newVec.end(), boardVec.begin() + firstIdx, boardVec.end());
249  }
250  }
251  boards.swap(newBoards);
252 
253  if (clearTrigEvents) {
254  // Clears the map with the processed triggers
255  auto& lastCompleteTrigIR = isTriggered ? mLastCompleteIRTrig : mLastCompleteIRSelfTrig;
256  auto low = mTrigEvents.begin();
257  auto up = mTrigEvents.upper_bound(lastCompleteTrigIR);
258  mTrigEvents.erase(low, up);
259  }
260 }

262 bool GBTRawDataChecker::isCompleteSelfTrigEvent(const o2::InteractionRecord& ir) const
263 {
265 
266  // The regional board information in self-triggered events is delayed
267  // compared to triggered events.
268  // So, we expect information from a previous orbit after having received an orbit trigger.
269  // Let us check that we have all boards with the same orbit
270 
271  bool isIncluded = false;
272  for (uint8_t ireg = 8; ireg < 10; ++ireg) {
273  auto item = mBoardsSelfTrig.find(ireg);
274  if (item != mBoardsSelfTrig.end()) {
275  if (item->second.back().interactionRecord.orbit == ir.orbit) {
276  return false;
277  }
278  if (item->second.front().interactionRecord.orbit <= ir.orbit) {
279  isIncluded = true;
280  }
281  }
282  }
283  return isIncluded;
284 }

286 unsigned int GBTRawDataChecker::getLastCompleteTrigEvent()
287 {
291 
292  // The information for an event comes at different times for different boards,
293  // depending on the length of the self-triggered event for that board.
294  // So, before testing the consistency of the event,
295  // we must wait to have received the information from all boards.
296  // This can be checked in triggered events, since all boards should be present.
297 
298  unsigned int completeMask = 0;
299 
300  // Check if we have a triggered event with the information from all active boards
301  mLastCompleteIRSelfTrig = o2::InteractionRecord();
302  uint16_t fullMask = (3 << 8) | mCrateMask;
303  auto trigEventIt = mTrigEvents.rbegin();
304  auto end = mTrigEvents.rend();
305  for (; trigEventIt != end; ++trigEventIt) {
306  if ((trigEventIt->second & fullMask) == fullMask) {
307  // The trigger events contain the unprocessed events for both triggered and self-triggered events
308  // These might not be synchronized (typically the latest complete self-triggered events lie behind)
309  // If the latest IR in memory is more recent than the current complete event found,
310  // then it means that we need to wait for more HBs.
311  if (mLastCompleteIRTrig.isDummy() || mLastCompleteIRTrig < trigEventIt->first) {
312  completeMask |= 1;
313  mLastCompleteIRTrig = trigEventIt->first;
314  }
315  auto trIt = trigEventIt;
316  while (trIt != end) {
317  if (isCompleteSelfTrigEvent(trIt->first)) {
318  completeMask |= (1 << 1);
319  mLastCompleteIRSelfTrig = trIt->first;
320  break;
321  }
322  ++trIt;
323  }
324 
325  return completeMask;
326  }
327  }
328 
329  return completeMask;
330 }

332 bool GBTRawDataChecker::runCheckEvents(unsigned int completeMask)
333 {
335 
336  bool isOk = true;
337 
338  if (completeMask & 0x1) {
339  sortEvents(true);
340  isOk &= checkEvents(true);
341  clearChecked(true, mBoardsSelfTrig.empty());
342  }
343 
344  if (completeMask & 0x2) {
345  sortEvents(false);
346  isOk &= checkEvents(false);
347  clearChecked(false, true);
348  }
349 
350  return isOk;
351 }

353 void GBTRawDataChecker::sortEvents(bool isTriggered)
354 {
356  auto& orderedIndexes = isTriggered ? mOrderedIndexesTrig : mOrderedIndexesSelfTrig;
357  auto& lastIndexes = isTriggered ? mLastIndexTrig : mLastIndexSelfTrig;
358  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
359  auto& lastCompleteTrigEventIR = isTriggered ? mLastCompleteIRTrig : mLastCompleteIRSelfTrig;
360  orderedIndexes.clear();
361  lastIndexes.clear();
362  for (auto& boardItem : boards) {
363  long int lastIdx = -1;
364  for (auto boardIt = boardItem.second.begin(), end = boardItem.second.end(); boardIt != end; ++boardIt) {
365  if (boardIt->interactionRecord > lastCompleteTrigEventIR) {
366  break;
367  }
368  lastIdx = std::distance(boardItem.second.begin(), boardIt);
369  orderedIndexes[boardIt->interactionRecord].emplace_back(boardItem.first, lastIdx);
370  }
371  lastIndexes[boardItem.first] = lastIdx;
372  }
373 }

375 bool GBTRawDataChecker::checkEvents(bool isTriggered)
376 {
378  bool isOk = true;
379  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
380  auto& orderedIndexes = isTriggered ? mOrderedIndexesTrig : mOrderedIndexesSelfTrig;
381  auto& busyFlag = isTriggered ? mBusyFlagTrig : mBusyFlagSelfTrig;
382  // Loop on the event indexes
383  for (auto& evtIdxItem : orderedIndexes) {
384  // All of these boards have the same timestamp
385  GBT gbtEvent;
386  bool busyRaised = false;
387  for (auto& evtPair : evtIdxItem.second) {
388  auto& boardInfo = boards[evtPair.first][evtPair.second];
389  uint8_t triggerId = boardInfo.board.triggerWord;
390  auto elinkId = getElinkId(boardInfo.board);
391 
392  bool isBusy = ((boardInfo.board.statusWord & raw::sREJECTING) != 0);
393  busyRaised |= isBusy;
394  busyFlag[elinkId] = isBusy;
395  if (isBusy && !isTriggered) {
396  // This is a special event that just signals a busy.
397  // Do not add the board to the events to be tested.
398  // Even because this event can have the same IR and triggerWord (0) of a self-triggered event
399  continue;
400  }
401  if (raw::isLoc(boardInfo.board.statusWord)) {
402  gbtEvent.locs.push_back(boardInfo.board);
403  } else {
404  gbtEvent.regs.push_back(boardInfo.board);
405  }
406  if (boardInfo.page >= 0) {
407  if (std::find(gbtEvent.pages.begin(), gbtEvent.pages.end(), boardInfo.page) == gbtEvent.pages.end()) {
408  gbtEvent.pages.push_back(boardInfo.page);
409  }
410  }
411  }
412  if (busyRaised && !isTriggered) {
413  ++mStatistics[2];
414  }
415  ++mStatistics[0];
416  if (!checkEvent(isTriggered, gbtEvent.regs, gbtEvent.locs)) {
417  std::stringstream ss;
418  ss << fmt::format("BCid: 0x{:x} Orbit: 0x{:x}", evtIdxItem.first.bc, evtIdxItem.first.orbit);
419  if (!gbtEvent.pages.empty()) {
420  ss << " [in";
421  for (auto& page : gbtEvent.pages) {
422  ss << std::dec << " page: " << page << " (line: " << 512 * page + 1 << ") ";
423  }
424  ss << "]";
425  }
426  ss << "\n";
427  isOk = false;
428  ss << mEventDebugMsg << "\n";
429  mDebugMsg += ss.str();
430  ++mStatistics[1];
431  }
432  }
433 
434  return isOk;
435 }

437 bool GBTRawDataChecker::process(gsl::span<const LocalBoardRO> localBoards, gsl::span<const ROFRecord> rofRecords, gsl::span<const ROFRecord> pageRecords)
438 {
440  mDebugMsg.clear();
441 
442  // Fill board information
443  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
444  if (rofIt->interactionRecord.orbit == 0xffffffff) {
445  // Protection for event with orbit 0
446  continue;
447  }
448  for (auto locIt = localBoards.begin() + rofIt->firstEntry; locIt != localBoards.begin() + rofIt->firstEntry + rofIt->nEntries; ++locIt) {
449  // Find what page this event corresponds to.
450  // This is useful for debugging.
451  long int page = -1;
452  for (auto& rofPage : pageRecords) {
453  if (rofIt->firstEntry >= rofPage.firstEntry && rofIt->firstEntry < rofPage.firstEntry + rofPage.nEntries) {
454  page = rofPage.interactionRecord.orbit;
455  break;
456  }
457  }
458 
459  // Store the information per local board.
460  // The information should be already ordered in time
461  auto id = getElinkId(*locIt);
462  auto& elinkVec = (locIt->triggerWord == 0) ? mBoardsSelfTrig[id] : mBoardsTrig[id];
463  elinkVec.push_back({*locIt, rofIt->interactionRecord, page});
464 
465  if (locIt->triggerWord == 0) {
466  continue;
467  }
468 
469  // Keep track of the busy
470  if (locIt->statusWord & raw::sREJECTING) {
471  auto& selfVec = mBoardsSelfTrig[id];
472  auto board = *locIt;
473  board.triggerWord = 0;
474  auto ir = rofIt->interactionRecord;
475  if (id >= crateparams::sMaxNBoardsInLink) {
476  uint16_t delayRegLocal = mElectronicsDelay.regToLocal;
477  if (rofIt->interactionRecord.bc < delayRegLocal) {
478  ir -= (constants::lhc::LHCMaxBunches - mResetVal - 1);
479  }
480  ir -= delayRegLocal;
481  }
482  selfVec.push_back({*locIt, ir, page});
483  }
484 
485  // Keep track of the trigger chosen for synchronisation
486  if (locIt->triggerWord & mSyncTrigger) {
487  mTrigEvents[rofIt->interactionRecord] |= (1 << id);
488  mResetVal = rofIt->interactionRecord.bc;
489  }
490 
491  // Compute the masks
492  if (locIt->triggerWord & raw::sSOX) {
493  if (raw::isLoc(locIt->statusWord)) {
494  auto maskItem = mMasks.find(locIt->boardId);
495  // Check if we have already a mask for this
496  if (maskItem == mMasks.end()) {
497  // If not, read the map
498  auto& mask = mMasks[locIt->boardId];
499  for (int ich = 0; ich < 4; ++ich) {
500  mask.patternsBP[ich] = locIt->patternsBP[ich];
501  mask.patternsNBP[ich] = locIt->patternsNBP[ich];
502  }
503  }
504  }
505  }
506  } // loop on local boards
507  } // loop on ROF records
508 
509  return runCheckEvents(getLastCompleteTrigEvent());
510 }
