216  bool StartObject()
217  {
218  enter("START_OBJECT");
219  if (in(State::IN_START)) {
220  push(State::IN_EXECUTION);
221  } else if (in(State::IN_DATAPROCESSORS)) {
222  push(State::IN_DATAPROCESSOR);
223  dataProcessors.push_back(DataProcessorSpec{});
224  } else if (in(State::IN_DATAPROCESSOR)) {
225  dataProcessors.push_back(DataProcessorSpec{});
226  } else if (in(State::IN_INPUTS)) {
227  push(State::IN_INPUT);
228  inputHasSubSpec = false;
229  } else if (in(State::IN_OUTPUTS)) {
230  push(State::IN_OUTPUT);
231  outputHasSubSpec = false;
232  } else if (in(State::IN_OPTIONS)) {
233  push(State::IN_OPTION);
234  } else if (in(State::IN_INPUT_OPTIONS)) {
235  push(State::IN_OPTION);
236  } else if (in(State::IN_WORKFLOW_OPTIONS)) {
237  push(State::IN_OPTION);
238  } else if (in(State::IN_METADATA)) {
239  push(State::IN_METADATUM);
240  metadata.push_back(DataProcessorInfo{});
241  } else if (in(State::IN_METADATUM)) {
242  metadata.push_back(DataProcessorInfo{});
243  }
244  return true;
245  }
246 
247  bool EndObject(SizeType memberCount)
248  {
249  enter("END_OBJECT");
250  if (in(State::IN_INPUT)) {
251  if (inputHasSubSpec) {
252  dataProcessors.back().inputs.push_back(InputSpec(binding, origin, description, subspec, lifetime, inputOptions));
253  } else {
254  dataProcessors.back().inputs.push_back(InputSpec(binding, {origin, description}, lifetime, inputOptions));
255  }
256  inputOptions.clear();
257  inputHasSubSpec = false;
258  } else if (in(State::IN_OUTPUT)) {
259  if (outputHasSubSpec) {
260  dataProcessors.back().outputs.push_back(OutputSpec({binding}, origin, description, subspec, lifetime));
261  } else {
262  dataProcessors.back().outputs.push_back(OutputSpec({binding}, {origin, description}, lifetime));
263  }
264  outputHasSubSpec = false;
265  } else if (in(State::IN_OPTION)) {
266  std::unique_ptr<ConfigParamSpec> opt{nullptr};
267 
268  using HelpString = ConfigParamSpec::HelpString;
269  std::stringstream is;
270  is.str(optionDefault);
271  switch (optionType) {
272  case VariantType::String:
273  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, optionDefault.c_str(), HelpString{optionHelp});
274  break;
275  case VariantType::Int:
276  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stoi(optionDefault, nullptr), HelpString{optionHelp});
277  break;
278  case VariantType::Int64:
279  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stol(optionDefault, nullptr), HelpString{optionHelp});
280  break;
281  case VariantType::Float:
282  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stof(optionDefault, nullptr), HelpString{optionHelp});
283  break;
284  case VariantType::Double:
285  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stod(optionDefault, nullptr), HelpString{optionHelp});
286  break;
287  case VariantType::Bool:
288  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, (bool)std::stoi(optionDefault, nullptr), HelpString{optionHelp});
289  break;
290  case VariantType::ArrayInt:
291  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayInt>(is), HelpString{optionHelp});
292  break;
293  case VariantType::ArrayFloat:
294  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayFloat>(is), HelpString{optionHelp});
295  break;
296  case VariantType::ArrayDouble:
297  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayDouble>(is), HelpString{optionHelp});
298  break;
299  // case VariantType::ArrayBool:
300  // opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayBool>(is), HelpString{optionHelp});
301  // break;
302  case VariantType::ArrayString:
303  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayString>(is), HelpString{optionHelp});
304  break;
305  case VariantType::Array2DInt:
306  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DInt>(is), HelpString{optionHelp});
307  break;
308  case VariantType::Array2DFloat:
309  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DFloat>(is), HelpString{optionHelp});
310  break;
311  case VariantType::Array2DDouble:
312  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DDouble>(is), HelpString{optionHelp});
313  break;
314  case VariantType::LabeledArrayInt:
315  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayInt>(is), HelpString{optionHelp});
316  break;
317  case VariantType::LabeledArrayFloat:
318  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayFloat>(is), HelpString{optionHelp});
319  break;
320  case VariantType::LabeledArrayDouble:
321  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayDouble>(is), HelpString{optionHelp});
322  break;
323  default:
324  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, optionDefault, HelpString{optionHelp});
325  }
326  // Depending on the previous state, push options to the right place.
327  if (previousIs(State::IN_OPTIONS)) {
328  dataProcessors.back().options.push_back(*opt);
329  } else if (previousIs(State::IN_WORKFLOW_OPTIONS)) {
330  metadata.back().workflowOptions.push_back(*opt);
331  } else if (previousIs(State::IN_INPUT_OPTIONS)) {
332  inputOptions.push_back(*opt);
333  } else {
334  assert(false);
335  }
336  }
337  pop();
338  return true;
339  }

374  pop();
375  }
376  pop();
377  return true;

450  bool String(const Ch* str, SizeType length, bool copy)
451  {
452  enter("STRING");
453  enter(str);
454  auto s = std::string(str, length);
455  if (in(State::IN_DATAPROCESSOR_NAME)) {
456  assert(dataProcessors.size());
457  dataProcessors.back().name = s;
458  } else if (in(State::IN_METADATUM_NAME)) {
459  assert(metadata.size());
460  metadata.back().name = s;
461  } else if (in(State::IN_METADATUM_EXECUTABLE)) {
462  assert(metadata.size());
463  metadata.back().executable = s;
464  } else if (in(State::IN_INPUT_BINDING)) {
465  binding = s;
466  } else if (in(State::IN_INPUT_ORIGIN)) {
467  origin.runtimeInit(s.c_str(), std::min(s.size(), 16UL));
468  } else if (in(State::IN_INPUT_DESCRIPTION)) {
469  description.runtimeInit(s.c_str(), std::min(s.size(), 16UL));
470  } else if (in(State::IN_OUTPUT_BINDING)) {
471  binding = s;
472  } else if (in(State::IN_OUTPUT_ORIGIN)) {
473  origin.runtimeInit(s.c_str(), std::min(s.size(), 16UL));
474  } else if (in(State::IN_OUTPUT_DESCRIPTION)) {
475  description.runtimeInit(s.c_str(), std::min(s.size(), 16UL));
476  } else if (in(State::IN_OPTION_NAME)) {
477  optionName = s;
478  } else if (in(State::IN_OPTION_TYPE)) {
479  optionType = (VariantType)std::stoi(s, nullptr);
480  } else if (in(State::IN_OPTION_DEFAULT)) {
481  optionDefault = s;
482  } else if (in(State::IN_OPTION_HELP)) {
483  optionHelp = s;
484  } else if (in(State::IN_METADATUM_ARG)) {
485  metadata.back().cmdLineArgs.push_back(s);
486  // This is in an array, so we do not actually want to
487  // exit from the state.
488  push(State::IN_METADATUM_ARG);
489  } else if (in(State::IN_METADATUM_CHANNEL)) {
490  metadata.back().channels.push_back(s);
491  // This is in an array, so we do not actually want to
492  // exit from the state.
493  push(State::IN_METADATUM_CHANNEL);
494  }
495  pop();
496  return true;
497  }

549  State pop()
550  {
551  if (states.empty()) {
552  states.push_back(State::IN_ERROR);
553  return State::IN_ERROR;
554  }

594 void WorkflowSerializationHelpers::import(std::istream& s,
595  std::vector<DataProcessorSpec>& workflow,
596  std::vector<DataProcessorInfo>& metadata)
597 {
598  // Skip any line which does not start with '{'
599  // If we do not find a starting {, we simply assume that no workflow
600  // was actually passed on the PIPE.
601  // FIXME: not particularly resilient, but works for now.
602  // FIXME: this will fail if { is found at char 1024.
603  char buf[1024];
604  while (s.peek() != '{') {
605  if (s.eof()) {
606  return;
607  }
608  if (s.fail() || s.bad()) {
609  throw std::runtime_error("Malformatted input workflow");
610  }
611  s.getline(buf, 1024, '\n');
612  }
613  rapidjson::Reader reader;
614  rapidjson::IStreamWrapper isw(s);
615  WorkflowImporter importer{workflow, metadata};
616  bool ok = reader.Parse(isw, importer);
617  if (ok == false) {
618  throw std::runtime_error("Error while parsing serialised workflow");
619  }
620 }

641  w.Key("inputs");

659  if (subSpec.has_value()) {

661  w.Uint64(*subSpec);
662  }
663  w.Key("lifetime");
664  w.Uint((int)input.lifetime);
665  if (input.metadata.empty() == false) {
666  w.Key("metadata");
667  w.StartArray();
668  for (auto& metadata : input.metadata) {
669  w.StartObject();
670  w.Key("name");
671  w.String(metadata.name.c_str());
672  auto s = std::to_string(int(metadata.type));
673  w.Key("type");
674  w.String(s.c_str());

678  w.String(oss.str().c_str());
679  w.Key("help");
680  w.String(metadata.help.c_str());
681  w.EndObject();
682  }
683  w.EndArray();
684  }
685  w.EndObject();
686  }
687  w.EndArray();
688 
689  w.Key("outputs");
690  w.StartArray();
691  for (auto& output : processor.outputs) {
692  w.StartObject();
693  w.Key("binding");
694  if (output.binding.value.empty()) {
695  auto autogenerated = DataSpecUtils::describe(output);
696  w.String(autogenerated.c_str());
697  } else {
698  w.String(output.binding.value.c_str());
699  }
700  ConcreteDataTypeMatcher dataType = DataSpecUtils::asConcreteDataTypeMatcher(output);
701  w.Key("origin");
702  w.String(dataType.origin.str, strnlen(dataType.origin.str, 4));
703  w.Key("description");
704  w.String(dataType.description.str, strnlen(dataType.description.str, 16));
705  // FIXME: this will have to change once we introduce wildcards for
706  // OutputSpec
707  auto subSpec = DataSpecUtils::getOptionalSubSpec(output);
708  if (subSpec.has_value()) {
709  w.Key("subspec");
710  w.Uint64(*subSpec);
711  }
712  w.Key("lifetime");
713  w.Uint((int)output.lifetime);
714  w.EndObject();
715  }
716  w.EndArray();
717 
718  w.Key("options");
719  w.StartArray();
720  for (auto& option : processor.options) {
721  if (option.name == "start-value-enumeration" || option.name == "end-value-enumeration" || option.name == "step-value-enumeration") {
722  continue;
723  }
724  w.StartObject();
725  w.Key("name");
726  w.String(option.name.c_str());
727  auto s = std::to_string(int(option.type));
728  w.Key("type");
729  w.String(s.c_str());

731  switch (option.type) {
732  case VariantType::ArrayInt:
733  case VariantType::ArrayFloat:
734  case VariantType::ArrayDouble:
735  case VariantType::ArrayBool:
736  case VariantType::ArrayString:
737  case VariantType::Array2DInt:
738  case VariantType::Array2DFloat:
739  case VariantType::Array2DDouble:
740  case VariantType::LabeledArrayInt:
741  case VariantType::LabeledArrayFloat:
742  case VariantType::LabeledArrayDouble:
743  VariantJSONHelpers::write(oss, option.defaultValue);
744  break;
745  default:
746  oss << option.defaultValue;
747  break;
748  }
749  w.Key("defaultValue");
750  w.String(oss.str().c_str());
751  w.Key("help");
752  w.String(option.help.c_str());
753  w.EndObject();
754  }
755  w.EndArray();
756  w.Key("rank");

758  w.Key("nSlots");
759  w.Int(processor.nSlots);
760  w.Key("inputTimeSliceId");

764 
765  w.EndObject();
766  }
767  w.EndArray();
768 
769  w.Key("metadata");
770  w.StartArray();
771  for (auto& info : metadata) {
772  w.StartObject();
773  w.Key("name");
774  w.String(info.name.c_str());
775  w.Key("executable");
776  w.String(info.executable.c_str());

778  w.StartArray();

780  w.String(arg.c_str());
781  }
782  w.EndArray();
783  w.Key("workflowOptions");
784  w.StartArray();
785  for (auto& option : info.workflowOptions) {
786  w.StartObject();
787  w.Key("name");
788  w.String(option.name.c_str());
789  auto s = std::to_string(int(option.type));
790  w.Key("type");

799  }
