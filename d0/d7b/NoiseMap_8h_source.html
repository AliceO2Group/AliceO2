54  float getNoiseLevel(int chip, int row, int col) const
55  {
56  if (chip > mNoisyPixels.size()) {
57  return 0;
58  }
59  auto key = row * 1024 + col;
60  const auto keyIt = mNoisyPixels[chip].find(key);
61  if (keyIt != mNoisyPixels[chip].end()) {

64  return 0;

76  int dumpAboveThreshold(int t = 3) const
77  {
78  int n = 0;
79  auto chipID = mNoisyPixels.size();
80  while (chipID--) {
81  const auto& map = mNoisyPixels[chipID];
82  for (const auto& pair : map) {
83  if (pair.second <= t) {
84  continue;
85  }
86  n++;
87  auto key = pair.first;
88  auto row = key / 1024;
89  auto col = key % 1024;
90  std::cout << "chip, row, col, noise: " << chipID << ' ' << row << ' ' << col << ' ' << pair.second << '\n';
91  }
92  }
93  return n;
94  }
95  int dumpAboveProbThreshold(float p = 1e-7) const
96  {
97  return dumpAboveThreshold(p * mNumOfStrobes);
98  }
99 
100  void applyProbThreshold(float t, long int n)
101  {
102  // Remove from the maps all pixels with the firing probability below the threshold
103  mProbThreshold = t;
104  mNumOfStrobes = n;
105  for (auto& map : mNoisyPixels) {
106  for (auto it = map.begin(); it != map.end();) {
107  float prob = float(it->second) / mNumOfStrobes;
108  if (prob < mProbThreshold) {
109  it = map.erase(it);
110  } else {
111  ++it;
112  }
113  }
114  }
115  }
116  float getProbThreshold() const { return mProbThreshold; }
117  long int getNumOfStrobes() const { return mNumOfStrobes; }
118 
119  bool isNoisy(int chip, int row, int col) const
120  {
121  if (chip > mNoisyPixels.size()) {
122  return false;
123  }
124  auto key = row * 1024 + col;
125  const auto keyIt = mNoisyPixels[chip].find(key);
126  if (keyIt != mNoisyPixels[chip].end()) {
127  return true;
128  }
129  return false;
130  }
